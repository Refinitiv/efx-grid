Simple Image Formatter

This formatter creates an image from the column's field by default.

Specific options

There are no specific options for this formatter.

Example

atlas-blotter {
    height: 320px;
}

<atlas-blotter id="grid"></atlas-blotter>

import { halo } from './theme-loader.js'; // This line is only required for demo purpose. It is not relevant for your application.
await halo(); // This line is only required for demo purpose. It is not relevant for your application.

/* ---------------------------------- Note ----------------------------------
  DataGenerator, Formatters and extensions are exposed to global scope
  in the bundle file to make it easier to create live examples.

var fields = ["image", "s_image"];
var records = new Array(40);
for(var i = 0; i < 40; ++i) {
    var record = records[i] = {};
    record[fields[0]] = "https://dummyimage.com/40x40&text=" + i;
    record[fields[1]] = "https://dummyimage.com/15x15&text=" + i;
}
var configObj = {
    sorting: {
        sortableColumns: true
    },
    columns: [
        {
            name: "Image",
            field: fields[0],
            alignment: "c",
            binding: tr.SimpleImageFormatter.create()
        },
        {
            name: "Small Image",
            field: fields[1],
            alignment: "c",
            binding: tr.SimpleImageFormatter.create()
        }
    ],
    staticDataRows: records
};

var grid = document.getElementById("grid");
grid.config = configObj;
