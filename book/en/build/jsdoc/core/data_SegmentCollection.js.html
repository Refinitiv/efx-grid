import Segment from "./Segment.js";

/** @constructor
*/
var SegmentCollection = function() {
	this._segments = {};
	this._childToSegmentId = {};
};


/** @type {!Object.<string, Segment>}
* @private
*/
SegmentCollection.prototype._segments;
/** @type {number}
* @private
*/
SegmentCollection.prototype._segmentCount = 0;
/** @type {Object}
* @private
*/
SegmentCollection.prototype._collapsedRids = null;
/** @type {boolean}
* @private
*/
SegmentCollection.prototype._dirtyCollapsingState = false;
/** @type {!Object.<string, string>}
* @private
*/
SegmentCollection.prototype._childToSegmentId;


/** @public
* @param {string} rid
* @return {boolean} Returns true if there is any change. Otherwise, returns false
*/
SegmentCollection.prototype.addSegment = function(rid) {
	if(rid && !this._segments[rid]) {
		if(this._childToSegmentId[rid]) {
			console.log("child of a segment cannot be set as a segment separator");
		}
		this._segments[rid] = new Segment(rid);
		++this._segmentCount;
		return true;
	}
	return false;
};
/** @public
* @param {Array.<string>} rids
* @return {boolean} Returns true if there is any change. Otherwise, returns false
*/
SegmentCollection.prototype.addSegments = function(rids) {
	var rowIds = Array.isArray(rids) ? rids : [rids];
	var rowCount = rowIds.length;
	var dirty = 0;
	for(var i = 0; i < rowCount; ++i) {
		dirty |= this.addSegment(rowIds[i]);
	}
	return dirty ? true : false;
};
/** @public
* @param {string} rid
* @return {boolean}
*/
SegmentCollection.prototype.containsSegment = function(rid) {
	return this._segments[rid] ? true : false;
};
/** @public
* @param {string} rid
* @return {boolean} Returns true if there is any change. Otherwise, returns false
*/
SegmentCollection.prototype.removeSegment = function(rid) {
	var segment = this._segments[rid];
	if(segment) {
		if(segment.isCollapsed()) {
			this._dirtyCollapsingState = true;
		}
		segment.removeAllChildren(this._childToSegmentId);
		delete this._segments[rid]; // Slow
		--this._segmentCount;
		return true;
	}
	return false;
};
/** @public
* @return {boolean} Returns true if there is any change. Otherwise, returns false
*/
SegmentCollection.prototype.removeAllSegments = function() {
	if(this._segmentCount) {
		this._segments = {};
		this._childToSegmentId = {};
		this._segmentCount = 0;
		this._dirtyCollapsingState = true;
		return true;
	}
	return false;
};

/** @public
* @return {number}
*/
SegmentCollection.prototype.getSegmentCount = function() {
	return this._segmentCount;
};
/** @public
* @param {string} rid
* @return {Segment}
*/
SegmentCollection.prototype.getSegment = function(rid) {
	return this._segments[rid] || null;
};
/** @public
* @return {Object.<string, Segment>}
*/
SegmentCollection.prototype.getSegments = function() {
	return this._segmentCount ? this._segments : null; // WARNING: Return private member
};
/** @public
* @return {!Array.<string>}
*/
SegmentCollection.prototype.getSegmentIds = function() {
	return Object.keys(this._segments);
};


/** @public
* @param {string} segmentId
* @param {boolean=} bool
* @return {boolean} Returns true if there is any change. Otherwise, returns false
*/
SegmentCollection.prototype.collapseSegment = function(segmentId, bool) {
	var segment = this._segments[segmentId];
	if(segment) {
		var dirty = segment.collapse(bool);
		if(dirty) {
			this._dirtyCollapsingState = true;
			return true;
		}
	}
	return false;
};
/** @public
* @param {string} segmentId
* @param {boolean=} bool
* @return {boolean}
*/
SegmentCollection.prototype.expandSegment = function(segmentId, bool) {
	return this.collapseSegment(segmentId, bool === false);
};
/** @public
* @return {boolean} Returns true if there is any change. Otherwise, returns false
*/
SegmentCollection.prototype.expandAllSegments = function() {
	if(this._segmentCount) {
		var segmentSeparators = this._segments;
		var dirty = 0;
		for(var rid in segmentSeparators) {
			dirty |= segmentSeparators[rid].expand();
		}
		if(dirty) {
			this._dirtyCollapsingState = true;
			return true;
		}
	}
	return false;
};
/** @public
* @param {string} segmentId
* @return {boolean} Returns true if there is any change. Otherwise, returns false
*/
SegmentCollection.prototype.isCollapsedSegment = function(segmentId) {
	var segment = this._segments[segmentId];
	if(segment) {
		return segment.isCollapsed();
	}
	return false;
};
/** @public
* @return {Object}
*/
SegmentCollection.prototype.getCollapsedRows = function() {
	if(this._dirtyCollapsingState) {
		this._dirtyCollapsingState = false;
		var collapsedRids = null;
		var count = 0;
		if(this._segmentCount) {
			var segmentSeparators = this._segments;
			collapsedRids = {};
			for(var rid in segmentSeparators) {
				if(segmentSeparators[rid].isCollapsed()) {
					segmentSeparators[rid].getCollapsingStates(collapsedRids);
					++count;
				}
			}
		}
		this._collapsedRids = count ? collapsedRids : null;
	}
	return this._collapsedRids;
};

/** @public
* @param {string} segmentId
* @param {string} rid
* @return {boolean} Returns true if there is any change. Otherwise, returns false
*/
SegmentCollection.prototype.addSegmentChild = function(segmentId, rid) {
	var segment = this._segments[segmentId];
	if(segment) {
		var dirty = segment.addChild(rid, this._childToSegmentId);
		if(dirty) {
			if(segment.isCollapsed()) {
				this._dirtyCollapsingState = true;
			}
			return true;
		}
	}
	return false;
};
/** @public
* @param {string} segmentId
* @param {Array.<string>} rids
* @return {boolean} Returns true if there is any change. Otherwise, returns false
*/
SegmentCollection.prototype.addSegmentChildren = function(segmentId, rids) {
	var segment = this._segments[segmentId];
	if(segment) {
		var dirty = segment.addChildren(rids, this._childToSegmentId);
		if(dirty) {
			if(segment.isCollapsed()) {
				this._dirtyCollapsingState = true;
			}
			return true;
		}
	}
	return false;
};
/** @public
* @param {string} segmentId

* @return {boolean} Returns true if there is any change. Otherwise, returns false
*/
SegmentCollection.prototype.containsSegmentChild = function(segmentId, rid) {
	var segment = this._segments[segmentId];
	if(segment) {
		return segment.containsChild(rid);
	}
	return false;
};

*/
SegmentCollection.prototype.removeSegmentChildren = function(segmentId, rids) {
	var segment = this._segments[segmentId];
	if(segment) {
		var dirty = segment.removeChildren(rids, this._childToSegmentId);
		if(dirty) {

			}
			return true;
		}
	}
	return false;
};
/** @public
* @return {boolean} Returns true if there is any change. Otherwise, returns false
*/
SegmentCollection.prototype.removeAllSegmentChildren = function() {
	this._childToSegmentId = {};
	var segmentSeparators = this._segments;
	var dirtyCollapsingState = this._dirtyCollapsingState;
	var dirty = false;
	for(var rid in segmentSeparators) {
		var segment = this._segments[rid];
		if(segment.removeAllChildren()) {
			dirty = true;
			if(!dirtyCollapsingState && segment.isCollapsed()) {
				dirtyCollapsingState = this._dirtyCollapsingState = true;
			}
		}
	}

	return dirty;
};

/** @public
* @param {string} segmentId
* @param {Array.<string>} rids
*/
SegmentCollection.prototype.fillSegment = function(segmentId, rids) {
	var rowCount = Array.isArray(rids) ? rids.length : 0;
	if(!rowCount) {
		return;
	}
	var segmentSeparators = this._segments;
	var childToSegmentId = this._childToSegmentId;
	var curSegment = segmentSeparators[segmentId];
	if(curSegment) {
		var segmentAt = rids.indexOf(segmentId);
		if(segmentAt >= 0) {
			if(curSegment.isCollapsed()) {
				this._dirtyCollapsingState = true;
			}
			curSegment.removeAllChildren(childToSegmentId);
			for(var r = segmentAt + 1; r < rowCount; ++r) {
				var rid = rids[r];
				var segmentSeparator = segmentSeparators[rid];
				if(segmentSeparator) {
					break;
				}
				curSegment.addChild(rid, childToSegmentId);
			}
		}
	}
};
/** @public
* @param {Array.<string>} rids
*/
SegmentCollection.prototype.fillSegments = function(rids) {
	var rowCount = Array.isArray(rids) ? rids.length : 0;
	this._childToSegmentId = {};
	var segmentSeparators = this._segments;
	var childToSegmentId = this._childToSegmentId;
	var curSegment = null;
	for(var r = 0; r < rowCount; ++r) {
		var rid = rids[r];
		var segmentSeparator = segmentSeparators[rid];
		if(segmentSeparator) {
			curSegment = segmentSeparator;
			if(curSegment.isCollapsed()) {
				this._dirtyCollapsingState = true;
			}
			curSegment.removeAllChildren();
		} else if(curSegment) {
			curSegment.addChild(rid, childToSegmentId);
		}
	}
};
/** @public
* @param {!Array.<string>} rids Array of row ids
* @return {Array.<number>} Returns Array of segment values, if there are at least two segments, otherwise returns null
*/
SegmentCollection.prototype.getSegmentValues = function(rids) {
	var rowCount = rids ? rids.length : 0;
	if(!rowCount) {
		return null;
	}
	var segmentSeparators = this._segments;
	var childToSegmentId = this._childToSegmentId;
	var curSegment = null;
	var segmentValues = new Array(rowCount);
	var curSegmentCount = 0;
	var segmentVal = 0;
	var r = 0;
	var childToIndex = {};
	while(r < rowCount) {
		var rid = rids[r];
		if(segmentSeparators[rid]) {
			++curSegmentCount;
			curSegment = segmentSeparators[rid];
			segmentVal += 100;
			segmentValues[r] = segmentVal;
			curSegment.setValue(segmentVal);
			if(curSegmentCount >= this._segmentCount) {
				++r;
				break;
			}
		} else {
			if(curSegmentCount) {
				segmentValues[r] = segmentVal + 10;
			} else {
				segmentValues[r] = 0;
			}
			if(childToSegmentId[rid]) {
				childToIndex[rid] = r;
			}
		}
		++r;
	}
	if(curSegmentCount) {
		while(r < rowCount) {
			rid = rids[r];
			segmentValues[r] = segmentVal + 10;
			if(childToSegmentId[rid]) {
				childToIndex[rid] = r;
			}
			++r;
		}
		for(var childId in childToIndex) {
			curSegment = segmentSeparators[childToSegmentId[childId]];
			segmentValues[childToIndex[childId]] = curSegment.getValue() + 1;
		}
	}

	return curSegmentCount ? segmentValues : null;
};


export default SegmentCollection;
export { SegmentCollection };
