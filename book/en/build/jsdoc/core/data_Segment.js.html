/** @constructor
* @param {string} rid
*/
var Segment = function(rid) {
	this._rid = rid;
	this._children = {};
};


/** @type {string}
* @private
*/
Segment.prototype._rid;
/** @type {!Object}
* @private
*/
Segment.prototype._children;
/** @type {number}
* @private
*/
Segment.prototype._childCount = 0;
/** @type {boolean}
* @private
*/
Segment.prototype._collapsed = false;
/** @type {number}
* @private
*/
Segment.prototype._value = 0;


/** @public
* @return {string}
*/
Segment.prototype.getId = function() {
	return this._rid;
};
/** @public
* @param {string} rid
* @param {Object=} objMap
* @return {boolean}
*/
Segment.prototype.addChild = function(rid, objMap) {
	if(rid && !this._children[rid]) {
		if(objMap) {
			objMap[rid] = this._rid;
		}
		this._children[rid] = 1;
		++this._childCount;
		return true;
	}
	return false;
};
/** @public
* @param {Array.<string>} rids
* @param {Object=} objMap
* @return {boolean}
*/
Segment.prototype.addChildren = function(rids, objMap) {
	var rowIds = Array.isArray(rids) ? rids : [rids];
	var rowCount = rowIds.length;
	var dirty = 0;
	for(var i = 0; i < rowCount; ++i) {
		dirty |= this.addChild(rowIds[i], objMap);
	}
	return dirty ? true : false;
};
/** @public
* @param {string} rid
* @return {boolean}
*/
Segment.prototype.containsChild = function(rid) {
	return this._children[rid] ? true : false;
};
/** @public
* @param {string} rid
* @param {Object=} objMap
* @return {boolean}
*/
Segment.prototype.removeChild = function(rid, objMap) {
	if(this._childCount && this._children[rid]) {
		if(objMap) {
			delete objMap[rid];
		}
		delete this._children[rid]; // Slow
		--this._childCount;
		return true;
	}
	return false;
};
/** @public
* @param {Array.<string>} rids
* @param {Object=} objMap
* @return {boolean}
*/
Segment.prototype.removeChildren = function(rids, objMap) {
	if(!this._childCount) {
		return false;
	}
	var rowIds = Array.isArray(rids) ? rids : [rids];
	var rowCount = rowIds.length;
	var dirty = 0;
	for(var i = 0; i < rowCount; ++i) {
		dirty |= this.removeChild(rowIds[i], objMap);
	}
	return dirty ? true : false;
};
/** @public
* @param {Object=} objMap
* @return {boolean}
*/
Segment.prototype.removeAllChildren = function(objMap) {
	if(this._childCount) {
		if(objMap) {
			var chdr = this._children;
			for(var rid in chdr) {
				if(objMap[rid]) {
					delete objMap[rid];
				}
			}
		}
		this._children = {};
		this._childCount = 0;
		return true;
	}
	return false;
};
/** @public
* @return {!Array.<string>}
*/
Segment.prototype.getChildIds = function() {
	return this._childCount ? Object.keys(this._children) : [];
};


/** @public
* @param {boolean=} bool
* @return {boolean}
*/
Segment.prototype.collapse = function(bool) {
	bool = (bool !== false);
	if(this._collapsed !== bool) {
		this._collapsed = bool;
		return true;
	}
	return false;
};
/** @public
* @param {boolean=} bool
* @return {boolean}
*/
Segment.prototype.expand = function(bool) {
	return this.collapse(bool === false);
};
/** @public
* @return {boolean}
*/
Segment.prototype.isCollapsed = function() {
	return this._collapsed;
};
/** @public
* @param {Object=} objMap
* @return {!Object}
*/
Segment.prototype.getCollapsingStates = function(objMap) {
	if(!objMap) {
		objMap = {};
	}
	if(this._childCount) {
		var chdr = this._children;
		var collapsed = this._collapsed;
		for(var rid in chdr) {
			objMap[rid] = collapsed;
		}
	}
	return objMap;
};
/** @public
* @return {number}
*/
Segment.prototype.getValue = function() {
	return this._value;
};
/** @public
* @param {number} val
*/
Segment.prototype.setValue = function(val) {
	this._value = val;
};


export default Segment;
export { Segment };
