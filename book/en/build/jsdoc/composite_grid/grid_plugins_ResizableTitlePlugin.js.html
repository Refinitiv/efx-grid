goog.provide("tr.grid.ResizableTitlePlugin");

goog.require("tr.res");

/** @constructor
 * @extends {tr.grid.Plugin}
 * @implements {tr.grid.IPlugin}
 */
tr.grid.ResizableTitlePlugin = function () {
	var _t = this;
	_t._onPreSectionRender = _t._onPreSectionRender.bind(_t);
	_t._onColumnAdded = _t._onColumnAdded.bind(_t);

	_t._onDrag = _t._onDrag.bind(_t);
	_t._onDragEnd = _t._onDragEnd.bind(_t);
	_t._updateHitBoxes = _t._updateHitBoxes.bind(_t);

	_t._hosts = [];
	
	var guideline = tr.grid.ResizableTitlePlugin._guideline;
	if (!guideline) {
		guideline = tr.grid.ResizableTitlePlugin._guideline = document.createElement("svg");
		guideline.className = "tr-guideline";
		guideline.style.position = "absolute";
		guideline.style.pointerEvents = "none";
		guideline.style.width = "3px";
		guideline.style.zIndex = "1";
		
		var guidelineColor = tr.grid.ResizableTitlePlugin._guidelineColor;
		if (guidelineColor) {
			guideline.style.backgroundColor = guidelineColor;
		}
	}
	
	_t._addEvents(
		"columnResized",
		"resizerDblClicked",
		"dragEnd",
		"dragStart"
	);
};
goog.inherits(tr.grid.ResizableTitlePlugin, tr.grid.Plugin);

/** @override */
tr.grid.ResizableTitlePlugin.prototype.getName = function () {
	return "ResizableTitlePlugin"; // Read Only
};
/** @override */
tr.grid.ResizableTitlePlugin.prototype.initialize = function (host) {
	if (this._hosts.indexOf(host) >= 0) { return; }

	this._hosts.push(host);
	if (this._hosts.length === 1) { 
		host.listen(tr.res.columnAdded, this._onColumnAdded);
	}
	host.listen("preSectionRender", this._onPreSectionRender);

	this._requestHitBoxUpdate();
};
/** @override */
tr.grid.ResizableTitlePlugin.prototype.unload = function (host) {
	var at = this._hosts.indexOf(host);
	if (at < 0) { return; }

	this._hosts.splice(at, 1);
	host.unlisten("preSectionRender", this._onPreSectionRender);
	host.unlisten(tr.res.columnAdded, this._onColumnAdded);

	this._removeHitBoxes(host);
	
	if (!this._hosts.length) {
		if (this._timerId) {
			clearTimeout(this._timerId);
			this._timerId = 0;
		}
	}
};

/** @public
* @param {Object=} options Composite grid object model
*/
tr.grid.ResizableTitlePlugin.prototype.config = function(options) {
	if (!options) {
		return;
	}
	
	var columnResizability = options["columnResizability"];
	if (columnResizability != null) {
		this._resizability = columnResizability ? true : false;
	}
	
	var extOptions = options["resizableTitle"];
	if (extOptions) {
		var func = extOptions["columnResized"];
		if (typeof func == "function") {
			this.listen("columnResized", func);
		}
		func = extOptions["resizerDblClicked"];
		if (typeof func == "function") {
			this.listen("resizerDblClicked", func);
		}
	}
	
	var columns = options["columns"];
	if (!columns) {
		return;
	}
	
	var colCount = columns.length;
	for (var c = 0; c < colCount; ++c) {
		this._setColumnOptions(c, columns[c]);

	}
};
/** @private
* @param {number} colIndex
* @param {Object=} options 
*/
tr.grid.ResizableTitlePlugin.prototype._setColumnOptions = function(colIndex, options) {
	var bool = options["resizable"];
	if (bool != null) {
		this.disableResizing(colIndex, !bool);
	}
};

/**
* @public
* @param {number} colIndex
* @param {boolean=} disabled
*/
tr.grid.ResizableTitlePlugin.prototype.disableResizing = function(colIndex, disabled) {
	var dirty = false;
	var enabled = disabled === false;
	var host = this._hosts[0];
	if (host && colIndex >= 0) {
		var colData = host.newColumnData(colIndex);
		var val = colData["resizable"];
		if (val !== enabled) {
			colData["resizable"] = enabled;
			dirty = true;
		}
	}
	
	if (dirty) {
		this._requestHitBoxUpdate();
	}
};

/** This function is used for set the size of multiple columns
 * @public
 * @param {Array.<number>} cols
 * @param {Array.<number>|number} width
 * @param {boolean=} opt_scalable If not defined false value is used instead
 */
tr.grid.ResizableTitlePlugin.prototype.setColumnWidths = function (cols, width, opt_scalable) {
	var shouldFreezeLayout = cols.length > 1;
	for (var i = this._hosts.length; --i >= 0;) {
		var host = this._hosts[i];
		var frozen;
		if (shouldFreezeLayout) {
			frozen = host.freezeLayout();
		}
		for (var index = cols.length; --index >= 0;) {
			var colIndex = cols[index];
			var colWidth = (typeof width == "number") ? width : width[index];
			host.setColumnWidth(colIndex, colWidth, opt_scalable); // events will be fired
		}
		if (shouldFreezeLayout) {
			host.freezeLayout(frozen);
		}
	}
};

	if(e["context"]) {
		this._setColumnOptions(e[tr.res.colIndex], e["context"]);
	}
	
	this._requestHitBoxUpdate();
};

	if (e[tr.res.sectionType] === tr.res.title) {
		this._requestHitBoxUpdate();
	}
};

/** @private
* @param {tr.Grid} host
* @param {Event} e
*/
tr.grid.ResizableTitlePlugin.prototype._onDragStart = function (host, e) {
	if (this._dragIndex >= 0) { return; }

	this._dragIndex = this._getResizerIndex(host, e);
	if (this._dragIndex < 0) { return; }

	window.addEventListener("mousemove", this._onDrag, false);
	window.addEventListener("mouseup", this._onDragEnd, false);

	document.body.classList.add("tr-dragging"); // Prevent text selection
	document.body.classList.add("tr-resize-cursor");
	this._draggedGrid = host;
	this._anchorX = e.clientX;
	this._guideX = (this._draggedGrid.getColumnRight(this._dragIndex) - 1) * this._draggedGrid["zoomFactor"];
	this._topLeft = tr.getRelativePosition(this._draggedGrid.getElement(), document.body);
	
	var guideline = tr.grid.ResizableTitlePlugin._guideline;
	if (guideline) {
		guideline.style.top = this._topLeft["y"] + "px";
		guideline.style.left = (this._topLeft["x"] + this._guideX) + "px";
		guideline.style.height = (host.getElement().parentNode.clientHeight) + "px";
		document.body.appendChild(guideline);
	}
	this._dispatch("dragStart", e);
	tr._preventDefault(e);
};
/** @private
* @param {Event} e
*/
tr.grid.ResizableTitlePlugin.prototype._onDrag = function (e) {
	var diff = e.clientX - this._anchorX;
	var width = this._draggedGrid.getColumnWidth(this._dragIndex);
	var minWidth = this._draggedGrid.getMinimumColumnWidth(this._dragIndex);

	if (width + diff < minWidth) {
		diff = minWidth - width;
	}

	var guideline = tr.grid.ResizableTitlePlugin._guideline;
	if (guideline) {
		guideline.style.left = (this._topLeft["x"] + this._guideX + diff) + "px";
	}

	tr._preventDefault(e);
};
/** @private
* @param {Event} e
*/
tr.grid.ResizableTitlePlugin.prototype._onDragEnd = function (e) {
	window.removeEventListener("mousemove", this._onDrag, false);
	window.removeEventListener("mouseup", this._onDragEnd, false);
	document.body.classList.remove("tr-dragging");
	document.body.classList.remove("tr-resize-cursor");

	var diff = e.clientX - this._anchorX;
	
	var minWidth = this._draggedGrid.getMinimumColumnWidth(this._dragIndex);
	var newWidth = this._draggedGrid.getColumnWidth(this._dragIndex) + (diff / this._draggedGrid["zoomFactor"]);
	if (newWidth < minWidth) { newWidth = minWidth; }

	var host = this._hosts[0];
	if (this._keepScaleColumn) {
		var colCount = host.getColumnCount();
		var totalWeight = 0;
		var totalWidth = 0;
		for (var j = 0;j < colCount;++j) {
			var colWidth = host.getColumnWidth(j);
			var colWeight = colWidth > 0 ? host.getColumnLayout().getCustomLaneSize(j) : 0;
			totalWeight += colWeight;
			totalWidth += colWidth;
		}
		var oldWeight = host.getColumnLayout().getCustomLaneSize(this._dragIndex);
		var oldWidth = host.getColumnWidth(this._dragIndex);
		var newWeight = ((newWidth * totalWeight )- (newWidth * oldWeight)) / (totalWidth - newWidth);
		this.setColumnWidths([this._dragIndex], newWeight, true);
		
		this._dispatch("columnResized", { 
			"colIndex": this._dragIndex,
			"width": newWeight,
			"scalability": true
		});
	}else{
		var cols = this._getResizingCols(this._dragIndex);
		this.setColumnWidths(cols, newWidth);
		
		this._dispatch("columnResized", { 
			"colIndex": this._dragIndex,
			"colIndices": cols,
			"width": newWidth
		});
	}

	this._dispatch("dragEnd", e);

	var guideline = tr.grid.ResizableTitlePlugin._guideline;
	if (guideline) {
		if (guideline.parentNode) {
			guideline.parentNode.removeChild(guideline);
		}
	}
	this._anchorX = 0;
	this._guideX = 0;
	this._dragIndex = -1;
	this._draggedGrid = null;
	this._topLeft = null;

	tr._preventDefault(e);
};

/** @private
* @param {tr.Grid} host
* @param {Event} e
*/
tr.grid.ResizableTitlePlugin.prototype._onDblClick = function(host, e){
	var colIndex = this._getResizerIndex(host, e);
	
	var plugin = this._getPlugin("ColumnFitterPlugin");
	if (!plugin) {
		plugin = this._getPlugin("ColumnWidthAdjustingPlugin");
	}
	
	if (plugin) {
		var cols = this._getResizingCols(colIndex); // Get All selected columns at once
		var selectedColCount = cols.length;
		var widths = new Array(selectedColCount);
		for (var i = selectedColCount; --i >= 0; ) {
			var selectedCol = cols[i];
			var fitWidth = this._getColumnFitWidth(selectedCol);
			widths[i] = (fitWidth == null) ? plugin["getAdjustedColumnWidth"](selectedCol) : /** @type{number} */(fitWidth);
		}
		this.setColumnWidths(cols, widths);
		tr._preventDefault(e);
	}
	this._dispatch("resizerDblClicked", {"colIndex": colIndex});
};

tr.grid.ResizableTitlePlugin._stopPropagation = function (e) {
	e.stopPropagation();
};

/** @private
*/
tr.grid.ResizableTitlePlugin.prototype._requestHitBoxUpdate = function () {
	if (!this._timerId) {
		this._timerId = setTimeout(this._updateHitBoxes, 0);
	}
};
/** @private
*/
tr.grid.ResizableTitlePlugin.prototype._updateHitBoxes = function () {
	this._timerId = 0;
	
	var host = this._hosts[0];
	if(!host) { 
		return;
	}
	
	var colCount = host.getColumnCount();
	var resizabilities = new Array(colCount);
	for (var c = 0; c < colCount; ++c) {
		resizabilities[c] = this.isColumnResizable(c);
	}
	
	for (var i = this._hosts.length; --i >= 0;) {
		host = this._hosts[i];
		var titleSects = host.getAllSections("title");
		var titleCount = titleSects.length;
		for (var s = 0; s < titleCount; s++) {
			var titleSect = titleSects[s];
			for (c = 0; c < colCount; ++c) {
				var rowCount = titleSect.getRowCount();
				
				var resizability = resizabilities[c];
				var col = titleSect.getColumn(c);
				
				for (var r = 0; r < rowCount; r++) {
					var cell = col.getCell(r);
					var box = /** @type{Element} */(cell._resizer);
					if (!box) {
						box = cell._resizer = document.createElement("div");
						box.setAttribute("draggable", "true");
						box.className = "resizer";
						box.addEventListener("mousedown", tr.grid.ResizableTitlePlugin._stopPropagation, false);
						box.addEventListener("click", tr.grid.ResizableTitlePlugin._stopPropagation, false);
						box.addEventListener("dragstart", this._onDragStart.bind(this, host), false);
						box.addEventListener("dblclick", this._onDblClick.bind(this, host), false);
						
						var cellElem = cell.getElement();
						cellElem.insertBefore(box, cellElem.firstChild); // Always add box element as the first child
					}
					
					box._disabled = !resizability; // disabled is an internal flag
					box.style.display = resizability ? "" : "none";
				}
			}
		}
	}
};
/** @private
* @param {tr.Grid} host
*/
tr.grid.ResizableTitlePlugin.prototype._removeHitBoxes = function (host) {
	var colCount = host.getColumnCount();
	var titleSects = host.getAllSections("title");
	var titleCount = titleSects.length;
	for (var s = 0; s < titleCount; s++) {
		var titleSect = titleSects[s];
		for (var c = 0; c < colCount; ++c) {
			var rowCount = titleSect.getRowCount();
			var col = titleSect.getColumn(c);
			
			for (var r = 0; r < rowCount; r++) {
				var cell = col.getCell(r);
				if (cell._resizer) {
					cell._resizer = null;
				}
			}
		}
	}
};

/** @private
* @param {tr.Grid} host
* @param {Event} e
*/
tr.grid.ResizableTitlePlugin.prototype._getResizerIndex = function (host, e) {
	var box = e["currentTarget"];
	if (box._disabled) { // disabled is an internal flag
		return -1;
	}

	var pos = host.getRelativePosition(e);
	var colIndex = pos[tr.res.colIndex];
	if (colIndex != null) {
		var colSpan = pos["section"].getCellColSpan(colIndex, pos[tr.res.rowIndex]);
		return colIndex + colSpan - 1;
	}
	return -1;
};

/** @private
 * @param {number} clickIndex
 */
tr.grid.ResizableTitlePlugin.prototype._getResizingCols = function (clickIndex) {
	var csp = this._getPlugin("ColumnSelectionPlugin");
	if (csp && csp["isEnabled"]()) {
		var cols = csp["getSelectedColumns"]();
		if (cols.indexOf(clickIndex) >= 0) { // if clickIndex does not exist in selected columns then return clickIndex.
			return cols;
		}
	}
	return [clickIndex];
};

/** @public
* @param {number} colIndex
* @return {boolean}
*/
tr.grid.ResizableTitlePlugin.prototype.isColumnResizable = function (colIndex) {
	var host = this._hosts[0];
	if (host) {
		if (colIndex < 0 || colIndex >= host.getColumnCount()) {
			return false;
		}
		
		var colData = host.getColumnData(colIndex);
		if (colData) {
			var resizable = colData["resizable"];
			if (resizable != null) {
				return resizable ? true : false;
			}
		}
	}
	return this._resizability; // By default all columns are resizable
};
/** @private
* @param {number} colIndex
* @return {*}
*/
tr.grid.ResizableTitlePlugin.prototype._getColumnFitWidth = function (colIndex) {
	var host = this._hosts[0];
	if (host) {
		var colData = host.getColumnData(colIndex);
		if (colData) {
			return colData["fitWidth"];
		}
	}
	return null;
};
