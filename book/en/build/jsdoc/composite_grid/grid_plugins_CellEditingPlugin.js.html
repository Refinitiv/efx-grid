/** Happens before the opening of text editor. This allow user to disable the editor on case by case basis
* @event tr.grid.CellEditingPlugin#preEditorOpening
* @example
* var cep = new grid.CellEditingPlugin();
* cep.listen("preEditorOpening", function(e) {
*     e["cancel"] = true; // Cancel the editing process
* });
*/

/** Happens after the opening of text editor. Any visual customization can be done at this stage, including the text shown in the editor.
* @event tr.grid.CellEditingPlugin#editorOpened
*/

/** Fired after user text has been entered and the text editor has been closed
* @event tr.grid.CellEditingPlugin#change
*/

/** Fired when a user cancels the operation and the text editor has been closed. If  tr.grid.CellEditingPlugin#change had been fired, "cancelled" event would not be fired.
* @event tr.grid.CellEditingPlugin#cancelled
*/

/** Fired after the text editor has been closed and all operations are done. This is useful to clean up left over resource.
* @event tr.grid.CellEditingPlugin#closed
*/


/** Double-click to activate edit mode
 * @constructor
 * @param {Array.<number|boolean>|null=} opt_restrictions By default all columns will be processed by this plugin
 * @param {Array.<number|boolean>|null=} opt_exclusions By default no column will be excluded by this plugin
 * @extends {tr.grid.Plugin}
 * @implements {tr.grid.IPlugin}
 */
tr.grid.CellEditingPlugin = function (opt_restrictions, opt_exclusions) {
	var _t = this;
	_t._onColumnAdded = _t._onColumnAdded.bind(_t);
	_t._onColumnMoved = _t._onColumnMoved.bind(_t);
	_t._onColumnRemoved = _t._onColumnRemoved.bind(_t);

	_t._onDoubleClick = _t._onDoubleClick.bind(_t);
	_t._onTextBlur = _t._onTextBlur.bind(_t);
	_t._onClickedOutside = _t._onClickedOutside.bind(_t);
	_t._onTextKeyDown = _t._onTextKeyDown.bind(_t);
	_t._onTextKeyUp = _t._onTextKeyUp.bind(_t);
	_t._onSuggestionSelected = _t._onSuggestionSelected.bind(_t);
	_t._onSuggestionTextEntered = _t._onSuggestionTextEntered.bind(_t);
	_t._onTouchStart = _t._onTouchStart.bind(_t);
	_t._onTouchEnd = _t._onTouchEnd.bind(_t);
	_t._onTimer = _t._onTimer.bind(_t);

	_t._onComboBoxBlur = _t._onComboBoxBlur.bind(_t);

	_t._hosts = [];

	var textBox = document.createElement("input");
	textBox.setAttribute("type", "text");
	textBox.addEventListener(tr.res.blur, _t._onTextBlur, false);
	textBox.addEventListener(tr.res.keydown, _t._onTextKeyDown, false);
	textBox.addEventListener(tr.res.keyup, _t._onTextKeyUp, false);
	textBox.addEventListener(tr.res.mousedown, _t._stopPropagation, false);
	textBox.addEventListener(tr.res.mouseup, _t._stopPropagation, false);
	textBox.addEventListener(tr.res.click, _t._stopPropagation, false);
	textBox.addEventListener("copy", _t._stopPropagation, false);
	textBox.addEventListener("cut", _t._stopPropagation, false);
	textBox.addEventListener("paste", _t._stopPropagation, false);

	_t._textBox = textBox;

	_t._marker = new grid.IndexMarker(opt_restrictions, opt_exclusions);

	_t._addEvent("preEditorOpening");
	_t._addEvent("editorOpened");
	_t._addEvent(tr.res.change);
	_t._addEvent("cancelled");
	_t._addEvent("closed");
};
goog.inherits(tr.grid.CellEditingPlugin, tr.grid.Plugin);

/** @override */
tr.grid.CellEditingPlugin.prototype.getName = function () { return "CellEditingPlugin"; };

/** @override */
tr.grid.CellEditingPlugin.prototype.initialize = function (host) {
	if(this._hosts.indexOf(host) >= 0) { return; }

	this._hosts.push(host);
	host.listen(tr.res.columnAdded, this._onColumnAdded);
	host.listen(tr.res.columnMoved, this._onColumnMoved);
	host.listen(tr.res.columnRemoved, this._onColumnRemoved);

	if(this._dblClick) {
		this._listenForTriggering(host, true);
	}
};

/** @override */
tr.grid.CellEditingPlugin.prototype.unload = function (host) {
	var at = this._hosts.indexOf(host);
	if(at < 0) { return; }
	this._hosts.splice(at, 1);

	host.unlisten(tr.res.columnAdded, this._onColumnAdded);
	host.unlisten(tr.res.columnMoved, this._onColumnMoved);
	host.unlisten(tr.res.columnRemoved, this._onColumnRemoved);

	this._listenForTriggering(host, false);
};

/**
* @public
* @param {tr.Grid} grid
* @param {tr.Grid.SectionReference} sectionRef
* @param {number} colIndex
* @param {number} rowIndex
* @fires tr.grid.CellEditingPlugin#editorOpened
*/
tr.grid.CellEditingPlugin.prototype.openCellEditor = function (grid, sectionRef, colIndex, rowIndex) {
	if(!grid) { return; }
	var cell = grid.getCell(sectionRef, colIndex, rowIndex);
	if (!cell) { return; }
	if(cell === this._activeCell) { return; } // The same cell is open twice

	if(this._activePos) {
		this.closeCellEditor(); // Close previously opened cell editor
	}

	grid.scrollToColumn(colIndex);
	this._onDoubleClick(cell, grid);
};

/** @public
* @param {boolean=} opt_disabled
*/
tr.grid.CellEditingPlugin.prototype.disableAutoSuggest = function (opt_disabled) {
	this._suggesting = (opt_disabled === false);
	if (this._suggesting) {
		var newAS = !this._autoSuggest;

		if(newAS) { // Create the Autosuggest if it does not exist
			// Check if UIFR Autosuggest is available
			var el = document.createElement("ef-autosuggest");
			if (el["is"] === "ef-autosuggest") {
				this._autoSuggest = el;
				this._autoSuggest["uifr"] = true;
			}

			/*
			 * Then try nova autosuggest
			 */
			if (!this._autoSuggest && window["nova"] && window["nova"]["AutoSuggest"]) {
				this._autoSuggest = new window["nova"]["AutoSuggest"]();
			}

			/*
			 * Default autosuggest
			 */
			if(!this._autoSuggest) {
				if(window["tr"]["AutoSuggest"]) {
					this._autoSuggest = new window["tr"]["AutoSuggest"]();
				}
			}
		}

		if(newAS && this._autoSuggest) {
			if (this._autoSuggest["uifr"]) {
				this._autoSuggest["setInput"](this._textBox);
				this._autoSuggest["addEventListener"]("item-select", this._onSuggestionSelected);
			} else {
				this._autoSuggest["setTextBox"](this._textBox);
				this._autoSuggest["listen"]("itemSelected", this._onSuggestionSelected);
				this._autoSuggest["listen"]("enter", this._onSuggestionTextEntered);

				var pm = this._autoSuggest["getPopupMenu"]();
				pm["setAutoHide"](true);
				pm["listen"]("clickedOutside", this._onClickedOutside);
			}
		}
		this._disableAutoSuggest(false);
	} else {
		this._disableAutoSuggest(true);
	}
};

	}
};

* @description Support only HTMLSelectElement or ef-combo-box.

*/
tr.grid.CellEditingPlugin.prototype.setDataFeedComponent = function (el) {
	if (el["is"] === "ef-combo-box") {
		el.addEventListener("item-selected", this._onComboBoxBlur, false);
		el.addEventListener(tr.res.blur, this._onComboBoxBlur, false);
		this._useCoralComboBox = true;
	} else {
		el.addEventListener("change", this._onComboBoxBlur, false);
	}
	this._comboBox = el;
};

/** @public
* @return {string}
*/
tr.grid.CellEditingPlugin.prototype.getDataFeed = function () {
	var feed = "Q";
	if (this._dataFeed && this._comboBox) {
		feed = this._useCoralComboBox ? this._comboBox["inputBoxText"] : this._comboBox.value;
	}
	return feed || "Q";
};
/** @public
* @param {string=} feed
*/
tr.grid.CellEditingPlugin.prototype.setDataFeed = function (feed) {
	if (this._dataFeed && this._comboBox) {
		if(!feed){
			feed = "Q";
		}
		if (this._useCoralComboBox) {
			this._comboBox["inputBoxText"] = feed;
			this._comboBox["selectItemById"](feed);
		} else {
			this._comboBox.value = feed;
		}
	}
};

/** @private
 * @param {boolean=} opt_disabled
 */
tr.grid.CellEditingPlugin.prototype._disableAutoSuggest = function (opt_disabled) {
	if(this._autoSuggest) {
		var b = (opt_disabled !== false);
		if (this._autoSuggest["uifr"]) {
			this._autoSuggest["disabled"] = b;
		} else if(this._autoSuggest["disableSuggestion"]) {
			this._autoSuggest["disableSuggestion"](b);
		} else if(this._autoSuggest["disable"]) {
			this._autoSuggest["disable"](b);
		}
	}
};

/** @private
* @param {Event|tr.IElementControl} e
* @param {tr.Grid=} opt_host
* @fires tr.grid.CellEditingPlugin#editorOpened
*/
tr.grid.CellEditingPlugin.prototype._onDoubleClick = function (e, opt_host) {
	var _t = this;
	var host = opt_host || this.getRelativeGrid(e);

	if (_t.isEditing() || !host) { return; }

	var arg = host.getRelativePosition(/** @type{Event} */(e));

	_t._dispatch("preEditorOpening", arg);

	if (arg["cancel"] || !arg["hit"]) { return; }

	var sectionType = arg[tr.res.sectionType];
	if (sectionType === tr.res.header) { return; }

	var cell = null;
	var colIndex = arg[tr.res.colIndex];
	var rowIndex = arg[tr.res.rowIndex];
	var section = arg[tr.res.section];

	if (sectionType === tr.res.title) {
		if (!_t._editableTitle) { return; }
		cell = section.getCell(colIndex, rowIndex);
	} else {
		if (!_t._editableContent) { return; }
		if (!_t._marker.isAllowed(colIndex)) { return; }

		cell = section.getCell(colIndex, rowIndex);
	}
	if (!cell || cell.hasClass("no-editing") || cell.hasClass("folder")) { return; }

	// Everything has been verified. Start opening editor process
	if(e["preventDefault"]) {
		tr._preventDefault(/** @type{Event} */(e));
	}
	if (_t._suggesting) {
		_t._disableAutoSuggest(sectionType === tr.res.header);
	}
	if (_t._dataFeed) {
		_t.enableDataFeed(sectionType !== tr.res.header);
	}

	if (rowIndex + 1 >= section.getRowCount()) {
		_t._textBox.classList.add("bottom");
	} else {
		_t._textBox.classList.remove("bottom");
	}

	// Initiate text box value
	_t._prevContent = cell.getContent();
	_t._textNode = null;
	_t._initialText = "";
	_t._initialFeed = "";
	if(_t._prevContent) {
		var len = _t._prevContent.childNodes.length;
		for(var i = 0; i < len; ++i) {
			if(_t._prevContent.childNodes[i].nodeType === 3) {
				_t._textNode = _t._prevContent.childNodes[i]; // To preserve current content structure
				break;
			}
		}
	}

	var cellContent;
	if(_t._textNode) {
		cellContent = _t._textNode.textContent;
	} else if(_t._prevContent) {
		cellContent = _t._prevContent.textContent;
	} else {
		cellContent = "";
	}

	_t.setDataFeed(_t._lastUsedFeed);
	_t._textBox.value = cellContent;


	_t._activePos = arg;
	_t._activeCell = cell;
	_t._usingBalloon = false;

	var cellElem = cell.getElement();
	var compStyles = (cellElem) ? window.getComputedStyle(cellElem) : {};
	if((_t._dataFeed && _t._comboBox) || _t._isBalloonMode || compStyles.display === "none" || compStyles.visibility === "hidden") {
		if(!_t._balloon) {
			_t._balloon = new ElementWrapper();
			_t._balloon.setClass("cell editing balloon");
		}
		var rel = host.getRelativePosition(cell);
		_t._balloon.setLeft(rel["x"] / rel["zoomFactor"]);
		_t._balloon.setTop(rel["y"] / rel["zoomFactor"]);
		var rowH = section.getRowHeight(rowIndex);
		rowH += rowH % 2;
		_t._balloon.setHeight(rowH);
		_t._balloon.setStyle("line-height", rowH + "px");

		_t._balloon.setParent(host);
		_t._usingBalloon = true;
	}

	section.addClass("edit-mode");
	cell.addClass("editing");
	this._editing = true;

	if(_t._usingBalloon) {
		var balloonElm = _t._balloon.getElement();
		balloonElm.appendChild(_t._textBox);

		if (this._dataFeed && _t._comboBox) { // Data feed selector will be shown when using balloon mode.
			_t._balloon.addClass("dataFeed");
			_t._textBox.parentNode.insertBefore(_t._comboBox, _t._textBox);
		}
	} else {
		cell.setContent(_t._textBox);
		cell.lockContent(true);
	}

	_t._textBox.focus();
	_t._textBox.select();

	if (_t._hasListener("editorOpened")) {
		arg[tr.res.content] = _t._prevContent;
		_t._dispatch("editorOpened", arg);
	}

	_t._initialText = _t._textBox.value; // After all modifications, cache the string for text change event
	_t._initialFeed = _t.getDataFeed();
};
/** @private
*/
tr.grid.CellEditingPlugin.prototype._onClickedOutside = function () {
	this._commitText(false);
};
/** @private
*/
tr.grid.CellEditingPlugin.prototype._isPopupVisible = function () {
	if (this._autoSuggest) {
		if (this._autoSuggest["uifr"]) {
			return this._autoSuggest["isSuggestionsShown"]();
		} else {
			var popup = this._autoSuggest["getPopupMenu"]();
			return popup["isVisible"]();
		}
	}
	return false;
};

/** @private
* @param {Object} e
*/
tr.grid.CellEditingPlugin.prototype._onTextBlur = function (e) {
	if (this._dataFeed && this._usingBalloon) {
		var rel = e.relatedTarget;
		if(this._balloon.getElement().contains(rel)) {
			return;
		}
		var children = document.body.children; // WORKAROUND: To find Uifr popup element
		var popup;
		for(var i = children.length; --i >= 0;) {
			var el = children[i];
			if(el.classList.contains("popup")) {
				popup = el;
				break;
			}
		}
		if(popup && popup.contains(rel)) {
			return;
		}
	} else {
		if (this._isPopupVisible()) {
			this._textBox.focus();
			return;
		}
	}
	this._commitText(this._autoCommitText);
};
/** @private
* @param {Object} e
*/
tr.grid.CellEditingPlugin.prototype._onTextChange = function (e) {
	//If select an item from the list, the popup will be invisible
	//Only need when use with non-uifr autosuggest
	if (this._autoSuggest && !this._autoSuggest["uifr"]) {
		if (this._isPopupVisible()) {
			return;
		}
	}

	var firstSuggestedItem = null;
	if (this._autoSuggest) {
		if (this._autoSuggest["uifr"]) {
			var list = this._autoSuggest["getList"]();
			if (list && list["data"] && list["data"]["length"] > 0) {
				firstSuggestedItem = list["data"][0]["value"];
			}
		} else {
			var popup = this._autoSuggest["getPopupMenu"]();
			var menu = popup["getMenu"]();
			firstSuggestedItem = menu["getItem"](0);
		}
	}
	this._commitText(true, {'firstSuggestedItem': firstSuggestedItem});
};

/** @private
* @param {Object} e
*/
tr.grid.CellEditingPlugin.prototype._onTextKeyDown = function (e) {
	if (e.keyCode === 27) { // Esc
		this._onTextBlur(e);
	}
	if (e.keyCode === 9) { // Tab
		e.preventDefault();
	}
	e.stopPropagation();
};
/** @private
* @param {Object} e
*/
tr.grid.CellEditingPlugin.prototype._onTextKeyUp = function (e) {
	if(e.keyCode === 13 || e.keyCode === 9) { // Enter or Tab
		if(this._initialText !== this._textBox.value || this._initialFeed !== this.getDataFeed()) {
			this._onTextChange(e);
			e.preventDefault();
		} else {
			this._commitText(false);
		}
	} else {
		var manualInput = e.currentTarget.value.charAt(0) === "'";
		if(this._manualInput !== manualInput) { // Disable AutoSuggest when textbox has a leading single quote at first char
			this._manualInput = manualInput;
			this._disableAutoSuggest(this._manualInput || !this._suggesting);
		}
	}
	e.stopPropagation();
};

/** @private
* @param {Object} e
*/
tr.grid.CellEditingPlugin.prototype._onComboBoxBlur = function (e) {
	var _t = this;

	if (_t._comboBox && _t._useCoralComboBox) {
		// WORKAROUND: Collapse the list when clicking on the selected item
		window.setTimeout(function() {
			_t._comboBox["closeList"]();
		}, 500);
	}
	_t._textBox.focus();
};

/** @private
* @param {boolean} bool
* @param {Object=} opt_e
*/
tr.grid.CellEditingPlugin.prototype._commitText = function (bool, opt_e) {
	var _t = this;
	if (!_t.isEditing()) { return; }

	_t._activePos[tr.res.section].removeClass("edit-mode");
	_t._activeCell.removeClass("editing");
	_t._activeCell.lockContent(false);
	this._editing = false;

	if (_t._dataFeed && _t._comboBox) {
		_t._balloon.removeClass("dataFeed");
	}

	_t._activePos["grid"].focus();

	_t._textBox.removeEventListener(tr.res.blur, _t._onTextBlur, false);
	if(_t._textBox.parentNode) {
		_t._textBox.parentNode.removeChild(_t._textBox);
	}
	if(_t._usingBalloon) {
		_t._balloon.setParent(null); // Remove from the document
		_t._usingBalloon = false;
	}

	var enteredText = _t._textBox.value;
	_t._activeCell.setContent(_t._prevContent);

	if (bool && _t._dataBinding) {
		var sectionSettings = _t._activePos[tr.res.sectionSettings];
		var bindable = sectionSettings.isDataBindable();
		if (bindable) {
			var dataColumnName = _t._activePos["grid"].getDataColumnName(_t._activePos[tr.res.colIndex]);
			sectionSettings.setData(dataColumnName, _t._activePos[tr.res.rowIndex], enteredText); // TODO: Handle header row
		}
		if(_t._textNode) {
			_t._textNode.textContent = enteredText;
		} else {
			if(bindable) {
				if (_t._prevContent && _t._prevContent.classList.contains("text")) { // Bind data only if previous content is a build in text
					_t._activeCell.setContent(enteredText);
				}
			} else {
				_t._activeCell.setContent(enteredText);
			}
		}
	}

	var eventType = (bool) ? "change" : "cancelled";
	if (_t._hasListener(eventType)) {
		_t._activePos["type"] = eventType;
		_t._activePos[tr.res.content] = _t._prevContent;
		_t._activePos["textEntered"] = enteredText; // Alias
		_t._activePos["enteredText"] = enteredText;
		if (_t._dataFeed) {
			var feedValue = this.getDataFeed();
			if(eventType === "change") {
				_t._lastUsedFeed = feedValue;
			}
			_t._activePos["dataFeed"] = feedValue;
		}
		_t._activePos["cell"] = _t._activeCell;
		_t._activePos["allItems"] = [];

		//Pass all items from auto suggest results with the event so it could be used by the event listener
		if (_t._autoSuggest) {
			if (_t._autoSuggest["uifr"]) {
				var list = this._autoSuggest["getList"]();
				if (list && list["data"]) {
					for(var i = 0; i < list["data"].length; i++){
						_t._activePos["allItems"].push(list["data"][i]["value"]);
					}
				}
			} else {
				var items = _t._autoSuggest['getPopupMenu']()['getMenu']()['getItems']();
				for(var i = 0, j = items.length; i < j; i++){
					_t._activePos["allItems"].push(items[i]["getValue"]());
				}
			}
		}

		if (opt_e) {
			if (_t._autoSuggest && _t._autoSuggest["uifr"]) {
				if (opt_e["detail"]) {
					var selectedItem = opt_e["detail"]["item"];
					var selectedValue = selectedItem["value"];
					_t._activePos["selectedItem"] = selectedItem;
					_t._activePos["searchResult"] = selectedValue;
					if (selectedValue["p"]) {
						_t._activePos["field"] = selectedValue["p"]["fn"];
					}
				}
			} else {
				var menuItem = opt_e["item"];
				if (menuItem) {
					_t._activePos["selectedItem"] = menuItem;
					var searchResult = menuItem["getValue"]();
					_t._activePos["searchResult"] = searchResult;
					if (searchResult && searchResult["p"] && searchResult["p"]["fn"]) {
						_t._activePos["field"] = searchResult["p"]["fn"];
					}
				}
			}
			var firstSuggestedItem = opt_e["firstSuggestedItem"];
			if (firstSuggestedItem) {
				_t._activePos["firstSuggestedItem"] = firstSuggestedItem;
			}
		}
		_t._dispatch(eventType, _t._activePos);
	}
	_t._dispatch("closed", _t._activePos);

	_t._textBox.addEventListener(tr.res.blur, _t._onTextBlur, false);

	if (_t._autoSuggest) {
		if (_t._autoSuggest["uifr"]) {
			_t._autoSuggest["hideSuggestions"]();
		} else {
			var popup = _t._autoSuggest["getPopupMenu"]();
			var menu = popup["getMenu"]();
			menu["setItems"](null);
		}
	}
	if (_t._comboBox) {
		if (_t._useCoralComboBox) {
			_t._comboBox["closeList"]();
		} // HTML select element will collapse the list automatically.
	}
	_t._disableAutoSuggest();
	_t.enableDataFeed(false);
	_t._manualInput = false;
	_t._activePos = null;
	_t._activeCell = null;
	_t._prevContent = null;
	_t._textNode = null;
};
/** @private
* @param {Event} e
*/
tr.grid.CellEditingPlugin.prototype._stopPropagation = function (e) {
	e.stopPropagation();
};

/**
* @private
* @param {tr.Grid} host
* @param {boolean} bool
*/
tr.grid.CellEditingPlugin.prototype._listenForTriggering = function (host, bool) {
	if(bool) {
		host.listen("dblclick", this._onDoubleClick);
		if(tr.isTouchDevice) { // Make a comparable functionality for touchscreen
			host.listen("touchstart", this._onTouchStart);
			host.listen("touchend", this._onTouchEnd);
			host.listen("touchcancel", this._onTouchEnd);
		}
	} else {
		host.unlisten("dblclick", this._onDoubleClick);
		host.unlisten("touchstart", this._onTouchStart);
		host.unlisten("touchend", this._onTouchEnd);
		host.unlisten("touchcancel", this._onTouchEnd);
	}
};
/**
* @private
* @param {Object} e
*/
tr.grid.CellEditingPlugin.prototype._onTouchStart = function (e) {
	this._resetTimer();
	this._timerId = setTimeout(this._onTimer, this._longPressTime);
};
/**
* @private
* @param {Object} e
*/
tr.grid.CellEditingPlugin.prototype._onTimer = function (e) {
	this._timerId = 0;
};
/**
* @private
* @param {Event} e
*/
tr.grid.CellEditingPlugin.prototype._onTouchEnd = function (e) {
	if(!this._resetTimer()) {
		this._onDoubleClick(e); // Trigger cell editor
	}
};
/**
* @private
*/
tr.grid.CellEditingPlugin.prototype._resetTimer = function () {
	if(this._timerId) {
		clearTimeout(this._timerId);
		this._timerId = 0;
		return true;
	}
	return false;
};
