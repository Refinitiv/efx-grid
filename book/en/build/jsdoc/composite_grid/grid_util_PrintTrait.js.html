goog.provide("tr.PrintTrait");

goog.require("tr.EventDispatcher");

/** Trigger before the printing and print preview.
* @event tr.PrintTrait#beforeprint
* @property {number} pageWidth Width of the print page
* @property {number} pageHeight Height of the print page
*/
/** Trigger after the printing is completed or cancelled. This event is fired only once at the end of printing
* @event tr.PrintTrait#afterprint
* @property {number} pageWidth Width of the current client window (not the print page)
* @property {number} pageHeight Height of the current client window (not the print page)
*/
/** Trigger before actual print command and PrintTrait#beforeprint event. This event is mainly used for setting number of pages in the initial state. The given paper size is NOT from the print device.
* @event tr.PrintTrait#pageCounting
* @property {number} pageWidth Width of portrait "letter" paper type (739px)
* @property {number} pageHeight Height of portrait "letter" paper type (978px)
*/

/** tr.PrintTrait simplify print process to 2 simple events ("beforeprint" and "afterprint")
* @constructor
* @extends{tr.EventDispatcher}
* @suppress {checkTypes}
* @param {HTMLIFrameElement=} opt_iframe Current Window object will be used if iframe is not specified
*/
tr.PrintTrait = function(opt_iframe) {
	this._onKeyDown = this._onKeyDown.bind(this);
	this._onMediaQuery = this._onMediaQuery.bind(this);
	this._onBeforePrint = this._onBeforePrint.bind(this);
	this._onAfterPrint = this._onAfterPrint.bind(this);
	this.print = this.print.bind(this);

	this._window = window;
	if(opt_iframe && opt_iframe.contentWindow) {
		this._window = opt_iframe.contentWindow;
	}
	this._defaultPrint = this._window.print;

	this._window.print = this.print;
	this._window.addEventListener("keydown", this._onKeyDown, false);
	this._window.addEventListener("beforeprint", this._onBeforePrint, false);
	this._window.addEventListener("afterprint", this._onAfterPrint, false);
	this._printMedia = this._window.matchMedia('print');
	if(this._printMedia) {
		this._printMedia.addListener(this._onMediaQuery);
	}

	this._eventArg = {};

	this._addEvent("pageCounting");
	this._addEvent("beforeprint");
	this._addEvent("afterprint");
};
goog.inherits(tr.PrintTrait, tr.EventDispatcher);

* @private

tr.PrintTrait.prototype._defaultWidth = 739;
/** @type {number}
* @private
*/
tr.PrintTrait.prototype._defaultHeight = 978;
/** @type {Window}
* @private
*/
tr.PrintTrait.prototype._window = null;
/** @type {Function}
* @private
*/
tr.PrintTrait.prototype._defaultPrint = null;
/** @type {Object}
* @private
*/
tr.PrintTrait.prototype._eventArg = null;
/** @type {MediaQueryList}
* @private
*/
tr.PrintTrait.prototype._printMedia = null;

* @private
*/
tr.PrintTrait.prototype._correction = false;
/** @type {boolean}
* @private
*/
tr.PrintTrait.prototype._fixedSize = false;


/** @public
* @suppress {checkTypes}
*/
tr.PrintTrait.prototype.dispose = function() {
	this.unlistenAll();
	this._window.print = this._defaultPrint;
	this._defaultPrint = null;

	this._window.removeEventListener("keydown", this._onKeyDown, false);
	this._window.removeEventListener("beforeprint", this._onBeforePrint, false);
	this._window.removeEventListener("afterprint", this._onAfterPrint, false);
	this._window = null;

	if(this._printMedia) {
		this._printMedia.removeListener(this._onMediaQuery);
	}

	this._eventArg = null;
};
/** @public
*/
tr.PrintTrait.prototype.print = function() {
	if(tr.isEikonDesktop()) { // DPI pixel reduced
		this._defaultWidth = 672; // 692 for letter width
		this._defaultHeight = 917;
	} else {
		this._defaultWidth = 717; // A4 safe width
		this._defaultHeight = 978; // letter safe height
	}
	this._eventArg["pageWidth"] = this._defaultWidth;
	this._eventArg["pageHeight"] = this._defaultHeight;
	this._dispatch("pageCounting", /** @type{!Object} */(this._eventArg));

	this._defaultPrint.call(this._window);
};
/** @public
* @param {number} pageCount
* @return {!Element}
*/
tr.PrintTrait.prototype.createBlankPages = function(pageCount) {
	var rootElem = document.createElement("div");
	for(var i = 0; i < pageCount; ++i) {
		var bp = document.createElement("div");
		bp.className = "tr-blank-page";
		rootElem.appendChild(bp);
	}
	rootElem.style.display = "block"; // This will beat any CSS selector
	return rootElem;
};
/** @public
* @param {number} pw Page width
* @param {number} ph Page height
* @return {!Element}
*/
tr.PrintTrait.prototype.createClientBox = function(pw, ph) {
	var rootElem = document.createElement("div");
	var styObj = rootElem.style;
	styObj.width = pw + "px";
	styObj.height = ph + "px";

	styObj.display = "block";
	styObj.border = "2px solid red";
	styObj.boxSizing = "border-box";

	styObj.fontSize = "3em";
	rootElem.textContent = pw + " x " + ph;
	return rootElem;
};

* @param {boolean} opt_enabled
*/
tr.PrintTrait.prototype.enablePaperSizeCorrection = function(opt_enabled) {
	this._correction = (opt_enabled !== false);
};
/** @public
* @param {boolean} opt_enabled
*/
tr.PrintTrait.prototype.fixPaperSize = function(opt_enabled) {
	this._fixedSize = (opt_enabled !== false);
};


/** @private
* @param {KeyboardEvent} e
*/
tr.PrintTrait.prototype._onKeyDown = function(e) {
	var ctrlCmdKey = tr.isMac ? e.metaKey : e.ctrlKey;

	if(ctrlCmdKey) {
		if(e.altKey) {
			return;
		}
		if(e.keyCode === 80) { // P
			e.preventDefault();
			e.stopPropagation();
			this.print();
		}
	}
};
/** @private
* @param {Object} e
*/
tr.PrintTrait.prototype._onMediaQuery = function(e) {
	if (e["matches"]) {
		this._onBeforePrint();
	} else {
		this._onAfterPrint();
	}
};
/** @private
* @param {Object=} e
*/
tr.PrintTrait.prototype._onBeforePrint = function(e) {
	this._calculatePageSize(this._correction, this._fixedSize);
	this._dispatch("beforeprint", /** @type{!Object} */(this._eventArg));
};
/** @private
* @param {Object=} e
*/
tr.PrintTrait.prototype._onAfterPrint = function(e) {
	this._calculatePageSize();
	this._dispatch("afterprint", /** @type{!Object} */(this._eventArg));
};
/** @private
* @param {boolean=} sizeCorrection
* @param {boolean=} fixedSize
* @return {boolean} True if there is any change
*/
tr.PrintTrait.prototype._calculatePageSize = function(sizeCorrection, fixedSize) {
	if(tr.isIE) {
		return false; // Cannot calculate paper size in IE
	}
	var cw = this._window.document.documentElement.clientWidth;
	var	iw = this._window.innerWidth || 0;
	var ch = this._window.document.documentElement.clientHeight;
	var	ih = this._window.innerHeight || 0;
	var nw = (cw >= iw) ? cw : iw;
	var nh = (ch >= ih) ? ch : ih;

	if(fixedSize) {
		if(nw > nh) {
			nw = this._defaultHeight;
			nh = this._defaultWidth;
		} else {
			nw = this._defaultWidth;
			nh = this._defaultHeight;
		}
	} else if(sizeCorrection) {
		var portrait = nh >= nw;
		var ar = this._calculateAspectRatio(nw, nh);
		var paperSize = tr.PrintTrait._paperSizes[ar];
		if(!paperSize) {
			paperSize = tr.PrintTrait._paperSizes[ar - 1] || tr.PrintTrait._paperSizes[ar + 1];
		}
		if(paperSize) {
			if(portrait) {
				nw = paperSize[0];
				nh = paperSize[1];
			} else {
				nh = paperSize[0];
				nw = paperSize[1];
			}
		}
	}

	if(this._eventArg["pageWidth"] !== nw || this._eventArg["pageHeight"] !== nh) {
		this._eventArg["pageWidth"] = nw;
		this._eventArg["pageHeight"] = nh;
		return true;
	}
	return false;
};

/** @private
* @return {number}
*/
tr.PrintTrait.prototype._calculateAspectRatio = function(w, h) {
	var portrait = h >= w;
	var ratio = (portrait) ? w / h : h / w;
	return Math.floor(ratio * 1000); // Return 3 digit precision
};

/** @private
* @type {Object.<number, Array.<number>>}
*/
tr.PrintTrait._paperSizes = {
	702: [3101, 4414], // "A0"
	699: [2168, 3100], // "A1"
	697: [1511, 2167], // "A2"
	692: [1046, 1511], // "A3"
	685: [717, 1046], // "A4"
	674: [483, 715], // "A5"
	583: [739, 1266], // "legal"
	755: [739, 978], // "letter"
	629: [979, 1554]  // "tabloid"
};
