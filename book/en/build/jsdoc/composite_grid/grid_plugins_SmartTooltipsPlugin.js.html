goog.provide("tr.grid.SmartTooltipsPlugin");

goog.require("tr.grid.IPlugin");
goog.require("tr.grid.Plugin");

/**
 * @constructor
 * @extends {tr.grid.Plugin}
 * @implements {tr.grid.IPlugin}
 * @param {Object} options
 */
tr.grid.SmartTooltipsPlugin = function (options) {
	if (options) {
		this.tooltipTimeout = options["tooltipTimeout"] || this.tooltipTimeout;
	}
	//bind context to callbacks
	this._onGridMouseOver = this._onGridMouseOver.bind(this);
	this._onGridMouseOut = this._onGridMouseOut.bind(this);
};

goog.inherits(tr.grid.SmartTooltipsPlugin, tr.grid.Plugin);

/** @override */
tr.grid.SmartTooltipsPlugin.prototype.getName = function () {
	return "SmartTooltipsPlugin"; // Read Only
};

/** @override */
tr.grid.SmartTooltipsPlugin.prototype.initialize = function (host) {
	this._host = host;

	var tooltipId = "smartTooltip" + tr.grid.SmartTooltipsPlugin._runningId++;

	var tooltipStyleElement = document.createElement("style");
	tooltipStyleElement.innerHTML = this._tooltipDefaultStyles.replace(new RegExp("%TOOLTIP_ID%", "g"), tooltipId);
	this._tooltipStylesElement = new ElementWrapper(tooltipStyleElement);

	var tooltipContainerElement = document.createElement("div");
	tooltipContainerElement.id = tooltipId;
	this._tooltipContainerElement = new ElementWrapper(tooltipContainerElement);
	this._tooltipContainerElement.addClass("alert-tooltip"); // EikonWebUI class

	this._tooltipTextElement = new ElementWrapper(document.createElement("span"));
	this._tooltipTextElement.addClass("alert-tooltip-text"); // EikonWebUI class

	this._tooltipContainerElement.setContent(this._tooltipTextElement);
	this._tooltipContainerElement.setStyle("top", "-9999px"); // hide tooltip

	this._host.getElement().appendChild(this._tooltipContainerElement.getElement());
	this._host.getElement().appendChild(this._tooltipStylesElement.getElement());

	this._host.listen("mouseover", this._onGridMouseOver);
	this._host.listen("mouseout", this._onGridMouseOut);
};

/** @override */
tr.grid.SmartTooltipsPlugin.prototype.unload = function (host) {
	this._tooltipTextElement.dispose();
	this._tooltipContainerElement.dispose();
	this._tooltipStylesElement.dispose();

	this._host.unlisten("mouseover", this._onGridMouseOver);
	this._host.unlisten("mouseout", this._onGridMouseOut);

	this._host = null;
};

/** @private
 * @param {Event} event
 */
tr.grid.SmartTooltipsPlugin.prototype._onGridMouseOver = function (event) {
	var relativePosition = this._host.getRelativePosition(event);
	var cell = relativePosition["cellElement"];

	if (cell && relativePosition["hit"]) {
		var aligner = cell.getElementsByClassName("valigner")[0];

		if (aligner) {
			this._onGridMouseOverElement = aligner.firstChild;

			if (this._onGridMouseOverElement) {
				this._tooltipValue = this._onGridMouseOverElement.getAttribute("title");

				if (this._tooltipValue) {
					this._onGridMouseOverElement.removeAttribute("title"); // to prevent showing browser tooltip
					var cellRect = cell.getBoundingClientRect();

					clearTimeout(this._showTooltipTimeoutId);
					this._showTooltipTimeoutId = setTimeout((function () {
						this._tooltipTextElement.getElement().innerText = this._tooltipValue;
						// show tooltip above cell at cell center
						this._tooltipContainerElement.setStyle("top", cellRect.top - this._tooltipContainerElement.getHeight() - 10 + "px");
						this._tooltipContainerElement.setStyle("left", cellRect.left + ((cellRect.width - this._tooltipContainerElement.getWidth()) / 2) + "px");
					}).bind(this), this.tooltipTimeout);
				}
			}
		}
	}
};

/** @private
 * @param {Event} event
 */
tr.grid.SmartTooltipsPlugin.prototype._onGridMouseOut = function (event) {
	this._tooltipContainerElement.setStyle("top", "-9999px"); // hide tooltip
	if (this._onGridMouseOverElement && this._tooltipValue && this._tooltipValue.length) {
		this._onGridMouseOverElement.setAttribute("title", this._tooltipValue); // restore saved title value
	}
	clearTimeout(this._showTooltipTimeoutId);
};

/** @private
 * @type {tr.ElementWrapper}
 */
tr.grid.SmartTooltipsPlugin.prototype._tooltipStylesElement = null;

/** @private
 * @type {tr.ElementWrapper}
 */
tr.grid.SmartTooltipsPlugin.prototype._tooltipContainerElement = null;

/** @private
 * @type {tr.ElementWrapper}
 */
tr.grid.SmartTooltipsPlugin.prototype._tooltipTextElement = null;

/** @private
 * @type {tr.ElementWrapper}
 */
tr.grid.SmartTooltipsPlugin.prototype._onGridMouseOverElement = null;

/** @private
 * @type {string}
 */
tr.grid.SmartTooltipsPlugin.prototype._tooltipDefaultStyles =
	"#%TOOLTIP_ID% {" +
	"	display: block;" +
	"	position: fixed;" +
	"}" +
	"#%TOOLTIP_ID%:before {" +
	"	margin-left: -7px;" +
	"}" +
	"#%TOOLTIP_ID%:after {" +
	"	margin-left: -6px;" +
	"}";

/** @private
 * @type {string}
 */
tr.grid.SmartTooltipsPlugin.prototype._tooltipValue = "";

/** @private
 * @type {number}
 */
tr.grid.SmartTooltipsPlugin.prototype._showTooltipTimeoutId = -1;

/** @private
 * @type {number}
 */
tr.grid.SmartTooltipsPlugin._runningId = 0;

/** @public
 * @type {number}
 */
tr.grid.SmartTooltipsPlugin.prototype.tooltipTimeout = 500;

goog.exportSymbol("tr.grid.SmartTooltipsPlugin", tr.grid.SmartTooltipsPlugin);
tr.grid.SmartTooltipsPlugin._proto = tr.grid.SmartTooltipsPlugin.prototype;

goog.exportProperty(tr.grid.SmartTooltipsPlugin._proto, "getName", tr.grid.SmartTooltipsPlugin._proto.getName);
goog.exportProperty(tr.grid.SmartTooltipsPlugin._proto, "initialize", tr.grid.SmartTooltipsPlugin._proto.initialize);
goog.exportProperty(tr.grid.SmartTooltipsPlugin._proto, "unload", tr.grid.SmartTooltipsPlugin._proto.unload);

goog.exportProperty(tr.grid.SmartTooltipsPlugin._proto, "tooltipTimeout", tr.grid.SmartTooltipsPlugin._proto.tooltipTimeout);
