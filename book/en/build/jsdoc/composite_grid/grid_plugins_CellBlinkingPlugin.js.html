goog.provide("tr.grid.CellBlinkingPlugin");

goog.require("tr.grid.IPlugin");
goog.require("tr.grid.Plugin");
goog.require("tr.grid.Blinker");
goog.require("tr.grid.IndexMarker");

goog.require("tr.DataView");

/** @constructor
 * @param {Array.<number|boolean>|null=} opt_restrictions By default all columns will be processed by this plugin
 * @param {Array.<number|boolean>|null=} opt_exclusions By default no column will be excluded by this plugin
 * @extends {tr.grid.Plugin}
 * @implements {tr.grid.IPlugin}
 */
tr.grid.CellBlinkingPlugin = function (opt_restrictions, opt_exclusions) {
	this._onPostSectionDataBinding = goog.bind(this._onPostSectionDataBinding, this);
	this._onColumnAdded = goog.bind(this._onColumnAdded, this);
	this._onColumnMoved = goog.bind(this._onColumnMoved, this);
	this._onColumnRemoved = goog.bind(this._onColumnRemoved, this);

	this._marker = new grid.IndexMarker(opt_restrictions, opt_exclusions);

	this._defaultBlinker = new grid.Blinker();
	this._blinkers = [];
};
goog.inherits(tr.grid.CellBlinkingPlugin, tr.grid.Plugin);

/** @override */
tr.grid.CellBlinkingPlugin.prototype.getName = function () {
	return "CellBlinkingPlugin"; // Read Only
};

/** @override */
tr.grid.CellBlinkingPlugin.prototype.initialize = function (host) {
	this._host = host;

	this._host.listen(tr.res.postSectionDataBinding, this._onPostSectionDataBinding);	
	this._host.listen(tr.res.columnAdded, this._onColumnAdded);
	this._host.listen(tr.res.columnMoved, this._onColumnMoved);
	this._host.listen(tr.res.columnRemoved, this._onColumnRemoved);
};

/** @override */
tr.grid.CellBlinkingPlugin.prototype.unload = function (host) {
	this._host.unlisten(tr.res.postSectionDataBinding, this._onPostSectionDataBinding);	
	this._host.unlisten(tr.res.columnAdded, this._onColumnAdded);
	this._host.unlisten(tr.res.columnMoved, this._onColumnMoved);
	this._host.unlisten(tr.res.columnRemoved, this._onColumnRemoved);

	this._host = null;
};

/** @public
 * @return {Array.<tr.grid.Blinker>}
 */
tr.grid.CellBlinkingPlugin.prototype.getBlinkers = function() {
	return this._blinkers;
};

/** @public
 * @param {number} colIndex
 * @return {Object|null}
 */
tr.grid.CellBlinkingPlugin.prototype.getBlinkerAt = function(colIndex) {
	return (this._blinkers[colIndex] !== undefined) ? this._blinkers[colIndex] : null;
};

/** @private
 * @param {Array.<tr.grid.Blinker>=} blinkers
 */
tr.grid.CellBlinkingPlugin.prototype.setBlinkers = function(blinkers) {
	if(blinkers !== undefined && blinkers !== null) {
		this._blinkers = blinkers;
	} else {
		this._blinkers.length = 0;
	}
};

/** @public
 * @param {number} colIndex
 * @param {tr.grid.Blinker} blinker
 */
tr.grid.CellBlinkingPlugin.prototype.setBlinkerAt = function(colIndex, blinker) {
	this._blinkers[colIndex] = null;
	if (blinker !== null && blinker instanceof tr.grid.Blinker) {
		this._blinkers[colIndex] = blinker;
	}
};

/** @public
 * @return {tr.grid.Blinker}
 */
tr.grid.CellBlinkingPlugin.prototype.getDefaultBlinker = function() {
	return this._defaultBlinker;
};

/**
 * @public
 * @param {Array.<string>=} opt_dataMap index in the array will be matched with the display column
 */
tr.grid.CellBlinkingPlugin.prototype.setDataColumnMap = function (opt_dataMap) {
	this._dataMap = (opt_dataMap !== undefined && opt_dataMap !== null) ? opt_dataMap.concat() : null;
};
/** @public
 * @return {Array.<string>}
 */
tr.grid.CellBlinkingPlugin.prototype.getDataColumnMap = function () {
	return this._getDataColumnMap();
};

/** Specify column to be processed by this plugin
* @public
* @return {tr.grid.IndexMarker}
*/
tr.grid.CellBlinkingPlugin.prototype.getMarker = function () {
	return this._marker;
};

/** @private
* @param {Object} e
*/
tr.grid.CellBlinkingPlugin.prototype._onPostSectionDataBinding = function (e) {
	if (e[tr.res.sectionType] !== tr.res.content) { return; }
	if (!e["actualUpdate"]) { return; } //Don't blink when scrolling

	var dataMap = this._getDataColumnMap();
	if (dataMap === null) { return; }

	var section = /** @type{tr.LayoutGrid} */(e[tr.res.section]);
	var columnIndices = {}; // Build valid column list
	for (var c = section.getColumnCount(); --c >= 0; ) {
		if (this._marker.isAllowed(c)) {
			if (dataMap[c]) {
				columnIndices[dataMap[c]] = c;
			}
		}
	}

	var col, row;
	var cid, rid;
	var colIndex, rowIndex;
	var crMap = {};
	var dataView = /** @type{tr.DataView} */(e[tr.res.dataSource]);
	
	var fromR = /** @type{number} */(e[tr.res.fromRowIndex]);
	var toR = /** @type{number} */(e[tr.res.toRowIndex]);
	var updateEvents = /**@type{Array.<Object>} */(e["events"]);
	var updateCount = updateEvents.length;
	for (var i = 0; i < updateCount; ++i) {
		var ue = updateEvents[i];
		if(ue["type"] === "updated") {
			var changes = ue["changes"];
			for(cid in changes) {
				if(columnIndices[cid] !== undefined) {
					col = crMap[cid];
					if(!col) {
						col = {};
						crMap[cid] = col;
					}
					rid = ue["rid"];
					if(col[rid] === undefined) {
						rowIndex = dataView.getRowIndex(rid); // Slow
						if (rowIndex >= fromR && rowIndex < toR) { // Within a display range
							col[rid] = rowIndex;
						}
					}
				}
			}
		}
	}
	
	for(cid in crMap) {
		col = crMap[cid];
		colIndex = columnIndices[cid];
		var blinker = this._blinkers[colIndex] || this._defaultBlinker;
		for(rid in col) {
			var prevData = dataView.getPreviousData(rid, cid);
			if (prevData !== undefined) {
				rowIndex = col[rid];
				blinker.blink(section.getCell(colIndex, rowIndex), dataView.getData(rid, cid), prevData);
			}
		}
	}
};
