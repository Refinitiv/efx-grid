/** @constructor
*/
tr.grid.Blinker = function () {
	if (!this._defaults) {
		this._defaults = {};
		this._defaults["up"] = ["#00FF00", "#00FF00", "blink-up"];
		this._defaults["down"] = ["#FF0000", "#FF0000", "blink-down"];
		this._defaults[tr.res.nochange] = ["#DDDDDD", "#CCCCCC", "blink-nochange"];
		this._defaults[tr.res.none] = ["", "", ""];
		this._defaults[tr.res.enabled] = [true, false, true];
		tr.grid.Blinker.prototype._defaults = this._defaults;
	}

	this._styles = {};
	for (var key in this._defaults) {
		var defaultValues = this._defaults[key];
		this._styles[key] = {};
		this._styles[key][tr.res.text] = defaultValues[0];
		this._styles[key]["bg"] = defaultValues[1];
		this._styles[key]["css"] = defaultValues[2];
	}

	this._scopes = [];

	this.setNumericParser(null);
	this.setCondition(null);
};

//#region Public
/** @public
* @param {tr.IElementControl} ctrl
* @param {*} newVal
* @param {*=} opt_oldVal In case of previous value is not from the specified newVal
*/
tr.grid.Blinker.prototype.blink = function (ctrl, newVal, opt_oldVal) {
	if (!this._enabled) { return; }
	if (ctrl === null) { return; }

	newVal = this._numericParser(newVal);

	var scope = ctrl["blinkState"];
	if (scope === undefined || scope === null) {
		scope = {};
		scope.blinker = this;
		scope.id = this._runningId++;
		scope.oldVal = newVal;
		scope.oldSignal = tr.res.none;
		scope.control = ctrl;
		this._scopes[scope.id] = scope;
		ctrl["blinkState"] = scope;

		if (opt_oldVal === undefined) { return; }
	}
	if (opt_oldVal !== undefined) {
		opt_oldVal = this._numericParser(opt_oldVal);
	} else {
		opt_oldVal = scope.oldVal;
	}

	var userSignal = this._condition(opt_oldVal, newVal);
	// Convert external signal to internal signal index (// up down nochanged
	if(userSignal !== userSignal) {
		scope.signal = tr.res.none;
	} else if(userSignal > 0) {
		scope.signal = "up";	
	} else if(userSignal < 0) {
		scope.signal = "down";
	} else {
		scope.signal = (this._useTwoColorMode) ? tr.res.none : tr.res.nochange;
	}

	if (!this._retainChange && scope.signal === tr.res.nochange) {
		if (scope.oldSignal !== tr.res.none) {
			scope.signal = scope.oldSignal;
			return;
		}
	}

	this._applyColors(scope.control, scope.signal);

	if (scope.timerId >= 0) {
		clearInterval(scope.timerId);
		scope.timerId = -1;
	}
	if (scope.control.getElement() && !this._retainChange) {
		scope.count = 1;
		scope.timerId = setInterval(tr.grid.Blinker._onBlink.bind(null, scope), this._flashDuration);
	}

	scope.oldVal = newVal;
	scope.oldSignal = scope.signal;
};
/** Blinks without data actually changing
* @public
* @param {tr.IElementControl} ctrl
* @param {*} cf_tick
*/
tr.grid.Blinker.prototype.flash = function (ctrl, cf_tick) {
	if (!cf_tick) {
		this.blink(ctrl, 0, 0);
	} else {
		if (!tr.isNumber(cf_tick)) {
			cf_tick = Number(cf_tick);
		}
		if (cf_tick === 1) {
			this.blink(ctrl, 1, 0);
		} else {
			this.blink(ctrl, 0, 1);
		}
	}
};

tr.grid.Blinker.prototype.setTrend = function (ctrl, state) {
	if(ctrl["trendState"] !== state){
		ctrl["trendState"] = state;
		this._applyColors(ctrl, tr.res.none);
	}
};

/** @public
*/
tr.grid.Blinker.prototype.clearAllData = function () {
	var count = 0;
	for (var i in this._scopes) {
		var scope = this._scopes[i];

		if (scope.timerId >= 0) {
			this._applyColors(scope.control, tr.res.none);
			clearInterval(scope.timerId);
			scope.timerId = -1;
		}
		delete scope.control["blinkState"];
		++count;
	}
	if(count > 0) {
		this._scopes = {};
	}
};

/** @public
* @param {number} millisecond
* @param {number=} opt_count Flashing count during the specified period
*/
tr.grid.Blinker.prototype.setDuration = function (millisecond, opt_count) {
	if (millisecond < 0) { millisecond = 0; }
	if (this._duration !== millisecond) {
		this.clearAllData(); // For safety
		this._duration = millisecond;
	}
	if(opt_count !== undefined && opt_count !== null && opt_count > 1) {
		this._flashTarget = opt_count * 2; // blink in and out
		this._flashDuration = Math.floor(this._duration / opt_count);
	} else {
		this._flashTarget = 1; // only blink in
		this._flashDuration = this._duration;
	}
};

tr.grid.Blinker.prototype._parseFloat = function (val) {
	if(tr.isNumber(val)) {
		return /** @type{number} */(val);
	}
	if(val === null || val === "") {
		return NaN;
	}
	return Number(val);
};

* @param {number} oldVal
* @param {number} newVal
* @return {number}

tr.grid.Blinker.prototype._defaultCondition = function (oldVal, newVal) {

/** @private
* @param {Object} scope
*/
tr.grid.Blinker._onBlink = function (scope) {
	var blinker = /**@type {tr.grid.Blinker} */(scope.blinker);

	if (blinker._flashTarget <= 1 || scope.count >= blinker._flashTarget) {
		scope.oldSignal = tr.res.none;
		scope.signal = tr.res.none;

		blinker._applyColors(scope.control, tr.res.none);
		clearInterval(scope.timerId);
		scope.timerId = -1;
		return;
	}

	if(scope.afterFlash) {
		blinker._applyColors(scope.control, scope.signal);
	} else {
		blinker._applyColors(scope.control, tr.res.none);
	}
	scope.count++;

	scope.afterFlash = !scope.afterFlash;
};

/** @private
* @param {tr.IElementControl} ctrl
* @param {string} signal
*/
tr.grid.Blinker.prototype._applyColors = function (ctrl, signal) {
	if(!ctrl.getElement()) { // Element has been disposed
		return;
	}
	
	var style = this._styles[signal];
	var color;

	if (this._styles[tr.res.enabled][tr.res.text]) {
		if( style[tr.res.text]){
			ctrl.setStyle("color", style[tr.res.text]);
		} else {
			var trendStyle = this._styles[ctrl["trendState"]];
			if(trendStyle){
				ctrl.setStyle("color", trendStyle["textTrend"] || "");
			} else {
				ctrl.setStyle("color", "");
			}
		}
	}

	if (this._styles[tr.res.enabled]["bg"]) {
		ctrl.setStyle("backgroundColor", style["bg"] || "");
	}

	if (this._styles[tr.res.enabled]["css"]) {
		ctrl.removeClass(this._styles["up"]["css"]);
		ctrl.removeClass(this._styles["down"]["css"]);
		ctrl.removeClass(this._styles[tr.res.nochange]["css"]);
		if(signal !== tr.res.none) {
			ctrl.addClass(style["css"]);
		}
	}
};
