/** All Novakit's UI controls is implementing this interface, which provides common element manipulations for ease of use.  
 * All standard HTML element events are allowed and can be used through {@link tr.IElementControl#listen}  
 * See all available event list here {@link tr.IElementControl#event:blur}
 * @interface
 */
tr.IElementControl = function () { };
//#region Content Manipulation
/** 
* @public
* @param {boolean=} opt_allContent If true all children will be returned
* @return {Element|Node|NodeList} Returns the last child, if there are multiple children. Returns null if there is no child
*/
tr.IElementControl.prototype.getContent = function (opt_allContent) { };
/** Replaces all existing child elements with the specified content. This method automatically converts all non-element to string wrapped with <span> for proper display.
* @public
* @param {string|Element|tr.IElementControl|*} content Use null to clear content. If string is given, <span> tag is automatically added to host the string. 
* @param {string=} opt_tooltip If undefined, content is added without modification. If string is given, it is added to title attribute of the content
* @return {Element}
*/
tr.IElementControl.prototype.setContent = function (content, opt_tooltip) { };
/** Append the specified content to the existing children list. Note that string is not acceptable
* Use setParent() if you just want to put an element into the document. This method is specifically designed for user content, but not for element reordering
* @public
* @param {Element|Node|tr.IElementControl} content will be appended as a child of this element
* @return {Element}
*/
tr.IElementControl.prototype.addContent = function (content) { };
/** Removes the specified content from the existing children list
* @public
* @param {Element|Node|tr.IElementControl} content If null is given, nothing is done
* @return {boolean} true if the content is removed
*/
tr.IElementControl.prototype.removeContent = function (content) { };
//#endregion Content Manipulation

tr.IElementControl.prototype.unlisten = function (type, handler) { };
