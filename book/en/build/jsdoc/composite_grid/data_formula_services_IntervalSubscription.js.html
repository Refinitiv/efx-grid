goog.provide("tr.IntervalSubscription");

goog.require("tr.EventManager");

/**
 * @constructor
 * @extends {tr.EventManager}
 * @param {string=} funcName
 * @param {Function=} func
 * @param {number=} time
 */
tr.IntervalSubscription = function(funcName, func, time) {
	if(typeof func !== "function" || typeof funcName !== "string" || funcName === "") {
		console.warn("Invalid function or function name");
		return;
	} else {
		try {
			this._value = func() || null; // First execution
		} catch(err) {
			console.error(err);
			return;
		}
		
	}

	time = +time;
	if(time) { // Not NaN or 0
		this._intervalTime = time;
	}

	this._onValueUpdate = this._onValueUpdate.bind(this);

	this._func = func;
	this._funcName = funcName;
	this.isValid = true;
};
goog.inherits(tr.IntervalSubscription, tr.EventManager);

/** @private
 * @type {?Function}
 */
tr.IntervalSubscription.prototype._func = null;
/** @private
 * @type {string}
 */
tr.IntervalSubscription.prototype._funcName = "";
/** @private
 * @type {*}
 */
tr.IntervalSubscription.prototype._value = null;
/** @private
 * @type {string}
 */
tr.IntervalSubscription.prototype._state = "stop";
/** @private
* @type {number|null}
*/
tr.IntervalSubscription.prototype._intervalId = null;
/** @private
* @type {number}
*/
tr.IntervalSubscription.prototype._intervalTime = 3600000; // 1 Hour
/** @const {boolean}
 */
tr.IntervalSubscription.prototype.isValid;

//#region public function of interval service
/** @public
 * @description Set the interval time for retrieving data
 * @param {number} time Millisecond
 */
tr.IntervalSubscription.prototype.setIntervalTime = function(time) {
	time = +time;
	if(!time) { // NaN or 0
		return;
	}
	this._intervalTime = time;
	this._stop();
	this._start();
};
/** @public
 * @description Get the value of this interval service
 * @return {*}
 */
tr.IntervalSubscription.prototype.getValue = function() {
	if(this._value === null) {
		this._value = this._func() || null;
	}
	return this._value;
};
/** @public
 * @description Get the value of this interval service name
 * @return {string}
 */
tr.IntervalSubscription.prototype.getFunctionName = function() {
	return this._funcName;
};
//#endregion public function of interval service


//#region public function of event
/** @override
 * @description Add the function handler into the specific event. This method will start interval automatically.
 * @param {string} type Event name
 * @param {Function} handler Event handler
 */
tr.IntervalSubscription.prototype.addEventListener = function(type, handler) {
	tr.IntervalSubscription.base(this, "addEventListener", type, handler);
	this._start();
};
/** @override
 * @description Remove the function handler from the specific event. If there is not any event handlers, the service will stop automatically.
 * @param {string} type Event name
 * @param {Function} handler Event handler
 */
tr.IntervalSubscription.prototype.removeEventListener = function(type, handler) {
	tr.IntervalSubscription.base(this, "removeEventListener", type, handler);
	if(!this._hasListener()) {
		this._stop();
		this._value = null;
	}
};
/** @override
 * @description Remove all function handlers of all events and stop the interval of retrieving data of service.
 */
tr.IntervalSubscription.prototype.removeAllEventListeners = function() {
	tr.IntervalSubscription.base(this, "removeAllEventListeners");
	this._stop();
	this._value = null;
};
//#endregion public function of event


//#region private function of interval service
/** @private
 * @description Start the interval of retrieving data
 */
tr.IntervalSubscription.prototype._start = function() {
	if(this._state === "start") {
		return;
	}
	var calTime = this._intervalTime - (Date.now() % this._intervalTime);
	this._intervalId = window.setTimeout(this._onValueUpdate, calTime);
	this._state = "start";
};
/** @private
 * @description Stop the interval of retrieving data
 */
tr.IntervalSubscription.prototype._stop = function() {
	if(this._intervalId !== null) {
		window.clearTimeout(this._intervalId);
	}
	this._intervalId = null;
	this._state = "stop";
};
/** @private
 * @description Function for executing the function and publish the event to the listener.
 */
tr.IntervalSubscription.prototype._onValueUpdate = function() {
	this._value = this._func() || null;

	var rowData = {};
	rowData[this._funcName] = this._value;
	this._dispatch("dataChanged", { "function": this._funcName, "values": rowData, "sender": this }); // Publish data

	// Stop then start service for calculating the precious timer to update value (Especially DateTime function)
	this._stop();
	this._start();
};
//#endregion private function of interval service
