goog.provide("tr.grid.DataSourceTreePlugin");

goog.require("tr.DataView");

goog.require("tr.grid.IPlugin");
goog.require("tr.grid.Plugin");

/** @constructor
* @param {number=} opt_startSection
* @extends {tr.grid.Plugin}
* @implements {tr.grid.IPlugin}
*/
tr.grid.DataSourceTreePlugin = function (opt_startSection) {
	this._onPostDataSourceChanged = this._onPostDataSourceChanged.bind(this);
	this._criteria = [];
	
	if (opt_startSection) {
		this._startSection = opt_startSection;
	}
};
goog.inherits(tr.grid.DataSourceTreePlugin, tr.grid.Plugin);

//#region Public Methods
/** @override */
tr.grid.DataSourceTreePlugin.prototype.getName = function () {
	return "DataSourceTreePlugin"; // Read Only
};
/** @override */
tr.grid.DataSourceTreePlugin.prototype.initialize = function (host) {
	this._host = host;
	
	this._host.listen("postDataSourceChanged", this._onPostDataSourceChanged);
	this._host.enableAutoSetDataSource(false); // Important
};
/** @override */
tr.grid.DataSourceTreePlugin.prototype.unload = function (host) {
	this._host.unlisten("postDataSourceChanged", this._onPostDataSourceChanged);
	
	this._host = null;
};

tr.grid.DataSourceTreePlugin.prototype.setAutoSyncContentRowCount = function (val) {
	this._autoSyncRowCount = val;
};

tr.grid.DataSourceTreePlugin.prototype.setStartSection = function(val) {
	this._startSection = val;
};

tr.grid.DataSourceTreePlugin.prototype.getGroupingCriteria = function () {
	return this._criteria;
};
/** Each duplicate values in the given columns will be grouped together. Number of generated groups will equal to number of unique values in the given columns. 
* @public 
* @param {string|Function|Array.<string|Function>} criteria
* Null input will clear all existing grouping criteria.   
* String input is treat as column name in data table  
* Array input will create mutilple groups in tree like manner (nested group criteria)
* Any previous given values will be cleared out after calling this method
*/
tr.grid.DataSourceTreePlugin.prototype.setGroupingCriteria = function (criteria) {
	var dirty = false;
	if (this._criteria.length > 0) {
		dirty = true;
		this._criteria.length = 0;
	}
	if (tr.isString(criteria) || tr.isFunction(criteria)) {
		dirty = true;
		this._criteria.push(/** @type{string|Function} */(criteria));
	} else if (tr.isArray(criteria)) {
		dirty = true;
		this._criteria = /** @type{!Array.<string>} */(criteria.concat());
	}

	if (dirty) {
		if (this._host) {
			var dv = this._host.getDataSource();
			if(dv) {
				if(this._criteria.length > 0) {
					dv.groupBy(this._criteria);
					this.sortGroups();
				} else {
					dv.groupBy(null);
				}
			}
		}
	}
};

/**
 * @public
 * @param {Function} sortFunction Comparer function for comparing the order of 2 groups.  The function takes at least 2 parameters.  
 * The function should return -1 if the first parameter should comes first, 1 for the other way, and 0 if they are equal.  
 * The two first parameters passed to the function are group value that separates group.  
 * The two last parameters passed to the function are data views of corresponding group.
 * @example
 * var compare = function(grpA, grpB, dataViewA, dataViewB) {  
 *   return (grpA <= grpB) ? -1 : 1;
 * }  
 * var dstp = new tr.grid.DataSourceTreePlugin();
 * dstp.setGroupSortingLogic(compare);
 */
tr.grid.DataSourceTreePlugin.prototype.setGroupSortingLogic = function(sortFunction) {
	this._groupSortFunction = sortFunction;
};

/** Get the current group sort function. If none specified by setGroupSortingLogic() return
 * standard string sort function.
 * @public
 * @return {Function}
 */
tr.grid.DataSourceTreePlugin.prototype.getGroupSortingLogic = function() {
	if (this._groupSortFunction) {
		return this._groupSortFunction;
	} else {
		return tr.grid.DataSourceTreePlugin._standardGroupSortFunction;
	}
};

/** Specify empty group in case of grouping values are known prior to requesting data 
* @public 
* @param {Array|Array.<Array>} predefinedGroups 
* Values in the given array should be matched the value coming from a data table, or else no content is added to the group  
* Use null value to clear all predefined groups.  
* Use 2D array to specify nested groups.  
* Any previous given values will be cleared out after calling this method
*/
tr.grid.DataSourceTreePlugin.prototype.setPredefinedGroups = function (predefinedGroups) {
	this._predefinedGroups = predefinedGroups;
	
	if (this._host) {
		var dv = this._host.getDataSource();
		if(dv) {
			dv.setPredefinedGroups(this._predefinedGroups);
		}
	}
};

/** 
* @public

tr.grid.DataSourceTreePlugin.prototype.getGroupHeaders = tr._deprecatedFunction;

/** 
* @public
* @deprecated
*/
tr.grid.DataSourceTreePlugin.prototype.getGroupContents = tr._deprecatedFunction;

/**
 * Trigger group sorting logic to re-sort row grouping 
 * @public
 */
tr.grid.DataSourceTreePlugin.prototype.sortGroups = function() {
	if(!this._groupSortFunction || !this._host) { return; }
	
	var dv = this._host.getDataSource();
	if(dv) {
		dv.sortGroups(this._groupSortFunction);
	}
};

//#region Private Methods
/** @private
 */
tr.grid.DataSourceTreePlugin.prototype._onPostDataSourceChanged = function (e) {
	var dv = /** @type{tr.DataView} */(e[tr.res.dataSource]);
	this._populateSectionTree(dv);
};

/** @private
 * @param {tr.DataView} dataSource
 */
tr.grid.DataSourceTreePlugin.prototype._populateSectionTree = function (dataSource) {
	var frozen = this._host.freezeLayout();
	
	this._host.setScrollTop(0);
	
	if (this._initialized) { // Clean up previous resources
		this._initialized = false;
		var i;
		var sectionList = this._host.getAllSectionSettings();
		var autoGenCount = 0;
		for(i = sectionList.length; --i >= 0;) {
			if(sectionList[i]._autoGenerated) {
				autoGenCount++;
			}
		}
		if (sectionList.length !== autoGenCount) {
			for (i = sectionList.length; --i >= 0;) {
				this._host.removeSection(sectionList[i]); // Slow
			}
		} else {
			this._host.removeAllSections();
		}
		
		var dv = this._host.getDataSource();
		if(dv) {
			dv.setPredefinedGroups(null);
			dv.groupBy(null);
		}
	}

	if (dataSource) {
		this._initialized = true;
		var parentSettings = this._addSectionSettings(tr.res.title);
		var childSettings = this._addSectionSettings(tr.res.content);
		childSettings.setParent(parentSettings);
		
		if(this._criteria.length > 0) {
			dataSource.setPredefinedGroups(this._predefinedGroups);
			dataSource.groupBy(this._criteria);
			this.sortGroups();
		}
	}
	this._host.freezeLayout(frozen);
};

/** @private
 * @param {string} type
 * @return {tr.grid.SectionSettings} childSettings
 */
tr.grid.DataSourceTreePlugin.prototype._addSectionSettings = function (type) {
	var section = null;
	if (type !== tr.res.title) {
		var insertAt = this._startSection;
		var tmp = this._host.getSection(insertAt);
		while(tmp) {
			if(!tmp._autoGenerated) {
				break;
			}
			tmp = this._host.getSection(++insertAt);
		}
		section = this._host.addSectionAt(insertAt, type);
		section._autoGenerated = true;
	} else {
		var existingTitle = this._host.getSection(type); // WARNING: Assume that rootNode is the top first title
		if (existingTitle) {
			section = existingTitle;
		} else {
			section = this._host.addSectionAt(this._startSection, type);
			section._autoGenerated = true;
		}
		if (section.getRowCount() == 0) {
			section.setRowCount(1);
		}
	}
	
	var sectionSettings = this._host.getSectionSettings(section);
	if(section._autoGenerated) {
		sectionSettings._autoGenerated = true;
	}

	if (type === tr.res.content) {
		if (this._autoSyncRowCount !== undefined) {
			sectionSettings.setAutoSyncRowCount(this._autoSyncRowCount);
		}
	} else {
		sectionSettings.disableDataBinding();
	}

	sectionSettings.setDataSource(this._host.getDataSource()); // Trigger data binding event
	return sectionSettings;
};

/** @private
* @param {*} valA
* @param {*} valB
* @return {number}
*/
tr.grid.DataSourceTreePlugin._standardGroupSortFunction = function(valA, valB) {
	var strA = "" + valA;
	var strB = "" + valB;
	if (strA < strB) {
		return -1;
	}
	if (strA > strB) {
		return 1;
	}
	return 0;
};
