/** @event tr.grid.DraggableContentPlugin#drag
* @description Fired on each mousemove
* @type {Object}
*/
/** @event tr.grid.DraggableContentPlugin#dragStart
* @description Fired before the actual drag. Use cancelDrag() to stop the drag
* @type {Object}
*/
/** @event tr.grid.DraggableContentPlugin#dragEnd
* @description Fired only if the drop is valid (not always on mouseup)
* @type {Object}
*/

/** @constructor
 * @param {boolean=} opt_crossSection Can the dragging goes across from the start section to another section. Default is 'true'
 * @extends {tr.grid.Plugin}
 * @implements {tr.grid.IPlugin}
 */
tr.grid.DraggableContentPlugin = function (opt_crossSection) {
	this._onContentScrolled = goog.bind(this._onContentScrolled, this);
	this._onDragPulse = goog.bind(this._onDragPulse, this);

	// For each cell
	this._onMouseDown = goog.bind(this._onMouseDown, this);
	this._clearCache = goog.bind(this._clearCache, this);
	this._onMouseOut = goog.bind(this._onMouseOut, this);

	// For timer
	this._delayStart = goog.bind(this._delayStart, this);
	this._onDragStart = goog.bind(this._onDragStart, this);

	this._onDrag = goog.bind(this._onDrag, this);
	this._onDragEnd = goog.bind(this._onDragEnd, this);

	this._onJETDrop = goog.bind(this._onJETDrop, this);
	this._onJETDragOver = goog.bind(this._onJETDragOver, this);
	this.stopDrag = goog.bind(this.stopDrag, this);

	this._addEvent("dragStart"); // This is to allow external component register a listener to this event
	this._addEvent("drag");
	this._addEvent("dragInterval");
	this._addEvent("dragEnd");

	if (opt_crossSection !== undefined) {
		this._isCrossSectionAllowed = opt_crossSection;
	}

	this._hosts = [];

	this._dragBox = new tr.ElementWrapper(document.createElement("svg"));
	this._dragBox.setClass("tr-dragbox content-dragging");
};
goog.inherits(tr.grid.DraggableContentPlugin, tr.grid.Plugin);

//#region Public Methods
/** @override */
tr.grid.DraggableContentPlugin.prototype.getName = function () {
	return "DraggableContentPlugin"; // Read Only
};
/** @override */
tr.grid.DraggableContentPlugin.prototype.initialize = function (host) {
	if(this._hosts.indexOf(host) >= 0) { return; }

	this._hosts.push(host);

	host.listen("dragstart", tr._preventDefault);
	host.listen(tr.res.mousedown, this._onMouseDown);
	host.getVScrollbar().listen("scroll", this._onContentScrolled);
};
/** @override */
tr.grid.DraggableContentPlugin.prototype.unload = function (host) {
	this.stopDrag();
	this._clearCache();
	this._jetDragContent = null;

	var at = this._hosts.indexOf(host);
	if(at >= 0) {
		host.unlisten(tr.res.mousedown, this._onMouseDown);
		host.getVScrollbar().unlisten("scroll", this._onContentScrolled);

		this._hosts.splice(at, 1);
	}
};

* @public

*/
tr.grid.DraggableContentPlugin.prototype.allowDrag = function (opt_allowed) {
	if(this._pos) {// WARNING: Please note that this method is intent to be used by client, and not for internal use
		this._pos["invalidTarget"] = (opt_allowed === false);
	}
};

		this._pos["cancel"] = true;
	}
};

/** @public
* @param {boolean=} opt_enabled
*/
tr.grid.DraggableContentPlugin.prototype.enableJETDragAndDrop = function (opt_enabled) {
	var val = opt_enabled !== false;
	if(this._jetDnD === val) { return; }

	this._jetDnD = val;

	var jet = window["JET"];
	if(jet) {
		//If ContainerDescription is undefined, that means JET is not initalized. Mostlikely the app
		//is running the local development mode without container. Disable dnd in this case
		if(jet["ContainerDescription"] && (jet["ContainerDescription"]["name"] !== "EikonNowContainer")) {
			if(this._jetDnD) {
				jet["onDrop"](this._onJETDrop);
				jet["onDragOver"](this._onJETDragOver);
				jet["onDragLeave"](this.stopDrag);
			} else {
				jet["onDrop"](null);
				jet["onDragOver"](null);
				jet["onDragLeave"](null);
			}
		} else {
			this._jetDnD = false;
		}
	}
};
/** @public
* @return {Object}
*/
tr.grid.DraggableContentPlugin.prototype.getJETDragContent = function () {
	return this._jetDragContent;
};
/** @public
* @param {Object} content
*/
tr.grid.DraggableContentPlugin.prototype.setJETDragContent = function (content) {
	this._jetDragContent = content;

	var t = window["tr"];
	var g = (t) ? t["grid"] : null;
	var jdc = (g) ? g["JetDragContent"] : null;
	if(content && jdc) {
		var jetDragContentMgr = new jdc(content);
		var rics = jetDragContentMgr["getRicsToDrop"]();
		var fields = jetDragContentMgr["getFieldsToDrop"]();

		this._jetContentHasRic = (rics && rics.length > 0) ? true : false;
		this._jetContentHasField = (fields && fields.length > 0) ? true : false;
	} else {
		this._jetContentHasRic = false;
		this._jetContentHasField = false;
	}
};

/** @private
 * @param {Event|Element|tr.IElementControl} e
 * @return {!Object}
 */
tr.grid.DraggableContentPlugin.prototype._hitTest = function (e) {
	var	host = this.getRelativeGrid(e);
	if (host) {
		return host.getRelativePosition(e);
	} else {
		return /** @type{!Object} */(e);
	}
};

/** @private
 * @param {string} str
 */
tr.grid.DraggableContentPlugin.prototype._setEntryPoint = function (str) {
	this._entryPoint = str;
	this._dragBox.setContent(null); // Clear original content
};

/** @private
* @param {boolean=} delayStart
* @param {Element|MouseEvent|tr.IElementControl=} opt_startPos
* @param {boolean=} opt_suppressEvent
*/
tr.grid.DraggableContentPlugin.prototype._startDrag = function (delayStart, opt_startPos, opt_suppressEvent) {
	if(this._isDragging) { return; }
	if(this._hosts.length <= 0) { return; }

	if(!opt_startPos) {
		opt_startPos = this._hosts[0];
	}

	var pos = this._hitTest(opt_startPos);
	if(pos["hit"]) {
		this._clearCache();
		this._startPos = pos;
		this._pos = pos;
		this._activeGrid = pos["grid"];

		if(delayStart) {
			this._activeGrid.listen(tr.res.mousemove, this._delayStart);
			this._listenAbortActions();
		} else { // Start immediately
			this._onDragStart(opt_suppressEvent);
		}
	}
};

/** @private
* @param {MouseEvent} e
*/
tr.grid.DraggableContentPlugin.prototype._onMouseDown = function (e) {
	if(e.button) { return; }  // We don't allow drag with middle click (button == 1), or right click (button == 2)

	if (this._draggingDisabled && // check drag is enable
		this._hitTest(e).sectionType !== 'content') { // check section is content
		return;
	}

	this._setEntryPoint("grid");
	this._startDrag(true, e); // Start with delay start
};

/** @private
 * @param {Event} e
 */
tr.grid.DraggableContentPlugin.prototype._onMouseOut = function (e) {
	e = e ? e : window.event;
	var from = e.relatedTarget || e.toElement;
	if (!from || from.nodeName == "HTML") {
		// window.console.warn("Left window");

		/*
		 * TRGRID-1493
		 * The DragStart is delayed using a timer (_timerId created by _delayStart )
		 * The mouse may left the window before the drag start
		 * In this case we should force the stopDrag to stop the drag on the source
		 * Because the mouse up will not be handled by the source
		 */

		// cancel the timer - _onDragStart will not be called by the timer
		if(this._timerId >= 0) {
			clearTimeout(this._timerId);
			this._timerId = -1;
		}

		this._onDragStart();
		this.stopDrag();
	}
};

/** @private
*/
tr.grid.DraggableContentPlugin.prototype._delayStart = function (e) {
	this._activeGrid.unlisten(tr.res.mousemove, this._delayStart);

	if(this._timerId < 0) {
		this._timerId = window.setTimeout(this._onDragStart, 200);
	}
	if(this._jetDnD) {
		document.addEventListener("mouseout", this._onMouseOut, false);
	}
};

/** @private
* @param {boolean=} bool
*/
tr.grid.DraggableContentPlugin.prototype._listenAbortActions = function (bool) {
	if(bool !== false) {
		window.addEventListener(tr.res.mouseup, this._clearCache, false);
	} else {
		window.removeEventListener(tr.res.mouseup, this._clearCache, false);
	}
};

/** @private
* @param {boolean=} opt_suppressEvent
*/
tr.grid.DraggableContentPlugin.prototype._onDragStart = function (opt_suppressEvent) {
	if (this._isDragging) { return; }
	if(!this._pos) { return; }

	this._listenAbortActions(false); // Unlisen abort action

	var useJetDragStart = false;
	var jet = window["JET"];
	if(opt_suppressEvent) { // Note that if suppressEvent is true, it means that this event is activated from JET's dragover
		useJetDragStart = true;
	} else {
		this._dispatch("dragStart", /** @type{!Object} */(this._pos));
		if(this._isDragCancelled() || !this._hasValidTarget()) {
			return; // User does not allow this drag to happen
		}
		if(this._jetDnD && this._jetDragContent && jet) {
			useJetDragStart = true;
			jet["dragStart"](this._jetDragContent);
		}
	}

	this._isDragging = true;
	document.body.classList.add("tr-dragging"); // Prevent text selection

	if (!useJetDragStart) { // WARNING: JET wrongly intercepts all mouse events without app's consent
		window.addEventListener(tr.res.mousemove, this._onDrag, false);
	}
	window.addEventListener(tr.res.mouseup, this._onDragEnd, false);

	this._initializeGuideline();

	// For step scrolling
	this._scrollStep = Math.floor(this._activeGrid.getScrollHeight() / 25); // Update the scrollbar first
	if(this._scrollStep < 40) {
		this._scrollStep = 40;
	} else if(this._scrollStep > 400) {
		this._scrollStep = 400;
	}
	var scrollbar = this._activeGrid.getVScrollbar();
	this._cacheTop = scrollbar.getTop() + 20;
	this._cacheHeight = scrollbar.getTop() + scrollbar.getHeight() - 20;

	this._dragPulseId = window.setInterval(this._onDragPulse, 150); // Start pulse
};
/** @private
* @param {Object} e
*/
tr.grid.DraggableContentPlugin.prototype._onDrag = function (e) {
	if (!this._isDragging) { return; }

	if(!e || (e["which"] !== 1)) { // Left click is not down. Support in IE9 and up
		this.stopDrag();
		return;
	}

	this._pos = this._hitTest(/** @type{MouseEvent} */(e));

	this._calculateValidDropArea();

	// Dispatch drag event to let user determine valid drop target using allowDrag (allowDrop) method
	this._dispatch("drag", /** @type{!Object} */(this._pos));

	this._updateGuideVisibility();
	this._updateGuidePosition();

	if(e["preventDefault"]) {
		tr._preventDefault(/** @type{MouseEvent} */(e));
	}
};
/** @private
* @param {Object=} opt_e
*/
tr.grid.DraggableContentPlugin.prototype._onDragEnd = function (opt_e) {
	if (!this._isDragging) { return; }
	this._isDragging = false;

	if(opt_e && opt_e["preventDefault"]) {
		tr._preventDefault(/** @type{MouseEvent} */(opt_e));
	}

	if (this._guideline) {
		this._guideline.setParent(null);
		this._guideline = null;
	}

	this._dragBox.setParent(null);

	var len = this._hosts.length;
	for (var i = 0; i < len; i++) { // Clear all cursor styling
		this._hosts[i].getElement().style.cursor = "";
	}

	document.body.classList.remove("tr-dragging");
	window.removeEventListener(tr.res.mouseup, this._onDragEnd, false);
	window.removeEventListener(tr.res.mousemove, this._onDrag, false);

	if(!this._isDragCancelled()) {
		this._extendEventOutput(this._pos, opt_e);
		this._dispatch("dragEnd", /** @type{!Object} */(this._pos));
		this._jetDragContent = null;
	}

	this._clearCache();
};

/** @private
* @param {Event} e
*/
tr.grid.DraggableContentPlugin.prototype._onContentScrolled = function (e) {
	if(!this._isDragging || !this._pos["hit"]) { return; }

	var host = this._pos["grid"];
	if(!host) { return; }

	if(this._validView) {
		this._dispatch("drag", /** @type{!Object} */(this._pos));
	}
	this._updateGuidePosition();
};
/** @private
*/
tr.grid.DraggableContentPlugin.prototype._onDragPulse = function () {
	if(!this._isDragging) { return; }

	this._dispatch("dragInterval", /** @type{!Object} */(this._pos));

	// Manage vertical scrollbar
	if(!this._hosts[0].getVScrollbar().isActive()) { return; }

	var y = this._pos["y"];
	var scrollVal = 0;
	if(y < this._cacheTop) {
		scrollVal = -Math.floor(this._scrollStep * (0.8 + Math.random()));

	} else if(y > this._cacheHeight) {
		scrollVal = Math.floor(this._scrollStep * (0.8 + Math.random()));
	}

	this._scrollDown(scrollVal);
};

/** @private
*/
tr.grid.DraggableContentPlugin.prototype._clearCache = function () {
	if(this._timerId >= 0) {
		clearTimeout(this._timerId);
		this._timerId = -1;
	}
	document.removeEventListener("mouseout", this._onMouseOut, false);
	if(this._dragPulseId >= 0) {
		clearInterval(this._dragPulseId);
		this._dragPulseId = -1;
	}
	this._listenAbortActions(false);

	this._startPos = null;
	this._pos = null;

	if (this._activeGrid) {
		this._activeGrid.unlisten(tr.res.mousemove, this._delayStart);
		this._activeGrid = null;
	}

	this._validDropArea = false;
	this._validView = false;
};

/** Guideline will be visible only when mouse is invalid drop area and drag operation has not been cancelled
* @private
* @return {boolean}
*/
tr.grid.DraggableContentPlugin.prototype._isGuideVisible = function () {
	return this._validDropArea && this._hasValidTarget();
};
/** @private
* @return {boolean}
*/
tr.grid.DraggableContentPlugin.prototype._isDragCancelled = function () {
	return (this._pos && this._pos["cancel"]) ? true : false;
};
/** @private
* @return {boolean}
*/
tr.grid.DraggableContentPlugin.prototype._hasValidTarget = function () {
	return (this._pos && this._pos["invalidTarget"]) ? false : true;
};

/** @private
*/
tr.grid.DraggableContentPlugin.prototype._initializeGuideline = function () {
	this._calculateValidDropArea();
	this._updateGuideVisibility();
	this._updateGuidePosition();
};
/** @private
*/
tr.grid.DraggableContentPlugin.prototype._updateGuideMode = function () {
	var verticalGuideMode = false;
	if(this._entryPoint === 'JET') { // Vertical guide mode can be true only if there is a field in drag content and mouse is over title section
		if(this._pos[tr.res.sectionType] === tr.res.title && this._jetContentHasField) {
			verticalGuideMode = true;
		}
	}

	if(!this._guideline || verticalGuideMode !== this._verticalGuideMode) {
		if(this._guideline) {
			this._guideline.setParent(null);
		}
		this._verticalGuideMode = verticalGuideMode;
		if(this._verticalGuideMode) {
			if(!this._vGuideline) {
				this._vGuideline = new tr.ElementWrapper(document.createElement("svg")); // Why svg
				this._vGuideline.addClass("tr-guideline");
				this._vGuideline.setWidth(3);
			}

			this._guideline = this._vGuideline;
		} else { // Horizontal guide mode (default)
			if(!this._hGuideline) {
				this._hGuideline = new tr.ElementWrapper();
				this._hGuideline.setClass("tr-guideline");
				// this._hGuideline.setLeft(0);
				this._hGuideline.setHeight(3);
			}
			this._hGuideline.setWidth(this._activeGrid.getWidth());

			this._guideline = this._hGuideline;
		}
		if(this._pos["hit"]) {
			this._updateGuideParent();
		}
	}
};
/** @private
*/
tr.grid.DraggableContentPlugin.prototype._updateGuideParent = function () {
	this._dragBox.setParent(this._activeGrid);
	if(this._verticalGuideMode) {
		this._guideline.setHeight(this._activeGrid.getHeight()); // Different grid may have different height
		this._guideline.setParent(this._activeGrid);
	} else {
		this._guideline.setParent(this._pos[tr.res.section].getColumnHost());
	}
};
/** @private
*/
tr.grid.DraggableContentPlugin.prototype._calculateValidDropArea = function () {
	this._updateGuideMode();

	this._validView = (this._pos["hit"]) ? true : false;
	this._validDropArea = this._validView; // If the given object is within the grid, it is in the valid drop area
	if(this._validDropArea) {
		var grid = this._pos["grid"];

		if(grid !== this._activeGrid || this._guideline.getParent() !== this._activeGrid.getElement()) {
			this._activeGrid = grid; // Mouse cursor styling will be clear in dropEnd event
			this._updateGuideParent(); // Update the new parent
		}

		if(this._verticalGuideMode) {
			if(this._pos[tr.res.sectionType] !== tr.res.title) {
				this._validDropArea = false; // In vertical guide mode if the mouse is not in the title section, it is not the valid drop area
			}
		} else {
			if(this._pos[tr.res.sectionType] !== tr.res.content) {
				this._validDropArea = false; // In horizontal guide mode if the mouse is not in the content section, it is not the valid drop area
			}

			if(this._entryPoint === 'JET' && !this._jetContentHasRic) {
				this._validDropArea = false; // Since jet content doesn't has any RICs for droping
			}

			if (!this._isCrossSectionAllowed) { // If crossing section is not allowed
				if (this._startPos[tr.res.sectionIndex] != this._pos[tr.res.sectionIndex]) {
					this._validDropArea = false; // The mouse is not in the same initial drag section, make it a invalid drop area
				}
			}
		}
	}
};
/** @private
*/
tr.grid.DraggableContentPlugin.prototype._updateGuidePosition = function () {
	if(!this._isGuideVisible()) {
		return; // Since guideline is not visible there is no point in updating its position
	}

	if(this._verticalGuideMode) {
		var colIndex = this._pos["colIndex"];
		var cellElement = this._pos['cellElement'];
		var isRight = 0;

		if (cellElement) {
			var rect = cellElement.getBoundingClientRect();
			var offsetX = (this._pos['clientX'] - rect['left']) / this._pos['zoomFactor'];
			isRight = Math.round(offsetX / cellElement.offsetWidth);
		}

		if(colIndex + isRight < this._activeGrid.getColumnCount()) {
			var left = this._activeGrid.getColumnLeft(colIndex + isRight);
			this._guideline.setLeft(left);
		} else {
			var right = this._activeGrid.getColumnRight(colIndex);
			this._guideline.setLeft(right - 4);
		}

	} else {
		this._guideline.setTop(this._pos['cellElement']['offsetTop']);
	}

	this._dragBox.setLeft(this._pos['x'] + 15);
	this._dragBox.setTop(this._pos['y'] + 20);
};
/** @private
*/
tr.grid.DraggableContentPlugin.prototype._updateGuideVisibility = function () {
	if(this._isGuideVisible() && this._validView) {
		this._guideline.show();
	} else {
		this._guideline.hide();
	}

	// Mouse Cursor
	if(this._activeGrid) {
		var elem = this._activeGrid.getElement();
		var cur = this._isGuideVisible() ? "move" : "not-allowed";
		if(elem.style.cursor !== cur) {
			elem.style.cursor = cur;
		}
	}
};

/** @private
* @suppress {checkTypes}
* @param {Object} clientPos
*/
tr.grid.DraggableContentPlugin.prototype._onJETDragOver = function (clientPos) {
	if(!clientPos) { return; }

	if(!this._isDragging) {
		if(!clientPos["dropData"]) { return; }
		this._setEntryPoint("JET");
		this.setJETDragContent(clientPos["dropData"]); // WARNING: We have no way to distinguish between dropData from grid and other apps
		this._startDrag(false, clientPos['e'], true); // Event is suppressed
	} else {
		this._onDrag(clientPos['e']);
	}
};
/** @private
* @param {Object} jetObj
*/
tr.grid.DraggableContentPlugin.prototype._onJETDrop = function (jetObj) {
	if (!this._isDragging) { return; }

	if(!this._jetDragContent) {
		this._jetDragContent = jetObj; // To maintain original client signature
	}
	this._onDragEnd(jetObj['mouse']['srcEvent']);
};

/** @private
* @param {Object} pos
 * @param {Object=} e
*/
tr.grid.DraggableContentPlugin.prototype._extendEventOutput = function (pos, e) {
	var cellElement = pos['cellElement'];
	var isRight = 0;

	if (e && cellElement) {
		var rect = cellElement.getBoundingClientRect();
		var offsetX = (e['clientX'] - rect['left']) / pos['zoomFactor'];
		isRight = Math.round(offsetX / cellElement.offsetWidth);
	}

	pos['colIndex'] += isRight;
	pos['currentPosition'] = pos; // To support JET requirement
	pos['data'] = this._jetDragContent;
	pos['srcGrid'] = this._startPos['grid'];
	pos['dropTarget'] = e.target;
};
