/**
 * @constructor
 * @implements {tr.IElementControl}
 * @extends {tr.Popup}
 * @param {Element|Object=} opt_initializer
 * @param {Object=} opt_model
 */
tr.PopupMenu = function(opt_initializer, opt_model) {
	var _this = this;

	// Add custom events
	_this._addEvent("itemHighlighted");
	_this._addEvent("itemActivated");
	_this._addEvent("itemSelected");
	_this._addEvent("treeItemActivated");
	_this._addEvent("treeItemSelected");

	_this._onKeyDown = _this._onKeyDown.bind(_this);
	_this._onChildHidden = _this._onChildHidden.bind(_this);
	_this._onItemHovered = _this._onItemHovered.bind(_this);
	_this._onItemHighlighted = _this._onItemHighlighted.bind(_this);
	_this._onItemActivated = _this._onItemActivated.bind(_this);
	_this._onItemSelected = _this._onItemSelected.bind(_this);
	_this._onTreeItemActivated = _this._onTreeItemActivated.bind(_this);
	_this._onTreeItemSelected = _this._onTreeItemSelected.bind(_this);

	tr.PopupMenu.base(this, 'constructor', opt_initializer, opt_model);
};

goog.inherits(tr.PopupMenu, tr.Popup);

/**
 * @override
 * @public
 */
tr.PopupMenu.prototype.dispose = function() {
	this.disposeChildren();

	// Dispose menu
	var menu = this._menu;
	if (menu) {
		menu.dispose();
	}

	tr.PopupMenu.base(this, "dispose");
};

/**
 * @ignore
 * @public
 */
tr.PopupMenu.prototype.disposeChildren = function() {
	var _this = this;

	if (_this._child) {
		_this._child.dispose();
		_this._child = null;
	}
};

		menu.focus();

/**
 * @override
 * @public
 * @param {boolean=} opt_shown
 */
tr.PopupMenu.prototype.show = function(opt_shown) {
	tr.PopupMenu.base(this, "show", opt_shown);

	var _this, creator, menu;

	_this = this;
	creator = _this._creator;
	menu = _this._menu;

	if (opt_shown !== false) { // Show popup menu
		if (creator) {
			_this.setAutoResize(creator.isAutoResize()); // Auto-resize flag should be the same as of creator
		}

		_this._managePosition();

		// Set focus on menu to allow key navigation
		if (menu && _this._handleKeyDown) {
			menu.focus();
		}
	}
	else { // Hide popup menu
		// Clear highlighted item
		if (menu && !creator) {
			menu.setHighlightedItem(null);
		}

		_this.hideChild();
	}
};

/**
 * @ignore
 * @public
 */
tr.PopupMenu.prototype.showChild = function() {
	var _this, item, subMenu, child;

	_this = this;

	if (_this._menu) {
		item = _this._menu.getHighlightedItem();
		if (item && item.hasClass("has-child")) { // Also check if item has child before proceed
			subMenu = item.getSubMenu();
		}
	}

	if (subMenu) {
		child = _this._child;

		if (child) {
			child.hideChild();
		}
		else {
			child = new PopupMenu();
			child.setCreator(_this);
			child.setAutoHide(false);
			child.setPin("top-right", "top-left");
			child.listen("hidden", _this._onChildHidden);
			 _this._child = child;
		}

		child.setMenu(subMenu);
		child.setAnchor(/** @type {tr.ElementControl} */(item));
		child.show();
		child.updatePosition();

		// Always reset highlighted item of child
		subMenu.highlightItem(null);
	}

	// Always clear hovered item
	_this._hoveredItem = null;
};

/**
 * @ignore
 * @public
 */
tr.PopupMenu.prototype.hideChild = function() {
	var child = this._child;

	if (child) {
		child.hide();
	}
};

/**
 * @ignore
 * @public
 */
tr.PopupMenu.prototype.hideTree = function() {
	var creator = this._creator;

	if (creator) {
		creator.hideTree();
	}

	this.hide();
};

/** {@link tr.PopupMenu#getMenu} 
 * @public
 * @param {tr.ListMenu | Object} menu
 */
tr.PopupMenu.prototype.setMenu = function(menu) {
	var _this, prevMenu, menuMaxHeight;

	_this = this;
	prevMenu = _this._menu;

	_this._topNode.addClass("customvisual");

	if (menu === _this._menu) return;
	else if (!menu) menu = null; // Ensure null

	// Clear existing menu
	if (prevMenu) {
		prevMenu.unlisten("itemHovered", _this._onItemHovered);
		prevMenu.unlisten("itemHighlighted", _this._onItemHighlighted);
		prevMenu.unlisten("treeItemActivated", _this._onTreeItemActivated);
		prevMenu.unlisten("treeItemSelected", _this._onTreeItemSelected);
		prevMenu.unlisten("keydown", _this._onKeyDown);

		if (prevMenu.getElement()) {
			prevMenu.setParent(null);
		}
	}

	if (menu) { // Set menu
		// Create object if parameter is model
		if (!(menu instanceof tr.ListMenu)) {
			menu = new ListMenu(menu);
		}

		_this.setContent(menu);
		menu.setHighlightedItem(null);
		menu.listen("itemHovered", _this._onItemHovered);
		menu.listen("itemHighlighted", _this._onItemHighlighted);
		menu.listen("itemActivated", _this._onItemActivated);
		menu.listen("itemSelected", _this._onItemSelected);
		menu.listen("treeItemActivated", _this._onTreeItemActivated);
		menu.listen("treeItemSelected", _this._onTreeItemSelected);

		if (_this._handleKeyDown) {
			menu.listen("keydown", _this._onKeyDown);
		}

		menuMaxHeight = menu.getStyle("max-height");

		if (menuMaxHeight) {
			_this.setMaxHeight(menuMaxHeight);
		}
	}

	_this._menu = menu;
};

/** All custom or user settings should be done here
 * @protected
 * @ignore
 * @param {*} opt_model
 */
tr.PopupMenu.prototype._deserialize = function(opt_model) {
	if (!opt_model) opt_model = {};

	var pref;

	pref = opt_model["popup"];
	if (pref) {
		tr.PopupMenu.base(this, "_deserialize", pref); // For backward compat: model is set in property popup
	}
	else {
		tr.PopupMenu.base(this, "_deserialize", opt_model); // Model should actually be the same as normal popup
	}

	pref = opt_model["handleKeyDown"];
	if (pref === false) {
		this._handleKeyDown = false;
	}

	pref = opt_model["menu"];
	if (pref) {
		this.setMenu(pref);
	}
};

//#region private methods

/**
 * @private
 */
tr.PopupMenu.prototype._onKeyDown = function(event) {
	var _this = this;

	switch (event.keyCode) {
		case 37: // Left
			if (_this._creator) {
				_this.hide();
			}
			break;

		case 39: // Right
			_this.showChild();
			break;

		case 27: // ESC
			_this.hideTree();
			break;
	}
};

/**
 * @private
 */
tr.PopupMenu.prototype._onChildHidden = function(event) {
	this.focus();
};

/**
 * @private
 */
tr.PopupMenu.prototype._onItemHovered = function(event) {
	var _this, item;

	_this = this;
	item = event["item"];

	if (item === _this._hoveredItem) return; // Do nothing if hovered item is not changed

	_this._hoveredItem = item;
};

/**
 * @private
 */
tr.PopupMenu.prototype._onItemHighlighted = function(event) {
	var _this, item;

	_this = this;
	item = event["item"];

	// Dispatch itemHighlighted event
	_this._dispatch("itemHighlighted", event);

	if (item === _this._hoveredItem) {
		if (item.getSubMenu()) {
			_this.showChild();
		}
		else {
			_this.hideChild();
		}
	}
};

 * @private

tr.PopupMenu.prototype._onItemActivated = function(event) {
