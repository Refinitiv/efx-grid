goog.provide("tr.grid.Virtualizer");

goog.require("tr.res");
goog.require("tr.grid.TrackLayout");
goog.require("tr.EventDispatcher");

/** @constructor
* @param {!tr.grid.TrackLayout} trackLayout
* @extends {tr.EventDispatcher}
*/
tr.grid.Virtualizer = function (trackLayout) {
	this._layout = trackLayout;
	this._arg = {};
	this._addEvent("indexChanged");
};
goog.inherits(tr.grid.Virtualizer, tr.EventDispatcher); // Inherits() will replace prototype of the derived object

* @type {!tr.grid.TrackLayout}

tr.grid.Virtualizer.prototype._viewSize = 0;

tr.grid.Virtualizer.prototype._viewOffset = 0;

tr.grid.Virtualizer.prototype._startOffsetCount = 0;

tr.grid.Virtualizer.prototype._endOffsetCount = 0;

tr.grid.Virtualizer.prototype._bufferSize = 0;

tr.grid.Virtualizer.prototype._active = true;

tr.grid.Virtualizer.prototype._virtualizable = false;

/** @private
* @return {number}
*/
tr.grid.Virtualizer.prototype._getMinIndex = function () {
	return this._startOffsetCount;
};
/** @private
* @return {number}
*/
tr.grid.Virtualizer.prototype._getMaxIndex = function () {
	var index = this._layout.getLaneCount() - this._endOffsetCount;
	return index > this._startOffsetCount ? index : this._startOffsetCount;
};

/** @public
* @param {number} index
* @return {number}
*/
tr.grid.Virtualizer.prototype.getContentStart = function (index) {
	return this._layout.getLaneStart(index);
};
/** @public
* @return {number}
*/
tr.grid.Virtualizer.prototype.getContentInViewCount = function () {
	var first = this.getFirstIndexInView();
	var last = this.getLastIndexInView();
	if (first >= 0 && first < last) {
		return last - first + 1;
	}
	return 0;
};

/** @private
*/
tr.grid.Virtualizer.prototype._getContentSizeInBounds = function () {
	var minIndex = this._getMinIndex();
	var maxIndex = this._getMaxIndex();
	return this._layout.getLaneStart(maxIndex) - this._layout.getLaneStart(minIndex);
};

/** @public
* @return {number}
*/
tr.grid.Virtualizer.prototype.getViewSize = function () {
	return this._viewSize;
};


/** @public
* @return {number}
*/
tr.grid.Virtualizer.prototype.getViewOffset = function () {
	var minIndex = this._getMinIndex();
	return this._viewOffset + this._layout.getLaneStart(minIndex);
};
/** @public
* @return {number}
*/
tr.grid.Virtualizer.prototype.getIndexOffset = function () {
	return this._startOffsetCount;
};

/** @public
* @return {boolean}
*/
tr.grid.Virtualizer.prototype.isVirtualizable = function () {
	return this._virtualizable;
};

/** @public
* @return {number}
*/
tr.grid.Virtualizer.prototype.getFirstIndexInView = function () {
	if (!this._virtualizable) {
		return 0;
	}
	var vo = this.getViewOffset();
	var trackSize = this._layout.getTrackSize();
	if(vo >= trackSize) {
		vo = trackSize - this._viewSize;  // WARNING: View offset already includes minIndex and this will override that
	}
	if(vo < 0) {
		return 0;
	}

	return this._layout.hitTest(vo);
};
/** @public
* @return {number} INCLUSIVE last index in view
*/
tr.grid.Virtualizer.prototype.getLastIndexInView = function () {
	if (!this._virtualizable) {
		return this._layout.getLaneCount() - 1;
	} else {
		var viewEnd = this.getViewOffset() + this._viewSize;

		// prevent from invalid-index because of verical buffering
		var maxIndex = this._getMaxIndex();
		var endBound = this._layout.getLaneStart(maxIndex);
		if (viewEnd >= endBound) {
			return maxIndex - 1; // maxIndex is exclusive
		}

		return this._layout.hitTest(viewEnd);
	}
};

/** @public
* @return {number}
*/
tr.grid.Virtualizer.prototype.getPrevFirstIndexInView = function () {
	return (this._arg[tr.res.prevFirstIndex] !== undefined) ? this._arg[tr.res.prevFirstIndex] : -1;
};
/** @public
* @return {number}
*/
tr.grid.Virtualizer.prototype.getPrevLastIndexInView = function () {
	return (this._arg[tr.res.prevLastIndex] !== undefined) ? this._arg[tr.res.prevLastIndex] : -1;
};

/** @public
* @param {boolean=} opt_force
* @return {boolean}
*/
tr.grid.Virtualizer.prototype.update = function (opt_force) {
	if (opt_force !== true) { 
		if (!(this._enabled && this._active)) {
			return false; 
		}
	}

	if (this._getEvent("indexChanged").isDispatching()) { return false; }

	var fi = this.getFirstIndexInView();
	var li = this.getLastIndexInView(); // INCLUSIVE
	var changed = (this._arg[tr.res.prevFirstIndex] !== fi) || (this._arg[tr.res.prevLastIndex] !== li);

	if (opt_force || changed) {
		this._arg[tr.res.firstIndex] = fi;
		this._arg[tr.res.lastIndex] = li;
		
		this._dispatch("indexChanged", this._arg);

		this._arg[tr.res.prevFirstIndex] = fi;
		this._arg[tr.res.prevLastIndex] = li;
	}
	return changed;
};

/** @public
* @return {boolean}
*/
tr.grid.Virtualizer.prototype.isEnabled = function () {
	return this._enabled;
};
/** @public
* @param {boolean=} opt_enabled
*/
tr.grid.Virtualizer.prototype.enable = function (opt_enabled) {
	opt_enabled = (opt_enabled !== false);
	if (this._enabled !== opt_enabled) {
		this._enabled = opt_enabled;
		this.validateVirtualization();
		this.update(true);
	}
};
/** @public
* @param {boolean=} opt_disabled
*/
tr.grid.Virtualizer.prototype.disable = function (opt_disabled) {
	this.enable(opt_disabled === false);
};

		this.validateVirtualization();
	}
};

* @param {number} end
*/
tr.grid.Virtualizer.prototype.setViewBounds = function (start, end) {
	this._startOffsetCount = start > 0 ? start : 0;
	this._endOffsetCount = end > 0 ? end : 0;
	this.validateVirtualization(); // Everytime row height is changed
};
/** @public
* @ignore
* @param {number} px
*/
tr.grid.Virtualizer.prototype.setBufferSize = function (px) {
	if (px < 0) { px = 0; }
	if (this._bufferSize !== px) {
		this._bufferSize = px;
		this.validateVirtualization();
	}
};

/** @public
* @ignore
*/
tr.grid.Virtualizer.prototype.validateVirtualization = function () {
	this._virtualizable = false;
	if (this._enabled && this._active) {
		// If either content or view have no size, or content is smaller than view, entire content is contained within view
		if (this._viewSize > 0) {
			if (this._getContentSizeInBounds() > this._viewSize) {
				this._virtualizable = true;
			}
		}
	}
};
