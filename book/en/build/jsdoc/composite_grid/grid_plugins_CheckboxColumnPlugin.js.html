goog.provide("tr.grid.CheckboxColumnPlugin");

goog.require("tr.Grid");

/** Event
* @event tr.grid.CheckboxColumnPlugin#clicked
* @property {Element} checkbox
* @property {number} rowIndex
* @property {boolean} checked
* @property {string} checkState 
*/

/** @constructor
* @param {number=} opt_displayColumn
* @param {string=} opt_dataColumnName
* @param {boolean=} opt_autoRowSelection
* @param {string=} opt_exceptSectionName
* @extends {tr.grid.Plugin}
* @implements {tr.grid.IPlugin}
*/
tr.grid.CheckboxColumnPlugin = function (opt_displayColumn, opt_dataColumnName, opt_autoRowSelection, opt_exceptSectionName) {
	this._onPreSectionRender = this._onPreSectionRender.bind(this);
	this._onPostSectionDataBinding = this._onPostSectionDataBinding.bind(this);
	this._onCheckBoxClicked = this._onCheckBoxClicked.bind(this);

	this._onColumnMoved = goog.bind(this._onColumnMoved, this);
	this._onColumnAdded = goog.bind(this._onColumnAdded, this);
	this._onColumnRemoved = goog.bind(this._onColumnRemoved, this);

	if (opt_exceptSectionName) {
		this._exceptSectionName = {};
		for (var i=0;i<opt_exceptSectionName.length;i++) {
			this._exceptSectionName[opt_exceptSectionName[i]] = opt_exceptSectionName[i];
		}
	} else {
		this._exceptSectionName = {};
	}
	this._exceptSectionName["bottomPadding"] = true;

	if (opt_displayColumn !== undefined && opt_displayColumn > 0) {
		this._displayColumn = opt_displayColumn;
	}

	if (opt_dataColumnName) {
		this._dataColumnName = opt_dataColumnName || "";
	}
	if (opt_autoRowSelection) {
		this._autoRowSelection = opt_autoRowSelection;
	}

	this._addEvent(tr.res.clicked);
	this._addEvent("checkboxCreated");
};
goog.inherits(tr.grid.CheckboxColumnPlugin, tr.grid.Plugin);

//#region Public
/** @override */
tr.grid.CheckboxColumnPlugin.prototype.getName = function () {
	return "CheckboxColumnPlugin"; // Read Only
};
/** @override */
tr.grid.CheckboxColumnPlugin.prototype.initialize = function (host) {
	this._host = host;
	this._host.listen("preSectionRender", this._onPreSectionRender);
	this._host.listen(tr.res.postSectionDataBinding, this._onPostSectionDataBinding);

	this._host.listen(tr.res.columnMoved, this._onColumnMoved);
	this._host.listen(tr.res.columnAdded, this._onColumnAdded);
	this._host.listen(tr.res.columnRemoved, this._onColumnRemoved);
	this._host.enableColumnClass(this._displayColumn, "tr-checkbox-column", true);
};
/** @override */
tr.grid.CheckboxColumnPlugin.prototype.unload = function (host) {
	this._host.unlisten("preSectionRender", this._onPreSectionRender);
	this._host.unlisten(tr.res.postSectionDataBinding, this._onPostSectionDataBinding);

	this._host.unlisten(tr.res.columnMoved, this._onColumnMoved);
	this._host.unlisten(tr.res.columnAdded, this._onColumnAdded);
	this._host.unlisten(tr.res.columnRemoved, this._onColumnRemoved);
	this._host.enableColumnClass(this._displayColumn, "tr-checkbox-column", false);
	
	this._host = null;
};

/** @public
* @param {tr.Grid.SectionReference} sectionRef
* @param {number} rowIndex
* @return {Element|null}
*/
tr.grid.CheckboxColumnPlugin.prototype.getCheckbox = function (sectionRef, rowIndex) {
	var section = this._host.getSection(sectionRef);
	if (section) {
		return this._getCheckbox(section, rowIndex);
	}

	return null;
};

/** @public
* @return {number}
*/
tr.grid.CheckboxColumnPlugin.prototype.getDisplayColumn = function () {
	return this._displayColumn;
};
/** @public
* @param {number} colIndex
*/
tr.grid.CheckboxColumnPlugin.prototype.setDisplayColumn = function (colIndex) {
	if(this._displayColumn === colIndex) {
		return null;
	}
	
	this._host.enableColumnClass(this._displayColumn, "tr-checkbox-column", false);
	this._displayColumn = colIndex;
	this._host.enableColumnClass(this._displayColumn, "tr-checkbox-column", true);
	
	var sectionList = this._host.getAllSectionSettings();
	for(var i = sectionList.length; --i >= 0;) {
		this._createCheckboxes(sectionList[i]);
	}
};

* @return {string}

	return this._dataColumnName;
};
/** @public
* @param {string} dataColumnName
*/
tr.grid.CheckboxColumnPlugin.prototype.setDataColumnName = function (dataColumnName) {
	this._dataColumnName = dataColumnName || "";
};

/** @public
* @param {boolean} check
*/
tr.grid.CheckboxColumnPlugin.prototype.setAllCheckStates = function (check) {
	var state = this._toCheckState(check);
	var titles = this._host.getAllSectionSettings(tr.res.title);
	var rootSettings;
	for(var i = 0; i < titles.length; ++i) { // Find the root settings
		if(titles[i].getChildren()) {
			rootSettings = titles[i];
			break;
		}
	}
	if(!rootSettings) { return; }
	
	this._setSectionState(rootSettings, state);
	this._setDescendantCheckState(rootSettings, state);
};

/** @public
* @param {tr.Grid.SectionReference} sectionRef
* @param {string=} opt_state Specifies state for searching. Default is "checked"
* @return {Array.<number>} the indices are relative to the specified section
*/
tr.grid.CheckboxColumnPlugin.prototype.getAllCheckedIndices = function (sectionRef, opt_state) {
	if (opt_state === undefined) { opt_state = tr.res.checked; }

	var ary = [];
	var sectionSettings = this._host.getSectionSettings(sectionRef);
	var i;
	if (sectionSettings !== null) {
		if (this._isDataBound(sectionSettings)) {
			var dataSource = sectionSettings.getDataSource();
			var data = dataSource.getColumnData(this._dataColumnName); // Exclude header rows
			for (i = data.length; --i >= 0; ) {
				if (this._toCheckState(data[i]) === opt_state) {
					ary.push(i);
				}
			}
		} else {
			var checkboxes = this._getAllCheckboxes(sectionSettings);
			for (i = checkboxes.length; --i >= 0; ) {
				if (this._getCheckState(checkboxes[i]) === opt_state) {
					ary.push(i);
				}
			}
		}		
	}
	return ary;
};

/** @private
* @param {tr.grid.SectionSettings} sectionSettings
* @param {string|null} state
*/
tr.grid.CheckboxColumnPlugin.prototype._setDescendantCheckState = function (sectionSettings, state) {
	var children = sectionSettings.getChildren();
	if (children === null) { return; }

	for (var i = 0; i < children.length; ++i) {
		var child = children[i];
		this._setSectionState(child, state);
		this._setDescendantCheckState(child, state);
	}
};
/** @private
* @param {tr.grid.SectionSettings} sectionSettings
* @param {string|null=} opt_callerState
*/
tr.grid.CheckboxColumnPlugin.prototype._syncAscendantCheckState = function (sectionSettings, opt_callerState) {
	var parent = sectionSettings.getParent();
	if (parent === null || opt_callerState === null || !parent.getSection().isVisible()) { return; }

	var state = opt_callerState || null;
	if (state !== tr.res.partial) {
		if (opt_callerState === undefined) {
			sectionSettings = null;
		} 

		var children = parent.getChildren();
		if(children) {
			for (var i = children.length; --i >= 0; ) {
				var child = children[i];
				if (child !== sectionSettings) {
					var childState = this._getSectionState(child);
					state = this._mergeState(state, childState);
					if (state === tr.res.partial) {
						break;
					}
				}
			}
		}
	}

	this._setSectionState(parent, state);
	this._syncAscendantCheckState(parent, state);
};

/** @private
 * @param {tr.grid.SectionSettings} sectionSettings
 * @param {string|null} state
 */
tr.grid.CheckboxColumnPlugin.prototype._setSectionState = function (sectionSettings, state) {
	if (state === null) { return; }

	var i;
	if (this._isDataBound(sectionSettings)) {
		var bool = (state !== tr.res.unchecked);
		var dt = /** @type{tr.DataTable} */(sectionSettings.getRootDataSource());
		var rowIds = dt.getAllRowIds(true); // Getting shallow copy
		dt.setColumnData(this._dataColumnName, this._generateData(rowIds, bool), rowIds);
	} else {
		var checkboxes = this._getAllCheckboxes(sectionSettings);
		for (i = checkboxes.length; --i >= 0; ) {
			this._updateCheckbox(checkboxes[i], state);
		}
	}
};

/** @private
 * @param {tr.ILayoutGrid} section
 * @param {number} rowIndex
 * @return {Element}
 */
tr.grid.CheckboxColumnPlugin.prototype._getCheckbox = function (section, rowIndex) {
	var chkbox = section.getCellContent(this._displayColumn, rowIndex);
	if(chkbox) {
		if(chkbox.type === tr.res.checkbox) {
			return /** @type{Element} */(chkbox);
		}
	}
	return null;
};
/** @private
 * @param {tr.grid.SectionSettings} sectionSettings
 * @param {Array.<Element>=} opt_ary
 * @return {!Array.<Element>}
 */
tr.grid.CheckboxColumnPlugin.prototype._getAllCheckboxes = function (sectionSettings, opt_ary) {
	opt_ary = opt_ary || [];

	var section = sectionSettings.getSection();
	var fi = section.getFirstIndexInView();
	var li = section.getLastIndexInView();
	for(var r = fi; r <= li; ++r) {
		var chkbox = this._getCheckbox(section, r);
		if(chkbox) {
			opt_ary.push(chkbox);
		}
	}
	return opt_ary;
};
/** @private
 * @param {tr.grid.SectionSettings} sectionSettings
 * @return {string|null}
*/
tr.grid.CheckboxColumnPlugin.prototype._getSectionState = function (sectionSettings) {
	var section = sectionSettings.getSection();

	var cumulativeState = null;
	var i, state;
	if (this._isDataBound(sectionSettings)) {
		var dataSource = sectionSettings.getDataSource();
		var data = dataSource.getColumnData(this._dataColumnName); // Exclude header rows
		for (i = data.length; --i >= 0; ) {
			cumulativeState = this._mergeState(cumulativeState, this._toCheckState(data[i]));
			if (cumulativeState === tr.res.partial) {
				break;
			}
		}
	} else {
		var checkboxes = this._getAllCheckboxes(sectionSettings);
		for (i = checkboxes.length; --i >= 0; ) {
			var checkbox = checkboxes[i];
			state = this._getCheckState(checkbox);
			cumulativeState = this._mergeState(cumulativeState, state);
			if (cumulativeState === tr.res.partial) {
				break;
			}
		}
	}
	return cumulativeState;
};

/** @private
 * @param {string|null} curState
 * @param {string|null} state
 * @return {string|null}
 */
tr.grid.CheckboxColumnPlugin.prototype._mergeState = function(curState, state) {
	if(curState) {
		if(state && curState !== state) {
			return tr.res.partial;
		}
	} else {
		return state;
	}
	return state;
};

/** @private
 * @param {tr.grid.SectionSettings} sectionSettings
 * @return {boolean}
 */
tr.grid.CheckboxColumnPlugin.prototype._isDataBound = function (sectionSettings) {
	if (sectionSettings.getType() === tr.res.content) {
		if (this._dataColumnName) {
			if (sectionSettings.getDataSource()) {
				return true;
			}
		}
	}
	return false;
};

/** @private
 * @param {*} val
 * @return {string}
 */
tr.grid.CheckboxColumnPlugin.prototype._toCheckState = function (val) {
	if (val === 'partial') {
		return tr.res.partial;
	} else {
		return (val) ? tr.res.checked : tr.res.unchecked;		
	}
};
/** @private
 * @param {Element} checkbox
 * @param {string} state
 * @return {boolean}
 */
tr.grid.CheckboxColumnPlugin.prototype._updateCheckbox = function (checkbox, state) {
	var prevState = checkbox.checked;
	// Process change only when necessary
	if (state !== prevState) {
		// Indeterminate cannot be used with :checked CSS selector
		if(this._indetermination) {
			if(checkbox["indeterminate"] !== undefined) {
				checkbox.indeterminate = (state === tr.res.partial);
			}
			checkbox.checked = (state === tr.res.checked);
		} else {
			checkbox.checked = (state === tr.res.checked || state === tr.res.partial);
		}
		if(state === tr.res.partial) {
			checkbox.classList.add("square-shape");
		} else {
			checkbox.classList.remove("square-shape");
		}
		return true;
	}

	return false;
};
/** @private
 * @param {Element} checkbox
 * @return {string|null}
 */
tr.grid.CheckboxColumnPlugin.prototype._getCheckState = function (checkbox) {
	if(checkbox.style.display == "none") {
		return null;
	}
	
	if (checkbox.classList.contains("square-shape")) {
		return tr.res.partial;
	}
	else if (checkbox.checked) {
		return tr.res.checked;
	}

	return tr.res.unchecked;
};

/** @private
* @param {Event} e
*/
tr.grid.CheckboxColumnPlugin.prototype._onCheckBoxClicked = function (e) {
	var checkbox = /** @type{Element} */(e.currentTarget);
	var checkState = this._getCheckState(checkbox);
	var parentState = checkState;
	var pos = this._host.getRelativePosition(e); // Getting row index
	var sectionSettings = pos[tr.res.sectionSettings];
	
	if (this._isDataBound(sectionSettings)) {
		var rowIndex = pos[tr.res.rowIndex];
		var dv = sectionSettings.getDataSource();
		var rid = dv.getRowId(rowIndex);
		var bool = checkState !== tr.res.unchecked;
		if(dv.hasGroup()) {
			var dt = sectionSettings.getRootDataSource();
			dt.freeze();
		}
		if(dv.getRowData(rid)) {
			dv.setData(rid, this._dataColumnName, bool);
		} else { // Group header
			var subGroup = dv.getGroup(rid);
			if(subGroup) {
				var rids = subGroup.getAllRowIds();
				subGroup.setColumnData(this._dataColumnName, this._generateData(rids, bool), rids);
			}
		}
		if(dt) {
			dt.unfreeze(); // Force update to all header rows
		}
	}

	this._setDescendantCheckState(sectionSettings, checkState);
	if (sectionSettings.getType() === tr.res.content) {
		parentState = this._getSectionState(sectionSettings);
	}
	this._syncAscendantCheckState(sectionSettings, parentState);

	if (this._hasListener(tr.res.clicked)) {
		pos[tr.res.checkbox] = checkbox;
		pos["checkState"] = checkState;
		pos[tr.res.checked] = checkbox.checked;
		this._dispatch(tr.res.clicked, pos);
	}

	e.stopPropagation();
};
/** @private
* @param {Object} e
*/
tr.grid.CheckboxColumnPlugin.prototype._onPreSectionRender = function (e) {
	if(this._displayColumn >= e[tr.res.fromColIndex] && this._displayColumn < e[tr.res.toColIndex]) {
		this._createCheckboxes(e[tr.res.sectionSettings], e[tr.res.fromRowIndex], e[tr.res.toRowIndex]);
	}
};
/** @private
* @param {tr.grid.SectionSettings} settings
* @param {number=} opt_firstIndex
* @param {number=} opt_lastIndex
 */
tr.grid.CheckboxColumnPlugin.prototype._createCheckboxes = function (settings, opt_firstIndex, opt_lastIndex) {
	if (this._exceptSectionName[settings.getSectionName()] !== undefined) {
		return;
	}

	var section = settings.getSection();
	var fi = (opt_firstIndex !== undefined) ? opt_firstIndex : section.getFirstIndexInView();
	var li = (opt_lastIndex !== undefined) ? opt_lastIndex : section.getLastIndexInView() + 1;
	for (var r = fi; r < li; ++r) {
		var cell = section.getCell(this._displayColumn, r);
		if(cell && cell.getElement()) {
			var chkbox = this._getCheckbox(section, r);
			if(!chkbox) {
				var newId = "trgrid_cbcplugin_cb" + tr.grid.CheckboxColumnPlugin._runningCheckboxId++;
				chkbox = document.createElement("input");
				chkbox.type = tr.res.checkbox;
				chkbox.addEventListener(tr.res.click, this._onCheckBoxClicked, false);
				chkbox.id = newId;

				var event = {};
				event[tr.res.checkbox] = chkbox;
				event[tr.res.section] = section;
				event[tr.res.rowIndex] = r;
				this._dispatch("checkboxCreated", event);

				cell.setContent(chkbox);

				var label = document.createElement("label");
				label.setAttribute('for', newId);
				chkbox.parentElement.appendChild(label);
			}
		}
	}
};
/** @private
* @param {Object} e 
*/
tr.grid.CheckboxColumnPlugin.prototype._onPostSectionDataBinding = function (e) {
	if (!this._dataColumnName) { return; }
	if (e[tr.res.sectionType] !== tr.res.content) { return; }

	var section = /** @type{tr.ILayoutGrid} */(e[tr.res.section]);
	var dv = /** @type{tr.DataView} */(e[tr.res.dataSource]);
	var fromR = /** @type{number} */(e[tr.res.fromRowIndex]);
	var toR = /** @type{number} */(e[tr.res.toRowIndex]);
	var dirty = false;

	for (var r = fromR; r < toR; ++r) {
		var chkbox = this._getCheckbox(section, r);
		if(!chkbox) {
			continue;
		}
		
		var state = null;
		var rowId = dv.getRowId(r); // Slow
		var rowData = dv.getRowData(rowId);
		if(!rowData) { // Group row
			var subGroup = dv.getGroup(rowId);
			if(subGroup) {
				var rids = subGroup.getAllRowIds();
				for(var i = rids.length; --i >= 0;) {
					var data = subGroup.getData(rids[i], this._dataColumnName);
					state = this._mergeState(state, this._toCheckState(data));
					if (state === tr.res.partial) {
						break;
					}
				}
			}
			if(!state) {
				state = this._toCheckState(false);
			}
		} else {
			state = this._toCheckState(rowData[this._dataColumnName]);
		}
	
		dirty |= this._updateCheckbox(chkbox, state);
		if (this._autoRowSelection && rowData) { // Select only content row
			section.setSelectedRow(r, state === tr.res.checked);
		}
	}
	if (dirty) {
		this._syncAscendantCheckState(e[tr.res.sectionSettings]);
	}
};

/** @private
* @param {Array.<string>} rowIds
* @param {boolean} bool
* @return {Array.<boolean>}
*/
tr.grid.CheckboxColumnPlugin.prototype._generateData = function (rowIds, bool) {
	var data = [];
	for(var i = rowIds.length; --i >= 0;) { data[i] = bool; }
	return data;
};

	this._displayColumn = tr.moveIndex(this._displayColumn, colIndex);
};
