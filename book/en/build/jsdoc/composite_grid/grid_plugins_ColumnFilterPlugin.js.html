/**
 * @constructor
 * @extends {tr.grid.Plugin}
 * @implements {tr.grid.IPlugin}
 */
tr.grid.ColumnFilterPlugin = function () {
	var _t = this;

	_t._onPreSectionRender = _t._onPreSectionRender.bind(_t);
	_t._onColumnAdded = _t._onColumnAdded.bind(_t); // required
	_t._onColumnMoved = _t._onColumnMoved.bind(_t);
	_t._onColumnRemoved = _t._onColumnRemoved.bind(_t);  // TODO: Remove column Filters as column is removed

	_t._onIconClick = _t._onIconClick.bind(_t);
	_t._filterLogic = _t._filterLogic.bind(_t);
	_t._updateAllColumnIcons = _t._updateAllColumnIcons.bind(_t);
	_t.refresh = _t.refresh.bind(_t);
	_t._setTimerForUpdatingLogic = _t._setTimerForUpdatingLogic.bind(_t);

	_t._nameMap = {};

	_t._globalFilters = [];
	_t._columnFilters = [];
	_t._colData = [];
	_t._hosts = [];
	_t._dvs = [];

	_t._addEvent("click");
	_t._addEvent("iconCreated");
};
goog.inherits(tr.grid.ColumnFilterPlugin, tr.grid.Plugin);

/**
 * @override
 */
tr.grid.ColumnFilterPlugin.prototype.getName = function () {
	return "ColumnFilterPlugin"; // read only
};

/**
 * @override
 */
tr.grid.ColumnFilterPlugin.prototype.initialize = function (host) {
	if (this._hosts.indexOf(host) >= 0) {
		return;
	}

	this._hosts.push(host);

	host.listen("preSectionRender", this._onPreSectionRender);
	host.listen("postDataSourceChanged", this._setTimerForUpdatingLogic);
	host.listen(tr.res.columnAdded, this._onColumnAdded);
	host.listen(tr.res.columnMoved, this._onColumnMoved);
	host.listen(tr.res.columnRemoved, this._onColumnRemoved);
};

/** @override */
tr.grid.ColumnFilterPlugin.prototype.unload = function (host) {
	var at = this._hosts.indexOf(host);

	if (at < 0) {
		return;
	}

	this._hosts.splice(at, 1);

	host.unlisten("preSectionRender", this._onPreSectionRender);
	host.unlisten("postDataSourceChanged", this._setTimerForUpdatingLogic);
	host.unlisten(tr.res.columnAdded, this._onColumnAdded);
	host.unlisten(tr.res.columnMoved, this._onColumnMoved);
	host.unlisten(tr.res.columnRemoved, this._onColumnRemoved);

	if (this._hosts.length <= 0) {
		this.removeAllFilters(); // Discard all user references
		this._overridingFilter = null;
		this._columnFormatter = null;
		this._formatter = null;

		if (this._uiTimerId) {
			clearTimeout(this._uiTimerId);
			this._uiTimerId = 0;
		}

		if (this._dvTimerId) {
			clearTimeout(this._dvTimerId);
			this._dvTimerId = 0;
		}
	}
};


/**
 * @private
 */
tr.grid.ColumnFilterPlugin.prototype._onPreSectionRender = function (e) {
	if (e[tr.res.sectionType] !== tr.res.title) {
		return;
	}

	this._setTimerForUpdatingIcons();
};


/**
 * All filter logics in tr.DataView will be temporarily removed until re-enabling. This is useful when trying to insert
 * an item at the specified index. After re-enabling, both logics and icons will be restored.
 * @public
 * @param {boolean=} opt_disabled
 * @param {string=} opt_id For handling nested called
 * @param {boolean=} opt_delayRefresh For delay {@link #refresh} update.
 * @returns {boolean} true if the filtering is actually disabled or enabled, false if the state does not change
 * (nested actions)
 */
tr.grid.ColumnFilterPlugin.prototype.disable = function (opt_disabled, opt_id, opt_delayRefresh) {
	var disabled = opt_disabled !== false;

	if (!this._disablingIds) {
		this._disablingIds = {};
	}

	if (!opt_id) {
		opt_id = "_";
	}

	this._disablingIds[opt_id] = disabled;

	disabled = false;

	for (var key in this._disablingIds) { // Find cumulative state
		if (this._disablingIds[key]) {
			disabled = true;

			break;
		}
	}

	if (this._disabled !== disabled) {
		this._disabled = disabled;

		if (opt_delayRefresh) {
			return true;
		}

		if (disabled) {
			this.refresh(); // Take an effect immediately
			this._updateAllColumnIcons();
		} else {
			this._setTimerForUpdatingLogic(); // Take an effect later
			this._setTimerForUpdatingIcons();
		}

		return true;
	}

	return false;
};

/**
 * Add a global filter that is not related to any column. No filter icon is created in this way
 * @public
 * @param {Function} func
 * @param {string=} funcName Optional function name for referencing the given function
 * @return {string} If success, function name is returned, otherwise empty string
 */
tr.grid.ColumnFilterPlugin.prototype.addFilter = function (func, funcName) {
	if (!func) {
		return "";
	}

	if (!funcName) {
		funcName = "_" + this._filterId++;
	}

	if (this._nameMap[funcName] || this._globalFilters.indexOf(func) >= 0) {
		return ""; // Function is already exists
	}

	this._nameMap[funcName] = func;
	this._globalFilters.push(func);
	this._setTimerForUpdatingLogic();

	return funcName;
};

/**
 * Add a column filter. Column filter icon is created.
 * @public
 * @param {Function} func
 * @param {number} colIndex
 * @return {boolean} True if success
 */
tr.grid.ColumnFilterPlugin.prototype.addColumnFilter = function (colIndex, func) {
	if (!func) {
		return false;
	}

	var cdata = this._colData[colIndex];

	if (!cdata) {
		cdata = {};
		cdata._filters = [];

		this._colData[colIndex] = cdata;
	}

	if (cdata._filters.indexOf(func) >= 0 || this._columnFilters.indexOf(func) >= 0) {
		return false;
	}

	cdata._filters.push(func);

	this._columnFilters.push(func);
	this._setTimerForUpdatingLogic();
	this._updateColumnIcon(colIndex);

	return true;
};

/**
 * Remove the registered global filter. The column filter cannot be removed
 * @public
 * @param {Function|string} funcRef Function object or the function name can be used to remove
 * @return {boolean} True if success
 */
tr.grid.ColumnFilterPlugin.prototype.removeFilter = function (funcRef) {
	var func = null;
	var funcName = "";
	var funcIndex = -1;

	if (typeof funcRef === "string") {
		funcName = /** @type{string} */(funcRef);
		func = this._nameMap[funcRef];
		funcIndex = this._globalFilters.indexOf(func);
	} else {
		func = /** @type{Function} */(funcRef);
		funcIndex = this._globalFilters.indexOf(func);

		if (funcIndex >= 0) {
			for (var key in this._nameMap) {
				if (this._nameMap[key] === func) {
					funcName = key;
				}
			}
		}
	}

	if (funcIndex < 0) {
		return false;
	}

	delete this._nameMap[funcName];

	this._globalFilters.splice(funcIndex, 1);
	this._setTimerForUpdatingLogic();

	return true;
};

/**
 * Remove all filters in the specified column
 * @public
 * @param {number} colIndex
 * @return {boolean} True if there is any change
 */
tr.grid.ColumnFilterPlugin.prototype.removeColumnFilters = function (colIndex) {
	var cdata = this._colData[colIndex];

	if (!cdata) {
		return false;
	}

	for (var i = cdata._filters.length; --i >= 0;) {
		var funcIndex = this._columnFilters.indexOf(cdata._filters[i]);

		if (funcIndex >= 0) {
			this._columnFilters.splice(funcIndex, 1);
		}
	}

	this._colData[colIndex] = null;
	this._setTimerForUpdatingLogic();
	this._updateColumnIcon(colIndex);

	return true;
};

/**
 * Remove all column filters, excluding global filters
 * @public
 */
tr.grid.ColumnFilterPlugin.prototype.removeAllColumnFilters = function () {
	if (this._columnFilters.length) {
		this._columnFilters.length = 0;
		this._colData.length = 0;
		this._updateAllColumnIcons();

		this._setTimerForUpdatingLogic();
	}
};

/**
 * Remove all column and global filters. Exclusion is not removed by this method
 * @public
 */
tr.grid.ColumnFilterPlugin.prototype.removeAllFilters = function () {
	var totalFilter = this._globalFilters.length + this._columnFilters.length;

	if (totalFilter <= 0) {
		return;
	}

	if (this._globalFilters.length) {
		this._globalFilters.length = 0;
		this._nameMap = {};
	}

	this.removeAllColumnFilters();
	this._setTimerForUpdatingLogic();
};

/**
 * Used by compositeGrid to get current filter states.
 * @public
 * @return {Array.<Object.<string, Function>>}
 */
tr.grid.ColumnFilterPlugin.prototype.getFilterStates = function () {
	var filterState = [];

	for (var k in this._nameMap) {
		filterState.push({
			"filterId": k,
			"filterFunction": this._nameMap[k]
		});
	}

	return filterState;
};

/**
 * Column overriding function is processed before column filters and after global filters. If it returns non-zero, the result will override all other column filter results. If it returns 1, the row will be always included. If it returns -1, the row will be always excluded. If it returns 0, the column filters will be used as normal.
 * @public
 * @param {Function} func
 */
tr.grid.ColumnFilterPlugin.prototype.setOverridingFilter = function (func) {
	this._overridingFilter = func;
};

/**

 * @public

 */
tr.grid.ColumnFilterPlugin.prototype.setFormatter = function (func) {
	this._formatter = func;

tr.grid.ColumnFilterPlugin.prototype.setColumnFormatter = function (func) {
	this._columnFormatter = func;

/**
 * @public
 * @param {number} colIndex
 * @return {Array.<Element>} icons
 */
tr.grid.ColumnFilterPlugin.prototype.getColumnFilterIcons = function (colIndex) {
	var icons = [];

	for (var i = this._hosts.length; --i >= 0;) {
		var host = this._hosts[i];
		var tsect = host.getSection(tr.res.title);

		if (!tsect) {
			continue;
		}

		var cell = tsect.getCell(colIndex, 0); // Get the top most column

		if (cell && cell._filterIcon) {
			icons.push(cell._filterIcon);
		}
	}

	return icons;
};

/**
 * @public
 */
tr.grid.ColumnFilterPlugin.prototype.updateAllColumnIcons = function () {
	if (this._uiTimerId) {
		clearTimeout(this._uiTimerId);

		this._uiTimerId = 0;
	}

	this._updateAllColumnIcons();
};

	if (at < 0) {
		this._dvs.push(dv);
		this._setTimerForUpdatingLogic();
	}

tr.grid.ColumnFilterPlugin.prototype.removeDataView = function (dv) {
	var at = this._dvs.indexOf(dv);

	if (at >= 0) {
		this._dvs.splice(at, 1);
		this._setTimerForUpdatingLogic();
	}
};

/**
 * @private
 * @param {string} rid
 * @param {Object} rowData
 * @return {boolean}
 */
tr.grid.ColumnFilterPlugin.prototype._filterLogic = function (rid, rowData) {
	if (rowData) {
		if (this._formatter) {
			rowData = this._formatter(rowData);
		}

		var i;
		var len = this._globalFilters.length;

		for (i = 0; i < len; ++i) {
			if (!this._globalFilters[i](rowData, rid)) {
				return false;
			}
		}

		len = this._activeColFilters.length;

		if (len) {
			if (this._overridingFilter) {
				var ret = this._overridingFilter(rowData, rid);
				if (ret) {
					return (ret > 0);
				}
			}

			if (this._columnFormatter) {
				rowData = this._columnFormatter(rowData);
			}

			for (i = 0; i < len; ++i) {
				if (!this._activeColFilters[i](rowData, rid)) {
					return false;
				}
			}
		}
	}

	return true;
};

/**
 * @private
 */
tr.grid.ColumnFilterPlugin.prototype._setTimerForUpdatingLogic = function() {

	if (this._dvTimerId) {
		return;
	}

	if (this._filterDelay) {
		this._dvTimerId = setTimeout(this.refresh, this._filterDelay);
	} else {
		this.refresh();
	}
};

/**
 * @private
 */
tr.grid.ColumnFilterPlugin.prototype._setTimerForUpdatingIcons = function () {
	if (!this._uiTimerId) {
		this._uiTimerId = setTimeout(this._updateAllColumnIcons, 10);
	}
};

/**
 * @description Perform filtering with the same parameters
 * @public
 * @suppress {checkTypes}
 */
tr.grid.ColumnFilterPlugin.prototype.refresh = function () {
	this._dvTimerId = 0;

	var cstkp = this._getPlugin("ColumnStackingPlugin");

	this._activeColFilters = this._columnFilters;

	if (cstkp) {
		var colCount = this._colData.length;
		var hasInactiveColumn = false;

		for (var colIndex = 0; colIndex < colCount; ++colIndex) {
			var cdata = this._colData[colIndex];

			if (cdata) {
				if (cstkp.isInactiveStackedColumn(colIndex)) {
					hasInactiveColumn = true;

					break;
				}
			}
		}

		if (hasInactiveColumn) {
			this._activeColFilters = [];

			for (var colIndex = 0; colIndex < colCount; ++colIndex) {
				var cdata = this._colData[colIndex];

				if (cdata && !cstkp.isInactiveStackedColumn(colIndex)) {
					this._activeColFilters = this._activeColFilters.concat(cdata._filters);
				}
			}
		}
	}

	var totalFilter = this._globalFilters.length + this._activeColFilters.length;
	var logic = (!this._disabled && totalFilter) ? this._filterLogic : null;
	var i;
	var len = this._dvs.length;

	if (len <= 0) {
		len = this._hosts.length;

		for (i = len; --i >= 0;) {
			var host = this._hosts[i];
			var dv = host.getDataSource();

			if (dv) {
				dv.filterIn(logic);
			}
		}
	} else {
		for (i = len; --i >= 0;) {
			this._dvs[i].filterIn(logic);
		}
	}
};

/**
 * @private
 * @param {number} colIndex
 */
tr.grid.ColumnFilterPlugin.prototype._updateColumnIcon = function (colIndex) {
	var cdata = this._colData[colIndex];
	var hasFilter = cdata ? cdata._filters.length : 0;

	for (var i = this._hosts.length; --i >= 0;) {
		var host = this._hosts[i];
		var tsect = host.getSection(tr.res.title);

		if (!tsect) {
			continue;
		}

		var cell = tsect.getCell(colIndex, 0); // Get the top most column

		if (!cell) {
			continue;
		}

		// HACK: Update cell padding for all type of icons (sorting and column grouping icons)
		cell.updatePadding();

		if (!this._disabled && hasFilter) {
			if (!cell._filterIcon) {
				cell.addClass("filtering");

				var filterIcon;

				if (tr.ElementFrameWork.isElfv3()) {
					filterIcon = document.createElement("ef-icon");
					filterIcon.setAttribute("icon", "filter");
				} else {
					filterIcon = document.createElement("div");
				}

				filterIcon.className = "filter-symbol icon-filter ef-icon";
				filterIcon.addEventListener("click", this._onIconClick, false);
				this._dispatch("iconCreated", {"icon": filterIcon, "colIndex": colIndex, "grid": host});

				cell.insertFloatingIcon(filterIcon, 10);
				cell._filterIcon = filterIcon;
			}
		} else {
			if (cell._filterIcon) {
				cell.removeFloatingIcon(cell._filterIcon);
				cell._filterIcon = null;
				cell.removeClass("filtering");
			}
		}
	}
};

/**
 * @private
 */
tr.grid.ColumnFilterPlugin.prototype._updateAllColumnIcons = function () {
	this._uiTimerId = 0;

	var host = this._hosts[0];
	var colCount = host ? host.getColumnCount() : 0;

	for (var i = 0; i < colCount; ++i) {
		this._updateColumnIcon(i);
	}
};
