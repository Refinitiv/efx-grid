* @ignore

tr.grid.SelectionList = function () {
	this._selections = [];
};
/** Clone all members
* @public
* @return {!tr.grid.SelectionList}
*/
tr.grid.SelectionList.prototype.clone = function () {
	// IMPORTANT NOTE: If you add any field or member to this class, clone it over here too.
	var newObj = new tr.grid.SelectionList();
	newObj._selections = this._selections.concat();
	newObj._count = this._count;
	newObj._anchor = this._anchor;
	newObj._firstIndex = this._firstIndex;
	newObj._lastIndex = this._lastIndex;
	return newObj;
};

/** @public
* @return {string}
*/
tr.grid.SelectionList.prototype.toString = function() {
	return this._selections.join(", ");
};
/** If selection count reach its limit, new selection will not be set
* @public
* @param {number} at Negative index is not allowed
* @return {boolean} True new selection is added. False otherwise (i.e. selecting the selected index)
*/
tr.grid.SelectionList.prototype.select = function(at) {
	if(at < 0) {
		return false;
	}

	if(this._select(at)) {
		this._anchor = at;
		if(this._firstIndex < 0 || at < this._firstIndex) {
			this._firstIndex = at;
		}
		if(this._lastIndex < 0 || at > this._lastIndex) {
			this._lastIndex = at;
		}
		return true;
	}
	return false;
};
/** WARNING: Prefer clearAllSelections() to this method
* @public
* @param {number} at
* @return {boolean}
*/
tr.grid.SelectionList.prototype.deselect = function (at) {
	if (this._selections[at]) {
		this._selections[at] = false;
		--this._count;
		if (this._anchor === at) {
			this._anchor = -1; // No anchor
		}

		if (this._firstIndex === at) {
			this._firstIndex = this._findNextSelection(at);
		}
		if (this._lastIndex === at) {
			this._lastIndex = this._findPrevSelection(at);
		}
		return true;
	}
	return false;
};
/** @public
* @param {number} at
*/
tr.grid.SelectionList.prototype.toggleSelection = function(at) {
	if(this._selections[at]) {
		this.deselect(at);
	} else {
		this.select(at);
	}
};
/** Force selection to be left only the selected one, in case of containing multiple selections
* Additionally, move the previous selection to the new one, if one selection has been selected
* Anchor is also moved.
* @public
* @param {number} at Negative index will clear all selections
* @return {boolean} True new selection is added. False otherwise (i.e. selecting the selected index)
*/
tr.grid.SelectionList.prototype.singularlySelect = function (at) {
	if (this._count !== 1 || at < 0) {
		this.clearAllSelections();
		this.select(at);
		return true;
	}
	if (this._firstIndex === at) {
		return false;
	}
	// Move the old selection to the new place
	this._selections[this._anchor] = false;
	this._selections[at] = true;
	this._anchor = at;
	this._firstIndex = at;
	this._lastIndex = at;
	return true;
};

/** Superior performance than calling select multiple times
* Use this method to immitate selectAll() behavior
* Negative length is allowed. It will be inverse selectRange
* @public
* @param {number} anchor Negative value is allowed and will be adjusted to zero
* @param {number} length This will be adjusted according to max selection index
*/
tr.grid.SelectionList.prototype.selectRange = function (anchor, length) {
	if(length === 0) { return; }
	if (anchor < 0) { anchor = 0; }
	if (length === 1 || length === -1) {
		this.select(anchor);
		return;
	}
	
	var at = anchor;
	var endIndex = at + length;
	if (length < 0) {
		endIndex = at + 1;
		at = endIndex + length;
	}
	
	if (at >= endIndex) {
		return;
	}
	this._anchor = anchor;
	
	if (this._firstIndex < 0 || at < this._firstIndex) {
		this._firstIndex = at;
	}
	if (this._lastIndex < 0 || endIndex >= this._lastIndex) {
		this._lastIndex = endIndex - 1;
	}

	for (var i = at; i < endIndex; ++i) {
		this._select(i);
	}
};

/** Superior performance than calling deselect multiple times
* Negative length is allowed. It will be inverse selectRange
* @public
* @param {number} at Negative value is allowed and will be adjusted to zero
* @param {number} length This will be adjusted according to max selection index
*/
tr.grid.SelectionList.prototype.deselectRange = function (at, length) {
	if (this._count <= 0) { return; }
	if (at < 0) { at = 0; }
	if (length === 1 || length === -1) {
		this.deselect(at);
		return;
	}
	
	var endIndex = at + length;
	if (length < 0) {
		endIndex = at + 1;
		at = endIndex + length;
	}

	if (this._firstIndex >= at && this._lastIndex < endIndex) {
		this.clearAllSelections();
		return;
	}

	if (this._anchor >= at && this._anchor < endIndex) {
		this._anchor = -1;
	}
	if (this._firstIndex >= at && this._firstIndex < endIndex) {
		this._firstIndex = this._findNextSelection(endIndex - 1);
	}
	if (this._lastIndex >= at && this._lastIndex < endIndex) {
		this._lastIndex = this._findPrevSelection(at);
	}

	var i = 0;
	for (i = at; i < endIndex; ++i) {
		if (this._selections[i]) {
			this._selections[i] = false;
			--this._count;
		}
	}
};
/** @public
* @param {number} at
* @param {number} length
* @return {boolean}
*/
tr.grid.SelectionList.prototype.hasSelection = function(at, length) {
	if(this._count <= 0) {
		return false;
	}
	var endIndex = at + length;
	if (length < 0) {
		endIndex = at + 1;
		at = endIndex + length;
	}
	if(endIndex < this._firstIndex || at > this._lastIndex) {
		return false;
	}
	
	for(var i = at; i < endIndex; ++i) {
		if(this._selections[i]) {
			return true;
		}
	}
	return false;
};

tr.grid.SelectionList.prototype.getSelection = function(at) {
	return (this._selections[at] === true);

/** WARNING: It will creates a new(EXPENSIVE) defragmented array of selected Index. The selected indices will always be sorted in ascending order
* @public
* @return {Array.<number>}
*/
tr.grid.SelectionList.prototype.getAllSelections = function() {
	if(this._count > 0) {
		var ary = new Array(this._count); // Fastest way to create an array
		var count = 0;
		for(var i = this._firstIndex; i <= this._lastIndex; ++i) {
			if(this._selections[i]) {
				ary[count++] = i;
			}
		}
		return ary;
	}
	return [];
};

* @public
* @return {Array.<boolean>}

tr.grid.SelectionList.prototype.getSelectionMap = function () {
	return this._selections;
};
/** @public
* @return {number}
*/
tr.grid.SelectionList.prototype.clearAllSelections = function() {
	if(this._count > 0) {
		var count = 0;
		this._selections.length = 0;
		this._count = 0;
		this._anchor = -1;
		this._firstIndex = -1;
		this._lastIndex = -1;
		return count;
	}
	return 0;
};

/** Copy selections from the source selection list for the specified length at the specified index. Note that selection anchor won't be changed
* @public 
* @param {tr.grid.SelectionList} srcSelections
* @param {number} fromSrcIndex
* @param {number} offsetIndex
* @param {number} forLength Positive value only. negative valie is not allowed
*/
tr.grid.SelectionList.prototype.copyFrom = function (srcSelections, fromSrcIndex, offsetIndex, forLength) {
	if (forLength <= 0) { return; }

	var toThisIndex = fromSrcIndex + offsetIndex;
	if (srcSelections === null) {
		this.deselectRange(toThisIndex, forLength);
		return;
	}
	if (srcSelections.hasSelection(fromSrcIndex, forLength) === false) {
		this.deselectRange(toThisIndex, forLength);
		return;
	}


	for (var i = 0; i < forLength; ++i) {
		if (srcSelections._selections[fromSrcIndex + i]) {
			this.select(toThisIndex + i);
		} else {
			this.deselect(toThisIndex + i);
		}
	}
	this._anchor = -1;
	if (srcSelections._anchor >= 0) {
		if (srcSelections._anchor >= fromSrcIndex && srcSelections._anchor < (fromSrcIndex + forLength)) {
			this._anchor = srcSelections._anchor + offsetIndex;
		}
	}
};

/** @private
* @param {number} at 
*/
tr.grid.SelectionList.prototype._select = function (at) {
	if(this._selections[at]) { return false; }

	this._selections[at] = true;
	++this._count;
	
	return true;
};
/** WARNING: Perform LINEAR SEARCH
* @private
* @param {number} exclusiveStart
* @return {number} Negative if not found
*/
tr.grid.SelectionList.prototype._findNextSelection = function(exclusiveStart) {
	if(this._lastIndex >= 0) {
		while(++exclusiveStart <= this._lastIndex) {
			if(this._selections[exclusiveStart]) {
				return exclusiveStart;
			}
		}
	}
	return -1;
};
/** WARNING: Perform LINEAR SEARCH
* @private
* @param {number} exclusiveStart
* @return {number} Negative if not found
*/
tr.grid.SelectionList.prototype._findPrevSelection = function(exclusiveStart) {
	if(this._firstIndex >= 0) {
		while(--exclusiveStart >= this._firstIndex) {
			if(this._selections[exclusiveStart]) {
				return exclusiveStart;
			}
		}
	}
	return -1;
};
