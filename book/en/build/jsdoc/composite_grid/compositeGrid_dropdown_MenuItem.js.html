 * @param {Element|Object=} opt_initializer
 * @param {Object=} opt_model
 */
tr.MenuItem = function(opt_initializer, opt_model) {
	var _this = this;

	// Add custom events
	_this._addEvent("itemHovered");
	_this._addEvent("itemClicked");

	_this._onItemHovered = _this._onItemHovered.bind(_this);
	_this._onItemClicked = _this._onItemClicked.bind(_this);

		opt_model = opt_initializer;

/**
 * @ignore
 * @public
 * @return {number} 0 - not selectable, 1 - selectable, 2 -  selectable but as command
 */
tr.MenuItem.prototype.checkSelectionType = function() {
	var result = 1;

	switch (this.getType()) {
		// case "label":
		case "separator":
			result = 0;
			break;

		case "command":
			result = 2; // No break statement here in order to let it flow to the next condition

		default:
			if (!this.isVisible() || this.isDisabled() || !this.isSelectable()) {
				result = 0;
			}
	}

	return result;
};

/**
 * @private
 * @param {Object} e
 */
tr.MenuItem.defaultRenderer = function(e) {
	var item, content, criteria, regex;

	item = e["item"];
	content = item.getText();
	criteria = item.getMatchingCriteria();

	// When criteria is object, get value from text
	if (criteria && !tr.isString(criteria)) {
		criteria = criteria["text"];
	}

	if (criteria) {
		regex = ( item.getMatchingMaskFunction() || tr.MenuItem.defaultMatchingMaskFunction )(criteria);

		content =  content.replace(regex, function(str) {
			return '<span class="matched">' + str + '</span>';
		});
	}

	item.getElement().innerHTML = content;
};

/**
 * @private
 * @param {Object} e
 */
tr.MenuItem.defaultMatchingFunction = function(e) {
	var item, criteria, i, j, dummyCriteria, attr, regex, matchingRenderer;

	item = e["item"];
	criteria = e["criteria"];

	if (tr.isString(criteria) || tr.isNumber(criteria)) {
		criteria = [criteria]; // Set as array to enable multiple key matching in each field
	}
	if (tr.isArray(criteria)) {
		criteria = { "text": criteria }; // Set final criteria as object
	}

	if (!criteria) return true; // No criteria

	for (i in criteria) {
		dummyCriteria = criteria[i];

		switch (i) {
			case "text":
				attr = item.getText();
				break;
			case "value":
				attr = item.getValue();
				break;
		}

		if (tr.isString(attr) && attr) { // String and not empty
			for (j = 0; j < dummyCriteria.length; ++j) { // Each dummyCriteria is supposed to be an array
				regex = ( item.getMatchingMaskFunction() || tr.MenuItem.defaultMatchingMaskFunction )(dummyCriteria[j]);

				if (regex.test(attr)) {
					return true;
				}
			}
		}
	}

	return false;
};

/**
 * @public
 * @param {string | Array.<string>} key
 * @return {Object} Matching mask as an regular expression
 */
tr.MenuItem.defaultMatchingMaskFunction = function(key) {
	if (tr.isString(key)) {
		key = key.split(" ");
	}

	key = key.map(function(str) {
		str = str.replace(/ /g, "");
		return str ? "\\b(" + tr.escapeCharacters(str) + ")" : str;
	});

	var i = 0;

	do {
		if (key[i]) {
			++i;
		}
		else {
			key.splice(i, 1);
		}
	}
	while(i < key.length);

	return new RegExp(key.join("|"), "ig");
};

	if (!opt_element) {
		opt_element = document.createElement("li");

/** All custom or user settings should be done here
 * @protected
 * @ignore
 * @param {*} opt_model
 */
tr.MenuItem.prototype._deserialize = function(opt_model) {
	if (!opt_model) opt_model = {};

	var _this, pref;

	_this = this;

	pref = opt_model["id"];
	if (pref) {
		_this.setId(pref);
	}

	pref = opt_model["type"];
	if (pref) {
		_this.setType(pref);
	}

	pref = opt_model["text"];
	if (goog.isDefAndNotNull(pref)) {
		_this.setText(pref);
	}

	pref = opt_model["value"];
	if (goog.isDefAndNotNull(pref)) {
		_this.setValue(pref);
	}

	pref = opt_model["isDisabled"];
	if (pref) {
		_this.disable(pref);
	}

	pref = opt_model["isVisible"];
	if (pref === false) {
		_this.show(pref);
	}

	pref = opt_model["isSelectable"];
	if (pref === false) {
		_this.disableSelection();
	}

	pref = opt_model["renderer"] || opt_model["renderFunction"];
	if (pref) {
		_this.setRenderer(pref);
	}

	pref = opt_model["matchingFunction"];
	if (pref) {
		_this.setMatchingFunction(pref);
	}

	pref = opt_model["matchingMaskFunction"];
	if (pref) {
		_this.setMatchingMaskFunction(pref);
	}

	pref = opt_model["subMenu"];
	if (pref) {
		_this.setSubMenu(pref);
	}
	else {
		pref = opt_model["items"];
		if (pref) {
			_this.setSubMenu({ "items": opt_model["items"] });
		}
	}

	// FIXME: Just to be compatible with old API, has to be removed later
	if (opt_model["isSeparator"]) {
		_this.setType("separator");
	}
};

/**
 * @public
 */
tr.MenuItem.prototype._updateContent = function() {
	if (this.getType() === "separator") return;

	var _this, topNode, element, content;

	_this = this;
	topNode = _this._topNode;
	element = topNode.getElement();

	// Always clear existing content
	element.textContent = "";

	// Re-render item content
	content = ( _this._renderer || tr.MenuItem.defaultRenderer )({ "item": _this, "element": element }); // The property "element" is deprecated

	// In case that renderer return some value, assume that there's no content set into element
	if (goog.isDef(content)) {
		if (tr.isString(content)) {
			element.innerHTML = content;
		}
		else if (tr.isElement(content)) {
			topNode.setContent(content);
		}
	}
};
