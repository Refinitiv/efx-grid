/** @public
* @param {tr.IElementControl} cell
* @param {number} value
* @param {number} min
* @param {number} max
*/
tr.grid.HeatMapRenderer.prototype.render = function(cell, value, min, max) {
	if(!cell) { return; }

	var style = this.calculateStyle(value, min, max);	
	cell.setStyle('backgroundColor', style['background-color']);
	cell.setStyle('color', style['color']);
};

/** Calculate heatmap styles to be used to render a cell 
* @public
* @param {number} value
* @param {number} min
* @param {number} max
* @return {Object}
*/
tr.grid.HeatMapRenderer.prototype.calculateStyle = function(value, min, max) {
	var diff = (value || value === 0) ? value - this._midPoint : NaN;
	if (diff === 0 || diff !== diff) {
		return {
			'background-color': '',
			'color': ''
		};
	}

	var ratio, topColor;
	if(diff > 0) {
		var posDiffDenominator = max - this._midPoint;
		var posDiffColor = this._maxColor;
		if (posDiffDenominator <= 0) {
			posDiffDenominator = min - this._midPoint;
			posDiffColor = this._minColor;
		}
		ratio = diff/posDiffDenominator;
		topColor = posDiffColor;
	} else {
		var negDiffDenominator = min - this._midPoint;
		var negDiffColor = this._minColor;
		if (negDiffDenominator >= 0) {
			negDiffDenominator = max - this._midPoint;
			negDiffColor = this._maxColor;
		}
		
		ratio = diff/negDiffDenominator;
		topColor = negDiffColor;
	}
	
	var blendedColor;
	if (this._textRendering) {
		blendedColor = this._blendColor(this._baseTextColor, topColor, ratio);
		return {
			'background-color': '',
			'color': this._num2Hex(blendedColor)
		};
	} else {
		blendedColor = this._blendColor(this._baseBgColor, topColor, ratio);
		return {
			'background-color': this._num2Hex(blendedColor),
			'color': this._blackAndWhite(blendedColor)
		};
	}
};

/** @public
* @description Static method
* @param {string} rgbCode
* @return {string} resultColor
*/
tr.grid.HeatMapRenderer.rgb2Hex = function (rgbCode) {
	var rgb = rgbCode.match(/\d+/g);
	if(!rgb || rgb.length < 3) {
		return '';
	}

	var triplet = [];
	for(var i = 0; i < 3; i++) {
		triplet.push(parseInt(rgb[i], 10));
	}
	return tr.grid.HeatMapRenderer.num2Hex(triplet);
};
/** @public
* @description Static method
* @param {Array.<number>} triplet
* @return {string} resultColor
*/
tr.grid.HeatMapRenderer.num2Hex = function (triplet) {
	var rgb = triplet[2] | (triplet[1] << 8) | (triplet[0] << 16);
	return '#' + (0x1000000 + rgb).toString(16).slice(1);
};

//#region Private Methods
/** @private
* @param {string} baseColor
* @param {string} maxColor
* @param {number} ratio [0, 1]
* @return {Array.<number>} resultColor
*/
tr.grid.HeatMapRenderer.prototype._blendColor = function (baseColor, maxColor, ratio) {
	if (ratio > 1) {
		ratio = 1;
	} else if(ratio < 0) {
		ratio = 0;
	}
	
	var baseColorTriplet = this._hex2Num(baseColor);
	var maxColorTriplet = this._hex2Num(maxColor);
	var blendResult = [];
	for (var i = 0; i < 3; ++i) {
		var gap = (maxColorTriplet[i] - baseColorTriplet[i]) * ratio;
		blendResult.push(baseColorTriplet[i] + gap);
	}

	return blendResult;
};
/** @private
* @param {string} hex
* @return {Array.<number>} array of size 3 which contains [red, green, blue]
*/
tr.grid.HeatMapRenderer.prototype._hex2Num = function (hex) {
	var hexInt = parseInt(hex.replace(/[^0-9A-F]/gi, ''), 16);
	var r = (hexInt >> 16) & 255;
	var g = (hexInt >> 8) & 255;
	var b = hexInt & 255;
	return [r, g, b];
};
/** @private
* @param {Array.<number>} triplet
* @return {string} resultColor
*/
tr.grid.HeatMapRenderer.prototype._num2Hex = tr.grid.HeatMapRenderer.num2Hex;
/** @private
* @param {Array.<number>} triplet
* @return {string} resultColor
*/
tr.grid.HeatMapRenderer.prototype._blackAndWhite = function (triplet) {
	var brightness = Math.sqrt(triplet[0] * triplet[0] * 0.241 + triplet[1] * triplet[1] * 0.691 + triplet[2] * triplet[2] * 0.068);

	if (brightness >= 135) { // Brighter color has more impact to human eye than the darker color
		return 'black';
	}
	return 'white';
};
