/**
 * @constructor
 * @extends {tr.EventManager}

tr.ADCSubscription = function() {
	this._adcFieldMap = {};
	this._queue = {};
	this._onADCUpdate = this._onADCUpdate.bind(this);
	this._refreshData = this._refreshData.bind(this);
	this._processQueue = this._processQueue.bind(this);
	this._onActivate = this._onActivate.bind(this);
	this._onDeactivate = this._onDeactivate.bind(this);
};
goog.inherits(tr.ADCSubscription, tr.EventManager);

//#region public function of ADC service
/** @public
 * @description Set the settings of ADC
 * @param {string} userId UUID
 * @param {string} productId Application name
 * @param {string} url Url of ADC server
 * @param {string=} opt_lang Language of Eikon UI
 */
tr.ADCSubscription.prototype.setDataCloudSettings = function (userId, productId, url, opt_lang) {
	if(userId !== undefined) {
		this._userId = userId;
	}
	if(productId !== undefined) {
		this._productId = productId;
	}
	if(url !== undefined) {
		this._adcUrl = url;
	}
	if(opt_lang) {
		this._lang = opt_lang;
	}
};
/** @public
 * @description Set the interval time for retrieving data
 * @param {number} time Millisecond
 */
tr.ADCSubscription.prototype.setIntervalTime = function(time) {
	time = +time;
	if(!time) { // NaN or 0
		return;
	}
	this._intervalTime = time;
	this._stop();
	this._start();
};
/** @public
 * @description Subscribe symbol to the spcified field. This method will start the interval for retrieving data automatically.
 * @param {string} symbol RIC
 * @param {string} field ADC field name
 * @return {*} Return the value if there is existing. Return undefined when requesting the invalid field.
 */
tr.ADCSubscription.prototype.addSymbolToField = function (symbol, field) {
	if(field.indexOf("undefined") >= 0) { // Invalid field name
		return;
	}

	var fieldMap = this._adcFieldMap[field];
	if(!fieldMap) {
		fieldMap = {};
		this._adcFieldMap[field] = fieldMap;
	}
	
	var value = fieldMap[symbol];
	if(value !== undefined) {
		return value; // Return the current value
	}

	fieldMap[symbol] = tr.ADCSubscription.PENDING_DATA; // First value to cache
	this._addQueue(field);
	return tr.ADCSubscription.PENDING_DATA;
};
//#endregion public function of ADC service

tr.ADCSubscription.prototype.addEventListener = function(type, handler) {
	tr.ADCSubscription.base(this, "addEventListener", type, handler);
	this._start();
};
/** @override
 * @description Remove the function handler from the specific event. If there is not any event handlers, the service will stop automatically.
 * @param {string} type Event name
 * @param {Function} handler Event handler
 */
tr.ADCSubscription.prototype.removeEventListener = function(type, handler) {
	tr.ADCSubscription.base(this, "removeEventListener", type, handler);
	if(!this._hasListener()) {
		this._stop();
		this._adcFieldMap = {};
		this._queue = {};
	}
};
/** @override
 * @description Remove all function handlers of all events and stop the interval of retrieving data of service.
 */
tr.ADCSubscription.prototype.removeAllEventListeners = function() {
	tr.ADCSubscription.base(this, "removeAllEventListeners");
	this._stop();
	this._adcFieldMap = {};
	this._queue = {};
};

//#region private function of ADC service
/** @private
 * @description Start the interval of retrieving data
 */
tr.ADCSubscription.prototype._start = function() {
	if(this._state === "start") {
		return;
	}
	this._intervalId = window.setInterval(this._refreshData, this._intervalTime);
	this._state = "start";
};
/** @private
 * @description Stop the interval of retrieving data
 */
tr.ADCSubscription.prototype._stop = function() {
	if(this._timerId !== null) {
		window.clearTimeout(this._timerId);
	}
	if(this._intervalId !== null) {
		window.clearInterval(this._intervalId);
	}
	this._timerId = null;
	this._intervalId = null;
	this._state = "stop";
};
/** @private
 * @description Refresh all the ADC data that had been subscribed to the service.
 */
tr.ADCSubscription.prototype._refreshData = function() {
	if (!this._isActive){
		this._updateOnActivation = true;
		this._stop();
		return;
	}
	this._addQueue(Object.keys(/** @type{!Object} */(this._adcFieldMap)));
};

/**
* @private
*/
tr.ADCSubscription.prototype._onActivate = function () {
	this._isActive = true;
	if (this._updateOnActivation) {
		this._updateOnActivation = false;
		this._start();
	}
};

/**
* @private
*/
tr.ADCSubscription.prototype._onDeactivate = function (){
	this._isActive = false;
};

/** @private
 * @description Manage the queue to retrieve the ADC data. Prevent to call the same ADC field multiple time.
 * @param {string|Array.<string>} fields
 */
tr.ADCSubscription.prototype._addQueue = function(fields) {
	if(this._timerId !== null) {
		window.clearTimeout(this._timerId);
	}

	if(!Array.isArray(fields)) {
		fields = [fields];
	}
	if(!this._queue) {
		this._queue = {};
	}
	for(var i = fields.length; --i >= 0;) {
		this._queue[fields[i]] = true;
	}
	this._timerId = window.setTimeout(this._processQueue, 0);
};
/** @private
 * @description Process the data queue to retrieve ADC data.
 */
tr.ADCSubscription.prototype._processQueue = function() {
	var fields = Object.keys(/** @type{!Object} */(this._queue));
	for(var j = fields.length; --j >= 0;) {
		this._requestDataCloud(fields[j]);
	}
	this._queue = {}; // Clear queue
	this._timerId = null; // Clear timeout id
};
/** @private
 * @description Publish the ADC request to the ADC server.
 * @param {string} field ADC field name
 */
tr.ADCSubscription.prototype._requestDataCloud = function(field) {
	var RICs = Object.keys(this._adcFieldMap[field]);
	var payload = "formula=" + field + "&identifiers=" + RICs;
	if(this._productId) {
		payload += '&productid=' + this._productId;
	}
	if(this._userId) {
		payload += '&uuid=' + this._userId;
	}
	if(this._lang) {
		payload += '&lang=' + this._lang;
	}

	var http = new HttpRequest(this._adcUrl, payload);
	http.send({"Content-Type":"application/x-www-form-urlencoded"})
	.then(function(json) {
		json["requestField"] = field; // Attached requested field
		return json;
	})
	.then(this._onADCUpdate)
	.catch(function(err) { 
		console.error(err);
	});
};
/** @private
 * @description Validate type of value that are number, string, boolean or not.
 * @param {*} value
 */
tr.ADCSubscription.prototype._isValidValueType = function(value) {
	var t = typeof value;
	return (value === null || t == "string" || t == "number" || t == "boolean");
};
/** @private
 * @description Function for processing the ADC response and publish the event to the listener.
 * @param {Object} json JSON object of ADC response
 */
tr.ADCSubscription.prototype._onADCUpdate = function(json) {
	if(json["status"] && json["status"].toLowerCase() !== "ok") {
		console.warn("Response status is not ready");
		return;
	}

	var rows = json["rows"];
	var reqField = json["requestField"];
	var header = rows[0];
	var ricIndex = -1, fieldIndex = -1;

	var type, field;
	for(var idx in header) {
		type = header[idx]["i"];
		field = header[idx]["r"];
		if(type === "instrument") {
			ricIndex = idx;
		}
		if (type === "value" && reqField === field) {
			fieldIndex = idx;
		}
	}

	var outputHash = {};
	var rowsLen = rows.length;
	var symbol, value, rowData;
	for(var i = 1; i < rowsLen; i++) {
		symbol = rows[i][ricIndex];
		value = rows[i][fieldIndex];

		rowData = outputHash[symbol];
		if (!rowData) {
			rowData = {};
			outputHash[symbol] = rowData;
		}

		if((rowData[reqField] === undefined || rowData[reqField] === null) && this._isValidValueType(value)) {
			rowData[reqField] = value;
		}
	}

	// Publish data
	for(symbol in outputHash) {
		rowData = outputHash[symbol];
		if(rowData.hasOwnProperty(reqField)) { // Have the data to be updated
			this._adcFieldMap[reqField][symbol] = rowData[reqField]; // Cache the data in service
			this._dispatch("dataChanged", { "ric": symbol, "values": rowData, "sender": this });
		}
	}
};
