/** {@link tr.IElementControl#listen} 
* @override 
*/
tr.ElementControl.prototype.listen = function (type, handler, opt_priority) {
	if(this._events !== null && this._events[type] !== undefined) {
		this._events[type].listen(handler, opt_priority);
	} else {
		this._topNode.listen(type, handler, opt_priority);
	}
};
/** {@link tr.IElementControl#unlisten} 
* @override 
*/
tr.ElementControl.prototype.unlisten = function (type, handler) {
	if(this._events !== null && this._events[type] !== undefined) {
		this._events[type].unlisten(handler);
	} else {
		this._topNode.unlisten(type, handler);
	}
};
/** {@link tr.IElementControl#unlistenAll} 
* @override 
*/
tr.ElementControl.prototype.unlistenAll = function (opt_type) {
	if(opt_type === undefined) {
		if (this._topNode) {
			this._topNode.unlistenAll();
		}
		for (var key in this._events) {
			this._events[key].unlistenAll();
		}
	} else if(this._events !== null && this._events[opt_type] !== undefined) {
		this._events[opt_type].unlistenAll();
	} else {
		this._topNode.unlistenAll(opt_type);
	}
};
/** {@link tr.IElementControl#dispose} 
* @override 
*/
tr.ElementControl.prototype.dispose = function () {
	this._topNode.dispose();
	for (var key in this._events) {
		this._events[key].unlistenAll();
	}
	this._topNode = null;
	this._events = null;
};

/** {@link tr.IElementControl#getEvent} 
* @ignore
* @inheritDoc */
tr.ElementControl.prototype.getEvent = function (type) {
	if (this._events !== null && this._events[type] !== undefined) {
		return this._events[type];
	}
	return this._topNode.getEvent(type);
};
