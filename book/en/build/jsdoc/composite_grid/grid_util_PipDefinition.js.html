/** @private
 * @param {number|string|null=} value
 * @return {number} Number of precision digit
 */
tr.PipDefinition._getNumberOfDecimals = function(value) {
	var str;
	if(typeof value == "number") {
		str = "" + value; // Fastest way to convert to string
	} else if(!value) { // empty string, null, undefined, NaN
		return 0;
	} else {
		str = /** @type{string} */(value);
	}
	
	var pos = str.indexOf("."); // "12345.789" gives 5
	if(pos < 0) {
		return 0;
	}
	return (str.length - pos - 1); // "12345.789" >> 9 - 5 - 1 == 3
};

/** @public
 * @ignore 
 * @param {string} ric
 * @param {number|string|null=} opt_bid
 * @param {number|string|null=} opt_ask
 * @return {tr.PipFormat}
 */
tr.PipDefinition.getPipFormat = function(ric, opt_bid, opt_ask) {

	tr.PipDefinition._init();
	/*
	 * First, check specified format from cache
	 */
	var cache = tr.PipDefinition._cache;
	var value = cache[ric] && cache[ric]["format"];
	if(value) {
		return {	// Clone the value so that the caller cannot modify the entry
			"start": value["start"],
			"pipLength": value["pipLength"],
			"subPipLength": value["subPipLength"]
		};
	}

	var bidDec = tr.PipDefinition._getNumberOfDecimals(opt_bid);
	var askDec = tr.PipDefinition._getNumberOfDecimals(opt_ask);
	var numberOfDecimals = Math.max(bidDec, askDec);

	/*
	 * Otherwise, retrieve format depending of the number of decimals
	 */
	var fallback = tr.PipDefinition._fallback;
	value = fallback[numberOfDecimals];
	if(value) {
		return {	// Clone the value so that the caller cannot modify the fallback
			"start": value["start"],
			"pipLength": value["pipLength"],
			"subPipLength": value["subPipLength"]
		};
	}

	/*
	 * Default value
	 */
	return {
		"start": -2,
		"pipLength": 2,
		"subPipLength": 0
	};
};
