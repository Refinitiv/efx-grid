goog.provide("tr.grid.ColumnSelectionPlugin");

goog.require("tr.grid.IPlugin");
goog.require("tr.grid.Plugin");

/** @constructor
 * @param {Array.<number|boolean>|null=} opt_restrictions By default all columns will be processed by this plugin
 * @param {Array.<number|boolean>|null=} opt_exclusions By default no column will be excluded by this plugin
* @extends {tr.grid.Plugin}
* @implements {tr.grid.IPlugin}
*/
tr.grid.ColumnSelectionPlugin = function (opt_restrictions, opt_exclusions) {
	this._onSectionAdded = goog.bind(this._onSectionAdded, this);
	this._onClick = goog.bind(this._onClick, this);
	this._onKeyDown = this._onKeyDown.bind(this);
	this._onSelectionChanged = this._onSelectionChanged.bind(this);
	this._gotoContent = this._gotoContent.bind(this);

	this._marker = new tr.grid.IndexMarker(opt_restrictions, opt_exclusions);

	this._addEvent(tr.res.selectionChanged);
	this.listen(tr.res.selectionChanged, this._onSelectionChanged);

	this._hosts = [];
};
goog.inherits(tr.grid.ColumnSelectionPlugin, tr.grid.Plugin);

tr.grid.ColumnSelectionPlugin.prototype._anchorCol = -1;

/** @override */
tr.grid.ColumnSelectionPlugin.prototype.getName = function () {
	return "ColumnSelectionPlugin"; // Read Only
};
/** @override */
tr.grid.ColumnSelectionPlugin.prototype.initialize = function (host) {
	if(this._hosts.indexOf(host) >= 0) { return; }
	this._hosts.push(host);

	host.listen(tr.res.sectionAdded, this._onSectionAdded);
	host.listen(tr.res.keydown, this._onKeyDown);
};
/** @override */
tr.grid.ColumnSelectionPlugin.prototype.unload = function (host) {
	var at = this._hosts.indexOf(host);
	if(at < 0) { return; }

	host.unlisten(tr.res.sectionAdded, this._onSectionAdded);
	host.unlisten(tr.res.keydown, this._onKeyDown);
	this._hosts.splice(at, 1);
};

	this._disabled = opt_bool !== false;

tr.grid.ColumnSelectionPlugin.prototype.isEnabled = function () {
	return !this._disabled;

/** @public
 * @param {Array.<number>=} opt_cols
 */
tr.grid.ColumnSelectionPlugin.prototype.clearSelection = function(opt_cols){
	this._clearSelection(opt_cols);
	// this._anchorCol = -1; Keep Anchor column to be used as the last selected selection
	this._activeGrid = null;
};
/** @public
 * @return {!Array.<number>}
 */
tr.grid.ColumnSelectionPlugin.prototype.getSelectedColumns = function(){
	if(!this._activeGrid) {
		return [];
	}

	var cols = [];
	var layout = this._activeGrid.getSection(0);
	if(layout){
		var colCount = this._getColumnCount();
		for(var i = 0; i < colCount; ++i){
			if(this._isSelectedColumn(i, layout)){
				cols.push(i);
			}
		}
	}
	return cols;
};
/** @public
 * @return {number}
 */
tr.grid.ColumnSelectionPlugin.prototype.getFirstSelectedColumn = function() {
	if(this._activeGrid) {
		var layout = this._activeGrid.getSection(0);
		if(layout) {
			var colCount = this._getColumnCount();
			for(var i = 0; i < colCount; ++i) {
				if(this._isSelectedColumn(i, layout)) {
					return i;
				}
			}
		}
	}
	return -1;
};
/** @public
 * @return {number}
 */
tr.grid.ColumnSelectionPlugin.prototype.getLastSelectedColumn = function(){
	if(this._activeGrid) {
		var layout = this._activeGrid.getSection(0);
		if(layout) {
			var colCount = this._getColumnCount();
			for(var i = colCount; --i >= 0;) {
				if(this._isSelectedColumn(i, layout)) {
					return i;
				}
			}
		}
	}
	return -1;
};

/** Select or deselect a column on the active grid
 * @public
 * @param {number} colIndex
 * @param {boolean} opt_select
 * @param {tr.Grid=} opt_grid
 * @return {boolean} Return true for successful selection, otherwise false
 */
tr.grid.ColumnSelectionPlugin.prototype.setSelectedColumn = function(colIndex, opt_select, opt_grid){
	if(!(colIndex >= 0)) { return false; }

	var gridChanged = false;
	if(opt_grid) {
		if(opt_grid !== this._activeGrid) {
			this._activeGrid = opt_grid; // TODO: Check if the grid is one of the hosts
			gridChanged = true;
		}
	}
	if(!this._activeGrid) {
		this._activeGrid = this._hosts[0]; // Select the first grid
		if(!this._activeGrid) {
			return false;
		}
	}

	if(colIndex >= this._getColumnCount()) {
		return false;
	}

	if(!this._marker.isAllowed(colIndex)) { return false; }

	opt_select = !!opt_select; // undefined value means false
	var layouts = this._activeGrid.getAllSections();
	for(var i = layouts.length; --i >= 0;){ // Slow
		layouts[i].enableColumnClass(colIndex, "selected-column", opt_select);
	}

	if(gridChanged || opt_select) { // TODO: Dispatch the event for all the changes
		this._dispatch(tr.res.selectionChanged, { // Notify active grid is changed
			"colIndex": colIndex,
			"grid": this._activeGrid
		});
	}
	return true;
};
/** Select the specified column on an active grid, and clear all other column selections
 * @public
 * @param {number|null=} opt_colIndex
 * @param {tr.Grid=} opt_grid
 */
tr.grid.ColumnSelectionPlugin.prototype.selectSingleColumn = function(opt_colIndex, opt_grid){
	if(opt_colIndex == null) { // null or undefined
		opt_colIndex = this._anchorCol;
	}
	if(!(opt_colIndex >= 0)) {
		opt_colIndex = 0;
	}
	this._clearSelection();
	if(this.setSelectedColumn(opt_colIndex, true, opt_grid)) {
		this._anchorCol = opt_colIndex; // Update the anchor
	}
};
/** Select from the anchorIndex to the target by length
 * @public
 * @param {number} anchorIndex
 * @param {number} len If 1, two column will be selected (anchor and the one next to it). Negative len will select column to the left. 0 only anchor column is selected.
 * @param {tr.Grid=} opt_grid
 */
tr.grid.ColumnSelectionPlugin.prototype.selectRange = function(anchorIndex, len, opt_grid) {
	this._anchorCol = !(anchorIndex >= 0) ? 0 : anchorIndex;
	this._selectFromAnchorToTarget(this._anchorCol + len);
};
/** Select all column in a single grid
 * @public
 */
tr.grid.ColumnSelectionPlugin.prototype.selectAllColumns = function() {
	this.selectRange(0, this._getColumnCount());
};

/** @public
 * @param {number} colIndex
 */
tr.grid.ColumnSelectionPlugin.prototype.isSelectedColumn = function(colIndex){
	if(this._activeGrid) {
		return this._isSelectedColumn(colIndex, this._activeGrid.getSection(0));
	}
	return false;
};

tr.grid.ColumnSelectionPlugin.prototype.getActiveGrid = function() {
	return this._activeGrid;
};

tr.grid.ColumnSelectionPlugin.prototype.enableSingleSelectionMode = function(bool) {
	this._singleSelMode = bool !== false;
};

/** @private
 * @param {Object} e
 */
tr.grid.ColumnSelectionPlugin.prototype._onSectionAdded = function (e) {
	var section = /** @type{tr.ILayoutGrid} */(e[tr.res.section]);
	section.listen(tr.res.click, this._onClick);
	section.listen("contextmenu", this._onClick);
};

/** @private
 * @description Left click on title causes selection change.
 * Ctrl/Shift left click on title causes multi selection.
 * Left click anywhere, not title, causes deselection.
 * @param {Event} e
 */
tr.grid.ColumnSelectionPlugin.prototype._onClick = function (e) {
	if(this._disabled) { return; }

	var host = this.getRelativeGrid(e);
	if (!host) { return; }

	var info = host.getRelativePosition(e);

	if(!info["hit"] || info[tr.res.sectionType] !== tr.res.title) { return; }

	if(e.button === 2) { // Right click
		if(e.shiftKey || e.ctrlKey) {
			return;
		}
		if(this.isSelectedColumn(info[tr.res.colIndex])) { // Allow right clicking at the content for context menu
			return;
		}
	}

	var selectedCols = this.getSelectedColumns();
	var cIndex = info[tr.res.colIndex];
	var singleSelection = (!this._activeGrid) || this._activeGrid !== host; // New grid is clicked
	var section = info[tr.res.section];

	if(singleSelection || this._singleSelMode) {
		this._clearSelection(); // TODO: Check if this.selectSingleColumn(cIndex) can replace this
		this._activeGrid = host;
		this.setSelectedColumn(cIndex, true);
		this._anchorCol = cIndex;
	} else { // The same active grid case: Ctrl(windows)/Command(macOS) + Click
		if ((e.ctrlKey && !tr.isMac) || (tr.isMac && !e.ctrlKey && e.metaKey)) {
			this.setSelectedColumn(cIndex, selectedCols.indexOf(cIndex) < 0);
			this._anchorCol = cIndex;
		} else {
			if(e.shiftKey){
				this._selectFromAnchorToTarget(cIndex);
			} else {
				this.selectSingleColumn(cIndex);
			}
		}
	}

	this._dispatch(tr.res.selectionChanged, {
		"colIndex": cIndex,
		"grid": host
	});
};
/** @private
 * @param {KeyboardEvent} e
 */
tr.grid.ColumnSelectionPlugin.prototype._onKeyDown = function (e) {
	if(!this._activeGrid) { return; }
	switch(e.keyCode) {
		case 9: // Tab
			this._selectByTab(e);
			break;
		case 37: //Left arrow
			this._selectByKey(-1, e);
			break;
		case 39: //Right arrow
			this._selectByKey(1, e);
			break;
		case 40: //Down arrow
			window.setTimeout(this._gotoContent, 10); // RowSelection also listen to the keydown, and thus we have to use timer
			break;
	}
};

/** @private
 * @param {KeyboardEvent} e
 */
tr.grid.ColumnSelectionPlugin.prototype._selectByTab = function(e) {
	var direction = (e.shiftKey) ? -1 : 1;
	var next = (direction < 0) ? this.getFirstSelectedColumn() : this.getLastSelectedColumn();
	if(next < 0) {
		return; // No selected column
	}
	tr._preventDefault(e);
	next += direction;
	var colCount = this._getColumnCount();

	if(next < 0) {
		if (this._gotoPrevGrid()) {
			return;
		}
		next = 0;
	} else if(next >= colCount) {
		if(this._gotoNextGrid()) {
			return;
		}
		next = colCount - 1;
	}
	this.selectSingleColumn(next);
	this._activeGrid.scrollToColumn(next);
};

/** @private
 * @param {number} direction
 * @param {KeyboardEvent} e
 */
tr.grid.ColumnSelectionPlugin.prototype._selectByKey = function(direction, e) {
	var curSel = (direction < 0) ? this.getFirstSelectedColumn() : this.getLastSelectedColumn();
	if(curSel < 0) {
		return; // No selected column
	}

	var colCount = this._getColumnCount();
	var next = this._getNextSelection(curSel, direction);
	if(next < 0) {
		if (!e.shiftKey && this._gotoPrevGrid()) {
			tr._preventDefault(e);
			return;
		}
		next = 0;
	} else if(next >= colCount) {
		if(!e.shiftKey && this._gotoNextGrid()) {
			tr._preventDefault(e);
			return;
		}
		next = colCount - 1;
	}
	if(!this._activeGrid.isColumnVisible(next)) {
		return; // No further visible column can be selected
	}

	tr._preventDefault(e);
	if(e.shiftKey && !this._singleSelMode){
		this._selectFromAnchorToTarget(next);
	} else {
		this.selectSingleColumn(next);
	}
	this._activeGrid.scrollToColumn(next);
};
/** @private
 * @param {number} curSel
 * @param {number} direction Must not be zero
 * @return {number} next selection
 */
tr.grid.ColumnSelectionPlugin.prototype._getNextSelection = function(curSel, direction) {
	var next = curSel + direction;
	var colCount = this._getColumnCount();
	// isColumnVisible() with negative index or invalid index gives true as a result.
	while(!this._activeGrid.isColumnVisible(next)) { // Skip invisible columns
		if(next < 0 || next >= colCount) {
			break; // no further visible selection
		}
		next += direction;
	}
	return next;
};

/** @private
 * @return {boolean}
 */
tr.grid.ColumnSelectionPlugin.prototype._gotoPrevGrid = function() {
	if(this._activeGrid === this._hosts[0]) { // This the first grid, so go to the content
		return this._gotoContent();
	} else {
		var curIndex = this._hosts.indexOf(this._activeGrid);
		if(curIndex > 0) {
			this.selectSingleColumn(this._getNextSelection(this._getColumnCount(), -1), this._hosts[curIndex - 1]); // go to the next grid at the first column
			return true;
		}
	}
	return false;
};
/** @private
 * @return {boolean}
 */
tr.grid.ColumnSelectionPlugin.prototype._gotoNextGrid = function() {
	var lastIndex = this._hosts.length - 1;
	if(this._activeGrid === this._hosts[lastIndex]) { // This the last grid, so go to the content
		return this._gotoContent();
	} else {
		var curIndex = this._hosts.indexOf(this._activeGrid);
		if(curIndex >= 0) {
			this.selectSingleColumn(this._getNextSelection(-1, 1), this._hosts[curIndex + 1]); // go to the next grid at the first column
			return true;
		}
	}
	return false;
};
/** @private
 * @return {boolean}
 */
tr.grid.ColumnSelectionPlugin.prototype._gotoContent = function() {
	if(this._activeGrid) {
		var rsp = this._getPlugin("RowSelectionPlugin");
		if (rsp) {
			return rsp["selectSingleRow"](0, this._activeGrid.getSectionSettings(tr.res.content), this._activeGrid); // This will clear all column selections
		}
	}
	return false;
};

/** @private
 * @param {number} colIndex
 * @param {tr.ILayoutGrid} section
 * @return {boolean}
 */
tr.grid.ColumnSelectionPlugin.prototype._isSelectedColumn = function(colIndex, section){
	if(section) {
		return section.hasColumnClass(colIndex, "selected-column");
	}
	return false;
};

/** @private
 * @param {Event} e
 */
tr.grid.ColumnSelectionPlugin.prototype._onSelectionChanged = function(e) {
	if(!this._activeGrid || e[tr.res.sectionType] === tr.res.content) {
		return;
	}
	var rsp = this._getPlugin("RowSelectionPlugin");
	if (rsp) {
		rsp["clearSelectedRows"]();
	}
	var cep = this._getPlugin("CellEditingPlugin");
	if(cep) {
		cep.closeCellEditor();
	}
};

/** @private
 * @param {Array.<number>=} opt_cols
 */
tr.grid.ColumnSelectionPlugin.prototype._clearSelection = function(opt_cols){
	opt_cols = opt_cols || this.getSelectedColumns();
	for(var i = opt_cols.length; --i >= 0;){
		this.setSelectedColumn(opt_cols[i], false);
	}
};
/** @private
 * @param {number} targetIndex
 */
tr.grid.ColumnSelectionPlugin.prototype._selectFromAnchorToTarget = function(targetIndex) {
	if(this._anchorCol < 0) {
		this.selectSingleColumn(targetIndex);
		return;
	}

	var i;
	var grid = this._activeGrid;
	this.clearSelection();
	if(targetIndex <= this._anchorCol) {
		for(i = targetIndex; i <= this._anchorCol; ++i){
			this.setSelectedColumn(i, true, grid);
		}
	} else {
		for(i = this._anchorCol; i <= targetIndex; ++i){
			this.setSelectedColumn(i, true, grid);
		}
	}
};
