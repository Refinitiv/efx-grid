/**
 * @constructor
 * @implements {tr.IElementControl}
 * @extends {tr.ElementControl}
 * @param {Element|Object=} opt_initializer

 */

	var _this = this;

	// Initialize properties
	_this._items = [];
	_this._subMenus = [];

	// Add custom events
	_this._addEvent("itemHovered");
	_this._addEvent("itemClicked");
	_this._addEvent("itemActivated");
	_this._addEvent("itemSelected");
	_this._addEvent("itemHighlighted");
	_this._addEvent("treeItemActivated");
	_this._addEvent("treeItemSelected");

	_this._onKeyDown = _this._onKeyDown.bind(_this);
	_this._onItemHovered = _this._onItemHovered.bind(_this);
	_this._onItemClicked = _this._onItemClicked.bind(_this);

		opt_model = opt_initializer;
		opt_initializer = null;
	}

		opt_model = { "items": opt_model };
	}

 */

			_this._subMenus[i].dispose();
		}

		for (i = 0; i < _this._items.length; i++) {
			_this._items[i].dispose();
		}

	}

/**
 * @public
 * @param {Array.<tr.MenuItem> | Array.<Object>} items
 */
tr.ListMenu.prototype.setItems = function(items) {
	var _this, i;

	_this = this;

	// Clear existing items
	while (_this._items.length > 0) {
		_this.removeItem(0); // items.length will always shorten after remove item. So just keep removing the first item until no item exists
	}

	_this._items = []; // Ensure empty items

	// Add items
	if (items) {
		for (i = 0; i < items.length; i++) {
			_this.addItem(items[i]);
		}
	}
};

/** Add new item to menu
 * @public
 * @param {tr.MenuItem | Object} item
 */
tr.ListMenu.prototype.addItem = function(item) {
	var _this = this;

	// Create object if parameter is model
	if (!(item instanceof tr.MenuItem)) {
		if (!item["renderer"]) {
			if (item["renderFunction"]) {
				item["renderer"] = item["renderFunction"]; // Backward compat
			}
			else if (_this._itemRenderer) {
				item["renderer"] = _this._itemRenderer;
			}
		}

		item = new MenuItem(item);
	}
	else if (_this._items.indexOf(item) >= 0) {
		return false;
	}

	item.setMenu(_this);

	return true;
};

/** Remove existing item from menu
 * @public
 * @param {number | tr.MenuItem | null} indicator
 * @return {boolean}
 */
tr.ListMenu.prototype.removeItem = function(indicator) {
	var _this, index, subMenu;

	_this = this;
	index = null;

	if (typeof indicator === "number") {
		index = indicator;
		indicator = _this.getItem(index); // Removal should be done by using item
	}
	else if (indicator instanceof tr.MenuItem) {
		index = _this.getItemIndex(indicator);
	}

	if (goog.isNull(indicator) || goog.isNull(index)) return false; // Return false if no item to removed

	// Update highlighted / selected item
	if (_this._highlightedItem === indicator) {
		_this._highlightedItem = null;
	}
	if (_this._selectedItem === indicator) {
		_this._selectedItem = null;
	}

	indicator.dispose();

	return true;
};

/** Append item to menu (bind event and add to collection)
 * @ignore
 * @public
 * @param {tr.MenuItem} item
 */
tr.ListMenu.prototype.hookItem = function(item) {
	var _this, subMenu;

	_this = this;

	item.listen("itemHovered", _this._onItemHovered);
	item.listen("itemClicked", _this._onItemClicked);

	// Keep item to item list of menu
	if (_this.getItemIndex(item) < 0) {
		_this._items.push(item);
	}

	// Keep sub-menu to sub-menu list of menu
	subMenu = item.getSubMenu();
	if (subMenu && _this.getSubMenuIndex(subMenu) < 0) {
		_this._subMenus.push(subMenu);
	}
};

/** Release item from menu (unbind event and remove from collection)
 * @ignore
 * @public
 * @param {tr.MenuItem} item
 */
tr.ListMenu.prototype.releaseItem = function(item) {
	var _this, index;

	_this = this;

	item.unlistenAll();

	// Remove item from item list of existing menu
	if (_this.getItemIndex(item) >= 0) {
		_this._items.splice(_this.getItemIndex(item), 1);
	}

	// Remove sub-menu from sub-menu list of existing menu
	index = _this.getSubMenuIndex(item.getSubMenu());
	if (index >= 0) {
		_this._subMenus.splice(index, 1);
	}
};

 * @public
 * @param {*} value
 * @param {boolean=} opt_includeChildren
 * @return {tr.MenuItem}
 */
tr.ListMenu.prototype.findItemByValue = function(value, opt_includeChildren) {
	var _this, i, item, subMenu;

	_this = this;

	for (i = 0; i < _this._items.length; i++) {
		item = _this._items[i];
		if (item.getValue() === value) return item;

		if (opt_includeChildren !== false) {
			subMenu = _this._items[i].getSubMenu();

			if (subMenu) {
				item = subMenu.findItemByValue(value, true);
				if (item) return item;
			}
		}
	}

	return null;
};

/** Highlight item
 * @public
 * @param {tr.MenuItem} item
 * @return {boolean} True if new item got highlight, otherwise false
*/
tr.ListMenu.prototype.highlightItem = function(item) {
	var _this, index;

	_this = this;

	if (item === _this._highlightedItem) {
		return false;
	}

	// Check item type
	if (!goog.isNull(item)) {
		index = _this.getItemIndex(item);

		switch (item.getType()) {
			case "separator":
				return false;

			default:
				if (!item.isVisible() || item.isDisabled()) {
					return false;
				}
		}
	}

	// Clear previous highlighted index
	if (!goog.isNull(_this._highlightedItem)) {
		_this._highlightedItem.highlight(false);
	}

	_this._highlightedItem = item;

	if (!goog.isNull(item)) {
		item.highlight();
		
		// Dispatch itemHighlighted  event
		_this._dispatch("itemHighlighted", { "item": item });
	}

	return true;
};

/** Move index of highlighted item by step
 * @ignore
 * @public
 * @param {number} step
 * @param {number=} opt_index Reference for moving, default as highlighted index
 * @return {boolean}
 */
tr.ListMenu.prototype.moveHighlightedItem = function(step, opt_index) {
	var _this, item, items, result;

	_this = this;
	items = _this._items;

	if (tr.isNumber(opt_index)) {
		if (opt_index < 0 || opt_index >= items.length) opt_index = undefined;
	}

	if (!goog.isDef(opt_index)) {
		opt_index = _this.getItemIndex(_this._highlightedItem);
		if (goog.isNull(opt_index)) opt_index = -1;
	}

	do {
		opt_index += step;
		if (opt_index < 0) opt_index = items.length - 1;
		if (opt_index >= items.length) opt_index = 0;
		item = items[opt_index];
	}
	while (!item.isVisible() || item.isDisabled() || item.getType() === "separator");

	result = _this.highlightItem(item);
	_this._scrollToItem(_this._highlightedItem);

	return result;
};

/** Select item
 * @public
 * @param {tr.MenuItem} item
 * @param {Object=} opt_event
 * @param {boolean=} opt_includeChildren Whether to include all sub-menu items
 * @param {boolean=} opt_check Wheter to check if item is in menu and selectable
 * @return {boolean} True if new item got selected, otherwise false
 */
tr.ListMenu.prototype.selectItem = function(item, opt_event, opt_includeChildren, opt_check) {
	var i, subItem,
		selectionType,
		prevSelectedItem,
		isSelectionChanged,
		result = 1;

	if (item) {
		selectionType = item.checkSelectionType();

		if (opt_check !== false) {
			if (!this._hasItem(item, true)) return false;
			if (selectionType < 1) return false;
		}

		if (this.getItemIndex(item) < 0) {
			result = -1;
		}
		else {
			result = selectionType;
		}
	}

	// Clear filter and manage selection
	if (result !== 0 && result !== 2) {
		this.filterItems(null, false); // Clear filter only in this menu

		prevSelectedItem = this._selectedItem;

		if (item !== prevSelectedItem) {
			isSelectionChanged = true;

			// Clear existing selected item
			if (prevSelectedItem) {
				prevSelectedItem.select(false);
			}

			// Select new item
			if (item) {
				item.select();
			}

			this._selectedItem = item;
		}
	}

	if (opt_event) {
		opt_event = {
			"eventType": opt_event["type"],
			"altKey": opt_event["altKey"],
			"ctrlKey": opt_event["ctrlKey"],
			"shiftKey": opt_event["shiftKey"]
		};
	}
	else {
		opt_event = {};
	}

	opt_event["item"] = item;

	// Dispatch itemActivated event
	this._dispatch("itemActivated", opt_event);

	if (isSelectionChanged) {
		opt_event["prevSelectedItem"] = prevSelectedItem;

		// Dispatch itemSelected event
		this._dispatch("itemSelected", opt_event);
	}

	if (opt_includeChildren !== false) {
		subItem = result <= 0 ? item : null; // If item has already been selected, selected item in each sub-menu should be clear

		for (i = 0; i < this._subMenus.length; i++) {
			// Select item in each sub-menu. The item can be either initial item or null.
			if (this._subMenus[i].selectItem(subItem, opt_event, true, false)) { // Valid when item is null or item is selectable and in sub-menu
				if (result < 1) { // Valid when the item has not been selected before
					result = selectionType; // Update result
				}
			}
		}
	}

	if (result < 1) return false;

	// Dispatch tree event of root menu
	if (!this._parentItem) {
		prevSelectedItem = this._treeSelectedItem;
		this._treeSelectedItem = item;

		// Dispatch treeItemActivated event
		this._dispatch("treeItemActivated", opt_event);

		if (result === 1 && item !== prevSelectedItem) {
			opt_event["prevSelectedItem"] = prevSelectedItem;

			// Dispatch treeItemSelected event
			this._dispatch("treeItemSelected", opt_event);
		}
	}

	return true;
};

/** Filter only criteria-matched items in sub items
 * @public
 * @param {string | Array.<string> | Object | null} criteria
 * @param {boolean=} opt_includeChildren
 * @return {number} Total number of filtered items
 */
tr.ListMenu.prototype.filterItems = function(criteria, opt_includeChildren) {
	if (!criteria) criteria = null; // Ensure null criteria if not able to be used

	var _this, i, item, itemType, subMenu, isSubMatched, isMatched, totalMatched;

	_this = this;
	totalMatched = 0;

	_this.enableClass("no-selected", !goog.isNull(criteria));

	for (i = 0; i < _this._items.length; i++) {
		item = _this._items[i];
		itemType = item.getType();
		isMatched = false;

		if (itemType === "default" || itemType === "command") {
			isMatched = item.matchCriteria(criteria) && !item.isDisabled();
			isSubMatched = false;

			// Filter child
			subMenu = item.getSubMenu();

			if (subMenu) {
				isSubMatched = subMenu.filterItems(opt_includeChildren !== false ? criteria : null, true) > 0;
			}

			if (isSubMatched) {
				isMatched = true;
			}

			item.enableClass("has-child", isSubMatched === true);
		}

		item.show(!criteria || isMatched);

		if (isMatched) {
			++totalMatched;
		}
	}

	_this._filterCriteria = criteria;
	_this._totalMatched = totalMatched;

	return totalMatched;
};

/** Filter only criteria-matched items in tree items
 * @public
 * @param {string | Array.<string> | Object | null} criteria
 * @return {number} Total number of filtered items
 */
tr.ListMenu.prototype.filterTreeItems = function(criteria) {
	var _this = this;

	if (!criteria) criteria = null; // Ensure null criteria if not able to be used
	if (criteria === _this._filterCriteria) {
		return _this._totalMatched ? _this._totalMatched : _this._items.length;
	}

	return _this.getRootMenu().filterItems(criteria, true);
};

tr.ListMenu.prototype._init = function(opt_element) {
	if (!opt_element) {

	temp  = new ElementWrapper(/** @type{Element} */ (opt_element));

	temp.listen("keydown", _this._onKeyDown);
	_this._topNode = temp;

	temp = new ElementWrapper(document.createElement("ul"));
	temp.addClass("tr-list");

/** All custom or user settings should be done here
 * @protected
 * @ignore
 * @param {*} opt_model
 */
tr.ListMenu.prototype._deserialize = function(opt_model) {
	if (!opt_model) return;

	var _this, itemRenderer, items, maxHeight, i, subMenu;

	_this = this;

	itemRenderer = opt_model["itemRenderer"] || opt_model["itemRenderFunction"];
	if (itemRenderer) {
		_this.setItemRenderer(itemRenderer, false);
	}

	itemRenderer = opt_model["itemMatchingMaskFunction"];
	if (itemRenderer) {
		_this.setItemRenderer(itemRenderer, false);
	}

	items = opt_model["items"];
	if (items) {
		_this.setItems(items);
	}

	maxHeight = opt_model["maxHeight"] || opt_model["menuHeight"];
	if (maxHeight) {
		_this.setMaxHeight(maxHeight, false);
	}

	for (i = 0; i < _this._subMenus.length; i++) {
		subMenu = _this._subMenus[i];

		// Assign default renderer as of parent
		if (!subMenu.getItemRenderer()) {
			subMenu.setItemRenderer(itemRenderer);
		}

		// Assign default max-height as of parent
		if (!subMenu.getStyle("max-height")) {
			subMenu.setMaxHeight(maxHeight);
		}
	}
};

/**
 * @private
 */
tr.ListMenu.prototype._onKeyDown = function(event) {
	var _this, item;

	_this = this;

	switch (event.keyCode) {
		case 40: // Down
			_this.moveHighlightedItem(1);
			event.preventDefault();
			break;

		case 38: // Up
			_this.moveHighlightedItem(-1);
			event.preventDefault();
			break;

		case 13: // Enter
			item = _this.getHighlightedItem();
			if (item) {
				_this.selectTreeItem(item, event);
			}
			event.preventDefault();
			break;

		case 8: // Backspace
			event.preventDefault();
			break;
	}
};

/** Check if item is in menu
 * @private
 * @param {tr.MenuItem} item
 * @param {boolean=} opt_includeChildren
 * @return {boolean} True if item is in menu or in tree (the condition depends on the 2nd parameter)
 */
tr.ListMenu.prototype._hasItem = function(item, opt_includeChildren) {
	if (this.getItemIndex(item) >= 0) return true;

	var i;

	if (opt_includeChildren !== false) {
		for (i = 0; i < this._subMenus.length; ++i) {
			if (this._subMenus[i]._hasItem(item)) return true;
		}
	}

	return false;
};

/**
 * @private
 * @param {tr.MenuItem} item
 */
tr.ListMenu.prototype._scrollToItem = function(item) {
	var _this,
		element, topNodeElement,
		itemTop, itemHeight, itemBottom,
		menuTop, menuHeight, menuBottom;

	_this = this;
	element = item.getElement();

	if (goog.isNull(element)) return;

	itemTop = element.offsetTop;
	itemHeight = element.offsetHeight;
	itemBottom = itemTop + itemHeight;

	menuTop = _this._topNode.getScrollTop();
	menuHeight = _this._topNode.getHeight();
	menuBottom = menuTop + menuHeight;

	// Do nothing if item is already shown
	if (menuTop < itemTop && menuBottom > itemBottom) {
		return;
	}

	topNodeElement = _this._topNode.getElement();

	if (menuTop > itemTop) { // Scroll up to item
		topNodeElement.scrollTop = itemTop;
	} 
	else if (itemTop - menuBottom < itemHeight)  { // Scroll down and item show item fully.
		topNodeElement.scrollTop = itemBottom - menuHeight;
	}
};
