/**
 * @constructor
 * @implements {tr.IElementControl}
 * @extends {tr.ElementControl}
 * @param {Element|Object=} opt_initializer
 * @param {Object=} opt_model
 */
tr.DropDown = function(opt_initializer, opt_model) {
	var _this = this;

	// Add custom events
	_this._addEvent("itemActivated");
	_this._addEvent("itemSelected");
	_this._addEvent("itemClicked"); // Deprecated

	_this._onTrig = _this._onTrig.bind(_this);
	_this._onItemActivated = _this._onItemActivated.bind(_this);
	_this._onItemSelected = _this._onItemSelected.bind(_this);
	_this._onPopupMenuShown = _this._onPopupMenuShown.bind(_this);

	if (!tr.isElement(opt_initializer)) {
		opt_model = opt_initializer;
		opt_initializer = null;
	}

	_this._init(/** @type{Element} */ (opt_initializer));
	_this._deserialize(opt_model);
};

goog.inherits(tr.DropDown, tr.ElementControl);

/**
 * @inheritDoc
 */
tr.DropDown.prototype.dispose = function() {
	var _this = this;

	_this.unlistenAll();

	_this._popupMenu.dispose();
	_this._popupMenu = null;

	if (_this._externalInput) {
		_this.setInput(null);
	}
	else {
		_this._topNode.dispose();
		_this._topNode = null;
	}
};

/** Enable dropdown
 * @public
 * @param {boolean=} opt_value
 */
tr.DropDown.prototype.enable = function(opt_value) {
	opt_value = opt_value !== false;
	this._topNode.enable(opt_value);

	// Always hide menu if disabled
	if (!opt_value) {
		this._popupMenu.hide();
	}
};

/** Check if dropdown is disabled
 * @public
 * @return {boolean} True if disabled
 */
tr.DropDown.prototype.isDisabled = function() {
	return !this.isEnabled();
};

/**
 * @public
 * @param {boolean=} opt_shown

tr.DropDown.prototype.showMenu = function(opt_shown) {
	this._popupMenu.show(opt_shown);
};

/**
 * @public
 * @param {boolean=} opt_hidden
 */
tr.DropDown.prototype.hideMenu = function(opt_hidden) {
	this.showMenu(opt_hidden === false);
};

/** Set place holder text
 * @public
 * @param {string} text
 */
tr.DropDown.prototype.setPlaceHolderText = function(text) {
	if (!text) text = ""; // Ensure string type
	if (text === this._placeHolderText) return; // Do nothing if placeholder text is not changed

	this._placeHolderText = text;
	this._updateContent();
};

tr.DropDown.prototype.getPlaceHolderText = function() {

/**
 * Attach this dropdown menu to the specified input element or any nova control
 * @public
 * @param {Element | tr.ElementWrapper} input
 * @param {boolean=} opt_external
 */
tr.DropDown.prototype.setInput = function(input, opt_external) {
	var _this = this;

	// Set flag to indicate if external input
	_this._externalInput = opt_external !== false;

	if (tr.isElement(input)) {
		if (_this._externalInput) {
			input = new ElementWrapper( /** @type{Element} */ (input));
		}
		else {
			input = new ElementWrapper(/** @type{Element} */(input));
			input.addClass("nova-button");
			input.addClass("tr-dropdown");
		}
	}

	if (_this._topNode) {
		if (input && input.getElement() === _this._topNode.getElement()) return; // Do nothing if input element is not changed

		if (_this._externalInput) {  // External input: unlisten only dropdown handlers
			_this._topNode.unlisten("click", _this._onTrig);
		}
		else { // Internal input: unlisten all event and dispose
			_this._topNode.unlistenAll();
			_this._topNode.dispose();
		}
	}

	if (input) {
		_this._topNode =  /** @type{tr.ElementWrapper} */ (input); // << _this breaks all designs. All styles are lost
		_this._topNode.listen("click", _this._onTrig);
	}
	else {
		_this._topNode = null;
	}

	if(_this._popupMenu) {
		_this._popupMenu.setAnchor(_this._topNode);
		_this._popupMenu.listen("shown", _this._onPopupMenuShown);
	}
};

tr.DropDown.prototype.setMenu = function(menu) {
	var _this = this;

	_this._popupMenu.setMenu(menu);
	_this._resetState();
};

/**
 * Set menu of popup menu
 * @public
 * @param {Array.<tr.MenuItem> | Array.<Object>} items
 */
tr.DropDown.prototype.setItems = function(items) {
	var _this, menu;

	_this = this;
	menu = _this._popupMenu.getMenu();

	if (!menu) {
		menu = new ListMenu();
		_this._popupMenu.setMenu(menu);
	}

	menu.setItems(items);

	_this._resetState();
};

/** Set new model
* @public
* @param {*} model
*/
tr.DropDown.prototype.setModel = function(model) {
	this._resetState();
	this._deserialize(model);
};

/** All default settings should be done here
 * @ignore
 * @protected
 * @param {Element=} opt_element
 */
tr.DropDown.prototype._init = function(opt_element) {
	if (!opt_element) {
		opt_element = document.createElement("div");
	}

	var _this, temp;

	_this = this;

	// Popup menu
	temp = new PopupMenu();
	temp.listen("treeItemActivated", _this._onItemActivated);
	temp.listen("treeItemSelected", _this._onItemSelected);
	_this._popupMenu = temp;

	// Top node (input)
	_this.setInput(opt_element, false);
};

/** All custom or user settings should be done here
 * @ignore
 * @protected
 * @param {*} opt_model
 */
tr.DropDown.prototype._deserialize = function(opt_model) {
	if (!opt_model) opt_model = {};

	var _this, pref;

	_this = this;

	pref = opt_model["menu"];
	if (pref) {
		_this.setMenu(pref);
	}

	pref = opt_model["placeHolderText"];
	if (pref) {
		_this.setPlaceHolderText(pref);
	}
	else {
		_this._updateContent();
	}
};

//#region private methods

/**
 * @private
 */
tr.DropDown.prototype._onTrig = function(e) {
	// Prevent IE event in case of non-input top node
	if (!this.isEnabled()) {
		e.preventDefault();
		return;
	}

	this._popupMenu.toggleVisibility();
};

/**
 * @private
 */
tr.DropDown.prototype._onItemActivated = function(event) {
	var _this = this;

	_this._popupMenu.hideTree();

	// Dispatch itemActivated event
	_this._dispatch("itemActivated", event);
	_this._dispatch("itemClicked", event); // Deprecated

	// Focus textBox if item is activated by browser event
	if (event["eventType"]) {
		if (_this._topNode && _this._topNode.getElement()) {
			_this._topNode.getElement().focus();
		}
	}
};

/**
 * @private
 */
tr.DropDown.prototype._onItemSelected = function(event) {
	var _this, item;

	_this = this;
	item = event["item"];

	if (item === _this._selectedItem) return; // Do nothing if selected item is not changed

	_this._selectedItem = item;
	_this._updateContent();

	// Dispatch itemSelected event
	_this._dispatch("itemSelected", event);
};

/**
 * @private
 */
tr.DropDown.prototype._onPopupMenuShown = function(e) {
	this._popupMenu.updatePosition(); // Ensure correct position
	this._popupMenu.setMinWidth(this.getWidth()); // Ensure proper width
};

/**
 * @private
 */
tr.DropDown.prototype._updateContent = function() {
	var _this, text;

	_this = this;

	if (_this._externalInput) return;

	if (!goog.isNull(_this._selectedItem)) {
		text = _this._selectedItem.getText();
	}

	_this.setContent(text || _this._placeHolderText);
};

/**
 * @private
 */
tr.DropDown.prototype._resetState = function() {
	this._selectedItem = null;
	this._updateContent();
};
