goog.provide("tr.grid.PercentBar");
goog.require("tr.ElementControl");
goog.require("tr.IElementControl");
goog.require("tr.cgrid.util");

/**
 * @constructor
 * @extends {tr.ElementControl}
 * @implements {tr.IElementControl}
 * @param {Element=} opt_element
 */
tr.grid.PercentBar = function (opt_element) {
	this._topNode = new tr.ElementWrapper(opt_element);
	this._topNode.getElement()._percentBar = this; // For internal reference by PercentBarRenderer
	this._topNode.setClass("tr-percent-bar");

	this._bar = document.createElement("div");
	this._track = document.createElement("div");
	this._track.appendChild(this._bar);

	this._txt = document.createElement("span");
	this.setAlignment("c");
};
goog.inherits(tr.grid.PercentBar, tr.ElementControl);

/**
 * @public
 * @returns {number}
 */
tr.grid.PercentBar.prototype.getValue = function () {
	return this._absValue * this._sign;
};
/**
 * @public
 * @returns {number}
 */
tr.grid.PercentBar.prototype.getPercent = function () {
	return (this._absValue / this._maxValue) * this._sign;
};

/**
 * @public
 * @param {number|string} value
 * @param {number=} opt_max
 * @param {string=} opt_displayText
 */
tr.grid.PercentBar.prototype.setValue = function (value, opt_max, opt_displayText) {
	value = tr.isNumeric(value) ? Number(value) : NaN;

	var isEmptyValue = !value && (value !== 0 || opt_displayText === ""); // NaN, undefined, empty string
	var isValidMaxValue = opt_max !== 0; // should not be zero due to percent calculation: absValue / maxValue * 100

	if (isEmptyValue || !isValidMaxValue) {
		this._absValue = NaN;

		if(!this._invalid) {
			this._invalid = true;
			this._topNode.hide();
		}

		return;
	} else {
		if(this._invalid) {
			this._invalid = false;
			this._topNode.show();
		}
	}

	if(value >= 0) {
		this._absValue = value;
		this._dirtySign |= this._sign !== 1;
		this._sign = 1;
	} else {
		this._absValue = -value;
		this._dirtySign |= this._sign !== -1;
		this._sign = -1;
	}

	if(opt_max !== undefined) {
		this._maxValue = (opt_max >= 0) ? opt_max : -opt_max;
	}

	if(this._absValue > this._maxValue) {
		this._absValue = this._maxValue;
	}

	this._update(true, opt_displayText);
};
/**
 * @public
 * @param {number} absVal
 */
tr.grid.PercentBar.prototype.setMax = function (absVal) {
	this._maxValue = (absVal >= 0) ? absVal : -absVal;
	if(this._absValue > this._maxValue) {
		this._absValue = this._maxValue;
	}
	this._update();
};

/**
 * @public
 * @param {string} txt
 */
tr.grid.PercentBar.prototype.setText = function (txt) {
	if(!this._invalid) {
		if(this._txt._prevText !== txt) {
			this._txt._prevText = txt;
			this._txt.textContent = txt;
			if (this._tooltipEnabled) {
				this._topNode.setTooltip(txt);
			}
		}
	}
};
/**
 * @public
 * @param {boolean=} opt_hiden
 */
tr.grid.PercentBar.prototype.hideText = function (opt_hiden) {
	this._textHidden = opt_hiden !== false;
	this._txt.style.display = this.isTextHidden() ? "none" : "";
	this.setAlignment(this._alignment);
};
/**
 * @public
 * @return {boolean}
 */
tr.grid.PercentBar.prototype.isTextHidden = function () {
	return this._textHidden || (this._textWidth === "0px");
};

/** Use when alignment is not center. This will be a reserved space for text
 * @public
 * @param {number|string} width
 */
tr.grid.PercentBar.prototype.setTextWidth = function (width) {
	if(!width && width !== 0) { // undefined, NaN, null, and ""
		width = 56;
	}
	var txtWidth = (typeof width === "number") ? width + "px" : width;
	if(this._textWidth !== txtWidth) {
		this._textWidth = txtWidth;
		this._txt.style.display = this.isTextHidden() ? "none" : "";
		this.setAlignment(this._alignment);
	}
};
/** @public
 * @function
 * @param {number|string} width
 */
tr.grid.PercentBar.prototype.setMaxTextLength = tr.grid.PercentBar.prototype.setTextWidth;
/**
 * @public
 * @return {number} scrollWidth
 */
tr.grid.PercentBar.prototype.getTextScrollWidth = function() {
	if(this._txt.scrollWidth > this._txt.offsetWidth) {
		return this._txt.scrollWidth;
	}
	return NaN;
};

/**
 * @public
 * @returns {string} alignment Alignment can be "left", "center", "right"
 */
tr.grid.PercentBar.prototype.getAlignment = function () {
	switch(this._alignment) {
		case "l": return "left";
		case "r": return "right";
	}
	return "center";
};

/**
 * @public
 * @param {string} alignment Alignment can be "left", "center", "right"
 */
tr.grid.PercentBar.prototype.setAlignment = function (alignment) {
	if(alignment) {
		alignment = alignment.charAt(0).toLowerCase();
	} else {
		alignment = "c";
	}

	if(alignment === "c") {
		this._track.style.width = "50%";
		this._txt.style.width = '50%';
		this._txt.style.textOverflow = 'elipsis';
		this._dirtySign = true;
	} else {
		if (this.isTextHidden()) {
			this._track.style.width = "100%";
		} else {
			this._track.style.width = "calc(100% - " + this._textWidth + ")";
		}
		this._txt.style.width = this._textWidth; // Text width can be percent ("50%") or pixel ("50px")
		this._txt.style.textAlign = "right";

		this._track.style.left = ""; // Important
		this._txt.style.transform = ""; // Important

		if(alignment === "r") {
			this._bar.style.right = "0";
		} else {
			this._bar.style.right = "";
		}
	}

	if(this._alignment !== alignment) {
		this._alignment = alignment;
		if(alignment !== "l") {
			this._topNode.addContent(this._track);
			this._topNode.addContent(this._txt);
		} else { // align left
			this._topNode.addContent(this._txt);
			this._topNode.addContent(this._track);
		}
	}
	this._update(false);
};

/**
 * @public
 * @param {Object=} colorObject An object with property "up" and "down"
 */
tr.grid.PercentBar.prototype.setMovementColor = function (colorObject) {
	if(colorObject) {
		if(colorObject["up"] && colorObject["down"]) {
			this._movementColor = colorObject;
		}
	}
};
 /** @public
 * @param {boolean=} opt_enabled
 */
tr.grid.PercentBar.prototype.useMovementColor = function (opt_enabled) {
	this._movementColorUsed = opt_enabled !== false;
	this._updateBarColor();
};

/**
 * @public
 * @param {string=} color
 */
tr.grid.PercentBar.prototype.setTrackColor = function (color) {
	this._track.style.backgroundColor = color ? color : "";
};

 /** @public
 * @param {boolean=} opt_enabled
 */
tr.grid.PercentBar.prototype.enableTooltip = function (opt_enabled) {
	this._tooltipEnabled = opt_enabled !== false;
	this._topNode.setTooltip('');
};

/**
 * @private
 * @param {boolean=} opt_updateVal
 * @param {string=} opt_displayText
 */
tr.grid.PercentBar.prototype._update = function (opt_updateVal, opt_displayText) {
	if(this._dirtySign) {
		this._dirtySign = false;
		if(this._alignment === "c") {
			if(this._sign >= 0) {
				this._track.style.left = "50%";
				this._bar.style.right = "";
				this._txt.style.transform = "translateX(-100%)";
				this._txt.style.textAlign = "right";
			} else {
				this._track.style.left = "";
				this._bar.style.right = "0";
				this._txt.style.transform = "";
				this._txt.style.textAlign = "left";
			}
		}

		if(this._movementColorUsed) {
			this._updateBarColor();
		}
	}

	if (opt_updateVal !== false) {
		var percent = (this._absValue / this._maxValue) * 100;
		this._bar.style.width = percent + "%";
		if(opt_displayText) {
			this.setText(opt_displayText);
		} else {
			this.setText(Math.floor(percent * this._sign * 100) / 100 + "%");
		}
	}
};
/**
 * @private
 */
tr.grid.PercentBar.prototype._updateBarColor = function () {
	if(this._movementColorUsed) {
		if(this._sign >= 0) {
			this._bar.style.backgroundColor = (this._movementColor && this._movementColor["up"]) ? this._movementColor["up"] : "green";
		} else {
			this._bar.style.backgroundColor = (this._movementColor && this._movementColor["down"]) ? this._movementColor["down"] : "red";
		}
	} else {
		this._bar.style.backgroundColor = "";
	}
};
