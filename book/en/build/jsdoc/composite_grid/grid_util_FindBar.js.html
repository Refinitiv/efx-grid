goog.provide("tr.grid.FindBar");
goog.require("tr.ElementControl");
goog.require("tr.ElementWrapper");
goog.require("tr.IElementControl");
goog.require("tr.ElementFrameWork");

/**
 * @constructor
 * @extends {tr.ElementControl}
 * @implements {tr.IElementControl}
 */
tr.grid.FindBar = function () {
	this._onInputKeyUp = this._onInputKeyUp.bind(this);
	this._onInputKeyDown = this._onInputKeyDown.bind(this);
	this._onFindPrev = this._onFindPrev.bind(this);
	this._onFindNext = this._onFindNext.bind(this);
	this._onClose = this._onClose.bind(this);

	var curPos = document.createElement("span");
	var ofSpan = document.createElement("span");
	var totalMatches = document.createElement("span");
	
	curPos.textContent = "0";
	ofSpan.textContent = "of";
	totalMatches.textContent = "0";
	
	var separator = document.createElement("div");
	var prevButton = tr.ElementFrameWork.createIconButton("down");
	var nextButton = tr.ElementFrameWork.createIconButton("up");
	var closeButton = tr.ElementFrameWork.createIconButton("cross-mini");
	
	separator.className = "tr-separator";
	prevButton.setAttribute("disabled", "");
	nextButton.setAttribute("disabled", "");
	prevButton.title = "Previous";
	nextButton.title = "Next";
	closeButton.title = "Close find bar";
	
	prevButton.classList.add("prev-button");
	nextButton.classList.add("next-button");
    closeButton.classList.add("close-button");
	prevButton.addEventListener("click", this._onFindPrev, false);
	nextButton.addEventListener("click", this._onFindNext, false);
	closeButton.addEventListener("click", this._onClose, false);
	
	var posPanel = document.createElement("div");
	posPanel.className = "pos-panel";
	posPanel.style.display = "none";
	posPanel.appendChild(curPos);
	posPanel.appendChild(ofSpan);
	posPanel.appendChild(totalMatches);
	
	var lastCell = document.createElement("td");
	lastCell.appendChild(posPanel);
	lastCell.appendChild(separator);
	lastCell.appendChild(prevButton);
	lastCell.appendChild(nextButton);
	lastCell.appendChild(closeButton);
	
	var firstCell = document.createElement("td");
	var inputBox = document.createElement("input");
	inputBox.setAttribute("type", "text");
	inputBox.addEventListener("keyup", this._onInputKeyUp, false);
	inputBox.addEventListener("keydown", this._onInputKeyDown, false);
	firstCell.appendChild(inputBox);
	
	var row = document.createElement("tr");
	row.appendChild(firstCell);
	row.appendChild(lastCell);
	
	var tbody = document.createElement("tbody");
	tbody.appendChild(row);
	
	var cols = document.createElement("colgroup");
	cols.appendChild(document.createElement("col"));
	cols.appendChild(document.createElement("col"));
	
	var table = document.createElement("table");
	table.className = "tr-find-bar";
	table.appendChild(cols);
	table.appendChild(tbody);
	table._FindBar = this; // For internal reference

	this._topNode = new tr.ElementWrapper(table);
	this._userInput = inputBox;
	this._posPanel = posPanel;
	this._curPosSpan = curPos;
	this._totalSpan = totalMatches;
	this._prevButton = prevButton;
	this._nextButton = nextButton;
	this._closeButton = closeButton;
	this._results = [];
	
	this._addEvent("newFind");
	this._addEvent("nextFind");
	this._addEvent("closed");
};
goog.inherits(tr.grid.FindBar, tr.ElementControl);

/** @public
* @param {Array} ary
 */
tr.grid.FindBar.prototype.setResults = function (ary) {
	this._results = ary;
	if(this._results && this._results.length) {
		this._curPosSpan.textContent = 1;
		this._totalSpan.textContent = this._results.length;
		this._prevButton.removeAttribute("disabled");
		this._nextButton.removeAttribute("disabled");
	} else {
		this._curPosSpan.textContent = "0";
		this._totalSpan.textContent = "0";
		this._prevButton.setAttribute("disabled", "");
		this._nextButton.setAttribute("disabled", "");
	}
};

/** @public
* @param {string=} opt_initialInput
 */
tr.grid.FindBar.prototype.showInDocument = function (opt_initialInput) {
	var parent = this._defaultParent || document.body;
	this._topNode.setParent(parent);
	if(opt_initialInput !== undefined) {
		this._userInput.value = opt_initialInput;
	}
	this._userInput.focus();
	this._userInput.select();
	this._dispatch("newFind", {"type": "popup"});
};
/** @public
 */
tr.grid.FindBar.prototype.close = function () {
	this._topNode.setParent(null);
	this._dispatch("closed", {});
};


/** @public
 */
tr.grid.FindBar.prototype.findAll = function() {
	this._dispatch("newFind", {"type": "code"});
};
/** @public
 */
tr.grid.FindBar.prototype.findPrev = function () {
	var curPos = this.getCurrentMatch();
	var targetPos = this._getValidMatch(curPos - 1); // The same position can be returned
	if(targetPos < 0) { return; }
	
	this._curPosSpan.textContent = targetPos + 1;
	this._dispatch("nextFind", {"current": curPos, "target": targetPos});
};
/** @public
 */
tr.grid.FindBar.prototype.findNext = function () {
	var curPos = this.getCurrentMatch();
	var targetPos = this._getValidMatch(curPos + 1); // The same position can be returned
	if(targetPos < 0) { return; }
	
	this._curPosSpan.textContent = targetPos + 1;
	this._dispatch("nextFind", {"current": curPos, "target": targetPos});
};
/** @public
* @param {number} at
*/
tr.grid.FindBar.prototype.setCurrentMatch = function (at) {
	var targetPos = this._getValidMatch(at);
	if(targetPos < 0) { return; }
	
	var curPos = this.getCurrentMatch();
	this._curPosSpan.textContent = targetPos + 1;
	this._dispatch("nextFind", {"current": curPos, "target": targetPos});
};

tr.grid.FindBar.prototype.getCurrentMatch = function () { 
	return +(this._curPosSpan.textContent) - 1;
};
/** @private
* @param {number} pos
* @return {number}
 */
tr.grid.FindBar.prototype._getValidMatch = function (pos) {
	if(!this._results || this._results.length <= 0) {
		return -1;
	}

	if(pos >= this._results.length) {
		return 0; // To perform loop back search
	} else if(pos < 0) {
		return this._results.length - 1; // To perform loop back search
	}
	return pos;
};

/** @public
* @param {Element} elem
 */
tr.grid.FindBar.prototype.setDefaultParent = function (elem) {
	this._defaultParent = elem;
};


/** @private
* @param {Object} e
 */
tr.grid.FindBar.prototype._onInputKeyUp = function (e) {
	var searchedText = this._userInput.value;
	if(searchedText === this._searchedText) {
		return;
	}
	
	this._searchedText = searchedText;
	if(this._searchedText) {
		this._posPanel.style.display = "";
	} else {
		this._posPanel.style.display = "none";
	}
	
	e.stopPropagation();
	this._dispatch("newFind", e);
};
/** @private
* @param {Object} e
 */
tr.grid.FindBar.prototype._onInputKeyDown = function (e) {
	if(!e.ctrlKey) {
		e.stopPropagation();
	}
	if(e.keyCode === 114 || e.keyCode === 13) { // F3 or Enter key
		if(e.shiftKey) {
			this.findPrev();
		} else {
			this.findNext();
		}
		e.preventDefault();
	} else if(e.keyCode === 27) { // Escape key
		this.close();
		e.preventDefault();
	}
};
/** @private
* @param {Object} e
 */
tr.grid.FindBar.prototype._onFindPrev = function (e) {
	e.stopPropagation();
	this.findPrev();
};
/** @private
* @param {Object} e
 */
tr.grid.FindBar.prototype._onFindNext = function (e) {
	e.stopPropagation();
	this.findNext();
};
/** @private
* @param {Object} e
 */
tr.grid.FindBar.prototype._onClose = function (e) {
	e.stopPropagation();
	this.close();
};
