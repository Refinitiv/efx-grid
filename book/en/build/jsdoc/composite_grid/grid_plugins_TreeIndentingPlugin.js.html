/** @constructor
* @param {number=} opt_displayColumn
* @extends {tr.grid.Plugin}
* @implements {tr.grid.IPlugin}
*/
tr.grid.TreeIndentingPlugin = function (opt_displayColumn) {
	var _t = this;
	if (opt_displayColumn !== undefined && opt_displayColumn > 0) {
		_t._displayColumn = opt_displayColumn;
	}

	_t._onColumnMoved = _t._onColumnMoved.bind(_t);
	_t._onColumnAdded = _t._onColumnAdded.bind(_t);
	_t._onColumnRemoved = _t._onColumnRemoved.bind(_t);
	_t._onCellClicked = _t._onCellClicked.bind(_t);
	
	_t._onPostSectionDataBinding = _t._onPostSectionDataBinding.bind(_t);
	_t._onPostDataSourceChanged = _t._onPostDataSourceChanged.bind(_t);
	_t._onGroupCriteriaChanged = _t._onGroupCriteriaChanged.bind(_t);
	_t._onGroupAdded = _t._onGroupAdded.bind(_t);
	

	_t._addEvent(tr.res.clicked);
	_t._addEvent("headerBinding");
	_t._addEvent("footerBinding");
	_t._addEvent("contentBinding");
};
goog.inherits(tr.grid.TreeIndentingPlugin, tr.grid.Plugin);

//#region IPlugin
/** @override */
tr.grid.TreeIndentingPlugin.prototype.getName = function () {
	return "TreeIndentingPlugin"; // Read Only
};
/** @override */
tr.grid.TreeIndentingPlugin.prototype.initialize = function (host) {
	this._host = host;
	this._host.listen(tr.res.columnMoved, this._onColumnMoved);
	this._host.listen(tr.res.columnAdded, this._onColumnAdded);
	this._host.listen(tr.res.columnRemoved, this._onColumnRemoved);
	this._host.listen("postSectionDataBinding", this._onPostSectionDataBinding);
	this._host.listen("dataSourceChanged", this._onPostDataSourceChanged);
};
/** @override */
tr.grid.TreeIndentingPlugin.prototype.unload = function (host) {
	this._host.unlisten(tr.res.columnMoved, this._onColumnMoved);
	this._host.unlisten(tr.res.columnAdded, this._onColumnAdded);
	this._host.unlisten(tr.res.columnRemoved, this._onColumnRemoved);
	this._host.unlisten("postSectionDataBinding", this._onPostSectionDataBinding);
	this._host.unlisten("dataSourceChanged", this._onPostDataSourceChanged);
	// TODO unlilsten to the dataView

	this._host = null;
};

//#endregion IPlugin

//#region Public methods
/** @public
* @param {boolean=} opt_disabled
*/
tr.grid.TreeIndentingPlugin.prototype.disable = function (opt_disabled) {
	var val = opt_disabled !== false;
	if(val === this._disabled) {
		return;
	}
	if(val) {
		this._clearSectionStyles(this._displayColumn);
	}
	this._disabled = val;
};
/** @public
* @param {number} start of tree depth where the indent should begin
*/
tr.grid.TreeIndentingPlugin.prototype.setIndentStart = function (start) {
	start = start > 0 ? start : 0;
	this._indentStart = start;
	this.updateHeaders();
};
/**Negative number means no ends for indenting level. 
* Zero value means there will be no indent
* @public 
* @param {number} end of tree depth where the indent should be ended
*/
tr.grid.TreeIndentingPlugin.prototype.setIndentEnd = function (end) {
	this._indentEnd = end;
	this.updateHeaders();
};

/** @public
* @param {number} size
*/
tr.grid.TreeIndentingPlugin.prototype.setIndentSize = function (size) {
	if (this._indentSize !== size) {
		this._indentSize = size;
		this.updateHeaders();
	}
};

	if(this._displayColumn === index) { return; }
	
	this._clearSectionStyles(this._displayColumn);
	this._displayColumn = index;
	this.updateHeaders();

	this.collapseAll(opt_expanded === false);

* @param {boolean=} opt_collapsed

tr.grid.TreeIndentingPlugin.prototype.collapseAll = function (opt_collapsed) {
	if (!this._host) { return; }

	this._host.getDataSource().collapseAllGroups(opt_collapsed);
};

* @param {string} rowId
* @param {boolean=} opt_expanded
*/
tr.grid.TreeIndentingPlugin.prototype.expand = function (rowId, opt_expanded) {
	this.collapse(rowId, opt_expanded === false);
};
/** @public
* @param {string} rowId
* @param {boolean=} opt_collapsed
*/
tr.grid.TreeIndentingPlugin.prototype.collapse = function (rowId, opt_collapsed) {
	if (!this._host) { return; }
	var dv = this._host.getDataSource();
	var grp = dv.getGroup(rowId);
	if(grp) {
		grp.collapse(opt_collapsed);
	}
};

/**
* Refresh the indentation of all group headers
* @deprecated
* @public
*/
tr.grid.TreeIndentingPlugin.prototype.updateAllIndents = function () {
};
/**
* Refresh all group headers text by dispatching "headerBinding"
* @public
*/
tr.grid.TreeIndentingPlugin.prototype.updateHeaders = function () {
	if (this._host === null || this._disabled) { return; }
	
	var settingList = this._host.getAllSectionSettings(tr.res.content);
	for (var i = settingList.length; --i >= 0; ) {
		this._updateHeader(settingList[i]);
	}
};

/**
* @public
* @param {boolean=} opt_disabled
*/
tr.grid.TreeIndentingPlugin.prototype.disableClicking = function (opt_disabled) {
	this._clickDisabled = opt_disabled !== false;
};

/**
* @public
* @param {boolean=} opt_enabled
*/
tr.grid.TreeIndentingPlugin.prototype.enableHeaderRowSpanning = function (opt_enabled) {
	opt_enabled = opt_enabled !== false;
	if(this._headerSpanning === opt_enabled) {
		return;
	}
	
	if(!opt_enabled) {
		this._clearSectionStyles(this._displayColumn, true);
	}
	this._headerSpanning = opt_enabled;
	this.updateHeaders();
};

/**
* @public
* @param {boolean=} opt_enabled
*/
tr.grid.TreeIndentingPlugin.prototype.enableFooterRowSpanning = function (opt_enabled) {
	opt_enabled = opt_enabled !== false;
	if(this._footerSpanning === opt_enabled) {
		return;
	}
	
	if(!opt_enabled) {
		this._clearSectionStyles(this._displayColumn, true);
	}
	this._footerSpanning = opt_enabled;
	this.updateHeaders();
};

/**
* @public
* @param {boolean|number=} opt_num
*/
tr.grid.TreeIndentingPlugin.prototype.enableFooterRow = function (opt_num) {
	if(opt_num === undefined || opt_num === true) {
		opt_num = 1;
	} else if(opt_num <= 0) { // Including false and null
		opt_num = 0;
	}
	
	this._footerRows = /** @type{number} */(opt_num);
};

//#region Private methods
/** @private
* @param {Object} e
*/
tr.grid.TreeIndentingPlugin.prototype._onPostDataSourceChanged = function(e) {
	var prev = /** @type{tr.DataView} */(e["prevDataSource"]);
	if(prev) {
		prev.unlisten("groupCriteriaChanged", this._onGroupCriteriaChanged);
		prev.unlisten("groupAdded", this._onGroupAdded);
	}
	
	var dv = /** @type{tr.DataView} */(e[tr.res.dataSource]);
	if(dv) {
		dv.listen("groupCriteriaChanged", this._onGroupCriteriaChanged);
		dv.listen("groupAdded", this._onGroupAdded);
	}
};
/** @private
* @param {Object} e
*/
tr.grid.TreeIndentingPlugin.prototype._onGroupCriteriaChanged = function(e) {
	var dv = /** @type{tr.DataView} */(e["sender"]);
	if(!dv.hasGroup()) {
		this._clearSectionStyles(this._displayColumn, true);
	}
};
/** @private
* @param {Object} e
*/
tr.grid.TreeIndentingPlugin.prototype._onGroupAdded = function(e) {
	if(this._footerRows) {
		e["newGroup"].addFooter(this._footerRows);
	}
};

/** @private
* @param {Object} e
*/
tr.grid.TreeIndentingPlugin.prototype._onPostSectionDataBinding = function(e) {
	var settings = e[tr.res.sectionSettings];
	this._updateHeader(settings, e[tr.res.fromRowIndex], e[tr.res.toRowIndex] - 1);
};

/** @private
* @param {MouseEvent} e
*/
tr.grid.TreeIndentingPlugin.prototype._onCellClicked = function (e) {
	if(this._clickDisabled) {
		return;
	}
	
	var cellElem = e.currentTarget;
	if(!cellElem.classList.contains("folder")) {
		return;
	}
	
	var collapsed = cellElem.classList.contains("closed");
	if(collapsed) {
		cellElem.classList.remove("closed");
	} else {
		cellElem.classList.add("closed");
	}
	
	var info = this._host.getRelativePosition(e);
	var rowIndex = info[tr.res.rowIndex];
	var settings = /** @type{tr.grid.SectionSettings} */(info[tr.res.sectionSettings]);
	
	var dv = settings.getDataSource();
	var grpView = dv.getGroup(dv.getRowId(rowIndex));
	if(grpView) {
		grpView.collapse(!collapsed);
	}

	info["expanded"] = collapsed; // After toggled
	this._dispatch(tr.res.clicked, info);
};

/** @override 
* @ignore
*/
tr.grid.TreeIndentingPlugin.prototype._onColumnMoved = function (e) {
	var from = /** @type{number} */(e[tr.res.fromColIndex]);
	var to = /** @type{number} */(e[tr.res.toColIndex]);
	
	if(this._fixedDisplayColumn) {
		this._clearSectionStyles(to);
	} else {
		this._clearSectionStyles(from);
		this._displayColumn = tr.moveIndex(this._displayColumn, from, to);
	}
	this.updateHeaders();
};

/** @override 
* @ignore
*/
tr.grid.TreeIndentingPlugin.prototype._onColumnAdded = function (e) {
	if (!e["atTheMiddle"]) { return; }
	var colIndex = /** @type{number} */(e[tr.res.colIndex]);
	var shiftedIndex = tr.moveIndex(this._displayColumn, null, colIndex);
	
	if(this._fixedDisplayColumn) {
		this._clearSectionStyles(shiftedIndex);
	} else {
		this._displayColumn = shiftedIndex;
	}
	this.updateHeaders(); // Update current span
};

/** @override 
* @ignore
*/
tr.grid.TreeIndentingPlugin.prototype._onColumnRemoved = function (e) {
	if (!e["atTheMiddle"]) { return; }
	var colIndex = /** @type{number} */(e[tr.res.colIndex]);
	var shiftedIndex = tr.moveIndex(this._displayColumn, colIndex, null);
	
	if(this._fixedDisplayColumn) {
		this._clearSectionStyles(shiftedIndex);
		this.updateHeaders(); // Update current span
	} else {
		this._displayColumn = shiftedIndex;
	}
	
};

/**
* @private
* @param {tr.grid.SectionSettings} settings
* @param {number=} firstIndex
* @param {number=} lastIndex INCLUSIVE
*/
tr.grid.TreeIndentingPlugin.prototype._updateHeader = function (settings, firstIndex, lastIndex) {
	if (!this._isGroupHeader(settings)) {
		return;
	}
	
	var dataSource = settings.getDataSource();
	var section = settings.getSection();
	var fi = (firstIndex || firstIndex === 0) ? firstIndex : section.getFirstIndexInView(); 
	var li = (lastIndex || lastIndex === 0) ? lastIndex : section.getLastIndexInView();
	
	var colCount = section.getColumnCount();
	var headerListener = this._hasListener("headerBinding");
	var contentListener = this._hasListener("contentBinding");
	var curGroupLevel = dataSource.getGroupLevel(fi);
	var arg = {};
	arg[tr.res.section] = section;
	
	for(var r = fi; r <= li; r++) {
		var rowId = dataSource.getRowId(r); // Slow
		var cell = section.getCell(this._displayColumn, r, false); // Work only one row
		if(!cell) { // There may be no header to be updated
			continue;
		}
		arg[tr.res.cell] = cell;
		arg[tr.res.rowIndex] = r;
		arg[tr.res.rowId] = rowId;
		arg["rowData"] = dataSource.getRowData(rowId);
		
		var groupView = dataSource.getGroupByRowId(rowId);
		if(groupView) {
			var footerIndex = groupView.getFooterRowIndex(rowId);
			arg["footerIndex"] = footerIndex;
			this._addHeaderStyle(section, this._displayColumn, r, colCount, footerIndex);
			cell.enableClass("closed", groupView.isCollapsed());
			
			var groupId = groupView.getGroupId();
			curGroupLevel = groupView.getGroupLevel();
			if(footerIndex >= 0 || headerListener) {
				arg["groupId"] = groupId;
				arg[tr.res.dataSource] = groupView;
				
				if(footerIndex >= 0) {
					this._dispatch("footerBinding", arg);
				} else {
					this._dispatch("headerBinding", arg);
				}
			} else {
				arg[tr.res.dataSource] = dataSource;
				cell.setContent(groupId);
			}
		} else {
			this._removeHeaderStyle(section, this._displayColumn, r, colCount);
			
			if(contentListener) {
				arg["rowData"] = dataSource.getRowData(rowId);
				arg[tr.res.dataSource] = dataSource;

				this._dispatch("contentBinding", arg);
			}
		}
		
		var rowLevel = (groupView) ? curGroupLevel : curGroupLevel + 1;
		var indent = 0;
		if (this._indentEnd !== 0 && rowLevel >= this._indentStart) {
			if (this._indentEnd < 0 || rowLevel < this._indentEnd) {
				indent = (rowLevel - this._indentStart + 1) * this._indentSize + 12; //Add default indent 12px
			}
		}
		cell.setStyle("paddingLeft", (indent > 0) ? indent + "px" : "");
	}
};
/**
* @private
* @param {number} colIndex
* @param {boolean=} opt_force
*/
tr.grid.TreeIndentingPlugin.prototype._clearSectionStyles = function (colIndex, opt_force) {
	if (this._host === null) { return; }
	
	var settingList = this._host.getAllSectionSettings(tr.res.content);
	for (var i = settingList.length; --i >= 0; ) {
		var settings = settingList[i];
		if (!opt_force && !this._isGroupHeader(settings)) {
			continue;
		}
		var section = settings.getSection();
		var colCount = section.getColumnCount();
		var fi = section.getFirstIndexInView(); 
		var li = section.getLastIndexInView();
		for(var r = fi; r <= li; r++) {
			this._removeHeaderStyle(section, colIndex, r, colCount);
		}
	}
};

/**
* @private
* @param {tr.ILayoutGrid} section
* @param {number} c
* @param {number} r
* @param {number} colCount
* @param {number} footerIndex
*/
tr.grid.TreeIndentingPlugin.prototype._addHeaderStyle = function (section, c, r, colCount, footerIndex) {
	var cell = section.getCell(c, r, false);
	if(!cell) { return; }
	
	if(this._headerSpanning && footerIndex < 0) {
		section.stretchCell(c, r, true, true); // No left stretching
	} else if(this._footerSpanning && footerIndex >= 0) {
		section.stretchCell(c, r, true, true); // No left stretching
	} else {
		for(var j = colCount; --j >= c;) {
			section.getCell(j, r, false).addClass("header");
		}
	}
	
	if(this._clickDisabled || footerIndex >= 0) {
		cell.addClass("header");
		cell.removeClass("folder");
	} else {
		cell.addClass("folder");
		if(!cell._collapsingListener) {
			cell._collapsingListener = true;
			cell.listen("click", this._onCellClicked);
		}
	}
};
/**
* @private
* @param {tr.ILayoutGrid} section
* @param {number} c
* @param {number} r
* @param {number} colCount
*/
tr.grid.TreeIndentingPlugin.prototype._removeHeaderStyle = function (section, c, r, colCount) {
	var cell = section.getCell(c, r, false);
	if(!cell) { return; }
	
	section.unstretchCell(c, r);
	for(var j = colCount; --j >= 0;) {
		section.getCell(j, r, false).removeClass("header"); // Very slow
	}
	cell.removeClass("folder");
	cell.removeClass("closed");
	cell.setStyle("paddingLeft", "");
};

/**
* @private
* @param {tr.grid.SectionSettings} settings
* @return {boolean}
*/
tr.grid.TreeIndentingPlugin.prototype._isGroupHeader = function (settings) {
	if(settings && settings.getType() === tr.res.content) {
		var dv = settings.getDataSource();
		if(dv) {
			return dv.hasGroup();
		}
	}
	
	return false;
};
