goog.provide("tr.grid.ColumnStackingPlugin");

goog.require("tr.res");
goog.require("tr.ElementWrapper");
goog.require("tr.ILayoutGrid");
goog.require("tr.grid.IPlugin");
goog.require("tr.grid.Plugin");
goog.require("tr.grid.Conflator");

/** Fires after clicking stack button
* @event tr.grid.ColumnStackingPlugin#clicked
* @property {tr.grid.Cell} cell
* @property {number} activeIndex Active column in the stack list
* @property {number} colIndex Primary column index that is being shown (left most column in the stack)
* @property {Array.<number>} stackList Indices of all columns in the stack list
*/
/** Fires after a new column stack icon is created.
* @event tr.grid.ColumnStackingPlugin#iconCreated
*/

/** @constructor
 * @extends {tr.grid.Plugin}
 * @implements {tr.grid.IPlugin}
 */
tr.grid.ColumnStackingPlugin = function () {
	var _t = this;

	_t._onPreSectionRender = _t._onPreSectionRender.bind(_t);

	_t._onColumnAdded = _t._onColumnAdded.bind(_t);
	_t._onColumnMoved = _t._onColumnMoved.bind(_t);
	_t._onColumnRemoved = _t._onColumnRemoved.bind(_t);
	_t._onStackButtonClicked = _t._onStackButtonClicked.bind(_t);
	_t._updateUI = _t._updateUI.bind(_t);

	_t._colData = []; // For storing column state
	_t._hosts = [];
	_t._stacks = {};
	
	_t._conflator = new grid.Conflator(100, _t._updateUI);


	_t._addEvent("clicked");
	_t._addEvent("iconCreated");
};
goog.inherits(tr.grid.ColumnStackingPlugin, tr.grid.Plugin);


/** @type {!Object.<string, Object>}
* @private
*/
tr.grid.ColumnStackingPlugin.prototype._stacks;

/** @type {number}
* @private
*/
tr.grid.ColumnStackingPlugin._runningId = 0;


/** @override */
tr.grid.ColumnStackingPlugin.prototype.getName = function () {
	return "ColumnStackingPlugin"; // read only
};
/** @override */
tr.grid.ColumnStackingPlugin.prototype.initialize = function (host) {
	if(this._hosts.indexOf(host) >= 0) { return; }
	this._hosts.push(host);
	if (this._hosts.length === 1) {
		host.listen(tr.res.columnAdded, this._onColumnAdded);
		host.listen(tr.res.columnMoved, this._onColumnMoved);
		host.listen(tr.res.columnRemoved, this._onColumnRemoved);
	}
	host.listen("preSectionRender", this._onPreSectionRender);
};
/** @override */
tr.grid.ColumnStackingPlugin.prototype.unload = function (host) {
	var at = this._hosts.indexOf(host);
	if(at < 0) { return; }

	this._hosts.splice(at, 1);

	host.unlisten("preSectionRender", this._onPreSectionRender);
	host.unlisten(tr.res.columnAdded, this._onColumnAdded);
	host.unlisten(tr.res.columnMoved, this._onColumnMoved);
	host.unlisten(tr.res.columnRemoved, this._onColumnRemoved);
	
	if(this._hosts.length <= 0) {
		this._conflator.reset();
	}
};


/** @public
 * @param {Array.<number>=} colIndices Selected columns will be used, if it is not given
 * @param {string=} opt_stackId Must be unique
 * @return {boolean} Return true if all of the given columns is stacked together
 */
tr.grid.ColumnStackingPlugin.prototype.stackColumns = function(colIndices, opt_stackId) {
	if(!colIndices) {
		colIndices = this._getSelectedColumns();
	} else {
		colIndices.sort(function(a, b) { return a - b; });
	}
	
	var len = colIndices.length;
	if(len <= 1) {
		return false; // Only two or more columns can be stacked
	}
	if(!this.isColumnStackable(colIndices)) {
		return false;
	}
	var stackId;
	if(opt_stackId) {
		if(this._stacks[opt_stackId]) {
			return false; // Cannot store the same stack Id
		}
		stackId = opt_stackId;
	} else {
		stackId = "Stack" + tr.grid.ColumnStackingPlugin._runningId++;
		while(this._stacks[stackId]) {
			stackId = "Stack" + tr.grid.ColumnStackingPlugin._runningId++;
		}
	}
	
	// Collecting data
	var host = this._hosts[0];
	var activeIndex = colIndices[0];
	var stack = {};
	stack.stackId = stackId;
	stack.colIds = new Array(len);
	for(var i = 0; i < len; ++i) {
		var colIndex = colIndices[i];
		this._colData[colIndex] = stack; // Column in the same stack stores the same data
		stack.colIds[i] = host._getColumnId(colIndex);
		if(colIndex !== activeIndex) {
			this._setColumnVisibility(colIndex, false);
		}
	}
	// Make sure that all columns stay packed together
	this._moveStackedColumns(stack.colIds);
	
	stack.activeColumn = stack.colIds[0]; // Left most column is the active column
	this._stacks[stackId] = stack;

	var csp = this._getPlugin("ColumnSelectionPlugin");
	if(csp && csp["isEnabled"]()) {
		csp["selectSingleColumn"](activeIndex);
	}
	var cfp = this._getPlugin("ColumnFilterPlugin");
	if(cfp) {
		cfp["refresh"]();
	}
	this._updateUI(); // asyncronuos
	return true;
};
/** @public
 * @param {Array.<number>=} colIndices Selected columns will be used, if it is not given
 * @return {boolean} 
 */
tr.grid.ColumnStackingPlugin.prototype.unstackColumns = function(colIndices) {
	if(!colIndices) {
		colIndices = this._getSelectedColumns();
	}
	
	var i, len = colIndices.length;
	var stacks = {};
	
	for(i = 0; i < len; ++i) {
		var colIndex = colIndices[i];
		var colData = this._colData[colIndex];
		if(!colData) {
			continue; // Invalid index
		}
		stacks[colData.stackId] = 1; // Exclude duplicate stacks
	}
	
	var dirty = false;
	var selFrom, selLen;
	
	for(var stackId in stacks) {
		var stack = this._stacks[stackId];
		if(!stack) {
			continue; // Invalid stackId
		}
		dirty = true;
		
		var colIds = stack.colIds;
		len = colIds.length;
		selFrom = this._getColumnIndex(stack.activeColumn);
		selLen = len - 1;
		
		for(i = 0; i < len; ++i) {
			var colId = colIds[i];
			var colIndex = this._getColumnIndex(colId);
			this._colData[colIndex] = null;
			this._setColumnVisibility(colIndex, true);
		}
		
		delete this._stacks[stackId]; // Remove all reference to the stack
	}
	if(dirty) {
		var csp = this._getPlugin("ColumnSelectionPlugin");
		if(csp && csp["isEnabled"]()) {
			csp["selectRange"](selFrom, selLen);
		}
		var cfp = this._getPlugin("ColumnFilterPlugin");
		if(cfp) {
			cfp["refresh"]();
		}
		this._updateUI(); // asyncronuos
	}
	
	return dirty;
};
/** Intended for determining if the selected columns can be stacked from the context menu
 * @public
 * @param {Array.<number>=} colIndices Selected columns will be used, if it is not given
 * @return {boolean} Return true if all of the given columns can be stacked together
 */
tr.grid.ColumnStackingPlugin.prototype.isColumnStackable = function(colIndices) {
	if(!colIndices) {
		colIndices = this._getSelectedColumns();
	}
	var cgp = this._getPlugin("ColumnGroupingPlugin");

	var len = colIndices.length;
	for(var i = 0; i < len; ++i) {
		var colIndex = colIndices[i];
		var colData = this._colData[colIndex];
		if(colData) {
			if(colData.colIds) {
				return false; // The given column is already stacked 
			}
		}
		// TODO: no isColumnInGroup API in ColumnGrouping external extension
		if(cgp && cgp.isColumnInGroup(colIndex)) {
			return false; // The column cannot be in a stack
		}
	}
	return true;
};
/** Intended for determining if the selected columns can be unstacked from the context menu
 * @public
 * @param {Array.<number>=} colIndices
 * @return {boolean} Return true if one of the columns is being stacked
 */
tr.grid.ColumnStackingPlugin.prototype.isColumnStacked = function(colIndices) {
	if(!colIndices) {
		colIndices = this._getSelectedColumns();
	}
	var len = colIndices.length;
	for(var i = 0; i < len; ++i) {
		var colData = this._colData[colIndices[i]];
		if(colData && colData.colIds) {
			return true;
		}
	}
	return false;
};
/** This is used for removing some column state while the column is inactive
 * @public
 * @param {number} colIndex
 * @return {boolean} Return true if the columns is stacked and inactive
 */
tr.grid.ColumnStackingPlugin.prototype.isInactiveStackedColumn = function(colIndex) {
	var host = this._hosts[0];
	if(host) {
		var colData = this._colData[colIndex];
		if(colData && colData.colIds) {
			return host._getColumnId(colIndex) !== colData.activeColumn;
		}
	}
	return false;
};
/** @public
* @param {number|Event} colRef
* @param {number} swappingIndex
* @return {boolean}
 */
tr.grid.ColumnStackingPlugin.prototype.swapColumn = function(colRef, swappingIndex) {
	var colIndex = -1;
	if(typeof colRef == "number") {
		colIndex = colRef;
	} else {
		var grid = this.getRelativeGrid(colRef);
		if(grid) {
			var pos = grid.getRelativePosition(colRef);
			colIndex = pos[tr.res.colIndex];
		}
	}
	
	var colData = this._colData[colIndex];
	if(!colData) { 
		return false; // Invalid column index
	}
	var colIds = colData.colIds;
	if(!colIds) {
		return false; // Invalid column type
	}
	var newActiveColumn = colIds[swappingIndex];
	if(!newActiveColumn) {
		return false; // Invalid selected index
	}
	
	if(newActiveColumn === colData.activeColumn) {
		return false; // The given index is already active stacked column
	}
	
	colData.colIds.splice(swappingIndex, 1);
	colData.colIds.unshift(newActiveColumn); // Move data to front
	var prevActiveColumn = colData.activeColumn;
	this._moveColumn(this._getColumnIndex(newActiveColumn), this._getColumnIndex(prevActiveColumn)); // Move UI to front
	
	this._setColumnVisibility(prevActiveColumn, false); // Hide current active column
	colData.activeColumn = newActiveColumn; // Change active column
	var activeIndex = this._setColumnVisibility(newActiveColumn, true);
	
	var csp = this._getPlugin("ColumnSelectionPlugin");
	if(csp && csp["isEnabled"]()) {
		csp["selectSingleColumn"](activeIndex);
	}
	var cfp = this._getPlugin("ColumnFilterPlugin");
	if(cfp) {
		cfp["refresh"]();
	}
	
	return true;
};

/** @public
 */
tr.grid.ColumnStackingPlugin.prototype.removeAllStacks = function() {
	var dirty = false;
	for(var stackId in this._stacks) {
		var stack = this._stacks[stackId];
		dirty = true;
		
		var colIds = stack.colIds;
		var len = colIds.length;
		for(var i = 0; i < len; ++i) {
			var colId = colIds[i];
			var colIndex = this._getColumnIndex(colId);
			this._colData[colIndex] = null;
			this._setColumnVisibility(colIndex, true);
		}
	}
	if(dirty) {
		this._colData.length = 0;
		this._stacks = {};
		this._updateUI(); // asyncronous
	}
	return dirty;
};
/** @public
 * @param {number} colIndex
 * @return {string}
 */
tr.grid.ColumnStackingPlugin.prototype.getStackId = function(colIndex) {
	var colData = this._colData[colIndex];
	if(colData) {
		return colData.stackId;
	}
	return "";
};


/** @private */
tr.grid.ColumnStackingPlugin.prototype._onPreSectionRender = function (e) {
	if (e[tr.res.sectionType] !== tr.res.title) {
		return;
	}
	this._updateUI(); // asyncronuos
};

/** @override */
tr.grid.ColumnStackingPlugin.prototype._onColumnMoved = function (e) {
	tr.grid.ColumnStackingPlugin.base(this, "_onColumnMoved", e); // Move markers and colData
	
	var colId = e["colId"];
	var colIndex = this._getColumnIndex(colId);
	var colData = this._colData[colIndex];
	if(colData) {
		if(colData.activeColumn === colId) { // Active stacked column has been moved
			setTimeout(this._moveStackedColumns.bind(this, colData.colIds), 10);
		}
	}
};
/** @override */
tr.grid.ColumnStackingPlugin.prototype._onColumnRemoved = function (e) {
	var colData = this._colData[e[tr.res.colIndex]]; // Cache data before the removal
	
	tr.grid.ColumnStackingPlugin.base(this, "_onColumnRemoved", e); // Move markers and colData
	
	var colId = e["colId"];
	if(colData) {
		if(colData.activeColumn === colId) { // Active stacked column has been removed
			setTimeout(this._removeStackedColumns.bind(this, colData.colIds), 10);
		}
	}
};

/** @private
 * @return {!Array.<number>} Sorted column indices
 */
tr.grid.ColumnStackingPlugin.prototype._getSelectedColumns = function() {
	var csp = this._getPlugin("ColumnSelectionPlugin");
	return (csp && csp["isEnabled"]()) ? csp["getSelectedColumns"]() : [];
};
/** @private
 * @param {string} colId
 * @return {number}
 */
tr.grid.ColumnStackingPlugin.prototype._getColumnIndex = function(colId) {
	var host = this._hosts[0];
	for(var c = host.getColumnCount(); --c >= 0;) {
		if(host._getColumnId(c) === colId) {
			return c;
		}
	}
	return -1;
};
/** @private
 * @param {number|string} colRef
 * @param {boolean} shown
 * @return {number} Affected column index
 */
tr.grid.ColumnStackingPlugin.prototype._setColumnVisibility = function(colRef, shown) {
	var colIndex = (typeof colRef == "number") ? colRef : this._getColumnIndex(colRef);
	if(colIndex >= 0) {
		for(var g = this._hosts.length; --g >= 0;) {
			this._hosts[g].setColumnVisibility(colIndex, shown, 2);
		}
	}
	return colIndex;
};
/** @private
* @param {Event} e
 */
tr.grid.ColumnStackingPlugin.prototype._onStackButtonClicked = function(e) {
	var elem = e.currentTarget;
	if(e.button) { // right click or middle click
		return;
	}
	
	var grid = this.getRelativeGrid(e);
	var pos = grid.getRelativePosition(e);
	var colIndex = pos[tr.res.colIndex];
	var colData = this._colData[colIndex];
	if(!colData) { return; }
	
	e.stopPropagation();
	
	var colIds = colData.colIds;
	var len = colIds.length;
	var colIndices = new Array(len);
	for(var i = len; --i >= 0;) {
		colIndices[i] = this._getColumnIndex(colIds[i]);
	}
	pos["activeIndex"] = colIds.indexOf(colData.activeColumn);
	pos["columnIndices"] = colIndices;
	pos["event"] = e;
	
	elem.focus();
	this._dispatch("clicked", pos);
};
/** @private
 */
tr.grid.ColumnStackingPlugin.prototype._updateUI = function() {
	if(this._conflator.conflate()) {
		return;
	}
	
	var colCount = this._getColumnCount();
	var gridCount = this._hosts.length;
	for(var c = 0; c < colCount; ++c) {
		var colData = this._colData[c];
		var colIds = (colData) ? colData.colIds : null;
		for(var g = 0; g < gridCount; ++g) {
			var host = this._hosts[g];
			var tSect = host.getSection("title");
			var cell = tSect.getCell(c, 0);
			if(!cell) { continue; }
			
			var stackIcon = cell._stackIcon;
			if(colIds) {
				if(!stackIcon) {
					cell.addClass("stacking");
                
                    stackIcon = tr.ElementFrameWork.createIconButton("down");
                    stackIcon.classList.add("stack-button");
                    stackIcon.classList.add("ef-icon");
					stackIcon.addEventListener("click", this._onStackButtonClicked, false);
					stackIcon.addEventListener("dblclick", tr._preventDefault, false);
					stackIcon.addEventListener("mousedown", tr._preventDefault, false);
					stackIcon.addEventListener("keydown", tr._preventDefault, false);
					
					cell.insertFloatingIcon(stackIcon, 20);
					cell._stackIcon = stackIcon;
					
					this._dispatch("iconCreated", {"icon": stackIcon, "colIndex": c, "grid": host, "cell": cell});
				}
			} else if(stackIcon) {
				cell.removeFloatingIcon(cell._stackIcon);
				cell._stackIcon = null;
				cell.removeClass("stacking");
			}
		}
		if(colData) {
			this._setColumnVisibility(c, !this.isInactiveStackedColumn(c)); // a little slow
		}
	}
};


/** @private
* @param {Array.<string>} colIds
 */
tr.grid.ColumnStackingPlugin.prototype._moveStackedColumns = function (colIds) {
	var len = colIds.length;
	for(var i = 1; i < len; ++i) {
		var fromIndex = this._getColumnIndex(colIds[i]);
		var destIndex = this._getColumnIndex(colIds[0]); // Active Column may be shifted due to the move
		this._moveColumn(fromIndex, destIndex + i);
	}
};
/** @private 
* @param {Array.<string>} colIds
*/
tr.grid.ColumnStackingPlugin.prototype._removeStackedColumns = function (colIds) {
	var len = colIds.length;
	for(var i = 1; i < len; ++i) {
		this._removeColumn(colIds[i]);
	}
};
/** @private
* @param {number} colIndex
* @param {number} destIndex
 */
tr.grid.ColumnStackingPlugin.prototype._moveColumn = function(colIndex, destIndex) {
	if(colIndex >= 0 && destIndex >= 0 && colIndex !== destIndex) {
		for(var i = this._hosts.length; --i >= 0;) {
			this._hosts[i].moveColumn(colIndex, destIndex);
		}
	}
};
/** @private
* @param {string} colId
 */
tr.grid.ColumnStackingPlugin.prototype._removeColumn = function(colId) {
	var colIndex = this._getColumnIndex(colId);
	if(colIndex >= 0) {
		for(var i = this._hosts.length; --i >= 0;) {
			this._hosts[i].removeColumnAt(colIndex);
		}
	}
};
