goog.provide("tr.grid.FindBarPlugin");

goog.require("tr.res");
goog.require("tr.grid.IPlugin");
goog.require("tr.grid.Plugin");

goog.require("tr.grid.FindBar");

/** Event
* @event tr.grid.FindBarPlugin#clicked
* @property {number} sectionIndex
* @property {boolean} expanded
*/

/** @constructor
* @extends {tr.grid.Plugin}
* @implements {tr.grid.IPlugin}
*/
tr.grid.FindBarPlugin = function () {
	this._onDocKeyDown = this._onDocKeyDown.bind(this);
	this._onNewFind = this._onNewFind.bind(this);
	this._onNextFind = this._onNextFind.bind(this);
	this._internalSearchLogic = this._internalSearchLogic.bind(this);
	this._focusInputBox = this._focusInputBox.bind(this);
	
	this._hosts = [];
	this._findBar = new grid.FindBar();
	this._findBar.listen("newFind", this._onNewFind);
	this._findBar.listen("nextFind", this._onNextFind);
	this._cumulativeResults = [];
	// this._findBar("closed", this._onFindBoxClosed);
	
	this._addEvent("popup");
};
goog.inherits(tr.grid.FindBarPlugin, tr.grid.Plugin);

/** @type {boolean}
* @private
*/
tr.grid.FindBarPlugin.prototype._enabled; // Intentional left undefined
/** @type {!tr.grid.FindBar}
* @private
*/
tr.grid.FindBarPlugin.prototype._findBar;
/** @type {function(string, Object):boolean|null}
* @private
*/
tr.grid.FindBarPlugin.prototype._userSearchLogic = null;
/** @type {string}
* @private
*/
tr.grid.FindBarPlugin.prototype._searchedText = "";
/** @type {boolean}
* @private
*/
tr.grid.FindBarPlugin.prototype._caseSensitivity = true;
/** @type {Array.<number>}
* @private
*/
tr.grid.FindBarPlugin.prototype._cumulativeResults = null;


/** @override */
tr.grid.FindBarPlugin.prototype.getName = function () {
	return "FindBarPlugin"; // Read Only
};
/** @override */
tr.grid.FindBarPlugin.prototype.initialize = function (host) {
	if(this._hosts.indexOf(host) >= 0) { return; }
	this._hosts.push(host);
	
	if(this._enabled === undefined) {
		this.enable();
	}
};
/** @override */
tr.grid.FindBarPlugin.prototype.unload = function (host) {
	var at = this._hosts.indexOf(host);
	if(at < 0) { return; }
	
	this._hosts.splice(at, 1);
	if(this._hosts.length <= 0) {
		this.disable();
	}
};


/** @public 
* @param {boolean=} opt_enabled
*/
tr.grid.FindBarPlugin.prototype.enable = function (opt_enabled) {
	this.disable(opt_enabled == false);
};
/** @public 
* @param {boolean=} opt_disabled
*/
tr.grid.FindBarPlugin.prototype.disable = function (opt_disabled) {
	this._enabled = opt_disabled === false;
	
	if(this._enabled) {
		window.addEventListener("keydown", this._onDocKeyDown, false);
	} else {
		window.removeEventListener("keydown", this._onDocKeyDown, false);
		this._findBar.close();
	}
};

/** @public 
* @param {function(string, Object):boolean} func
*/
tr.grid.FindBarPlugin.prototype.setSearchLogic = function (func) {
	this._userSearchLogic = func;
};

/** @public 
* @param {string=} opt_initialInput
*/
tr.grid.FindBarPlugin.prototype.showInDocument = function (opt_initialInput) {
	this._dispatch("popup", {"findBar": this._findBar});
	this._findBar.showInDocument(opt_initialInput);
};
/** @public 
* @param {boolean=} opt_disabled
*/
tr.grid.FindBarPlugin.prototype.disableCaseSensitivity = function (opt_disabled) {
	this._caseSensitivity = opt_disabled === false;
};


/** @private 
* @param {Object} e
*/
tr.grid.FindBarPlugin.prototype._onDocKeyDown = function (e) {
	if(!this._enabled) { return; }

	if(e.altKey) {
		return; // Nothing needs to be done with Alternate key
	}
	
	var activation = false;
	if(e.keyCode === 114) { // F3 key
		if(e.ctrlKey) {
			return;
		}
		activation = true; 
	} else if(e.keyCode === 70) { // F key
		if(!e.ctrlKey) {
			return;
		}
		activation = true; // Ctrl + F
	} else if (e.keyCode === 71) { // G key
		if(!e.ctrlKey) {
			return;
		}
		activation = !tr.isIE; // IE does not support Ctrl + G
	} else if(e.keyCode === 27) { // Esc
		if(this._findBar.getParent()) {
			this._findBar.close();
			return;
		}
	}
	
	if(activation) {
		e.preventDefault(); // Prevent default browser's find bar
		
		var inDoc = this._findBar.getParent();
		if(inDoc) {
			this._findBar.getInputBox().focus();
			if (e.shiftKey) {
				this._findBar.findPrev(); // Dispatch the event
			} else {
				this._findBar.findNext(); // Dispatch the event
			}
		} else {
			this.showInDocument(); // Dispatch the event
		}
	}
};

/** @private 
* @param {Object} e
*/
tr.grid.FindBarPlugin.prototype._onNewFind = function (e) {
	//var prevSearch = this._searchedText;
	this._searchedText = this._findBar.getInputText();
	if(!this._isSearchable()) { 
		this._findBar.setResults([]);
		return; 
	}
	
	if(!this._caseSensitivity) { // Convert every input to lowercase for easier searching
		this._searchedText = this._searchedText.toLowerCase();
	}
	
	var results = this._findBar.getResults();
	results.length = 0;
	this._cumulativeResults.length = 0;
	
	var len = this._hosts.length;
	for(var i = 0; i < len; ++i) {
		var host = this._hosts[i];
		var dv = host.getDataSource();
		dv.searchAll(this._internalSearchLogic, results);
		this._cumulativeResults[i] = results.length;
	}
	
	this._findBar.setResults(results);
	if(e["type"] !== "popup") {
		this._selectResult(0);
	}
};
/** @private 
* @param {Object} e
*/
tr.grid.FindBarPlugin.prototype._onNextFind = function (e) {
	if(!this._isSearchable()) { return; }
	
	this._selectResult(e["target"]);
};

/** @private
* @param {Object} rowData
* @return {boolean}
*/
tr.grid.FindBarPlugin.prototype._internalSearchLogic = function (rowData) {
	return this._userSearchLogic(this._searchedText, rowData);
};

/** @private
* @return {boolean}
*/
tr.grid.FindBarPlugin.prototype._isSearchable = function () {
	if(!this._enabled) {
		return false; // Plugin must be enabled
	}
	if(!this._searchedText) {
		return false; // No text to be searched
	}
	if(!this._userSearchLogic) {
		return false; // User must specify the search logic
	}
	var parent = this._findBar.getParent();
	if(!parent) {
		return false; // FindBar must be in the document
	}
	return true;
};
/** @private
* @param {number} resultIndex
*/
tr.grid.FindBarPlugin.prototype._selectResult = function (resultIndex) {
	var results = this._findBar.getResults();
	var rowId = results[resultIndex];
	var targetGrid = this._searchGrid(resultIndex);
	if(!targetGrid) { return; }
	
	var dv = targetGrid.getDataSource();
	var rowIndex = dv.getRowIndex(rowId);
	if(rowIndex < 0) { return; }

	var rsp = this._getPlugin("RowSelectionPlugin");
	if(rsp) {
		rsp["selectSingleRow"](rowIndex, "content", targetGrid);
		setTimeout(this._focusInputBox, 500);
	}
};
/** @private
* @param {number} resultIndex
* @return {tr.Grid}
*/
tr.grid.FindBarPlugin.prototype._searchGrid = function (resultIndex) {
	var len = this._hosts.length;
	for(var i = 0; i < len; ++i) {
		if(resultIndex < this._cumulativeResults[i]) {
			return this._hosts[i];
		}
	}
	return this._hosts[0];
};
/** @private
*/
tr.grid.FindBarPlugin.prototype._focusInputBox = function () {
	if(this._findBar.getParent()) {
		this._findBar.getInputBox().focus();
	}
};
