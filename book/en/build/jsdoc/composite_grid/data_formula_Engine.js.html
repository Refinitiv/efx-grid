/** @constructor
* @extends {tr.EventDispatcher}
*/
tr.fn.Engine = function() {
	this._context = {};
	tr.fn.Engine._addContext(this._context, tr.fn.Engine.getContext()); // Clone context for supporting more than one Engine in App
	this._context["_DATA"] = this._getData.bind(this);  // _DATA is an internal context used for getting cached data
	this._setData = this._defaultFormulaDataSetter.bind(this); // Default setter for adding the result into rowData
	this._formulas = [];
	this._formulaMap = {};
	this._referenceMap = {};
	this._addEvent("dataChanged");
	this._addEvent("dataRequired");
	this._addEvent("formulaChanged");
};
goog.inherits(tr.fn.Engine, tr.EventDispatcher);

	var context = tr.fn.Engine._context;
	if(!context) {
		context = {};
		tr.fn.Engine._context = context;

		// tr.fn.Engine._addContext(context, tr.fn.AdFin);
		tr.fn.Engine._addContext(context, tr.fn.DateTime);
		// tr.fn.Engine._addContext(context, tr.fn.Excel);

		tr.fn.Engine._addContext(context, tr.fn.Logic);

		tr.fn.Engine._addContext(context, tr.fn.Stat);
		tr.fn.Engine._addContext(context, tr.fn.Text);
	}
	return context;
};

/** @private
* @param {tr.fn.Formula} formula
* @param {*} result
* @param {Object.<string, *>} rowData
* @param {Object.<string, *>} opt_changes
*/
tr.fn.Engine.prototype._defaultFormulaDataSetter = function(formula, result, rowData, opt_changes) {
	if(result !== "...") {			// Data update is pending - do not erase previous value
		var alias = formula.getAlias();
		rowData[alias] = result;
		if (opt_changes) {
			opt_changes[alias] = result;
		}
	}
};

/** @public
* @param {tr.fn.Formula|string} userInput
* @return {!tr.fn.Formula}
*/
tr.fn.Engine.prototype.addFormula = function(userInput) {
	var f;
	if(userInput instanceof tr.fn.Formula){
		f = userInput;
	} else {
		this._rowData = {}; // Formula will be tested in the initialization process. So we need to mock the cached data.
		f = new fn.Formula(/** @type{string}*/(userInput), this._context);
		this._rowData = null;
	}

	f.listen("dataCacheChanged", this._dispatch.bind(this, "dataChanged"));
	f.listen("dataRequired", this._dispatch.bind(this, "dataRequired"));
	f.listen("formulaChanged", this._dispatch.bind(this, "formulaChanged"));
	this._formulas.push(f); // The one in the back will be executed last
	return f;
};

/** @public
* @param {tr.fn.Formula|string|number=} ref
* @return {tr.fn.Formula}
*/
tr.fn.Engine.prototype.removeFormula = function(ref) {
	if(ref instanceof tr.fn.Formula) { // Convert tr.fn.Formula to index
		ref = this._formulas.indexOf(/** @type{tr.fn.Formula} */(ref));
	}

	var f;
	if(typeof ref === "number") {
		f = this._formulas[ref];
		if(f) {
			this._formulas.splice(ref, 1);
			if(f.getReference()) {
				delete this._formulaMap[f.getReference()];
				delete this._referenceMap[f.getReference()];
			}
			f.dispose();
		}
	} else if (typeof ref === "string") {
		f = this._formulaMap[ref];
		if(f) {
			var at = this._formulas.indexOf(f);
			if(at >= 0) {
				this._formulas.splice(at, 1);
				delete this._formulaMap[ref];
				delete this._referenceMap[ref];
			}
			f.dispose();
		}
	}

	return f || null;
};
/** @public
*/
tr.fn.Engine.prototype.removeAllFormulas = function() {
	var formulas = this._formulas;
	for(var i = formulas.length; --i >= 0;) {
		delete this._referenceMap[formulas[i].getReference()];
	}

	this._formulas.length = 0;
	this._formulaMap = {};
};

/** @private
 * @param {string} ref
 * @return {string}
 */
tr.fn.Engine.prototype._generateUniqueReference = function(ref) {
	var uniRef = ref;
	var index = 0;
	while(this._referenceMap[uniRef] !== undefined) {
		uniRef = ref + "_" + ++index;
	}
	return uniRef;
};
/** @public
 * @param {string|tr.fn.Formula} src
 * @param {string} ref
 * @return {string}
 */
tr.fn.Engine.prototype.addReference = function(src, ref) {
	if (!src || !ref) {
		return "";
	}
	ref = this._generateUniqueReference(ref);
	var colName = "";
	if (src instanceof tr.fn.Formula) {
		var res = this._setFormulaReference(src, ref);
		if (!res) {
			return "";
		}
		colName = src.getAlias();
	} else {
		colName = /** @type{string} */(src);
	}
	this._referenceMap[ref] = colName;
	return ref;
};
/** @public
 * @param {string} oldRef
 * @param {string} newRef
 * @return {boolean}
 */
tr.fn.Engine.prototype.renameReference = function(oldRef, newRef) {
	if (!newRef) {
		return this.removeReference(oldRef);
	}
	if (!oldRef || this._referenceMap[oldRef] === undefined || this._referenceMap[newRef] !== undefined || oldRef === newRef) {
		return false;
	}
	var f = this._formulaMap[oldRef];
	if (f && !this.isValidReference(f, newRef)) {
		return false;
	}

	// Change the reference being used in all formulas
	for (var i = this._formulas.length; --i >= 0;) {
		var formula = this._formulas[i];
		if (formula.changeField(oldRef, newRef)) {
			var ref = formula.getReference();
			if (ref) {
				this._referenceMap[ref] = formula.getAlias();
			}
		}
	}

	if (f) {
		this._setFormulaReference(f, newRef);
	}

	this._referenceMap[newRef] = this._referenceMap[oldRef];
	delete this._referenceMap[oldRef];
	return true;
};
/** @public
 * @param {string} ref
 * @return {boolean}
 */
tr.fn.Engine.prototype.removeReference = function(ref) {
	if (!ref || this._referenceMap[ref] === undefined) {
		return false;
	}
	var f = this._formulaMap[ref];
	if (f !== undefined) {
		f._setReference("");
		delete this._formulaMap[ref];
	}
	delete this._referenceMap[ref];
	return true;
};
/** @public
 * @param {string} refName
 * @return {string}
 */
tr.fn.Engine.prototype.resolveReference = function(refName) {
	return this._referenceMap[refName] || refName;
};

/** Get value from cached row data
 *  @private
 * @param {string} refName
 * @param {Object.<string, *>} rowData WARNING: This param will not be used!
 * @return {*}
 */
tr.fn.Engine.prototype._getData = function(refName, rowData) {
	//  TODO: Assign default value based on refName in case of non existing data
	if (!this._rowData) {
		return;
	}
	var value = this._rowData[this.resolveReference(refName)];

	if (value) {
		var num = Number(value);

		if (num === num) { // Convert text to number for formula calculation
			value = num;
		}
	}

	return value;
};
/** @private
* @param {tr.fn.Formula} formula
* @param {string} ref
* @return {boolean}
*/
tr.fn.Engine.prototype._setFormulaReference = function(formula, ref) {
	if(this._formulaMap[ref]) {
		console.warn("Duplicate formula reference detected.");
		return false;
	}
	var prevRef = formula.getReference();
	if(prevRef === ref) {
		return false;
	}

	if(ref) { // Empty string is no need to be in formula map
		this._formulaMap[ref] = formula;
	}
	formula._setReference(ref);
	var dependencies = this._sortDependencies(this._formulas, this._formulaMap);
	if (!dependencies) { // Circular dependencies detected
		formula._setReference(prevRef);
		delete this._formulaMap[ref];
		return false;
	}

	// Conclude
	this._formulas = dependencies;

	delete this._formulaMap[prevRef]; // Delete old reference of formula
	return true;
};
/** @public
* @param {Object.<string, *>} rowData
* @param {Object.<string, *>=} opt_changes
*/
tr.fn.Engine.prototype.calculate = function(rowData, opt_changes) {
	this._rowData = rowData; // Cache for resolving reference

	var len =  this._formulas.length;
	for(var i = 0; i < len; ++i) { // This must be done in normal order
		var f = this._formulas[i];
		// if opt_changes is undefined then recalculate all function
		// if opt_changes is exist then check formula need to recalculate
		if (opt_changes) {
			// Don't recalculate formula if fields are not changed and not update from realtime formula
			var needRecalculate = (opt_changes[f.getAlias()] !== undefined); // Update from realtime formula
			if (!needRecalculate) {
				var fields = f.getFieldNames();
				for (var j = fields.length; --j >= 0;) {
					if (opt_changes[this.resolveReference(fields[j])] !== undefined) {
						needRecalculate = true;
						break;
					}
				}
				if (!needRecalculate) {
					continue;
				}
			}
		}

		var result = f.calculate(rowData);
		this._setData(f, result, rowData, opt_changes);
	}

	this._rowData = null;
};

/**
 * @public
 * @param {tr.fn.Formula} formula
 * @param {string} ref
 * @return {boolean}
 */
tr.fn.Engine.prototype.isValidReference = function(formula, ref) {
	if(!ref) {
		return true;
	}
	if(this._referenceMap[ref]) {
		return false; // duplicate reference detected.
	}

	var prevRef = formula.getReference();
	var parentChildrenMap = this._createParentChildrenMap(this._formulas);

	// Replace the old reference with new reference
	if (prevRef) {
		delete parentChildrenMap[prevRef];
		for (var parent in parentChildrenMap) {
			var children = parentChildrenMap[parent];
			var index = children.indexOf(prevRef);
			if (index !== -1) {
				children = children.slice(0); // Clone. The orignal children is an array in formula object. We should not modify it.
				children.splice(index, 1);
				index = children.indexOf(ref);
				if (index === -1) {
					children.push(ref);
				}
				parentChildrenMap[parent] = children;
			}
		}
	}
	parentChildrenMap[ref] = formula.getFieldNames();

	return (!!this._sortParentChildrenMap(parentChildrenMap));
};

/** @private
 * @param {Array.<tr.fn.Formula>} formulas
 * @param {Object.<string, tr.fn.Formula>} formulaMap	Contains all initial formulas in a map form
 * @return {Array.<tr.fn.Formula>}
 */
tr.fn.Engine.prototype._sortDependencies = function(formulas, formulaMap) {
	var formulaCount = formulas.length;
	if (formulaCount === 1) {
		return formulas;
	}

	var parentChildrenMap = this._createParentChildrenMap(formulas);
	var sortedParents = this._sortParentChildrenMap(parentChildrenMap);
	if (!sortedParents) {
		return null;
	}

	var noDepFormulas = formulas.filter(function(f) {
		return (!f.getReference());
	});

	var results = [];
	if (formulaMap) {
		for(var i = sortedParents.length; --i >= 0;) {
			var f = formulaMap[sortedParents[i]];
			if(f) {
				results.push(f); // The one in the back will be executed last
			}
		}
		if(noDepFormulas.length > 0) {
			results = results.concat(noDepFormulas);
		}
	}
	return results;
};

/**
 * @private
 * @param {Array.<tr.fn.Formula>} formulas
 * @return {Object.<string, Array.<string>>}
 */
tr.fn.Engine.prototype._createParentChildrenMap = function(formulas) {
	var i, parentChildrenMap = {};
	for (i = formulas.length; --i >= 0;) {
		var f = formulas[i];
		var ref = f.getReference();
		if (ref) {
			parentChildrenMap[ref] = f.getFieldNames();
		}
	}
	return parentChildrenMap;
};

/** @description https://en.wikipedia.org/wiki/Topological_sorting
 * @private
 * @param {Object.<string, Array.<string>>} ParentChildrenMap
 * @return {Array.<string>}
 */
tr.fn.Engine.prototype._sortParentChildrenMap = function(ParentChildrenMap) {
	var edges = {};
	var i, child, children, childCount;
	var parent;
	for (parent in ParentChildrenMap) {
		children = ParentChildrenMap[parent];
		childCount = children.length;
		for(var j = 0; j < childCount; ++j) {
			child = children[j];
			edges[child] = (edges[child] + 1) || 1;
		}
	}

	// Collect nodes with no edge (root nodes)
	var rootNodes = [];
	for(parent in ParentChildrenMap) {
		if(!edges[parent]) {
			rootNodes.push(parent);
		}
	}

	// Navigate from root nodes down to all the leaf nodes
	var sortedParents = [];
	while(rootNodes.length > 0) {
		parent = rootNodes.shift();
		sortedParents.push(parent);

		children = ParentChildrenMap[parent];
		childCount = (children) ? children.length : 0;
		for(i = 0; i < childCount; ++i) {
			child = children[i];
			if(!(--edges[child] > 0)) {
				rootNodes.push(child); // This child has no more edge; and hence, it becomes root node
			}
		}
	}

	// Check for circular reference
	for(child in edges) {
		if(edges[child]) {
			return null;
		}
	}

	return sortedParents;
};
