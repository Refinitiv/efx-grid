goog.provide("tr.fn.Text");

/**
 * @namespace Text
 * @description
 *    {lang=en} Text manipulation functions
 *    {lang=ja} 文字列に関する関数
 *    {lang=zh} 文本控制函数
 *    {lang=ko} 텍스트 조작 함수
 *    {lang=ru} Text manipulation functions
 */
tr.fn.Text = {};

tr.fn._toString = function(text) {
	if(!text && text !== 0) {
		return "";
	}

	return text + "";
};
tr.fn._isNull = function(text) {
	return !text && text !== 0;
};
/**
 * @function EXACT
 * @memberof Text
 * @description
 *   {lang=en} The EXACT function returns 1(TRUE) if text string 1 is same as text string 2. Otherwise returns 0(FALSE). Use operand = for a shorthand.
 *   {lang=ja} EXACT 関数は、2つの文字列を比較して同じ場合は 1(TRUE) を返します。異なる場合は 0(FALSE) を返します。オペランド [ = ] を使用します。
 *   {lang=zh} 如果文本字符串 1 与文本字符串 2 相同，则 EXACT 函数返回 1(TRUE)。否则返回 0(FALSE)。快速记录请使用操作符 =
 *   {lang=ko} EXACT 함수는 텍스트 문자열 1이 텍스트 문자열 2와 같으면 1(TRUE)을 반환합니다. 다르면 0(FALSE)을 반환합니다. 약칭으로 피연산자 =를 사용합니다.
 *   {lang=ru} The EXACT function returns 1(TRUE) if text string 1 is same as text string 2. Otherwise returns 0(FALSE). Use operand = for a shorthand.
 * @param  {string} text1
 *   {lang=en} The text string 1.
 *   {lang=ja} 文字列 1
 *   {lang=zh} 文本字符串 1。
 *   {lang=ko} 텍스트 문자열 1.
 *   {lang=ru} The text string 1.
 * @param  {string} text2
 *   {lang=en} The text string 2.
 *   {lang=ja} 文字列 2
 *   {lang=zh} 文本字符串 2。
 *   {lang=ko} 텍스트 문자열 2.
 *   {lang=ru} The text string 2.
 * @return {number}
 * @example
 *   =EXACT(CF_CURR,"USD")
 *   =CF_CURR="USD"
 */
tr.fn.Text["EXACT"] = function(text1, text2) {
	text1 = text1 + "";
	text2 = text2 + "";
	return (text1 === text2) ? 1 : 0;
};
/**
 * @function FIND
 * @memberof Text
 * @description
 *   {lang=en} The FIND function returns the position of a specified character or sub-string within a supplied text string.
 *             REMARK: The function is case-sensitive. The first character is counted as 0.
 *   {lang=ja} FIND 関数は、指定された文字列内で指定された文字の位置を返します。
 *             注記: この関数は、大文字と小文字を区別します。最初の文字は [0] としてカウントされます。
 *   {lang=zh} FIND 函数返回指定的字符或子字符串在所提供的文本字符串中的位置。
 *   {lang=ko} FIND 함수는 제공된 텍스트 문자열 내에서 지정된 문자 또는 하위 문자열의 위치를 반환합니다.
 *   {lang=ru} The FIND function returns the position of a specified character or sub-string within a supplied text string.
 *             注意: 该函数区分大小写。首个字符计为 0。
 * @param  {string} findText
 *   {lang=en} The character or sub-string that you wish to find.
 *   {lang=ja} 検索文字列
 *   {lang=zh} 您希望查找的字符或子字符串。
 *   {lang=ko} 찾으려는 문자나 하위 문자열.
 *   {lang=ru} The character or sub-string that you wish to find.
 * @param  {string} withinText
 *   {lang=en} The text string that is to be searched.
 *   {lang=ja} 検索する文字列
 *   {lang=zh} 要搜索的文本字符串。
 *   {lang=ko} 검색할 텍스트 문자열.
 *   {lang=ru} The text string that is to be searched.
 * @param  {number=} opt_startNum
 *   {lang=en} An optional argument that specifies the position in the within_text string, from which the search should begin.
 *   {lang=ja} 検索文字列内で検索を開始する位置を指定するオプショナル引数。
 *   {lang=zh} 可选参数，指定搜索开始于 within_text 字符串中的位置。
 *   {lang=ko} 검색을 시작해야 하는 within_text 문자열의 위치를 지정하는 선택적 인수.
 *   {lang=ru} An optional argument that specifies the position in the within_text string, from which the search should begin.
 * @return {number}
 * @example
 *    =FIND("r","surfthewave",5)
 */
tr.fn.Text["FIND"] = function(findText, withinText, opt_startNum) {
	findText = tr.fn._toString(findText);
	if(!findText) { // Invalid findText
		return 0;
	}
	withinText = tr.fn._toString(withinText);
	if(!withinText) { // Invalid base text
		return 0;
	}
	opt_startNum = +opt_startNum;
	if (!opt_startNum || opt_startNum < 0) {
		opt_startNum = 0;
	}
	var pos = withinText.indexOf(findText, opt_startNum);
	return pos;
};

 *   {lang=ko} LEFT 함수는 제공된 텍스트 문자열의 시작 부분에서 지정된 수의 문자를 반환합니다.
 *   {lang=ru} The LEFT function returns a specified number of characters from the start of a supplied text string.
 *   {lang=de} Die Funktion LINKS gibt eine bestimmte Anzahl von Zeichen ab dem Beginn einer bereitgestellten Textzeichenfolge zurück.
 *             REMARK: If opt_numChars, (the number of returned characters), is more than the length of text string, return the whole string.
 * @param {string} text
 *   {lang=en} The original text string.
 *   {lang=ja} 元の文字列
 *   {lang=zh} 初始文本字符串。
 *   {lang=hant} 初始文本字符串。
 *   {lang=ko} 원본 텍스트 문자열.
 *   {lang=ru} The original text string.
 *   {lang=de} Die ursprüngliche Textzeichenfolge.
 * @param {number} opt_numChars
 *   {lang=en} An optional argument that specifies the number of characters to be returned from the start of the supplied text.
 *   {lang=ja} 指定された文字列の先頭から指定された文字数を返すオプショナル引数です。
 *   {lang=zh} 可选参数，指定从所提供的文本起始处要返回的字符数。
 *   {lang=hant} 可選參數，指定從所提供的文本起始處要返回的字符數。
 *   {lang=ko} 제공된 텍스트의 시작 부분에서 반환할 문자 수를 지정하는 선택적 인수.

 * @return {string}
 * @example
 *    =LEFT("Surf the Wave",4)
 */
tr.fn.Text["LEFT"] = function(text, opt_numChars) {
	opt_numChars = +opt_numChars;
	if(!opt_numChars) {
		opt_numChars = 0;
	}
	if(opt_numChars < 0) {
		opt_numChars = 1;
	}
	return tr.fn._toString(text).substr(0, opt_numChars);
};

/**
 * @function REPLACE
 * @memberof Text
 * @description
 *   {lang=en} The REPLACE function replaces all or part of a text string with another string.
 *   {lang=ja} REPLACE 関数は、指定された数の文字を新しい文字で置き換えた文字列を返します。
 *   {lang=zh} REPLACE 函数将全部或部分文本字符串替代为另一字符串。

tr.fn.Text["REPLACE"] = function(oldText, numChars, startNum, newText) {
	newText = newText + "";
	oldText = tr.fn._toString(oldText);
	var arrOldText = oldText.split("");
	arrOldText.splice(startNum, numChars, newText);
	return arrOldText.join("");
};

 */
tr.fn.Text["RIGHT"] = function(text, opt_numChars) {
	text = tr.fn._toString(text);
	opt_numChars = +opt_numChars;
	if(!opt_numChars) {
		opt_numChars = 0;
	}
	if(opt_numChars < 0) {
		opt_numChars = 1;
	}
	var startNum = text.length - opt_numChars;
	return text.substr(startNum, opt_numChars);
};

 *   {lang=ko} CLEAN 함수는 문자열에서 비 ASCII 문자(32-127 범위에 없는 문자)를 제거합니다.
 *   {lang=ru} The CLEAN function removes non-ASCII characters, (characters not in the range 32-127 inclusive), from the string.
 *   {lang=de} Die Funktion SÄUBERN entfernt Nicht-ASCII-Zeichen (Zeichen, die nicht im Bereich 32-127 einschließlich liegen) aus der Zeichenfolge.
 * @param {string} text
 *   {lang=en} Text
 *   {lang=ja} 文字列
 *   {lang=zh} 文本
 *   {lang=hant} 文本
 *   {lang=ko} 텍스트
 *   {lang=ru} Text
 *   {lang=de} Text
 * @return {string}
 * @example
 *    =CLEAN("Kobérña")
 */
tr.fn.Text["CLEAN"] = function(text) {
	var arrText = tr.fn._toString(text).split("");
	for(var i = arrText.length; --i >= 0;) {
		var ascii = arrText[i].charCodeAt(0);
		if(ascii < 32 || ascii > 127) {
			arrText[i] = "";
		}
	}
	return arrText.join("");
};

/**
 * @function PROPER
 * @memberof Text
 * @description
 *   {lang=en} The PROPER function converts the first letter of all words in sentence to uppercase, and converts all remaining letters to lowercase.
 *   {lang=ja} ROPER 関数は、文中の各単語の先頭文字を大文字に変換し、残りの文字をすべて小文字に変換します。
 *   {lang=zh} PROPER 函数将句子中全部单词的首字母转换成大写，其他字母全部保持小写。
 *   {lang=hant} PROPER 函數將句子中全部單詞的首字母轉換成大寫，其他字母全部保持小寫。
 *   {lang=ko} PROPER 함수는 문장의 모든 단어의 첫 글자를 대문자로 변환하고 나머지 문자를 모두 소문자로 변환합니다.
 *   {lang=ru} The PROPER function converts the first letter of all words in sentence to uppercase, and converts all remaining letters to lowercase.
 *   {lang=de} Die Funktion RICHTIG konvertiert den ersten Buchstaben aller Wörter im Satz in Großbuchstaben und alle verbleibenden Buchstaben in Kleinbuchstaben.
 * @param  {string} text
 *   {lang=en} Text
 *   {lang=ja} 文字列
 *   {lang=zh} 文本
 *   {lang=hant} 文本
 *   {lang=ko} 텍스트
 *   {lang=ru} Text
 *   {lang=de} Text
 * @return {string}
 * @example
 *    =PROPER("A STRANGE BLACK CAT ON YOUR PORCH BRINGS PROSPERITY.")
 */
tr.fn.Text["PROPER"] = function(text) {
	text = tr.fn._toString(text);
	var arrTokens = text.split(/[\W_]/g);
	var arrSyms = text.match(/[\W_]/g);
	var arrLen = arrTokens.length;
	var result = "";
	for(var i = 0; i < arrLen; i++) {
		var token = arrTokens[i];
		var len = token.length;
		if(len === 1) {
			result += token.toUpperCase();
		} else if(len !== 0) {
			result += token.charAt(0).toUpperCase() + token.substr(1).toLowerCase();
		}

		var sym = arrSyms[i];
		if(sym) {
			result += sym;
		}
	}
	return result;
};

 *   {lang=en} Text
 *   {lang=ja} 文字列
 *   {lang=zh} 文本
 *   {lang=hant} 文本
 *   {lang=ko} 텍스트
 *   {lang=ru} Text
 *   {lang=de} Text
 * @param  {number} numRep
 *   {lang=en} Number of times
 *   {lang=ja} 回数
 *   {lang=zh} 次数

 *   {lang=ko} 횟수
 *   {lang=ru} Number of times
 *   {lang=de} Mehrfach
 * @return {string}
 * @example
 *    =REPEAT("Ho! ",3)

tr.fn.Text["REPEAT"] = function(text, numRep) {
	numRep = +numRep;
	if(!numRep || numRep < 0) {
		return "";
	}

	text = tr.fn._toString(text);
	if(text === "") {
		return "";
	}

	var newText = text;
	for(var i = 1; i < numRep; i++) {
		newText += text;
	}
	return newText;
};
