goog.provide("tr.TSISubscription");

goog.require("tr.HttpRequest");
goog.require("tr.RequestQueue");
goog.require("tr.EventManager");

/**
 * @constructor
 * @extends {tr.EventManager}
 */
tr.TSISubscription = function() {
	this._refreshData = this._refreshData.bind(this);

	// Old TSI service
	this._onTSIUpdate = this._onTSIUpdate.bind(this);
	this._requestTimeSeriesData = this._requestTimeSeriesData.bind(this);

	// New metadata service
	this._manageMetadataQ = this._manageMetadataQ.bind(this);
	this._extractMetadata = this._extractMetadata.bind(this);
	this._onRequestMetadata = this._onRequestMetadata.bind(this);

	// New TSI service
	this._manageTSIDataQ = this._manageTSIDataQ.bind(this);
	this._extractTSIData = this._extractTSIData.bind(this);
	this._onRequestTSIData = this._onRequestTSIData.bind(this);

	// Temporary Map
	this._metadataMap = {};
	this._metadataPendingMap = {};
	this._tsiDataMap = {};
	this._tsiDataPendingMap = {};

	// Data Map
	this._tsiPeriodMap = {};
	this._tsiMetadata = {};

	this._metadataQ = new RequestQueue(this._onRequestMetadata, 80, 10000); // 80 requests in 10 seconds
	this._metadataQ.setConcurrentLimit(2); // To prevent blocking request due to long running request time.

	this._tsiDataQ = new RequestQueue(this._onRequestTSIData, 80, 10000); // 80 requests in 10 seconds
	this._tsiDataQ.setConcurrentLimit(4); // To prevent blocking request due to long running request time.

	this._oldTsiDataQ = new RequestQueue(this._requestTimeSeriesData, 80, 10000); // 80 requests in 10 seconds
	this._oldTsiDataQ.setConcurrentLimit(4); // To prevent blocking request due to long running request time.
};
goog.inherits(tr.TSISubscription, tr.EventManager);

//#region public function of TSI service
/** @public
 * @description Set the interval time for retrieving data
 * @param {number} time Millisecond
 */
tr.TSISubscription.prototype.setIntervalTime = function(time) {
	time = +time;
	if(!time) { // NaN or 0
		return;
	}
	this._intervalTime = time;
	this._stop();
	this._start();
};
/** @public
 * @description Set the settings of TSI
 * @param {string=} sharedServiceUrl Url of TSI shared-service
 * @param {string=} taServiceUrl Url of TSI TAservice
 */
tr.TSISubscription.prototype.setTSISettings = function(sharedServiceUrl, taServiceUrl) {
	if(sharedServiceUrl !== undefined) {
		this._tsiSharedServiceUrl = sharedServiceUrl;
	}
	if(taServiceUrl !== undefined) {
		this._tsiTaServiceUrl = taServiceUrl;
	}
};
/** @public
 * @description Subscribe symbol and field to the specified period. This method will start the interval for retrieving data automatically.
 * @param {string} symbol RIC
 * @param {string} field TSI field name
 * @param {string} period Specified Date(YYYY-MM-DD) or period (NNNNNP where N is a digit between 0 and 9, and P can be: D for days, W for weeks, M for months, Q for quarters, Y for years)
 * @return {*} Return the value if there is existing. Return undefined when requesting the invalid field.
 */
tr.TSISubscription.prototype.addSymbolAndFieldToPeriod = function (symbol, field, period) {
	var periodRIC = this._getPeriodRICData(period, symbol);
	if(periodRIC["dataCache"][field] !== undefined) { // Already have data
		return periodRIC["dataCache"][field];
	}

	var periodMap = this._getPeriodMap(period);
	var isNewSymbol = periodMap["symbol"][symbol] === undefined;
	if(isNewSymbol) {
		periodMap["symbol"][symbol] = true;
	}

	var isNewField = periodMap["field"][field] === undefined;
	if(isNewField) {
		periodMap["field"][field] = true;
	}

	if(isNewSymbol || isNewField) { // Request if it is new symbol or new field
		this._oldTsiDataQ.add({ period: period });
	}

	this._start(); // Start interval

	/*
	 *  1: No valid data from service
	 *  2: Waiting required data
	 */
	return tr.TSISubscription.PENDING_DATA;
};
//#endregion public function of TSI service


//#region public function of event
/** @override
 * @description Add the function handler into the specific event. This method will start interval automatically.
 * @param {string} type Event name
 * @param {Function} handler Event handler
 */
tr.TSISubscription.prototype.addEventListener = function(type, handler) {
	tr.TSISubscription.base(this, "addEventListener", type, handler);
	this._start();
};
/** @override
 * @description Remove the function handler from the specific event. If there is not any event handlers, the service will stop automatically.
 * @param {string} type Event name
 * @param {Function} handler Event handler
 */
tr.TSISubscription.prototype.removeEventListener = function(type, handler) {
	tr.TSISubscription.base(this, "removeEventListener", type, handler);
	if(!this._hasListener()) {
		this.removeAllEventListeners();
	}
};
/** @override
 * @description Remove all function handlers of all events and stop the interval of retrieving data of service.
 */
tr.TSISubscription.prototype.removeAllEventListeners = function() {
	tr.TSISubscription.base(this, "removeAllEventListeners");
	this._stop();
	if(this._metadataTimer > 0) {
		window.clearTimeout(this._metadataTimer);
	}
	this._metadataTimer = 0;
	if(this._tsiDataTimer > 0) {
		window.clearTimeout(this._tsiDataTimer);
	}
	this._tsiDataTimer = 0;

	this._tsiPeriodMap = {};
	this._metadataMap = {};
	this._tsiDataMap = {};
	this._metadataQ.reset();
	this._tsiDataQ.reset();
};
//#endregion public function of event


//#region private function of TSI service
/** @private
 * @description Start the interval of retrieving data
 */
tr.TSISubscription.prototype._start = function() {
	if(this._state === "start") {
		return;
	}
	this._intervalId = window.setInterval(this._refreshData, this._intervalTime);
	this._state = "start";
};
/** @private
 * @description Stop the interval of retrieving data
 */
tr.TSISubscription.prototype._stop = function() {
	if(this._intervalId > 0) {
		window.clearInterval(this._intervalId);
	}

	this._intervalId = 0;
	this._state = "stop";
};
/** @private
 * @description Refresh all the TSI data that had been subscribed to the service.
 */
tr.TSISubscription.prototype._refreshData = function() {
	for(var period in this._tsiPeriodMap) {
		// Request to old TSI service
		this._oldTsiDataQ.add({ period: period }); // If it has error code, it will fallback to new service automatically.
	}
};
/** @private
 * @description Get object that related to period. It has symbol and field properties to store the requested field and RIC to old service (shared-service).
 * @param {string} period Specified Date(YYYY-MM-DD) or period (NNNNNP where N is a digit between 0 and 9, and P can be: D for days, W for weeks, M for months, Q for quarters, Y for years)
 * @return {Object} Object that related to period
 */
tr.TSISubscription.prototype._getPeriodMap = function(period) {
	if(!this._tsiPeriodMap[period]) {
		this._tsiPeriodMap[period] = {
			"symbol": {}, // These 2 properties is used for requesting the old service (shared-service)
			"field": {}
		};
	}
	return this._tsiPeriodMap[period];
};
/** @private
 * @description Get data object that related to period and RIC. It has view property to store the requested view and dataCache property to store data.
 * @param {string} period Specified Date(YYYY-MM-DD) or period (NNNNNP where N is a digit between 0 and 9, and P can be: D for days, W for weeks, M for months, Q for quarters, Y for years)
 * @param {string} symbol RIC
 * @return {Object} Object that related to period and RIC
 */
tr.TSISubscription.prototype._getPeriodRICData = function(period, symbol) {
	var tsiPeriodMap = this._getPeriodMap(period);
	if(!tsiPeriodMap[symbol]) {
		tsiPeriodMap[symbol] = {
			"view": {}, // This property will used for new TSI service
			"dataCache": {}
		};
	}

	return tsiPeriodMap[symbol];
};

//#region private function of calling TS metadata service
/** @private
 * @description Check symbol has its metadata or not
 * @param {string} symbol RIC
 * @return {boolean}
 */
tr.TSISubscription.prototype._hasMetadata = function(symbol) {
	return !symbol || (typeof this._tsiMetadata[symbol] === "object");
};
/** @private
 * @description Method for requesting the metadata of symbol. Prevent requesting with duplicated RIC.
 * @param {string} symbol RIC
 */
tr.TSISubscription.prototype._requestMetadata = function(symbol) {
	if(this._metadataPendingMap[symbol] || this._hasMetadata(symbol) || this._metadataMap[symbol]) { // Has metadata || pending request
		 return;
	}

	this._metadataMap[symbol] = true;
	if(this._metadataTimer <= 0) {
		this._metadataTimer = window.setTimeout(this._manageMetadataQ, 200);
	}
};
/** @private
 * @description Manage the request queue of TSI metadata. Split 300 RICs per request.
 */
tr.TSISubscription.prototype._manageMetadataQ = function() {
	this._metadataTimer = 0; // Reset

	var symbolMap = this._metadataMap;
	this._metadataMap = {};

	var symbols = [];
	for(var key in symbolMap) {
		if(this._metadataPendingMap[key]) {
			continue;
		}
		symbols.push(key);
		this._metadataPendingMap[key] = true;

		if(symbols.length >= tr.TSISubscription.MAX_CONCURRENT_OBJECT) {
			this._metadataQ.add({ rics: symbols }); // Add full capcity to queue
			symbols = [];
		}
	}
	if(symbols.length > 0) {
		this._metadataQ.add({ rics: symbols }); // Add remaining to queue
	}
};
/** @private
 * @description Publish TSI metadata request to server.
 * @param {Object} obj Specified RICs
 * @param {Array.<Object>} q All RICs that are waiting in a queue
 * @return {tr.HttpRequest|Promise}
 */
tr.TSISubscription.prototype._onRequestMetadata = function(obj, q) {
	var reqRics = obj.rics;
	var len = reqRics.length;
	if(len <= 0) {
		return Promise.resolve("");
	}

	// Merge request
	var qi = q.length;
	var remaining = tr.TSISubscription.MAX_CONCURRENT_OBJECT - len;
	while(remaining > 0  && --qi >= 0) {
		var qObj = q[qi];
		reqRics = reqRics.concat( qObj.rics.splice(0, remaining) );

		len = reqRics.length; // Update length
		remaining = tr.TSISubscription.MAX_CONCURRENT_OBJECT - len;
		if(qObj.rics.length === 0) { // No data anymore
			q.pop();
		}
	}

	var payload = { "rics": reqRics };
	var pendingMap = this._metadataPendingMap;
	var http = new HttpRequest(this._tsiTaServiceUrl + "/GetMetadata", payload);
	return http.send().then(this._extractMetadata)
			.catch(function(err) {
				console.error(err);
				for(var i = len; --i >= 0;) {
					delete pendingMap[reqRics[i]]; // Remove key from pending list due to error
				}
			});
};
/** @private
 * @description Extract the metadata response from service. This will trig to call service for getting data automatically.
 * @param {Object} serviceResponse
 */
tr.TSISubscription.prototype._extractMetadata = function(serviceResponse) {
	var instruments = serviceResponse["instruments"];
	if(instruments) { // Successful request
		var aryLen = Object.keys(/** @type {!Object} */(tr.TSISubscription.INTERVAL_INDEX)).length;
		var ary = new Array(aryLen); // Reused array
		for(var i = instruments.length; --i >= 0;) {
			var instrument = instruments[i];

			var data = null;
			var viewList = instrument["viewList"] || [];
			for(var j = viewList.length; --j >= 0;) {
				var view = viewList[j];
				var intervals = view["intervals"];
				var k, interval, len;
				for(k = 0, len = intervals.length; k < len; k++) {
					interval = intervals[k];
					var idx = tr.TSISubscription.INTERVAL_INDEX[ interval["name"] ];
					if(idx !== undefined) {
						if(ary[idx] == null || tr.TSISubscription.INTERVAL_MAIN[ interval["name"] ]) { // Main interval has more priority than other interval in same category
							ary[idx] = interval;
						}
					}
				}

				var viewData = null;
				var columns;
				for(k = 0; k < aryLen; k++) {
					interval = ary[k];
					ary[k] = null; // Clear array for reused
					if(interval != null && viewData === null) {
						columns = 0; // Reset

						for(var l = interval["fields"].length; --l >= 0;) { // Column mask for view + interval
							columns |= tr.TSISubscription.COLUMN_MASK_ENUM[ interval["fields"][l] ]; // undefined will be same as 0
						}

						if(columns !== 0) { // Valid interval to retrieve data
							viewData = {
								interval: tr.TSISubscription.INTERVAL_ENUM[ interval["name"] ],
								columnMask: columns
							};
						}
					}
				}

				if(viewData !== null) {
					if(data === null) {
						data = {};
					}
					data[view["name"]] = viewData;
				}
			}

			this._tsiMetadata[instrument["ric"]] = data;
			delete this._metadataPendingMap[instrument["ric"]]; // Remove RIC from pending list
		}

		// Trig to call service for getting data
		if(this._tsiDataTimer <= 0) {
			this._tsiDataTimer = window.setTimeout(this._manageTSIDataQ, 0);
		}
	}
};
//#endregion private function of calling TS metadata service


//#region private function of calling new TS data service
/** @private
 * @description Get the string of key by symbol, view and period.
 * @param {string} symbol RIC
 * @param {string} view TSI view name
 * @param {string} period Specified Date(YYYY-MM-DD) or period (NNNNNP where N is a digit between 0 and 9, and P can be: D for days, W for weeks, M for months, Q for quarters, Y for years)
 * @return {string} String of key
 */
tr.TSISubscription.prototype._getKey = function(symbol, view, period) {
	return symbol + "â™«" + view + "â™«" + period;
};
/** @private
 * @description Build the request information for metadata
 * @param {!Object} reqPayload Object that contains symbol, view and period
 * @param {!Object.<string, number>} dateMap Date calculation map. Reduce duplicated date to calculate.
 * @return {Object.<string, Array>}
 */
tr.TSISubscription.prototype._buildRequestInfo = function(reqPayload, dateMap) {
	var ric = reqPayload.ric;
	var view = reqPayload.view;
	var viewData = this._tsiMetadata[ric] && this._tsiMetadata[ric][view];
	if(!viewData) {
		return null;
	}

	var period = dateMap[reqPayload.period];
	if(period == null) {
		period = tr.TSISubscription._dateCalculation(reqPayload.period).getTime();
		dateMap[reqPayload.period] = period;
	}

	return {
		"RequestInfos": [{
			"__type": "TSRequest:http://www.thomsonreuters.com/ns/2012/01/01/webservices/TAService_1/",
			"RIC": ric,
			"Interval": viewData.interval,
			"Field": view,
			"ColumnMask": viewData.columnMask,
			"DataPoints": "1",
			"EndDate": "/Date(" + period +")/"
		}]
	};
};
/** @private
 * @description Method for requesting the data of symbol and view with specified period. Prevent requesting with duplicated symbol, view and period.
 * @param {string} symbol RIC
 * @param {string} view TSI view name
 * @param {string} period Specified Date(YYYY-MM-DD) or period (NNNNNP where N is a digit between 0 and 9, and P can be: D for days, W for weeks, M for months, Q for quarters, Y for years)
 */
tr.TSISubscription.prototype._requestTSIData = function(symbol, view, period) {
	if(!symbol || !view || !period) {
		return;
	}

	var key = this._getKey(symbol, view, period);
	if(this._tsiDataPendingMap[key] || this._tsiDataMap[key]) { // Already in list
		return;
	}

	this._tsiDataMap[key] = {
		ric: symbol,
		view: view,
		period: period
	};
	if(this._tsiDataTimer <= 0) {
	   this._tsiDataTimer = window.setTimeout(this._manageTSIDataQ, 200);
	}
};
/** @private
 * @description Manage the request queue of TSI data. Split 300 objects per request. If the requested object dose not have metadata, it will request metadata first.
 */
tr.TSISubscription.prototype._manageTSIDataQ = function() {
	this._tsiDataTimer = 0; // Reset

	var listMap = this._tsiDataMap;
	this._tsiDataMap = {};

	var obj;
	var listArr = [];
	for(var key in listMap) {
		obj = listMap[key];
		if(this._hasMetadata(obj.ric)) {
			if(this._tsiDataPendingMap[key]) {
				continue;
			}
			listArr.push( obj );
			this._tsiDataPendingMap[key] = true;
		} else { // No metadata
			this._requestMetadata(obj.ric);
			this._tsiDataMap[key] = obj; // Add to pending list again
			continue;
		}

		if(listArr.length >= tr.TSISubscription.MAX_CONCURRENT_OBJECT) {
			this._tsiDataQ.add({ list: listArr }); // Add full capcity to queue
			listArr = [];
		}
	}
	if(listArr.length > 0) {
		this._tsiDataQ.add({ list: listArr }); // Add remaining to queue
	}
};
/** @private
 * @description Publish TSI data request to server.
 * @param {Object} obj Specified list of objects
 * @param {Array.<Object>} q All list of objects that are waiting in a queue
 * @return {tr.HttpRequest|Promise}
 */
tr.TSISubscription.prototype._onRequestTSIData = function(obj, q) {
	var reqPayloads = obj.list;
	var len = reqPayloads.length;
	if(len <= 0) {
		return Promise.resolve("");
	}

	// Merge request
	var qi = q.length;
	var remaining = tr.TSISubscription.MAX_CONCURRENT_OBJECT - len;
	while(remaining > 0  && --qi >= 0) {
		var qObj = q[qi];
		reqPayloads = reqPayloads.concat( qObj.list.splice(0, remaining) );

		len = reqPayloads.length; // Update length
		remaining = tr.TSISubscription.MAX_CONCURRENT_OBJECT - len;
		if(qObj.list.length === 0) { // No data anymore
			q.pop();
		}
	}

	var dateMap = {};
	var requestInfos = [];
	var reqInfo;
	for(var i = len; --i >= 0;) {
		var reqPayload = reqPayloads[i];
		var ric = reqPayload.ric;
		var view = reqPayload.view;
		var period = reqPayload.period;

		reqInfo = this._buildRequestInfo(reqPayload, dateMap);
		if(reqInfo) {
			requestInfos.push( reqInfo ); // BEWARE: This is the reverse loop
		} else {

			/*
			 * No metadata for request
			 * Complete the request so that it will not stay indefinitely as "pending"
			 * TRGRID-2335
			 */
			var key = this._getKey(ric, view, period);
			delete this._tsiDataPendingMap[key]; // Remove from pending map

			var periodRIC = this._getPeriodRICData(period, ric);
			var periodMap = this._getPeriodMap(period);
			var fields = periodMap["field"];
			var rowData = {};
			for(var field in fields) {
				if(!Object.prototype.hasOwnProperty.call(fields, field)) {
					continue;
				}
				if(periodRIC["dataCache"][field] === undefined) {
					periodRIC["dataCache"][field] = null;
				}
				rowData[period + "â™ " + field] = periodRIC["dataCache"][field];
			}
			this._dispatch("dataChanged", { "ric": ric, "values": rowData, "sender": this }); // Publish data
			reqPayloads.splice(i, 1);
		}
	}

	if(requestInfos.length === 0) {
		return Promise.resolve("");
	}

	var self = this;
	var payload = { "messages": requestInfos.reverse() }; // Reverse array back for normal order
	var http = new HttpRequest(this._tsiTaServiceUrl + "/CalculateAllData", payload);
	http.setResponseType("text"); // Prevent JSON parsing with invalid data
	return http.send()
		.then(function(serviceResponse) {
			return {
				payload: reqPayloads, // attached payload
				response: serviceResponse
			};
		})
		.then(this._extractTSIData)
		.catch(function(err) {
			console.error(err);
			for(var i = len; --i >= 0;) {
				var key = self._getKey(reqPayloads[i].ric, reqPayloads[i].view, reqPayloads[i].period);
				delete self._tsiDataPendingMap[key]; // Remove from pending map
			}
		});
};
/** @private
 * @description Extract the TSI data response from service. Cache the data and dispatch an event.
 * @param {Object} res Object that contains response from service in response property and requested payload in payload property
 * @return {Object}
 */
tr.TSISubscription.prototype._extractTSIData = function(res) {
	var response = res.response.replace(/\[NaN(,NaN)*\]/g, "[]"); // Handler for return the invalid JSON syntax due to no data
	var json;
	try {
		json = /** @type {Object} */(JSON.parse(response));
	} catch(ex) {
		console.error(ex);
		json = null;
	}

	if(json) {
		var payload = res.payload;
		var len = payload.length;
		var periodRIC, dataCache, view, symbol, period, rowData, field, periodMap;
		for(var i = 0; i < len; i++) {
			view = payload[i].view;
			symbol = payload[i].ric;
			period = payload[i].period;

			periodRIC = this._getPeriodRICData(period, symbol);
			periodRIC["view"][view] = true; // Mark as already requested
			dataCache = periodRIC["dataCache"];

			var data = json[i]["Columns"] || [];
			if(data.length === 0) {
				continue;
			}

			rowData = {};
			var val;
			for(var j = 0, dataLen = data.length; j < dataLen; j++) {
				val = data[j]["Values"][0];
				if(val == null || val !== val) {
					/*
					 * undefined, null, NaN
					 * Complete the request so that it will not stay indefinitely as "pending"
					 * TRGRID-2335
					*/
					periodMap = this._getPeriodMap(period);
					for(field in periodMap["field"]) {
						if(!periodMap["field"].hasOwnProperty(field)) continue;
						dataCache[field] = null;
						rowData[period + "â™ " + field] = dataCache[field];
					}
					continue;
				}

				switch(data[j]["Name"]) { // Cache the data in service
					case "Date":
						field = view + ".TIMESTAMP";
						dataCache[field] = new Date(+val.substring(6, val.length - 2)).toJSON().substr(0, 19);
						rowData[period + "â™ " + field] = dataCache[field];
						break;
					case "Open":
						field = view + ".OPEN";
						dataCache[field] = val;
						rowData[period + "â™ " + field] = dataCache[field];
						break;
					case "High":
						field = view + ".HIGH";
						dataCache[field] = val;
						rowData[period + "â™ " + field] = dataCache[field];
						break;
					case "Low":
						field = view + ".LOW";
						dataCache[field] = val;
						rowData[period + "â™ " + field] = dataCache[field];
						break;
					case "Close":
						field = view + ".CLOSE";
						dataCache[field] = val;
						rowData[period + "â™ " + field] = dataCache[field];

						field = view + ".VALUE";
						dataCache[field] = val; // WORKAROUND: Service return VALUE in factlist CLOSE
						rowData[period + "â™ " + field] = dataCache[field];
						break;
					case "Volume":
						field = view + ".VOLUME";
						dataCache[field] = val;
						rowData[period + "â™ " + field] = dataCache[field];
						break;
					case "Count":
						field = view + ".COUNT";
						dataCache[field] = val;
						rowData[period + "â™ " + field] = dataCache[field];
						break;
					default:
						field = (view + "." + data[j]["Name"]).toUpperCase();
						dataCache[field] = val; // Rarely case: new fields
						rowData[period + "â™ " + field] = dataCache[field];
				}
			}

			this._dispatch("dataChanged", { "ric": symbol, "values": rowData, "sender": this }); // Publish data
			delete this._tsiDataPendingMap[ this._getKey(symbol, view, period) ]; // Remove from pending map
		}
	}

	return json;
};
//#endregion private function of calling new TS data service

/** @private
 * @description Publish the TSI request to the TSI server.
 * @param {Object} obj Specified Date(YYYY-MM-DD) or period (NNNNNP where N is a digit between 0 and 9, and P can be: D for days, W for weeks, M for months, Q for quarters, Y for years)
 * @param {Array.<Object>} q All periods that are waiting in a queue
 * @return {tr.HttpRequest}
 */
tr.TSISubscription.prototype._requestTimeSeriesData = function(obj, q) {
	var period = obj.period;
	for(var i = q.length; --i >= 0;) {
		var qObj = /** @type{Object} */(q[i]);
		if(!qObj.rics && qObj.period === period) {
			q.splice(i, 1); // Remove any duplicate period that is being queued
		}
	}

	var periodMap = this._getPeriodMap(period);
	var fields = Object.keys(periodMap["field"]);
	var symbols = obj.rics ? obj.rics : Object.keys(periodMap["symbol"]);

	if(fields.length === 0 || symbols.length === 0) {
		return null;
	}

	var reqDate = tr.TSISubscription._dateCalculation(period).toJSON();
	if(reqDate === null) { // Invalid date/period parameter
		return null;
	} else {
		reqDate = reqDate.slice(0, 10);
	}

	if(symbols.length > 300) { // TSI cannot handle more than 300 rics in a single request
		obj.rics = symbols;
		symbols = obj.rics.splice(0, 300); // Split the rics
		q.unshift(obj); // Push the request back at the front of the queue
	}

	var payload = {
		"rics": symbols.join(" "),
		"factList": fields.join(","),
		"date": reqDate
	};

	var http = new HttpRequest(this._tsiSharedServiceUrl, payload);

	return http.send()
	.then(function(result) {
		return {
			"period": period, // Attach requested period
			"reqDate": reqDate, // Attach requested date for comparison with the returned data
			"result": result
		};
	})
	.then(this._onTSIUpdate)
	.catch(function(err) {
		console.error(err);
	});
};
/** @private
 * @description Function for processing the TSI response and publish the event to the listener.
 * @param {Object} json JSON object of TSI response
 */
tr.TSISubscription.prototype._onTSIUpdate = function(json) {
	var period = json["period"];
	var result = json["result"];
	var periodMap = this._getPeriodMap(period);
	var fields = periodMap["field"];

	var header = result[0];
	var ricIdx = -1;
	var returnStatusIdx = -1;
	var timestampIdx = -1;
	var fieldIdxMap = {};
	for(var i = header.length; --i >= 0;) {
		if(header[i] === "RIC") {
			ricIdx = i;
		} else if(header[i] === "ReturnStatus") {
			returnStatusIdx = i;
		} else if(header[i] === "TimeStamp") {
			timestampIdx = i;
		} else if(fields[header[i]]) { // Requested fields
			fieldIdxMap[header[i]] = i;
		}
	}
	fields = Object.keys(fieldIdxMap); // Collect the fields of result

	var rowLen = result.length;
	var symbol, value, rowData, dataCache;
	// var isCorrectDate;
	for(i = 1; i < rowLen; i++) {
		symbol = result[i][ricIdx];

		if(result[i][returnStatusIdx] !== "0") { // Have an error in response, then fallback to new TS data service
			// Request to new service
			for(var k = fields.length; --k >= 0;) {
				var view = fields[k].substring(0, fields[k].lastIndexOf("."));
				if(!this._hasMetadata(symbol) || (this._tsiMetadata[symbol] && this._tsiMetadata[symbol][view])) { // No metadata || Ready to make a request
					this._requestTSIData(symbol, view, period);
				}
			}
			continue;
		}

		// isCorrectDate = (reqDate === result[i][timestampIdx]);
		dataCache = this._getPeriodRICData(period, symbol)["dataCache"];
		rowData = {};

		for(var j = fields.length; --j >= 0;) {
			// value = (isCorrectDate) ? (result[i][fieldIdxMap[fields[j]]] || null) : null; // If result is same as the requested timestamp then return the value. Otherwise return null.
			value = result[i][fieldIdxMap[fields[j]]] || null; // If return empty string, it should be null
			if(value !== null) {
				value = +value; // Convert to number
			}
			dataCache[fields[j]] = value; // Cache the data in service
			rowData[period + "â™ " + fields[j]] = dataCache[fields[j]];
		}

		this._dispatch("dataChanged", { "ric": symbol, "values": rowData, "sender": this }); // Publish data
	}
};
//#endregion private function of calling old TS data service


//#region private function of date calculation
/** @private
 * @description Add/Subtract days to the given JS Date object.
 * @param {!Date} date Javascript Date object
 * @param {number} day Day to add or subtract. Plus value is add operation. Minus value is subtract operation.
 * @return {!Date}
 */
tr.TSISubscription._addDays = function(date, day) {
	date.setDate(date.getDate() + day);

	return date;
};
/** @private
 * @description Add/Subtract months to the given JS Date object.
 * @param {!Date} date Javascript Date object
 * @param {number} month Month to add or subtract. Plus value is add operation. Minus value is subtract operation.
 * @return {!Date}
 */
tr.TSISubscription._addMonths = function(date, month) {
	// Reference: http://stackoverflow.com/questions/7937233/how-do-i-calculate-the-date-in-javascript-three-months-prior-to-today
	var expectedMonth = ((date.getMonth() + month) % 12 + 12) % 12;
	date.setMonth(date.getMonth() + month);
	if (date.getMonth() !== expectedMonth) { // If it has an error of native date subtraction
		date.setDate(0); // Backward to the last day of previous month
	}
	return date;
};
/** @private
 * @description Add/Subtract months to the given JS Date object.
 * @param {!Date} date Javascript Date object
 * @param {number} num
 * @param {string=} dateUnit
 * @return {!Date}
 */
tr.TSISubscription._addDate = function(date, num, dateUnit) {
	if(dateUnit === "Y") {
		return tr.TSISubscription._addMonths(date, num * 12);
	} else if(dateUnit === "Q") {
		return tr.TSISubscription._addMonths(date, num * 3);
	} else if(dateUnit === "M") {
		return tr.TSISubscription._addMonths(date, num);
	} else if(dateUnit === "W") {
		return tr.TSISubscription._addDays(date, num * 7);
	}

	return tr.TSISubscription._addDays(date, num);
};
/** @private
 * @description Calculate the given period or transform the given date to string of date
 * @param {string} period Specified Date(YYYY-MM-DD) or period (NNNNNP where N is a digit between 0 and 9, and P can be: D for days, W for weeks, M for months, Q for quarters, Y for years)
 * @return {Date} Date object
 */
tr.TSISubscription._dateCalculation = function(period) {
	var goBackPeriods = period.match(/\d+(D|W|M|Q|Y)/g);
	if(goBackPeriods) { // Relative date pattern
		var now = new Date();
		for(var i = goBackPeriods.length; --i >= 0;){
			var p = goBackPeriods[i];
			var dateUnit = p.substr(p.length - 1);
			var num = -(p.substr(0, p.length - 1));
			now = tr.TSISubscription._addDate(now, num, dateUnit);
		}
		return now;
	}

	// Specified date
	var temp = new Date(period);
	if(temp.toJSON() === null) {
		console.warn("Invalid period parameter: " + period);
		return null;
	}
	var date = null;
	if(period.match(/^\d{1,4}(-\d{2}(-\d{2})?)?$/)) { // If ISO format (YYYY-MM-DD)
		date = temp;
	} else { // If non ISO format
		date = new Date(temp.getTime() - temp.getTimezoneOffset() * 60000); // Adjust timezone
	}
	return date; // Return the date in string format e.g. 2017-02-21
};
//#endregion private function of date calculation
