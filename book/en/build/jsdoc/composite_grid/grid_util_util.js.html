/** Move an item in the given array to the specified destination. If the destination equal or greater than array length, the array will be expanded
* @public
* @ignore
* @param {Array} ary
* @param {number} from INCLUSIVE
* @param {number} to INCLUSIVE
* @return {Array} ary
*/
tr.moveArrayItem = function (ary, from, to) {
	if (!ary) { return null; }

	var len = ary.length;
	if (from >= len && to >= len) {
		return /** @type{Array} */(ary); // No item needs to be moved
	}

	var item = ary.splice(from, 1)[0];
	if (to < len) {
		ary.splice(to, 0, item);
	} else {
		ary[to] = item;
	}
	return ary;
};
/** Insert an item in the given array at the specified position. If the position exceeds the array length, the array will be expanded
* @public
* @ignore
* @param {Array|*} ary
* @param {number} at INCLUSIVE
* @param {*} item
*/
tr.insertArrayItem = function (ary, at, item) {
	if (!ary) { return; }

	if (at < 0) {
		return; // Negative index cannot be added
	}
	if (at < ary.length) {
		ary.splice(at, 0, item); // any subsequence item is shifted to the right
	} else {
		ary[at] = item;
	}
};
/** Remove the specified item from the given array
* @public
* @ignore
* @param {Array|*} ary
* @param {number} at INCLUSIVE
* @return {*} removed item. If not existed, null is returned
*/
tr.removeArrayItem = function (ary, at) {
	if (!ary) { return null; }

	if (at >= 0 && at < ary.length) {
		return ary.splice(at, 1)[0];
	}
	return null;
};
/** @public
* @function
* @param {Array} ary
* @param {Array} items
* @return {Array}
*/
tr.pushArrayItems = function(ary, items) {
	var len = items ? items.length : 0;
	if(len > 0) {
		if(len < 1e5) { // Maximum number of arguments that the spread operator can take is exactly 125052
			if(len > 1) {
				Array.prototype.push.apply(ary, items);
			} else {
				ary.push(items[0]);
			}
		} else {
			var start = 0;
			var end = 0;
			while(start < len) {
				end += 1e5;
				Array.prototype.push.apply(ary, items.slice(start, end));
				start = end;
			}
		}
	}
	return ary;
};
/** Insert multiple empty items to an array
* @public
* @function
* @param {Array} ary
* @param {number} at Zero or positive number
* @param {number} chunkSize
* @example
* insertEmptyItems(["a", "b"], 1, 2); // ["a", , , "b"]
*/
tr.insertEmptyItems = function (ary, at, chunkSize) {
	if(chunkSize > 0) {
		if(chunkSize > 1) {
			var tailEnd = ary.slice(at);
			ary.length = at; // Remove the tail end part
			ary.length += chunkSize;
			tr.pushArrayItems(ary, tailEnd);
		} else {
			ary.splice(at, 0, null);
		}
	}
};
/** Move all items from the specify index to the front/top of the array
* @public
* @function
* @param {Array} ary
* @param {number} numItems Index of the first item to be moved to the top
* @return {Array}
* @example
* cycleArray(["a", "b", "c"], 2); // ["c", "a", "b"]
*/
tr.cycleArray = function (ary, numItems) {
	if(numItems > 0 && ary) {
		if(numItems < ary.length) {
			var front = ary.splice(0, numItems);
			return ary.concat(front);
		}
	}
	return ary || null;
};

/** Correct the index when associated item is moved by the specified indices
* @public
* @ignore
* @param {number|null} num
* @param {number|null} from INCLUSIVE removed index
* @param {number|null=} opt_to INCLUSIVE added index
* @return {number} Negative if the index is removed
*/
tr.moveIndex = function (num, from, opt_to) {
	if (!tr.isNumber(num)) { return -1; }
	if (num === from) {
		if (tr.isNumber(opt_to)) {
			return /** @type{number} */(opt_to);
		}
		return -1;
	}
	if (from !== null) {
		if (num > from) {
			--num;
		}
	}
	if (opt_to !== null) {
		if (num >= opt_to) {
			++num;
		}
	}
	return /** @type{number} */(num);
};

/** STL implementation of lowerBound() of the C++ vector class with no recursion
* @public
* @ignore
* @param {Array.<*>} ary The given array must already be sorted in ascending order
* @param {*} value the value to be searched
* @param {number=} left the begin index of the search range is INCLUSIVE
* @param {number=} right the end index of the search range is EXCLUSIVE
* @param {function(?, ?):boolean=} opt_lessFunc "Less than comparison functor"
* @return {number} Return the lowest index that gives equal or greater than the searched value
*/
tr.lowerBound = function (ary, value, left, right, opt_lessFunc) {
	if (left === undefined || left < 0) {
		left = 0;
	}
	if (right === undefined || right < 0 || right > ary.length) {
		right = ary.length;
	}
	if (left > right) {
		left = right;
	}
	if (opt_lessFunc === undefined) {
		opt_lessFunc = tr._defaultLessComparator;
	}

	var mid = 0;
	while (right > left) {
		mid = (left + right) >> 1;
		if (opt_lessFunc(ary[mid], value)) {
			left = mid + 1;
		} else {
			right = mid;
		}
	}
	return left; // May return ary.length
};

/** Upper bouund is the greatest index that has value that is always greater than the given value
* @public
* @ignore
* @param {Array.<*>} ary The given array must already be sorted in ascending order
* @param {*} value the value to be searched
* @param {number=} left the begin index of the search range is INCLUSIVE
* @param {number=} right the end index of the search range is EXCLUSIVE
* @param {function(?, ?):boolean=} opt_lessFunc "Less than comparison functor"
* @return {number} Return the highest index that gives equal or greater than the searched value
 */
tr.upperBound = function (ary, value, left, right, opt_lessFunc) {
	var at = tr.lowerBound(ary, value, left, right, opt_lessFunc);
	while (ary[at] === value) { ++at; }
	return at;
};

 * @param {number} lowVal
 * @param {number} lastVal
 * @param {number} highVal
 * @return {number}
 */
tr.calculatePercent = function (lowVal, lastVal, highVal) {
	if ((!lowVal && lowVal !== 0) ||
		(!lastVal && lastVal !== 0) ||
		(!highVal && highVal !== 0)) {
		return NaN;
	}
	if (highVal === lowVal) {
		return 50;
	}
	var percent = (lastVal - lowVal) / (highVal - lowVal);
	if (percent < 0) {
		return 0;
	}
	if (percent > 1) {
		return 100;
	}
	return percent * 100;
};

	if (!array1 || !array2) {
		return false;
	}

		return false;
	}

	for (var i = 0, l = array1.length; i < l; i++) {

/** @public
* @function
* @param {string|Function} expression
* @return {Function}
* @example
* var fn = tr.parseRowCondition("[CF_BID] >= 10 && [CF_BID] <= 100");
* window.console.log(fn(25));
*/
tr.parseRowCondition = function (expression) {
	if (!expression) {
		return null;
	}
	if (typeof expression !== "string") {
		return tr.isFunction(expression) ? /** @type {Function} */(expression) : null;
	}

	var regEx = new RegExp(/\[[^[]*\]/g); // Regex for matching the field token E.g. '[CF_BID]'
	var brackets = expression.match(regEx); // Retrieving field tokens (anything in between brackets)
	var map = {}; // For checking duplication
	var fields = [];
	var len = brackets ? brackets.length : 0;
	for (var i = len; --i >= 0;) {
		var field = brackets[i];
		if (!map[field]) {
			map[field] = "f[" + fields.length + "]"; // Create mapping of field token to array with index E.g. { "[CF_BID]": "f[0]" }
			fields.push(field.substring(1, field.length - 1)); // Strip '[' and ']' of field token to get field name
		}
	}

	// Replace the field tokens with rowData of array with index E.g. [CF_BID] ==> rowData[f[0]]
	expression = expression.replace(regEx, function (match) {
		return "rowData[" + map[match] + "]";
	});
	var finalExp = "(function(f, rowData) { return (" + expression + "); })";

	var fn = null;
	try {
		fn = eval(finalExp);
		fn = fn.bind(null, fields); // Bind fields for matching field name
	} catch (err) {
		window.console.log(err.message);
	}
	return (fn) ? /** @type{Function} */(fn) : null;
};
goog.exportSymbol("tr.parseRowCondition", tr.parseRowCondition);
