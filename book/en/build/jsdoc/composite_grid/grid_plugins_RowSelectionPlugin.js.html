* @event tr.grid.RowSelectionPlugin#selectionChanged

/** @constructor
* @extends {tr.grid.Plugin}
* @implements {tr.grid.IPlugin}
*/
tr.grid.RowSelectionPlugin = function () {
	this._onMouseDown = this._onMouseDown.bind(this);
	this._onClick = this._onClick.bind(this);
	this._onKeyDown = this._onKeyDown.bind(this);
	this._onSelectionChanged = this._onSelectionChanged.bind(this);

	this._addEvent(tr.res.selectionChanged);
	this._addEvent(tr.res.keydown);
	this.listen(tr.res.selectionChanged, this._onSelectionChanged);

	this._hosts = [];
};
goog.inherits(tr.grid.RowSelectionPlugin, tr.grid.Plugin);

//#region Implement IPlugin
/** @override */
tr.grid.RowSelectionPlugin.prototype.getName = function () {
	return "RowSelectionPlugin"; // Read Only
};
/** @override */
tr.grid.RowSelectionPlugin.prototype.initialize = function (host) {
	if(this._hosts.indexOf(host) >= 0) { return; }
	this._hosts.push(host);

	host.listen(tr.res.mousedown, this._onMouseDown);
	host.listen(tr.res.click, this._onClick);
	host.listen(tr.res.keydown, this._onKeyDown);
};
/** @override */
tr.grid.RowSelectionPlugin.prototype.unload = function (host) {
	var at = this._hosts.indexOf(host);
	if(at < 0) { return; }

	host.unlisten(tr.res.mousedown, this._onMouseDown);
	host.unlisten(tr.res.click, this._onClick);
	host.unlisten(tr.res.keydown, this._onKeyDown);
	this._hosts.splice(at, 1);
};

/** @public
 * @param {number} rowIndex
 * @param {tr.Grid.SectionReference=} opt_sectRef
 * @param {tr.Grid=} opt_ActiveGrid
 * @return {boolean}
 * @fires tr.grid.RowSelectionPlugin#selectionChanged
 */
tr.grid.RowSelectionPlugin.prototype.selectSingleRow = function(rowIndex, opt_sectRef, opt_ActiveGrid){
	if(!(rowIndex >= 0)) { return false; }

	var prevGrid = this._activeGrid;
	if(opt_ActiveGrid) {
		this._clearSelectedRows();
		this._activeGrid = opt_ActiveGrid; // HACK
	}
	var section = this._getSection(opt_sectRef); // TODO: Check if the given section is in the active grid
	if(this._selectSingleRow(section, rowIndex)) {
		this._dispatchSelectionChanged();

		this._scrollToRow(rowIndex);
		return true;
	} else if(opt_ActiveGrid) {
		this._activeGrid = prevGrid;
	}
	return false;
};

/** @public
 * @param {tr.Grid=} opt_ActiveGrid
 * @fires tr.grid.RowSelectionPlugin#selectionChanged
 */
tr.grid.RowSelectionPlugin.prototype.selectAllRows = function(opt_ActiveGrid){
	if(opt_ActiveGrid) {
		this._activeGrid = opt_ActiveGrid;
	}

	if(!this._activeGrid) { return; }

	var sections = this._activeGrid.getAllSections(tr.res.content);
	for(var i=sections.length;  --i >= 0;) {
		this._selectRowRange(0, sections[i].getRowCount(), sections[i]);
		if(i == 0) {
			this._anchorSection = sections[i];
		}
	}

	this._dispatchSelectionChanged();
};

/**
 * @public

 */
tr.grid.RowSelectionPlugin.prototype.selectRowRange = function(rowIndex, length, opt_sectRef, opt_scrollToRow) {
	this._selectRowRange(rowIndex, length, opt_sectRef);
	this._dispatchSelectionChanged();

	if (opt_scrollToRow) {
		this._scrollToRow(rowIndex);
	}
};

/** @private
 * @param {Event} e
 */
tr.grid.RowSelectionPlugin.prototype._onClick = function (e) {
	if(this._pendingClickIndex >= 0 && this._anchorSection) {
		this._selectSingleRow(this._anchorSection, this._pendingClickIndex);
		this._pendingClickIndex = -1;

		this._dispatchSelectionChanged(e);
	}
};
/** @private
 * @description Left click will cause single row selection
 * Shift + left click will cause range selection
 * Ctrl + left click will cause inclusion selection
 * Shift/Ctrl + right click will cause single row selection
 * Middle click has no effect
 * @suppress {checkTypes}
 * @param {Event} e
 */
tr.grid.RowSelectionPlugin.prototype._onMouseDown = function (e) {
	if (!this._clickAllowed) {
		return;
	}
	if(e.button === 2) { // Right click
		if(e.shiftKey || e.ctrlKey) {
			return;
		}
	}

	this._pendingClickIndex = -1;

	var host = this.getRelativeGrid(e);
	var	info;

	if (!host) {
		return;
	}

	info = host.getRelativePosition(e);

	if(!info["hit"] || info[tr.res.sectionType] !== tr.res.content) {return;}

	if(e.button === 2) { // Right click at the selected column should override clicking row to select;
		var csp = this._getPlugin("ColumnSelectionPlugin");
		if (csp && csp['isEnabled']()) {
			if(csp['isSelectedColumn'](info[tr.res.colIndex])) {
				return;
			}
		}
	}

	var singleSelection = (!this._activeGrid) || this._activeGrid !== host; // New grid is clicked
	var section = info[tr.res.section];
	var rowIndex = info[tr.res.rowIndex];

	//If max selection count is 1, it performs like selecting signle row
	singleSelection |= !this._multiSelect;

	if(!singleSelection) {
		if(e.shiftKey) { // Shift + Click: selects every rows start from anchor to clicked point. Anchor point doesn't change.
			this._selectFromAnchorToTarget(section, rowIndex);
		} else if ((e.ctrlKey && !tr.isMac) || (tr.isMac && !e.ctrlKey && e.metaKey)) { // Ctrl(windows)/Command(macOS) + Click: additionally selects a row without deselect all others.
			section.setSelectedRow(rowIndex, !section.isSelectedRow(rowIndex));
		} else {
			singleSelection = true;
		}
	}

	if(singleSelection) { // Click: selects a single row
		this._anchorSection = section;
		if(section.isSelectedRow(rowIndex)) { // Click on the selected row
			this._pendingClickIndex = rowIndex; // Delay the operation to allow drag and drop operation
			return;
		} else { // Perform single click immediately

			this._selectSingleRow(this._anchorSection, rowIndex);
		}
		this._activeGrid = host; // active grid must be set after selecting a row to allow previous grid selection to be cleared
	}

	this._dispatchSelectionChanged(e, rowIndex, section);
};
/** @private
 * @param {KeyboardEvent} e
 */
tr.grid.RowSelectionPlugin.prototype._onKeyDown = function (e) {
	if(!this._activeGrid) { return; }
	if(this._hasListener(tr.res.keydown)) {
		var obj = {"event": e};
		this._dispatch(tr.res.keydown, obj); // Ask clients if they want to cancel this event
		if(obj["cancel"]) {
			return;
		}
	}

	//Arrow Up/Down move a row selection up/down by one.
	//Arrow Up/Down and shift additionally select a row that is the one before first or after last selected row.
	//Page Up/Down move a row selection up/down by page scroll.
	//Page Up/Down and shift select every rows from anchor point to the next page scroll, compared to the current position.
	switch(e.keyCode){
		case 37: //Left arrow
			if(this._gotoPrevGrid()) {
				tr._preventDefault(e);
			}
			break;
		case 38: //Up arrow
			this._selectByKey(-1, e);
			break;
		case 39: //Right arrow
			if(this._gotoNextGrid()) {
				tr._preventDefault(e);
			}
			break;
		case 40: //Down arrow
			this._selectByKey(1, e);
			break;
		case 33: //Page up
			this._selectByKey(-1, e, true);
			break;
		case 34: //Page down
			this._selectByKey(1, e, true);
			break;
		case 36: //Home
			this._selectByKey(-1, e, true);
			break;
		case 35: //End
			this._selectByKey(1, e, true);
			break;
	}
};

/**
 * Clear column selection after row selection changed.
 *@private
 */
tr.grid.RowSelectionPlugin.prototype._onSelectionChanged = function (e) {
	if(!this._activeGrid) { return; }
	var csp = this._getPlugin("ColumnSelectionPlugin");
	if (csp && csp['isEnabled']()) {
		csp['clearSelection']();
	}
	var cep = this._getPlugin("CellEditingPlugin");
	if(cep) {
		cep.closeCellEditor();
	}
};

/** @private
 * @param {tr.Grid.SectionReference=} opt_sectRef
 * @return {tr.ILayoutGrid}
 */
tr.grid.RowSelectionPlugin.prototype._getSection = function(opt_sectRef){
	if(!this._activeGrid) {
		this._activeGrid = this._hosts[0];
		if(this._activeGrid) {
			this._anchorSection = this._activeGrid.getSection("content");
			if(!this._anchorSection) {
				this._activeGrid = null;
			}
		}
	}
	if(opt_sectRef){
		return this._activeGrid.getSection(opt_sectRef);
	}

	if (this._anchorSection) {
		if (!this._anchorSection.getElement()) { // The section may has been disposed.
			this._anchorSection = null;
		}
	}

	return this._anchorSection;
};

/** @private
 * @param {boolean=} opt_preserveAnchor
 */
tr.grid.RowSelectionPlugin.prototype._clearSelectedRows = function(opt_preserveAnchor){ // Slow
	if(!this._activeGrid) { return; }

	var sections = this._activeGrid.getAllSections(tr.res.content);
	for(var i = sections.length; --i >= 0;){
		sections[i].clearSelectedRows();
	}

	if (!opt_preserveAnchor) {
		this._anchorSection = null;
	}
	this._pendingClickIndex = -1;
};
/** @private
 * @suppress {missingProperties}
 * @param {number} direction
 * @param {KeyboardEvent} e
 * @param {boolean=} pageKey
 */
tr.grid.RowSelectionPlugin.prototype._selectByKey = function(direction, e, pageKey) {
	e.preventDefault();

	var shiftKey = e.shiftKey;
	var next = 0;
	var rowLimit = this._anchorSection.getRowCount() - 1;
	if(pageKey) {
		var info = this._activeGrid.getVScrollView();
		if(direction === -1){ // Up
			if(info) {
				next = this._anchorSection.getFirstSelectedRow() <= info["firstFullRow"] ? info["prevPageRowIndex"] : info["firstFullRow"];
			}
		} else {
			if(info) {
				next = this._anchorSection.getLastSelectedRow() >= info["lastFullRow"] ? info["nextPageRowIndex"] : info["lastFullRow"];
			} else {
				next = rowLimit;
			}
		}
	} else {
		if(direction === -1) {
			next = this._anchorSection.getFirstSelectedRow() + direction;
		} else {
			next = this._anchorSection.getLastSelectedRow() + direction;
		}
	}
	if(next < 0) {
		var csp = this._getPlugin("ColumnSelectionPlugin");
		if (!shiftKey && csp && csp['isEnabled']()) {
			csp['selectSingleColumn'](null, this._activeGrid);
			return; // ColumnSelectionPlugin has cleared all row selections at this point
		}
		next = 0;
	} else if(next > rowLimit) {
		next = rowLimit;
	}

	if(shiftKey && this._multiSelect){
		this._selectFromAnchorToTarget(this._anchorSection, next);
	} else {
		this._selectSingleRow(this._anchorSection, next);
	}

	this._scrollToRow(next);

	this._dispatchSelectionChanged(e, next);
};
/** @private
 * @param {tr.ILayoutGrid} section
 * @param {number} rowIndex
 * @return {boolean}
 */
tr.grid.RowSelectionPlugin.prototype._selectSingleRow = function(section, rowIndex){
	if(!section) { return false; }

	this._clearSelectedRows(); // Deselect all rows
	section.setSelectedRow(rowIndex, true); // The anchor row is set
	this._anchorSection = section;
	return true;
};


/** @private
 * @param {number} rowIndex
 * @param {number} length
 * @param {tr.Grid.SectionReference=} opt_sectRef
 */
tr.grid.RowSelectionPlugin.prototype._selectRowRange = function(rowIndex, length, opt_sectRef){
	var section = this._getSection(opt_sectRef);
	if(!section) { return; }
	section.selectRowRange(rowIndex, length);
};

/** @private
 * @param {tr.ILayoutGrid} section
 * @param {number} rowIndex
 */
tr.grid.RowSelectionPlugin.prototype._selectFromAnchorToTarget = function(section, rowIndex) {
	var anchorRow = this._anchorSection.getRowAnchor();
	if(this._anchorSection !== section || anchorRow !== rowIndex){
		this._clearSelectedRows(true); // Keep anchor section
		var anchorSectIndex = this._anchorSection.getIndex();
		var targetSectIndex = section.getIndex();
		if (anchorSectIndex === targetSectIndex) {
			var selectionCount = (anchorRow - rowIndex) + (anchorRow > rowIndex ? 1 : -1);
			section.selectRowRange(rowIndex, selectionCount);
		} else if (anchorSectIndex < targetSectIndex) {
			this._selectSectionRange(anchorSectIndex+1, targetSectIndex-1);
			this._anchorSection.selectRowRange(anchorRow, this._anchorSection.getRowCount() - anchorRow);
			section.selectRowRange(0, rowIndex + 1);
		} else if (anchorSectIndex > targetSectIndex) {
			this._selectSectionRange(targetSectIndex+1, anchorSectIndex-1);
			this._anchorSection.selectRowRange(0, anchorRow + 1);
			section.selectRowRange(rowIndex, section.getRowCount() - rowIndex);
		}
		this._anchorSection.setRowAnchor(anchorRow); //Make sure that the anchor stays at the same position
	}
};
/** @private
 * @param {number} fromIndex
 * @param {number} toIndex
 */
tr.grid.RowSelectionPlugin.prototype._selectSectionRange = function(fromIndex, toIndex){
	if(!this._activeGrid) { return; }
	for (var i = fromIndex; i <= toIndex; i++) {
		var settings = this._activeGrid.getSectionSettings(i);
		if (settings.getType() === tr.res.content) {
			var section = settings.getSection();
			section.selectRowRange(0, section.getRowCount());
		}
	}
};

/** @private
 * @return {boolean}
 */
tr.grid.RowSelectionPlugin.prototype._gotoPrevGrid = function() {
	if(!this._activeGrid) { return false; }

	if(this._activeGrid === this._hosts[0]) { // This is the first grid
		return false;
	} else {
		var curIndex = this._hosts.indexOf(this._activeGrid);
		if(curIndex > 0) {
			this._gotoGrid(curIndex - 1);
			return true;
		}
	}
	return false;
};
/** @private
 * @return {boolean}
 */
tr.grid.RowSelectionPlugin.prototype._gotoNextGrid = function() {
	if(!this._activeGrid) { return false; }

	var lastIndex = this._hosts.length - 1;
	if(this._activeGrid === this._hosts[lastIndex]) { // This is the last grid
		return false;
	} else {
		var curIndex = this._hosts.indexOf(this._activeGrid);
		if(curIndex >= 0) {
			this._gotoGrid(curIndex + 1);
			return true;
		}
	}
	return false;
};
/** @private
 * @param {number} gridIndex
 */
tr.grid.RowSelectionPlugin.prototype._gotoGrid = function(gridIndex) {
	var anchorRow = this._anchorSection.getRowAnchor();
	var sectionIndex = this._anchorSection.getIndex();
	this.clearSelectedRows(); // Clear all current grid's selections
	this.selectSingleRow(anchorRow, sectionIndex, this._hosts[gridIndex]); // go to the next grid at the first column
};

/** @private
 * @param {Event=} e
 * @param {number=} rowIndex
 * @param {tr.ILayoutGrid=} section
 */
tr.grid.RowSelectionPlugin.prototype._dispatchSelectionChanged = function(e, rowIndex, section) {
	if(this._hasListener(tr.res.selectionChanged)) {
		var obj = {};
		obj[tr.res.rowIndex] = (rowIndex !== undefined) ? rowIndex : this._anchorSection.getRowAnchor();
		obj[tr.res.section] = (section !== undefined) ? section : this._anchorSection;
		if(e !== undefined) {
			obj[tr.res.event] = e;
		}
		obj["grid"] = this._activeGrid;
		this._dispatch(tr.res.selectionChanged, obj);
	}
};
