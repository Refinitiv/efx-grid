 * @function FACT
 * @memberof Math
 * @description

 *    {lang=ja} 数値の階乗を返します
 *    {lang=zh} 返回一个数字的阶乘

 *    {lang=de} Gibt den faktoriellen Wert einer Zahl aus
 * @param  {number} num
 *    {lang=en} Number
 *    {lang=ja} 数値
 *    {lang=zh} 数字
 *    {lang=hant} 數字
 *    {lang=ko} 숫자
 *    {lang=ru} Number
 *    {lang=de} Zahl
 * @return {number}
 * @example
 *    =FACT(10)
 */
tr.fn.Math["FACT"] = function(num) {
	if(num < 0) { return NaN; }
	num = Math.floor(num);
	var ret = 1;
	while(num > 1) {
		ret *= num;
		--num;
	}
	return ret;
};

	var i;

	}
