goog.provide("tr.grid.MovableHeaderPlugin");

goog.require("tr.res");
goog.require("tr.ElementWrapper");
goog.require("tr.ILayoutGrid");

goog.require("tr.grid.IPlugin");
goog.require("tr.grid.Plugin");
goog.require("tr.grid.SectionSettings");
goog.require("tr.DataView");

/** Event
* @event tr.grid.MovableHeaderPlugin#dragged
* @type {MouseEvent}
* @property {*} * Depends on the browser's mouse event arguments
*/
/** Event
* @deprecated
* @event tr.grid.MovableHeaderPlugin#sectionMoved
* @property {number} startSectionIndex
* @property {number} destSectionIndex
* @property {number} movedChunkSize
* @property {tr.grid.SectionSettings} startSection
* @property {tr.grid.SectionSettings} destinationSection
*/
/** Event
* @event tr.grid.MovableHeaderPlugin#groupMoved
* @property {tr.grid.SectionSettings} sectionSettings
* @property {tr.DataView} group
* @property {string} groupName
*/

/** @constructor
 * @extends {tr.grid.Plugin}
 * @implements {tr.grid.IPlugin}
 */
tr.grid.MovableHeaderPlugin = function () {
	this._onSectionAdded = goog.bind(this._onSectionAdded, this);

	this._siblingRange = [];

	// For each cell
	this._onMouseDown = goog.bind(this._onMouseDown, this);

	// For timer
	this._onDragStart = goog.bind(this._onDragStart, this);

	// For mouse move
	this._onDrag = goog.bind(this._onDrag, this);
	this._onDragEnd = goog.bind(this._onDragEnd, this);

	this._guideline = new tr.ElementWrapper();
	this._guideline.setClass("tr-guideline");
	this._guideline.setLeft(0);
	this._guideline.setHeight(3);
	this._dragbox = new tr.ElementWrapper(document.createElement("svg"));
	this._dragbox.setClass("tr-dragbox content-dragging");

	this._addEvent("dragged");
	this._addEvent("groupMoved");
};
goog.inherits(tr.grid.MovableHeaderPlugin, tr.grid.Plugin);

//#region Public Methods
/** @override */
tr.grid.MovableHeaderPlugin.prototype.getName = function () {
	return "MovableHeaderPlugin"; // Read Only
};
/** @override */
tr.grid.MovableHeaderPlugin.prototype.initialize = function (host) {
	this._host = host;
	this._host.listen("sectionAdded", this._onSectionAdded);
};
/** @override */
tr.grid.MovableHeaderPlugin.prototype.unload = function (host) {
	this._host.unlisten("sectionAdded", this._onSectionAdded);
	
	this._onDragEnd();
	
	this._host = null;
};

//#endregion Public Methods

//#region Private Methods
/** @private
* @param {Object} e
*/
tr.grid.MovableHeaderPlugin.prototype._onSectionAdded = function (e) {
	if (e[tr.res.sectionType] !== tr.res.content) { return; }

	var section = /** @type{tr.ILayoutGrid} */(e[tr.res.section]);
	section.listen(tr.res.mousedown, this._onMouseDown);
	section.listen(tr.res.mouseup, this._onDragEnd);
	section.listen("dragstart", tr._preventDefault);
};

/** @private
* @param {MouseEvent} e
*/
tr.grid.MovableHeaderPlugin.prototype._onMouseDown = function (e) {
	if(e.button !== 0) { 
		return;
	}
	
	if (this._timerId >= 0) {
		return;
	}
	
	var info = this._host.getRelativePosition(e);
	if (!info) {
		return;
	}
	
	var dv = this._host.getDataSource();
	if(!dv) {
		return;
	}
	
	var rid = dv.getRowId(info[tr.res.rowIndex]);
	var grpView = dv.getGroup(rid);
	if(!grpView) {
		return;
	}
	
	this._draggedInfo = {};
	this._draggedInfo.group = grpView;
	this._draggedInfo.settings = info[tr.res.sectionSettings];

	this._timerId = window.setTimeout(this._onDragStart, 300);
};

/** @private
* @param {MouseEvent} e
*/
tr.grid.MovableHeaderPlugin.prototype._onDragStart = function (e) {
	if (this._isDragging || !this._host) { return; }
	this._isDragging = true;
	this._timerId = -1;
	
	document.body.classList.add("tr-dragging"); // Prevent text selection
	document.body.classList.add("tr-move-cursor"); 
	document.addEventListener(tr.res.mouseup, this._onDragEnd, false);

	var width = this._host.getWidth();
	
	var s = this._draggedInfo.settings.getSection();
	var dv = this._draggedInfo.settings.getDataSource();
	var curGrp = this._draggedInfo.group;
	var parentGrp = curGrp.getDataSource();
	var rowIndex = dv.getRowIndex(curGrp.getGroupRowId());
	
	var sectionInfo = tr.getRelativePosition(s, document.body);
	var left = sectionInfo["x"];
	var top = this._getCellTop(s, rowIndex);
	
	var columnIndex = this._getHeaderColumn();
	var cell = s.getCell(columnIndex, rowIndex, true);
	var content = (cell) ? cell.getContent() : null;
	var txtContent = (content) ? content.textContent : "";
	this._dragbox.setContent(txtContent);
	
	this._guideline.setWidth(width);
	this._guideline.setLeft(left);
	this._guideline.setTop(top + 1);
	
	var grpNames = parentGrp.getGroupList();
	var len = grpNames.length;
	var rowIndices = [];
	for(var i = 0; i < len; ++i) {
		var grp = parentGrp.getGroup(grpNames[i]);
		rowIndices.push(dv.getRowIndex(grp.getGroupRowId()));
	}
	rowIndices.push(rowIndices[0] + parentGrp.getVisibleRowCount());
	this._draggedInfo.availableRows = rowIndices;
	this._draggedInfo.startRowIndex = rowIndex;
	this._draggedInfo.destGroupIndex = grpNames.indexOf(curGrp.getGroupId());

	window.addEventListener(tr.res.mousemove, this._onDrag, false);
	this._guideline.setParent(document.body);
};

/** @private
* @param {Event} e
*/
tr.grid.MovableHeaderPlugin.prototype._onDrag = function (e) {
	if (!this._isDragging) { return; }

	var mousePos = tr.getRelativePosition(e, document.body);
	var mouseY = mousePos["y"];
	var top = mouseY;
	
	var s = this._draggedInfo.settings.getSection();
	var rowIndices = this._draggedInfo.availableRows;
	
	var gridPos = this._host.getRelativePosition(e);
	var rowIndex = gridPos[tr.res.rowIndex];
	var lastGrpIndex = rowIndices.length - 1;
	var grpIndex;
	
	if(rowIndex >= 0) { // undefined is not greater nor equal to 0
		if(rowIndex <= rowIndices[0]) {
			grpIndex = 0;
		} else if(rowIndex >= rowIndices[lastGrpIndex]) {
			grpIndex = lastGrpIndex;
		} else {
			grpIndex = tr.lowerBound(rowIndices, rowIndex);
			var grpRowIndex = rowIndices[grpIndex];
			if(grpRowIndex > rowIndex) {
				--grpIndex;
			}
		}
		top = this._getCellTop(s, rowIndices[grpIndex]);
	} else {
		top = this._getCellTop(s, rowIndices[0]);
		if(mouseY <= top) {
			grpIndex = 0;
		} else {
			grpIndex = lastGrpIndex;
			top = this._getCellTop(s, rowIndices[lastGrpIndex]);
		}
	}
	
	this._draggedInfo.destGroupIndex = grpIndex;
	
	this._guideline.setTop(top);

	this._dragbox.setLeft(mousePos["x"] + 5);
	this._dragbox.setTop(mouseY - 5);
	if (this._dragbox.getParent() === null) {
		this._fadeout(true);

		this._dragbox.setParent(document.body);
	}

	this._dispatch("dragged", /** @type{!Object} */(e));

	tr._preventDefault(e);
};

/** @private
*/
tr.grid.MovableHeaderPlugin.prototype._onDragEnd = function () {
	if(this._timerId >= 0) {
		window.clearTimeout(this._timerId);
		this._timerId = -1;
		this._draggedInfo = null;
	}
	
	if (!this._isDragging) { return; }
	this._isDragging = false;
	
	document.body.classList.remove("tr-dragging");
	document.body.classList.remove("tr-move-cursor");
	document.removeEventListener(tr.res.mouseup, this._onDragEnd, false);

	this._fadeout(false);
	this._guideline.setParent(null);
	this._dragbox.setParent(null);
	window.removeEventListener(tr.res.mousemove, this._onDrag, false);
	
	var dv = this._draggedInfo.settings.getDataSource();
	dv.moveGroup(this._draggedInfo.group.getGroupRowId(), this._draggedInfo.destGroupIndex);

	if (this._hasListener("groupMoved")) {
		var obj = {};
		obj[tr.res.sectionSettings] = this._draggedInfo.settings;
		obj["group"] = this._draggedInfo.group;
		obj["groupName"] = this._draggedInfo.group.getGroupId();
		this._dispatch("groupMoved", obj);
	}

	this._draggedInfo = null;
};

/** @private
* @param {boolean} bool
*/
tr.grid.MovableHeaderPlugin.prototype._fadeout = function (bool) {
	var s = this._draggedInfo.settings.getSection();
	s.enableRowClass(this._draggedInfo.startRowIndex, "drag-fadeout", bool);
};

/** @private
* @return {number}
*/
tr.grid.MovableHeaderPlugin.prototype._getHeaderColumn = function () {
	var plugin = this._getPlugin("TreeIndentingPlugin");
	if(plugin){
		return plugin["getDisplayColumnIndex"]();
	}
	plugin = this._getPlugin("CollapsibleHeaderPlugin");
	if(plugin){
		return plugin["getDisplayColumnIndex"]();
	}
	return 0;
};

/** @private
* @param {tr.ILayoutGrid} section
* @param {number} rowIndex
* @return {number}
*/
tr.grid.MovableHeaderPlugin.prototype._getCellTop = function (section, rowIndex) {
	var rowCount = section.getRowCount();
	if(rowIndex >= rowCount) {
		var info = tr.getRelativePosition(section.getElement(), document.body);
		return info["y"] + section.getHeight();
	}
	
	var cell;
	var colCount = section.getColumnCount();
	for(var c = 0; c < colCount; ++c) {
		cell = section.getCell(c, rowIndex);
		if(cell && cell.isVisible() && cell.getParent()) {
			break;
		}
	}
	if(!cell) {
		return -100;
	}
	
	return tr.getRelativePosition(cell.getElement(), document.body)["y"];
};
