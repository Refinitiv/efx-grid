goog.provide("tr.grid.GroupStatisticsPlugin");

goog.require("tr.res");
goog.require("tr.grid.IPlugin");
goog.require("tr.grid.Plugin");
goog.require("tr.DataView");
goog.require("tr.grid.SectionSettings");
goog.require("tr.grid.IndexMarker");

/** @constructor
 * @param {Array.<number|boolean>|null=} opt_restrictions By default all columns will be processed by this plugin
 * @param {Array.<number|boolean>|null=} opt_exclusions By default no column will be excluded by this plugin
* @extends {tr.grid.Plugin}
* @implements {tr.grid.IPlugin}
*/
tr.grid.GroupStatisticsPlugin = function (opt_restrictions, opt_exclusions) {
	this._onPreSectionDataBinding = goog.bind(this._onPreSectionDataBinding, this);
	this._onPreSectionRender = goog.bind(this._onPreSectionRender, this);
	this._onColumnAdded = goog.bind(this._onColumnAdded, this);
	this._onColumnMoved = goog.bind(this._onColumnMoved, this);
	this._onColumnRemoved = goog.bind(this._onColumnRemoved, this);

	this._marker = new tr.grid.IndexMarker(opt_restrictions, opt_exclusions);

	if(this._predefinedOperations === null) {
		this._predefinedOperations = {
			"sum": tr.grid.GroupStatisticsPlugin.sum,
			"min": tr.grid.GroupStatisticsPlugin.min,
			"max": tr.grid.GroupStatisticsPlugin.max,
			"mean": tr.grid.GroupStatisticsPlugin.mean,
			"median": tr.grid.GroupStatisticsPlugin.median
		};
		tr.grid.GroupStatisticsPlugin.prototype._predefinedOperations = this._predefinedOperations;
	}
};
goog.inherits(tr.grid.GroupStatisticsPlugin, tr.grid.Plugin);

/** @override */
tr.grid.GroupStatisticsPlugin.prototype.getName = function () {
	return "GroupStatisticsPlugin"; // Read Only
};

/** @override */
tr.grid.GroupStatisticsPlugin.prototype.initialize = function (host) {
	this._host = host;
	this._host.listen("preSectionRender", this._onPreSectionRender);
	this._host.listen(tr.res.preSectionDataBinding, this._onPreSectionDataBinding);	
	this._host.listen(tr.res.columnAdded, this._onColumnAdded);
	this._host.listen(tr.res.columnMoved, this._onColumnMoved);
	this._host.listen(tr.res.columnRemoved, this._onColumnRemoved);
};

/** @override */
tr.grid.GroupStatisticsPlugin.prototype.unload = function (host) {
	this._host.unlisten("preSectionRender", this._onPreSectionRender);
	this._host.unlisten(tr.res.preSectionDataBinding, this._onPreSectionDataBinding);
	this._host.unlisten(tr.res.columnAdded, this._onColumnAdded);
	this._host.unlisten(tr.res.columnMoved, this._onColumnMoved);
	this._host.unlisten(tr.res.columnRemoved, this._onColumnRemoved);
	this._host = null;
};

/** Specify column to be processed by this plugin
* @public
* @return {tr.grid.IndexMarker}
*/
tr.grid.GroupStatisticsPlugin.prototype.getMarker = function () {
	return this._marker;
};

//#region Public
/** @public
 * @param {string|number|null=} opt_col column name or column index
 * @param {string|function(Array.<number>)|null=} opt_fn function name or function object that takes array of number as an argument
 * @return {Object|number|null}
 */
tr.grid.GroupStatisticsPlugin.prototype.getColumnStat = function (opt_col, opt_fn) {
	return this._calculateStat(this._host.getDataSource(), opt_col, opt_fn);
};

/** @public
 * @param {tr.Grid.SectionReference} sectionRef
 * @param {string|number|null=} opt_col column name or column index
 * @param {string|function(Array.<number>)|null=} opt_fn function name or function object that takes array of number as an argument
 * @return {Object|number|null}
 */
tr.grid.GroupStatisticsPlugin.prototype.getSectionStat = function (sectionRef, opt_col, opt_fn) {
	var sectionSettings = this._host.getSectionSettings(sectionRef);
	if(sectionSettings === null) { return null; }

	return this._calculateStat(sectionSettings.getDataSource(), opt_col, opt_fn);
};

tr.grid.GroupStatisticsPlugin.prototype.getPredefinedOperations = function () {
	var ary = [];
	for(var key in this._predefinedOperations) {
		ary.push(key);
	}
	return ary;

/** @override 
* @ignore
*/
tr.grid.GroupStatisticsPlugin.prototype._onColumnAdded = function (e) {
	if (!e["atTheMiddle"]) { return; }
	
	var colIndex = /** @type{number} */(e[tr.res.colIndex]);

	this._marker.insertMarker(colIndex);
	tr.insertArrayItem(this._operations, colIndex, null);
};

/** @override 
* @ignore
*/
tr.grid.GroupStatisticsPlugin.prototype._onColumnMoved = function (e) {
	var from = /** @type{number} */(e[tr.res.fromColIndex]);
	var to = /** @type{number} */(e[tr.res.toColIndex]);

	this._marker.moveMarker(from, to);
	tr.moveArrayItem(this._operations, from, to);
};

/** @override 
* @ignore
*/
tr.grid.GroupStatisticsPlugin.prototype._onColumnRemoved = function (e) {
	if (!e["atTheMiddle"]) { return; }
	
	var colIndex = /** @type{number} */(e[tr.res.colIndex]);
	
	this._marker.removeMarker(colIndex);
	if(this._operations !== null) {
		this._operations.splice(colIndex, 1);
	}
};

/** @private
* @param {Object} e
*/
tr.grid.GroupStatisticsPlugin.prototype._onPreSectionRender = function (e) {
	if (!this._autoStatRendering) { return; }
	if (e[tr.res.sectionType] === tr.res.content) { return; }

	var sectionSettings = /** @type{tr.grid.SectionSettings} */(e["sectionSettings"]);
	var dataSource = /** @type{tr.DataView} */(sectionSettings.getDataSource());
	if (dataSource === null) {
		return; // This section has no data
	}
	var dataMap = this._getDataColumnMap();

	var section = /** @type{tr.ILayoutGrid} */(e[tr.res.section]);
	var toC = /** @type{number} */(e[tr.res.toColIndex]);
	var toR = /** @type{number} */(e[tr.res.toRowIndex]);
	for (var c = e[tr.res.fromColIndex]; c < toC; ++c) {
		if (!this._marker.isAllowed(c)) { continue; }

		var result = this._calculateStat(dataSource, c, null, true);
		if(result === undefined || result === null || result !== result) {
			result = "NaN";
		}

		var fnName = null;
		if (this._operations !== null) {
			fnName = this._operations[c];
		}
		if (fnName === undefined || fnName === null) {
			fnName = this._defaultOperation;
		}
		result = fnName.charAt(0).toUpperCase() + fnName.slice(1) + ": " + result;

		for (var r = e[tr.res.fromRowIndex]; r < toR; ++r) {
			section.setCellContent(c, r, result);
		}
	}
};

/** @private
* @param {Object} e
*/
tr.grid.GroupStatisticsPlugin.prototype._onPreSectionDataBinding = function (e) {
	if (!this._autoStatCalculating) { return; }

	var dataSource = /** @type{tr.DataView} */(e[tr.res.dataSource]);
	e["stat"] = this._calculateStat(dataSource, null, null, true);
};

/** @private
 * @param {tr.DataView} dataSource
 * @param {string|number|null=} opt_col column name or column index
 * @param {string|function(Array.<number>)|null=} opt_fn function name or function object that takes array of number as an argument
 * @param {boolean=} opt_userPreference
 * @return {Object|number|null}
 */
tr.grid.GroupStatisticsPlugin.prototype._calculateStat = function (dataSource, opt_col, opt_fn, opt_userPreference) {
	if(dataSource === undefined || dataSource === null) { return null; }

	var dataMap = this._getDataColumnMap();

	var colName = opt_col;
	var colIndex = -1;
	if(tr.isNumber(opt_col)) {
		colName = dataMap[/** @type {number} */(opt_col)];
		if(colName === undefined || colName === null) {
			return null;
		}
		if(opt_userPreference === true) {
			colIndex = /** @type{number} */(opt_col);
		}
	}

	var func = null;
	if(colName !== null) {
		func = this._getOperation(opt_fn, colIndex);
		return this._calculateColumnStat(this._getColumnValues(dataSource, /** @type{string} */(colName)), func);
	}

	if(opt_userPreference !== true) {
		func = this._getOperation(opt_fn, -1);
	}
	var stats = {}; 
	for (var c = dataMap.length; --c >= 0;) {
		if (!this._marker.isAllowed(c)) { continue; }

		colName = dataMap[c];
		if(colName !== undefined && colName !== null) {
			var colValues = this._getColumnValues(dataSource, colName);
			if(opt_userPreference === true) {
				func = this._getOperation(opt_fn, c);
			}
			stats[colName] = this._calculateColumnStat(colValues, func);
		}
	}
	return stats; // Object of column names
};

/** @private
 * @param {*} opt_fn
 * @param {number} colIndex
 * @return {function(Array.<number>)|null} func
 */
tr.grid.GroupStatisticsPlugin.prototype._getOperation = function (opt_fn, colIndex) {
	if(opt_fn === null) {
		if(colIndex >= 0) {
			if (this._operations !== null) {
				opt_fn = this._operations[colIndex];
				if(this._predefinedOperations[opt_fn] === undefined) {
					opt_fn = this._defaultOperation;
				}
			}
			if (opt_fn === undefined || opt_fn === null) {
				opt_fn = this._defaultOperation;
			}
		} else {
			return null;
		}
	}

	if(tr.isString(opt_fn)) {
		var func = /** @type{function(Array.<number>)} */(this._predefinedOperations[/** @type{string} */(opt_fn)]);
		if(func === undefined) {
			window.console.log("WARNING: " + opt_fn + " operation does not defined");
			return null;
		}
		return func;
	} else if(tr.isFunction(opt_fn)) {
		return /** @type{function(Array.<number>)} */(opt_fn);
	}
	return null;
};

/** @private
 * @param {Array.<number>} colValues
 * @param {function(Array.<number>)|null} func
 * @return {Object.<string, number>|number} Object of all stat or number
 */
tr.grid.GroupStatisticsPlugin.prototype._calculateColumnStat = function (colValues, func) {
	if(func !== null) {
		return func(colValues); // Number is returned
	}

	// Calculate all statistic
	var stat = {};
	var len = colValues.length;
	if(len <= 0) {
		stat["min"] = NaN;
		stat["max"] = NaN;
		stat["sum"] = NaN;
		stat["mean"] = NaN;
		stat["median"] = NaN;
		return stat;
	}
	var val = colValues[0];
	var min = val;
	var max = val;
	var sum = val;

	for(var i = len; --i > 0;) {
		val = colValues[i];
		if(val < min) { min = val; }
		if(max < val) { max = val; }
		sum += val;
	}
	stat["min"] = min;
	stat["max"] = max;
	stat["sum"] = sum;
	stat["mean"] = sum / len;
	stat["median"] = this._predefinedOperations["median"](colValues);
	return stat;
};

/** @private
 * @param {tr.DataView} dataSource
 * @param {string} colName
 * @return {!Array.<number>}
 */
tr.grid.GroupStatisticsPlugin.prototype._getColumnValues = function (dataSource, colName) {
	var i = 0;
	var colValues = [];
	var rowCount = dataSource.getRowCount();
	var rowId;
	if (this._filterDataColumn) {
		for (i = 0; i < rowCount; ++i) {
			rowId = dataSource.getRowId(i);
			if (!dataSource.getData(rowId, this._filterDataColumn)) {
				colValues.push(dataSource.getData(rowId, colName));
			}
		}
	} else {
		for (i = 0; i < rowCount; ++i) {
			rowId = dataSource.getRowId(i);
			colValues.push(dataSource.getData(rowId, colName));
		}
	}
	// Auto numeric convertion
	if (!tr.isNumber(colValues[0]) && // first
		!tr.isNumber(colValues[colValues.length - 1]) && // last
		!tr.isNumber(colValues[Math.floor(colValues.length / 2)]) // middle
		) {
		for (i = colValues.length; --i >= 0; ) {
			colValues[i] = Number(colValues[i]);
		}
	}

	return colValues;
};

/** @private
 * @param {Array.<number>} colValues
 * @return {number}
 */
tr.grid.GroupStatisticsPlugin.min = function (colValues) {
	// WARNING: Normal looping is much more faster than Math.min.apply()
	if(colValues.length <= 0) { return NaN; }
	var min = colValues[0];
	for(var i = colValues.length; --i > 0;) {
		if(colValues[i] < min) {
			min = colValues[i];
		}
	}
	return min;
};
/** @private
 * @param {Array.<number>} colValues
 * @return {number}
 */
tr.grid.GroupStatisticsPlugin.max = function (colValues) {
	// WARNING: Normal looping is much more faster than Math.max.apply()
	if(colValues.length <= 0) { return NaN; }
	var max = colValues[0];
	for(var i = colValues.length; --i > 0;) {
		if(max < colValues[i]) {
			max = colValues[i];
		}
	}
	return max;
};

/** @private
 * @param {Array.<number>} colValues
 * @return {number}
 */
tr.grid.GroupStatisticsPlugin.sum = function (colValues) {
	var sum = 0;
	for (var i = colValues.length; --i >= 0;) {
		sum += colValues[i];
	}
	return sum;
};
/** @private
 * @param {Array.<number>} colValues
 * @return {number}
 */
tr.grid.GroupStatisticsPlugin.mean = function (colValues) {
	if(colValues.length <= 0) { return NaN; }
	var sum = tr.grid.GroupStatisticsPlugin.sum(colValues);
	return sum / colValues.length;
};
/** @private
 * @param {number} a
 * @param {number} b
 * @return {number}
 */
tr.grid.GroupStatisticsPlugin.compareNumber = function (a, b) {
	return a - b;
};
/** @private
 * @param {Array.<number>} colValues
 * @return {number}
 */
tr.grid.GroupStatisticsPlugin.median = function (colValues) {
	// WARNING: This is a super slow operation
	var len = colValues.length;
	if(len <= 1) {
		return (len < 1) ? NaN : colValues[0];
	}

	var sortedValues = colValues.concat().sort(tr.grid.GroupStatisticsPlugin.compareNumber);
	var half = len >> 1;
	if (len & 1) {
		return sortedValues[half];
	}
	
	return (sortedValues[half - 1] + sortedValues[half]) / 2.0;
};
