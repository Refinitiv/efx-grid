goog.provide("tr.ElementWrapper");

goog.require("tr.IElementControl");
goog.require("tr.Event");

/** Please see {@link tr.IElementControl} for full APIs and parameter lists. This class is used to wrap any element for better management in JavaScript  
 * All standard HTML element events are allowed and can be used through {@link tr.IElementControl#listen}  
 * See all available event list here {@link tr.IElementControl#event:blur}
 * @constructor
 * @implements {tr.IElementControl}
 * @param {Element=} opt_element If not specified "div" element is instantly created.
 */
tr.ElementWrapper = function (opt_element) {
	if(opt_element) {
		this._element = opt_element;
	} else {
		this._element = document.createElement("div"); // For fastest speed
	}
};

/** @private
* @type {Object}
*/
tr.ElementWrapper._passiveEvents = null;

//#region Content Manipulation
/** {@link tr.IElementControl#getContent} 
* @inheritDoc 
*/
tr.ElementWrapper.prototype.getContent = function (opt_allContent) { 
	if(opt_allContent) {
		return this._element.children; // WARNING: Returns private member
	} 
	return /** @type{Element|Node} */(this._element.lastChild);
};
/** {@link tr.IElementControl#setContent} 
* @inheritDoc 
*/
tr.ElementWrapper.prototype.setContent = function (content, opt_tooltip) {
	if (content == null || content !== content) {
		content = null;
	} else if (content.nodeType !== 1 && content.nodeType !== 11) { // Element type checking (1 = DOM, 11 = DocumentFragment)
		if (content[tr.res.getElement] === undefined) {
			if (!tr.isString(content)) { content = content + ""; }

			if (this._textSpan === null) {
				this._textSpan = document.createElement("span"); // For fastest speed
			}
			this._textSpan.textContent = content;

			content = this._textSpan;
		} else {
			content = content[tr.res.getElement]();
		}
	}

	if (content !== this._content) {
		this._content = /** @type{Element} */(content); 
		while (this._element.firstChild !== null) {
			this._element.removeChild(this._element.firstChild);
		}
		if (this._content !== null) {
			this._element.appendChild(this._content);
		}
	}

	if (opt_tooltip && content) {
		content.setAttribute("title", opt_tooltip);
	}
	return /** @type{Element} */(this._content); // Can only be an element at this point
};
/** {@link tr.IElementControl#addContent} 
* @inheritDoc 
*/
tr.ElementWrapper.prototype.addContent = function (content) {
	content = tr.getElement(content);
	if (content === null) { return null; }

	this._content = this._element.children;
	this._element.appendChild(content);
	return content;
};
/** {@link tr.IElementControl#removeContent} 
* @inheritDoc 
*/
tr.ElementWrapper.prototype.removeContent = function (content) {
	content = tr.getElement(content);
	if (content === null) { return false; }

	this._content = this._element.children;
	this._element.removeChild(content);
	return true;
};
//#endregion Content Manipulation

//#region Event Related Methods
/** {@link tr.IElementControl#listen} 
* @inheritDoc
* @suppress {invalidCasts}   // because of opts that is seen as 'Object' instead of 'boolean'
*/
tr.ElementWrapper.prototype.listen = function (type, handler, opt_priority) {
	if (!this._events) {
		this._events = {};
	}
	var e = this._events[type];
	if (!e) {
		e = new Event();
		this._events[type] = e;
		
		e.dispatch = e.dispatch.bind(e);
		
		if (!tr.ElementWrapper._passiveEvents) {
			// https://github.com/WICG/EventListenerOptions/blob/gh-pages/explainer.md#feature-detection
			var supportsPassive = false;
			try {
				var opts = /** @type{boolean} */(Object.defineProperty({}, 'passive', {
					/**
					 * @return {boolean}
					 */
					"get": function() {
						supportsPassive = true;
						return false;
					}
				}));
				window.addEventListener("testPassive", null, opts);
				window.removeEventListener("testPassive", null, opts);
			} catch (err) {}
			
			if(supportsPassive) {
				var passiveOption = { "passive": true };
				
				// "wheel" and "mousewheel" have default behavior that needs to be prevented. We have to exclude them from the list
				tr.ElementWrapper._passiveEvents = {
					// "wheel": passiveOption,
					// "mousewheel": passiveOption,
					"touchstart": passiveOption,
					"touchmove": passiveOption
				};
			} else {
				tr.ElementWrapper._passiveEvents = {};
			}
		}
		
		this._element.addEventListener(type, e.dispatch, tr.ElementWrapper._passiveEvents[type] || false);
	}

	e.listen(handler, opt_priority);
};
/** {@link tr.IElementControl#unlisten} 
* @inheritDoc
*/
tr.ElementWrapper.prototype.unlisten = function (type, handler) {
	if (this._events) {
		var e = this._events[type];
		if (e) {
			e.unlisten(handler);
		}
	}
};
/** {@link tr.IElementControl#unlistenAll} 
* @inheritDoc
*/
tr.ElementWrapper.prototype.unlistenAll = function (opt_type) {
	if (this._events) {
		if (opt_type) {
			if (this._events[opt_type]) {
				this._unlistenAll(opt_type);
				delete this._events[opt_type];
			}
		} else {
			for (var key in this._events) {
				this._unlistenAll(key);
			}
			this._events = null;
		}
	}
};
/** {@link tr.IElementControl#dispose} 
* @inheritDoc
*/
tr.ElementWrapper.prototype.dispose = function () {
	this.setParent(null);
	this.unlistenAll();

	this._element = null;
	this._content = null;
	this._textSpan = null;
};
/** {@link tr.IElementControl#getEvent} 
* @inheritDoc
* @ignore
*/
tr.ElementWrapper.prototype.getEvent = function (type) {
	return this._events[type] || null;
};

	return this._element.parentNode;
};
/** {@link tr.IElementControl#setParent} 
* @inheritDoc 
*/
tr.ElementWrapper.prototype.setParent = function (parent, opt_prepend) {
	parent = tr.getElement(parent);
	if (parent !== null) {
		if (opt_prepend) {
			parent.insertBefore(this._element, parent.firstChild);
		} else {
			parent.appendChild(this._element);
		}
	} else if (this._element.parentNode !== null) {
		this._element.parentNode.removeChild(this._element);
	}
};
/** {@link tr.IElementControl#insertBefore} 
* @inheritDoc 
*/
tr.ElementWrapper.prototype.insertBefore = function (nextSibling) {
	nextSibling = tr.getElement(nextSibling);
	if (nextSibling === null) { 
		this.setParent(this.getParent());
		return;
	}

	var parent = nextSibling.parentNode;
	if (parent === null) { return; }

	parent.insertBefore(this._element, nextSibling);
};
/** {@link tr.IElementControl#insertAfter} 
* @inheritDoc 
*/
tr.ElementWrapper.prototype.insertAfter = function (previousSibling) { 
	previousSibling = tr.getElement(previousSibling);
	if(previousSibling) {
		if(previousSibling.nextSibling) {
			this.insertBefore(previousSibling.nextSibling);
		} else {
			this.setParent(previousSibling.parentNode);
		}
	} else {
		this.setParent(this.getParent(), true);
	}
};

	if (this.hasClass(str)) {
		this.removeClass(str);
		if (opt_replacement) {
			this.addClass(opt_replacement);
		}
		return false;
	}
	this._element.classList.add(str);
	return true;
};
