	var sub = this._subMap[subId];
	if(!sub) {
		return;
	}
	this._dispatch("subscriptionRemoved", {"subs": [sub]});
	delete this._subMap[subId];
	delete this._dataMap[subId + sub["ric"]];
	if(sub["chain"]) {
		for(var i = 0; i < 10; ++i) {
			delete this._dataMap[subId + "Child_" + String.fromCharCode(65 + i)];
		}
	}
};

tr.MockSubscriptions.prototype.removeAllSubscriptions = function() { // All channels will also be closed
	var working = this._working;
	this.stop();

	if(this._hasListener("subscriptionRemoved")) {
		var subs = [];
		for(var subId in this._subMap) {
			subs.push(this._subMap[subId]);

		}

	this._subMap = {};
	this._dataMap = {};
	if(working) {
		this.start();
	}
};

/** @private
* @param {string} ric
* @param {boolean} asChain
* @param {string|number=} opt_subId
* @return {string|null} subscription id
*/
tr.MockSubscriptions.prototype._addSymbol = function(ric, asChain, opt_subId) {
	if(!ric) { return null; }

	if(!opt_subId && opt_subId !== 0) {
		do {
			opt_subId = "_" + this._runningSubId++;
		} while(this._subMap[opt_subId]);
	} else {
		opt_subId += ""; // Convert to string
	}
	if(this._subMap[opt_subId]) {
		window.console.log("WARNING: Cannot add ric with duplicate id");
		return null;
	}

	if(asChain) {
		ric = ric.replace("0#", ""); // The receiver is not expecting 0# symbol
	}
	var sub = {};
	sub["ric"] = ric;
	sub["chain"] = asChain;
	sub["id"] = opt_subId;
	this._subMap[opt_subId] = sub; // Collect all user subscriptions
	this._dataMap[opt_subId + sub["ric"]] = sub;
	if(asChain) {
		for(var i = 0; i < 10; ++i) { // Mock 10 constituents
			sub = {};
			sub["ric"] = "Child_" + String.fromCharCode(65 + i);
			sub["id"] = opt_subId; // Child shares the same sub id as its chain parent
			this._dataMap[opt_subId + sub["ric"]] = sub;
		}
	}
	
	this._dispatch("subscriptionAdded", {"subs": [sub]});

	this._connect();
	return opt_subId;
};

/** @private
*/
tr.MockSubscriptions.prototype._connect = function() {
	if(!this._working || this._timerId >= 0) { return; }

	this._timerId = window.setTimeout(this._onSubscriptionResponse, this._updateInterval + this._rand(50, -50));
};

/** @private
*/
tr.MockSubscriptions.prototype._onSubscriptionResponse = function() {
	this._timerId = -1;

	var field;
	for(var dataId in this._dataMap) {
		if(this._rand(5) !== 0) {
			continue;
		}
		var sub = this._dataMap[dataId];
		var prevData = sub.prevData;

		var values = {};
		for(field in this._fields){
			var fInfo = tr.MockSubscriptions._fieldInfo[field] || {};
			var value, min = 100, max = 10000, prec = 0;
			if(fInfo.value){
				value = fInfo.value;
			} else {
				if(fInfo.type === "string"){
					value = sub["ric"].replace(".", "");
				} else { //default is number
					if(fInfo.min !== undefined){
						min = fInfo.min;
					}
					if(fInfo.max !== undefined){
						max = fInfo.max;
					}
					if(fInfo.prec !== undefined){
						prec = fInfo.prec;
					}
					value  = this._rand(max, min, prec);
				}
			}
			if(prevData) {
				if(fInfo.changeOnly) {
					if(prevData[field] === value) {
						continue;
					}
				}
				prevData[field] = value;
			}
			values[field] = value;

			if(fInfo.prefix){ value = fInfo.prefix + value; }
			if(fInfo.suffix){ value = value + fInfo.suffix; }
			values[field + "_FORMATTED"] = value + "";
		}

		var evtArg = {};
		evtArg["ric"] = sub["ric"];
		evtArg["values"] = values;
		evtArg["subId"] = sub["id"];
		if(!prevData) {
			prevData = {};
			sub.prevData = prevData;
			evtArg["type"] = "statusUpdated";

			values["STATUS"] =  1;
			values["STATUS_FORMATTED"] = this._statusMap["1"];
			values["SUB_ID"] = sub["id"];
			for(field in values) {
				prevData[field] = values[field];
			}
		}
		this._dispatch("dataChanged", evtArg);
	}

	this._connect();
};

/** @public
 * @param {number} max
 * @param {number=} min
 * @param {number=} prec
 * @return {number}
 */
tr.MockSubscriptions.prototype._rand = function(max, min, prec) {
	if(!min) { min = 0; }
	var v = (max - min) * Math.random() + min;

	if(prec > 0) {
		var pow = Math.pow(10, prec);
		v = Math.floor(v * pow) / pow;
	} else {
		v = Math.floor(v);
	}
	return v;
};

	if(!listeners) { return; }
	var len = listeners.length;
	for(var i = 0; i < len; ++i) {
		listeners[i](eventArg);
	}
};
