goog.provide("tr.grid.ColumnWidthAdjustingPlugin");

goog.require("tr.grid.IPlugin");
goog.require("tr.grid.Plugin");
goog.require("tr.grid.TextWidthCalculator");
goog.require("tr.grid.Conflator");

/** @constructor
 * @param {Array.<number|boolean>|null=} opt_restrictions By default all columns will be processed by this plugin
 * @param {Array.<number|boolean>|null=} opt_exclusions By default no column will be excluded by this plugin
 * @param {boolean=} opt_autoAdjust If false, don't update column min widths
 * @extends {tr.grid.Plugin}
 * @implements {tr.grid.IPlugin}
*/
tr.grid.ColumnWidthAdjustingPlugin = function (opt_restrictions, opt_exclusions, opt_autoAdjust) {
	var _t = this;
	_t._onColumnAdded = goog.bind(_t._onColumnAdded, _t);
	_t._onColumnMoved = goog.bind(_t._onColumnMoved, _t);
	_t._onColumnRemoved = goog.bind(_t._onColumnRemoved, _t);
	_t._onPostSectionDataBinding = goog.bind(_t._onPostSectionDataBinding, this);
	_t._onPreForcedUpdate = goog.bind(_t._onPreForcedUpdate, _t);

	if (_t._calculator === undefined || _t._calculator === null) {
		_t._calculator = new tr.grid.TextWidthCalculator();
		tr.grid.ColumnWidthAdjustingPlugin.prototype._calculator = _t._calculator;
	}

	_t._hosts = [];
	_t._updateConflator = new tr.grid.Conflator(300, _t._onPostSectionDataBinding);
	_t._marker = new tr.grid.IndexMarker(opt_restrictions, opt_exclusions);
	_t._dummyCells = [];
	_t._minWidths = [];
	_t._autoAdjust = (opt_autoAdjust !== false);
};
goog.inherits(tr.grid.ColumnWidthAdjustingPlugin, tr.grid.Plugin);

//#region Implement IPlugin
/** @override */
tr.grid.ColumnWidthAdjustingPlugin.prototype.getName = function () {
	return "ColumnWidthAdjustingPlugin"; // Read Only
};
/** @override */
tr.grid.ColumnWidthAdjustingPlugin.prototype.initialize = function (host) {
	if(this._hosts.indexOf(host) >= 0) { return; }
	this._hosts.push(host);
	if(!this._autoAdjust) {
		return;
	}
	
	if (this._hosts.length === 1) { 
		host.listen(tr.res.columnAdded, this._onColumnAdded);
		host.listen(tr.res.columnMoved, this._onColumnMoved);
		host.listen(tr.res.columnRemoved, this._onColumnRemoved);
	}
	host.listen("postSectionDataBinding", this._onPostSectionDataBinding);
	host.listen("preForcedUpdate", this._onPreForcedUpdate);
};
/** @override */
tr.grid.ColumnWidthAdjustingPlugin.prototype.unload = function (host) {
	var at = this._hosts.indexOf(host);
	if(at < 0) { return; }
	this._hosts.splice(at, 1);

	this._updateConflator.reset();
	host.unlisten(tr.res.columnAdded, this._onColumnAdded);
	host.unlisten(tr.res.columnMoved, this._onColumnMoved);
	host.unlisten(tr.res.columnRemoved, this._onColumnRemoved);
	host.unlisten("postSectionDataBinding", this._onPostSectionDataBinding);
	host.unlisten("preForcedUpdate", this._onPreForcedUpdate);
};
//#endregion

/** @override
* @ignore

tr.grid.ColumnWidthAdjustingPlugin.prototype._onColumnAdded = function (e) {
	tr.grid.ColumnWidthAdjustingPlugin.base(this, "_onColumnAdded", e);
	var colIndex = /** @type{number} */(e[tr.res.colIndex]);
	tr.insertArrayItem(this._minWidths, colIndex, undefined);
};
/** @override
* @ignore
*/
tr.grid.ColumnWidthAdjustingPlugin.prototype._onColumnMoved = function (e) {
	tr.grid.ColumnWidthAdjustingPlugin.base(this, "_onColumnMoved", e);
	var from = /** @type{number} */(e[tr.res.fromColIndex]);
	var to = /** @type{number} */(e[tr.res.toColIndex]);
	tr.moveArrayItem(this._minWidths, from, to);
};
/** @override
* @ignore
*/
tr.grid.ColumnWidthAdjustingPlugin.prototype._onColumnRemoved = function (e) {
	tr.grid.ColumnWidthAdjustingPlugin.base(this, "_onColumnRemoved", e);
	var colIndex = /** @type{number} */(e[tr.res.colIndex]);
	this._minWidths.splice(colIndex, 1);
};

/** Create temporary element and set text content to calculate size.
* @public
* @param {number} colIndex
* @return {number}
*/
tr.grid.ColumnWidthAdjustingPlugin.prototype.getAdjustedColumnWidth = function (colIndex) {
	if (this._hosts.length === 0) {
		return NaN;
	}
	var layout = this._hosts[0].getSection(tr.res.content);
	if(!layout) {
		return NaN;
	}
	
	var columnElem = layout.getColumn(colIndex).getElement();
	var cacheText = this._getColumnCacheText(colIndex);

	var dummyCell = this._createDummyCell();
	dummyCell.innerText = cacheText;
	columnElem.appendChild(dummyCell);

	// NOTE: if font isn't cached, comparison is being done against Arial instead of Proxima Nova, thus yielding wrong offsetwidth
	var width = dummyCell.offsetWidth;
	if (this._maxWidth >= 0 && width > this._maxWidth) {
		width = this._maxWidth;
	}
	columnElem.removeChild(dummyCell);

	return width;
};

/** Force update and calculation
* @public
* @param {Object=} opt_e
* @return {boolean} True if there is any change, otherwise false
*/
tr.grid.ColumnWidthAdjustingPlugin.prototype.update = function (opt_e) {
	if(!this._autoAdjust) { return false; }
	var host = (opt_e)? opt_e["grid"]: this._hosts[0];
	if (!host) { return false; }

	var layout = host.getLastSection();
	if(layout === null) { return false; }

	var columnCount = host.getColumnCount();
	var widths = [];
	var width = 0;
	var columns = [];
	var dummyCell = null;
	var cacheTexts = [];
	var c = 0;

	// calculate cacheText for each column
	for (c = 0; c < columnCount; ++c) {
		columns[c] = layout.getColumn(c).getElement();
		cacheTexts[c] = this._getColumnCacheText(c);
	}

	// append dummycell to each column
	for (c = 0; c < columnCount; ++c) {
		if(cacheTexts[c]) {
			dummyCell = this._dummyCells[c];
			if (dummyCell === undefined) {
				dummyCell = this._createDummyCell();
				this._dummyCells[c] = dummyCell;
			}
			dummyCell.innerText = cacheTexts[c];
			var colElem = columns[c];
			colElem.appendChild(this._dummyCells[c]);
		}
	}

	// calculate column width if update width
	for (c = 0; c < columnCount; ++c) {
		if(cacheTexts[c]) {
			width = this._dummyCells[c].offsetWidth + this._offsetWidth;
			if (this._maxWidth >= 0 && width > this._maxWidth) {
				width = this._maxWidth;
			}
			if (width < this._minWidths[c]) { // Undefined value is not greater than any number
				width = this._minWidths[c];
			}
			widths[c] = width;
		}
	}

	// remove dummyCell from each column
	for (c = 0; c < columnCount; ++c) {
		if(cacheTexts[c]) {
			columns[c].removeChild(this._dummyCells[c]);
		}
	}

	var dirty = false;
	var layoutX = host.getColumnLayout();
	for (c = 0; c < columnCount; ++c) {
		width = widths[c];
		if (width !== undefined && width > 0) {
			dirty |= layoutX.setMinimumLaneSize(c, width);
			if (this._proportionalScaling) {
				if (layoutX.getLaneScalability(c)) {
					dirty |= layoutX.setLaneSize(c, width);
				}
			}
		}
	}
	if (dirty) {
		host.updateColumnLayout();
	}

	return /** @type {boolean} */ (dirty);
};

/** @private
* @param {number} colIndex
* @return {string}
*/
tr.grid.ColumnWidthAdjustingPlugin.prototype._getColumnCacheText = function (colIndex) {
	var contentIncluded = this._marker.isAllowed(colIndex);
	var cacheText = "";

	for (var i = this._hosts.length; --i >= 0;) {
		var host = this._hosts[i];
		var sectionCount = host.getSectionCount();
		for (var s = 0; s < sectionCount; s++) {
			var sectionSettings = host.getSectionSettings(s);
			var section = sectionSettings.getSection();
			var rowCount = section.getRowCount();
			var sectionType = sectionSettings.getType();
			var r, text, cell;
			if (sectionType === tr.res.content) {
				if (contentIncluded) {
					for (r = 0; r < rowCount; r++) {
						cell = section.getCell(colIndex, r);
						if (cell && cell.getElement()) {
							text = /** @type{string} */(cell.getElement().textContent); // Content cannot be a number
							if (text) {
								this._addText(text);
							}
						}
					}
				}
			} else if(this._nonContent) {
				for (r = 0; r < rowCount; r++) {
					cell = section.getCell(colIndex, r);
					if (cell && cell.getElement()) {
						text = /** @type{string} */(cell.getElement().textContent); // Content cannot be a number
						if (text) {
							this._addText(text);
						}
					}
				}
			}
			if (!this._calculator.isEmpty()) {
				cacheText += this._calculator.getCacheText() + "\n";
			}
		}
	}

	this._calculator.removeAllText();
	return cacheText;
};
/**
* @private
* @param {string} content
*/
tr.grid.ColumnWidthAdjustingPlugin.prototype._addText = function (content) {
	if (this._maxCharacters > 0) {
		content = content.substr(0, this._maxCharacters - 1);
	}
	this._calculator.addText(content);
};
