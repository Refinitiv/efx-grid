goog.provide("tr.grid.ZoomPlugin");

goog.require("tr.grid.IPlugin");
goog.require("tr.grid.Plugin");
goog.require("tr.Grid");

/** @constructor
* @extends {tr.grid.Plugin}
* @implements {tr.grid.IPlugin}
* @suppress {checkTypes}
*/
tr.grid.ZoomPlugin = function () {
	this._updateZoomedLayout = this._updateZoomedLayout.bind(this);
	this._updateElementLayout = this._updateElementLayout.bind(this);
	this._onWheel = this._onWheel.bind(this);
	
	this._hosts = [];
	this._elems = [];
	
	this._addEvent(tr.res.zoomChanged);

	// Setter and getter is used to controll min - max zoom factor. Produce event report once zoom successfuly applied
	Object.defineProperty(this, "zoomFactor", {
		set: this.zoomTo,
		get: this.getZoomFactor
	});
};
goog.inherits(tr.grid.ZoomPlugin, tr.grid.Plugin);

/** @private
* @ignore
* @type {number}
*/
tr.grid.ZoomPlugin.prototype._zoomFactor = 1;
/** @private
 * @type {number}
 */
tr.grid.ZoomPlugin.prototype._step = 0.1;
/** @private
 * @type {boolean}
 */
tr.grid.ZoomPlugin.prototype._wheel = false;
/** @private
 * @type {!Array.<Element>}
 */
tr.grid.ZoomPlugin.prototype._elems;


/** @override */
tr.grid.ZoomPlugin.prototype.getName = function () {
	return "ZoomPlugin"; // Read Only
};
/** @override */
tr.grid.ZoomPlugin.prototype.initialize = function(host) {
	if (this._hosts.indexOf(host) >= 0) {
		return;
	}

	this._applyZoom(host);
	
	this._hosts.push(host);
	if(this._hosts.length === 1) {
		host.listen("widthChanged", this._updateZoomedLayout); // Grid needs to be updated its zoomed layout
	}
	if(this._wheel) {
		host.listen("wheel", this._onWheel);
	}
};
/** @override */
tr.grid.ZoomPlugin.prototype.unload = function(host) {
	var at = this._hosts.indexOf(host);
	if(at < 0) { return; }
	
	if(this._hosts.length === 1) {
		host.unlisten("widthChanged", this._updateZoomedLayout);
	}
	if(this._wheel) {
		host.unlisten("wheel", this._onWheel);
	}
	this._hosts.splice(at, 1);
};


/** @public
* @return {number} 
*/
tr.grid.ZoomPlugin.prototype.getZoomFactor = function() {
	return this._zoomFactor;
};
/** @deprecated
* @public
* @function
* @return {number}
*/
tr.grid.ZoomPlugin.prototype.getDefaultZoomFactor = tr.grid.ZoomPlugin.prototype.getZoomFactor;

/** @public
* @return {number} zoomFactor
*/
tr.grid.ZoomPlugin.prototype.zoomIn = function() {
	this["zoomFactor"] = this["zoomFactor"] + this._step;
	return this._zoomFactor;
};
/** @public
* @return {number} zoomFactor
*/
tr.grid.ZoomPlugin.prototype.zoomOut = function() {
	this["zoomFactor"] = this["zoomFactor"] - this._step;
	return this._zoomFactor;
};
/** @public
 * @param {number=} factor
 * @return {number} factor
 */
tr.grid.ZoomPlugin.prototype.zoomTo = function(factor) {
	if (!factor) { // 0, NaN, undefined, empty string, and null
		factor = 1;
	} else {
		if(typeof factor != "number") {
			factor = parseFloat(factor);
		}
		if (factor < 0.6) {
			factor = 0.6;
		} else if (factor > 2.5) {
			factor = 2.5;
		}
	}
	if (this._zoomFactor === factor) {
		return factor;
	}

	var prevZoomFactor = this._zoomFactor;
	this._zoomFactor = factor;
	
	// TODO: Refactor the logic, so that the zoom styles are applied directly to the parent element, eliminating margin-right styling
	var i;
	for(i = this._elems.length; --i >= 0;) {
		this._applyZoom(this._elems[i]);
	}
	for(i = this._hosts.length; --i >= 0;) {
		this._applyZoom(this._hosts[i]); // Due to possible virtualization, all grids need to be updated
	}

	//trigger event
	this._dispatch(tr.res.zoomChanged, {
		"zoomFactor":  this._zoomFactor,
		"prevZoomFactor": prevZoomFactor
	});
	return this._zoomFactor;
};
/** @public
 * @function
 * @param {number=} zoomFactor
 * @return {number} zoomFactor
 */
tr.grid.ZoomPlugin.prototype.zoomDefault = tr.grid.ZoomPlugin.prototype.zoomTo;

/** @public
 * @function
 * @param {Element} elem
 */
tr.grid.ZoomPlugin.prototype.addElement = function(elem) {
	if(elem && this._elems.indexOf(elem) < 0) {
		this._applyZoom(elem);
		this._elems.push(elem);
	}
};
/** @public
 * @function
 * @param {Element} elem
 */
tr.grid.ZoomPlugin.prototype.removeElement = function(elem) {
	var at = this._elems.indexOf(elem);
	if(at >= 0) {
		this._elems.splice(at, 1);
		this._clearZoomStyles(elem);
	}
	return elem;
};

/** @private
 */
tr.grid.ZoomPlugin.prototype._updateElementLayout = function(elem) {
	if(this._zoomFactor !== 1) {
		// After scaling there is empty space between elem and it's parent
		// So it needs to correct the transformation shift of an inline-element
		elem.style["marginRight"] = elem.offsetWidth * (this._zoomFactor - 1) + "px";
	} else {
		elem.style["marginRight"] = "";
	}
};
/** @public
 */
tr.grid.ZoomPlugin.prototype.updateZoomedLayout = function() {
	this._updateZoomedLayout();
};
/** @private
 */
tr.grid.ZoomPlugin.prototype._updateZoomedLayout = function() {
	var i, elem;
	for(i = this._elems.length; --i >= 0;) {
		elem = this._elems[i];
		this._updateElementLayout(elem);
	}
	for(i = this._hosts.length; --i >= 0;) {
		elem = this._hosts[i].getElement(); // Due to possible virtualization, all grids need to be updated
		this._updateElementLayout(elem);
	}
};
/** @private
 * @param {Element|tr.Grid} obj
 */
tr.grid.ZoomPlugin.prototype._applyZoom = function(obj) {
	if(!obj) { return; }
	
	var elem = (obj instanceof tr.Grid) ? /** @type{Element} */(obj.getElement()) : /** @type{Element} */(obj);
	
	// This type of css scale is most compatable options
	// alternative css can be more promising but not the most copatable (like: zoom)
	if (this._zoomFactor !== 1) {
		elem.style["transform"] =  "scale(" + this._zoomFactor + ")";
		elem.style["transformOrigin"] = "0 0 0";
		elem.style["overflow"] = "initial";
	} else { // Clear
		this._clearZoomStyles(elem);
	}
	this._updateElementLayout(elem);
	
	if (obj instanceof tr.Grid) {
		obj["zoomFactor"] =  this._zoomFactor; // Scroll event is fired
	} 
};
/** @private
 * @param {Element} elem
 */
tr.grid.ZoomPlugin.prototype._clearZoomStyles = function(elem) {
	var domStyle = /** @type{CSSStyleDeclaration} */(elem.style);
	domStyle["transform"] = "";
	domStyle["transformOrigin"] = "";
	domStyle["marginRight"] = "";
	domStyle["overflow"] = "";
};

/** @public
* @param {boolean=} bool
*/
tr.grid.ZoomPlugin.prototype.enableWheelToZoom = function(bool) { 
	bool = bool !== false;
	if(this._wheel !== bool) {
		this._wheel = bool;
		for(var i = this._hosts.length; --i >= 0;) {
			if(bool) {
				this._hosts[i].listen("wheel", this._onWheel);
			} else {
				this._hosts[i].unlisten("wheel", this._onWheel);
			}
		}
	}
};
/** @private
* @param {Event} e
*/
tr.grid.ZoomPlugin.prototype._onWheel = function(e) {
	if(!e.ctrlKey || e.shiftKey || e.altKey) { return; } // Only handle Ctrl+wheel
	
	var delta = e["deltaY"]; // Provided by WheelEvent >> https://developer.mozilla.org/en-US/docs/Web/API/WheelEvent
	if(!delta) { return; }
	
	tr._preventDefault(e); // Disable browser default zoom behaviour
	if(delta > 0) {
		this.zoomOut();
	} else {
		this.zoomIn();
	}
};
/** @public
* @return {function(Event)} Mouse wheel logic that is attached to this instance
*/
tr.grid.ZoomPlugin.prototype.getDefaultMouseWheelLogic = function () {
	return this._onWheel;
};
