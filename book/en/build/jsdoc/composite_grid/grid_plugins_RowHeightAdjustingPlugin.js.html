goog.provide("tr.grid.RowHeightAdjustingPlugin");

goog.require("tr.res");
goog.require("tr.ILayoutGrid");

goog.require("tr.grid.IPlugin");
goog.require("tr.grid.Plugin");

goog.require("tr.grid.SectionSettings");
goog.require("tr.grid.IndexMarker");
goog.require("tr.grid.Cell");

/** @constructor
* @param {Array.<number|boolean>|null=} opt_restrictions By default all columns will be processed by this plugin
* @param {Array.<number|boolean>|null=} opt_exclusions By default no column will be excluded by this plugin
* @extends {tr.grid.Plugin}
* @implements {tr.grid.IPlugin}
*/
tr.grid.RowHeightAdjustingPlugin = function (opt_restrictions, opt_exclusions) {
	var _t = this;
	_t._onPreSectionRender = _t._onPreSectionRender.bind(_t);
	_t._onContentWidthChanged = _t._onContentWidthChanged.bind(_t);
	_t._onPostSectionDataBinding = _t._onPostSectionDataBinding.bind(_t);
	_t._onColumnMoved = _t._onColumnMoved.bind(_t);
	_t._onColumnAdded = _t._onColumnAdded.bind(_t);
	_t._onColumnRemoved = _t._onColumnRemoved.bind(_t);
	_t._forceCalculations = [];

	_t._marker = new grid.IndexMarker(opt_restrictions, opt_exclusions);
};
goog.inherits(tr.grid.RowHeightAdjustingPlugin, tr.grid.Plugin);
/** @type {boolean}
* @private
*/
tr.grid.RowHeightAdjustingPlugin.prototype._enabled = true;
/** @type {boolean}
* @private
*/
tr.grid.RowHeightAdjustingPlugin.prototype._calculating = false;
/** @type {boolean}
* @private
*/
tr.grid.RowHeightAdjustingPlugin.prototype._widthChanging = false;

/** @type {boolean}
* @private
*/
tr.grid.RowHeightAdjustingPlugin.prototype._allSections = false;
/** @type {Object<string,boolean>}
* @private
*/
tr.grid.RowHeightAdjustingPlugin.prototype._forceCalculations = null;


//#region IPlugin
/** @override */
tr.grid.RowHeightAdjustingPlugin.prototype.getName = function () {
	return "RowHeightAdjustingPlugin"; // Read Only
};
/** @override */
tr.grid.RowHeightAdjustingPlugin.prototype.initialize = function (host) {
	this._host = host;

	this._host.listen("preSectionRender", this._onPreSectionRender);
	this._host.listen("widthChanged", this._onContentWidthChanged);
	this._host.listen(tr.res.postSectionDataBinding, this._onPostSectionDataBinding);

	this._host.listen(tr.res.columnMoved, this._onColumnMoved);
	this._host.listen(tr.res.columnAdded, this._onColumnAdded);
	this._host.listen(tr.res.columnRemoved, this._onColumnRemoved);
};
/** @override */
tr.grid.RowHeightAdjustingPlugin.prototype.unload = function (host) {
	this._host.unlisten("preSectionRender", this._onPreSectionRender);
	this._host.unlisten("widthChanged", this._onContentWidthChanged);
	this._host.unlisten(tr.res.postSectionDataBinding, this._onPostSectionDataBinding);

	this._host.unlisten(tr.res.columnMoved, this._onColumnMoved);
	this._host.unlisten(tr.res.columnAdded, this._onColumnAdded);
	this._host.unlisten(tr.res.columnRemoved, this._onColumnRemoved);

	this._host = null;
};
//#endregion IPlugin

//#region Public methods
/** Specify column to be processed by this plugin
* @public
* @return {tr.grid.IndexMarker}
*/
tr.grid.RowHeightAdjustingPlugin.prototype.getMarker = function () {
	return this._marker;
};

/** @public
* @return {boolean}
*/
tr.grid.RowHeightAdjustingPlugin.prototype.adjustAllRows = function () {
	if (this._host === null) { return false; }
	if(this._calculating) { return false; }
	
	this._calculating = true;
	var prevFrozenState = this._host.isLayoutFrozen();
	var frozen;
	if(!prevFrozenState && !this._widthChanging) {
		frozen = this._host.freezeLayout();
	}
	var i = this._host.getSectionCount();
	var dirty = false;
	while (--i >= 0) {
		dirty |= this.adjustAllRowsInSection(i);
	}
	if(!prevFrozenState && !this._widthChanging) {
		this._host.freezeLayout(frozen);
	}
	this._calculating = false;
	return /** @type {boolean} */ (dirty);
};

/** @public
* @param {tr.Grid.SectionReference} sectionRef
* @param {number=} from
* @param {number=} to
* @return {boolean}
*/
tr.grid.RowHeightAdjustingPlugin.prototype.adjustAllRowsInSection = function (sectionRef, from, to) {
	if (!this._host) { return false; }

	var sectionSettings = this._host.getSectionSettings(sectionRef);
	if (!sectionSettings) { return false; }

	var sectionType = sectionSettings.getType();
	if (!this._allSections && sectionType !== tr.res.content) {
		return false;
	}
	var forceCalculation = (this._forceCalculations[sectionType] === true);

	var dirty = false;
	var layout = sectionSettings.getSection();
	if(from === undefined) {
		from = 0;
	}
	if(to === undefined) {
		to = layout.getRowCount();
	}
	var prevFrozenState = layout.getFrozenLayout();
	if(!prevFrozenState && to - from > 1) {
		layout.setFrozenLayout(true);
	}
	for(var i = from; i < to; ++i) {
		dirty |= this._adjustRowHeightAt(layout, i, forceCalculation);
	}
	if(!prevFrozenState && to - from > 1) {
		layout.setFrozenLayout(false);
	}
	return /** @type {boolean} */ (dirty);
};

/** @public
* @param {tr.grid.SectionSettings} sectionSettings
* @param {number} rowIndex
* @return {boolean} true if the row height changes
*/
tr.grid.RowHeightAdjustingPlugin.prototype.adjustRowHeightAt = function (sectionSettings, rowIndex) {
	var forceCalculation = (this._forceCalculations[sectionSettings.getType()] === true);
	if (this._adjustRowHeightAt(sectionSettings.getSection(), rowIndex, forceCalculation)) {
		return true;
	}
	return false;
};

/** @public
* @param {boolean} enable
*/
tr.grid.RowHeightAdjustingPlugin.prototype.enableAutomaticAdjustment = function (enable) {
	if (this._enabled !== enable) {
		this._enabled = enable;
		if (this._enabled) {
			this.adjustAllRows();
		}
	}
};

/** @public
* @param {boolean} bool
*/
tr.grid.RowHeightAdjustingPlugin.prototype.includeAllSections = function (bool) {
	this._allSections = (bool !== false);
};
/** @public
* @param {string|Array.<string>|null=} types
*/
tr.grid.RowHeightAdjustingPlugin.prototype.includeAllColumnsFor = function (types) {
	this._forceCalculations = {};
	if (!types) { return; }

	var list = (typeof types === "string") ? [types] : types;
	
	for (var i = list.length; --i >= 0; ) {
		this._forceCalculations[list[i]] = true;
	}
};
//#endregion

//#region Private methods
/** @private
* @param {Object} e
*/
tr.grid.RowHeightAdjustingPlugin.prototype._onPreSectionRender = function (e) {
	var sectionSettings = /** @type{tr.grid.SectionSettings} */(e[tr.res.sectionSettings]);
	if (!this._allSections && sectionSettings.getType() !== tr.res.content) {
		return;
	}

	var section = sectionSettings.getSection();
	if (this._forceCalculations[sectionSettings.getType()] === true) {
		section.enableClass("wrap", true);
	} else {
		var toC = e[tr.res.toColIndex];
		for (var c = e[tr.res.fromColIndex]; c < toC; ++c) {
			if (this._marker.isAllowed(c)) {
				section.enableColumnClass(c, "wrap", true);
			}
		}
	}
};


/** @private
* @param {Object} e
*/
tr.grid.RowHeightAdjustingPlugin.prototype._onContentWidthChanged = function (e) {
	this._widthChanging = true;
	this.adjustAllRows();
	this._widthChanging = false;
};
/** @private
* @param {Object} e
*/
tr.grid.RowHeightAdjustingPlugin.prototype._onPostSectionDataBinding = function (e) {
	if(!this._enabled) { return; }
	if(this._calculating) { return; }
	
	this._calculating = true;
	
	var sectionSettings = e[tr.res.sectionSettings];
	if (!this._allSections && sectionSettings.getType() !== tr.res.content) { 
		return;
	}
	this.adjustAllRowsInSection(sectionSettings, e[tr.res.fromRowIndex], e[tr.res.toRowIndex]);
	
	this._calculating = false;
};

/**
* @private
* @param {number} height
* @return {number}
*/
tr.grid.RowHeightAdjustingPlugin.prototype._getSteppingHeight = function (height) {
	return Math.floor((height + 1) / 4) * 4 + 8;
};

/** @private
* @param {tr.ILayoutGrid} layout
* @param {number} rowIndex
* @param {boolean=} opt_force
* @return {boolean} true if the row height changes
*/
tr.grid.RowHeightAdjustingPlugin.prototype._adjustRowHeightAt = function (layout, rowIndex, opt_force) {
	var maxHeight = layout.getDefaultRowHeight();

	//Find maximum row height from all monitored columns
	var colCount = layout.getColumnCount();
	for (var c = 0; c < colCount; ++c) {
		if (opt_force !== true && !this._marker.isAllowed(c)) {
			continue; 
		}

		var contentHeight = 0; 
		var cell = layout.getCell(c, rowIndex);
		if(cell && cell.isVisible()) {
			var cellContent = cell.getContent();
			if(cellContent) {
				contentHeight = this._getSteppingHeight(cellContent.offsetHeight);
			}
		}
		if (contentHeight > maxHeight) { 
			maxHeight = contentHeight;
		}
	}
	if (layout.getRowHeight(rowIndex) !== maxHeight) {
		layout.setRowHeight(rowIndex, maxHeight);
		return true;
	}
	return false;
};
