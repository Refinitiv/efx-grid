goog.provide("tr.AdFinSubscription");

goog.require("tr.HttpRequest");
goog.require("tr.EventManager");

/**
 * @constructor
 * @extends {tr.EventManager}
 */
tr.AdFinSubscription = function() {
	this._cache = {};
	this._queue = [];
	this.onAdFinUpdate = this.onAdFinUpdate.bind(this);
	this._clearCache = this._clearCache.bind(this);
	this._processQueue = this._processQueue.bind(this);
};
goog.inherits(tr.AdFinSubscription, tr.EventManager);

/** @typedef {{Function:string, Parameters:Array.<Array.<string|number|null>>}} */
tr.AdFinSubscription.Request;

/** @type {Object.<string, {pending:boolean,value:*}>}
 * @description call signature -> value
 * @private
 */
tr.AdFinSubscription.prototype._cache = null;
/** @private
 * @type {Array.<tr.AdFinSubscription.Request>}
 */
tr.AdFinSubscription.prototype._queue = null;
/** @private
 * @type {string}
 */
tr.AdFinSubscription.prototype._state = "stop";
/** @private
* @type {string}
*/
tr.AdFinSubscription.prototype._userId = "GUEST";
/** @private
* @type {string}
*/
tr.AdFinSubscription.prototype._productId = "TMON";
/** @private
* @type {string}
*/
tr.AdFinSubscription.prototype._adfinUrl = ((window && window.location && window.location.origin)
	? window.location.origin
	: "" ) + "/tracsweb/plugin.svc/adf/CompositeMulti/json2";
/** @private
* @type {number|null}
*/
tr.AdFinSubscription.prototype._intervalId = null;
/** @private
* @type {number}
*/
tr.AdFinSubscription.prototype._intervalTime = 3600000; // 1 Hour
/** @private
* @type {number|null}
*/
tr.AdFinSubscription.prototype._timerId = null;
/** @private
* @type {number}
*/
tr.AdFinSubscription._chunckSize = 50;
/** @public
 * @description Conflator delay (in ms)
 * @type {number}
 */
tr.AdFinSubscription.ConflatorDelay = 1000;

//#region public function of Adfin service
/** @public
 * @description Set the settings of Adfin
 * @param {string} userId UUID
 * @param {string} productId Application name
 * @param {string} url Url of Adfin plugin
 */
tr.AdFinSubscription.prototype.setAdFinSettings = function (userId, productId, url) {
	if(userId) {
		this._userId = userId;
	}
	if(productId) {
		this._productId = productId;
	}
	if(url) {
		this._adfinUrl = url;
	}
};
/** @public
 * @description Set the interval time for retrieving data
 * @param {number} time Millisecond
 */
tr.AdFinSubscription.prototype.setIntervalTime = function(time) {
	time = +time;
	if(!time) { // NaN or 0
		return;
	}
	this._intervalTime = time;
	this._stop();
	this._start();
};
/** @public
 * @description Subscribe symbol to the specified field. This method will start the interval for retrieving data automatically.
 * @param {string} name the name of the function
 * @param {Array.<Array.<string|number|null>>} parameters the parameters of the function
 * @return {*} Return the value if there is existing. Return undefined when requesting the invalid field.
 */
tr.AdFinSubscription.prototype.call = function (name, parameters) {
	if(!name || !parameters) {
		throw "Invalid parameters";
	}

	for(var i = 0, length = parameters.length; i < length; ++i) {
		if(parameters[i][1] === "...") {
			return "...";
		}
	}

	var request = /** @type {tr.AdFinSubscription.Request} */({"Function":name, "Parameters":parameters});

	var signature = JSON.stringify(request);
	var entry = this._cache[signature];
	if(entry && (entry["pending"] === false)) {
		return entry["value"];
	}
	if(!entry) {
		this._cache[signature] = {
			"pending":true,
			"value":"..."
		};
		this._addQueue([request]);
	}

	return "...";
};
//#endregion public function of Adfin service

 * @param {string} type Event name
 * @param {Function} handler Event handler

tr.AdFinSubscription.prototype.addEventListener = function(type, handler) {
	tr.AdFinSubscription.base(this, "addEventListener", type, handler);
	this._start();
};
/** @override
 * @description Remove the function handler from the specific event. If there is not any event handlers, the service will stop automatically.
 * @param {string} type Event name
 * @param {Function} handler Event handler
 */
tr.AdFinSubscription.prototype.removeEventListener = function(type, handler) {
	tr.AdFinSubscription.base(this, "removeEventListener", type, handler);
	if(!this._hasListener()) {
		this._stop();
		this._cache = {};
		this._queue = [];
	}
};
/** @override
 * @description Remove all function handlers of all events and stop the interval of retrieving data of service.
 */
tr.AdFinSubscription.prototype.removeAllEventListeners = function() {
	tr.AdFinSubscription.base(this, "removeAllEventListeners");
	this._stop();
	this._cache = {};
	this._queue = [];
};
//#endregion public function of event


//#region private function of Adfin service
/** @private
 * @description Start the interval of retrieving data
 */
tr.AdFinSubscription.prototype._start = function() {
	if(this._state === "start") {
		return;
	}
	this._intervalId = window.setInterval(this._clearCache, this._intervalTime);
	this._state = "start";
};
/** @private
 * @description Stop the interval of retrieving data
 */
tr.AdFinSubscription.prototype._stop = function() {
	if(this._timerId !== null) {
		window.clearTimeout(this._timerId);
	}
	if(this._intervalId !== null) {
		window.clearInterval(this._intervalId);
	}
	this._timerId = null;
	this._intervalId = null;
	this._state = "stop";
};
/** @private
 * @description Clear the cache
 */
tr.AdFinSubscription.prototype._clearCache = function() {
	this._cache = {};
};
/** @private
 * @description Manage the queue to retrieve the Adfin results. Prevent to call the same Adfin function multiple time with same parameters.
 * @param {Array.<tr.AdFinSubscription.Request>} requests
 */
tr.AdFinSubscription.prototype._addQueue = function(requests) {
	if(this._timerId !== null) {
		window.clearTimeout(this._timerId);
	}

	if(!Array.isArray(requests)) {
		requests = [requests];
	}
	if(!this._queue) {
		this._queue = [];
	}
	this._queue = this._queue.concat(requests);
	this._timerId = window.setTimeout(this._processQueue, tr.AdFinSubscription.ConflatorDelay);
};
/** @private
 * @description Process the data queue to retrieve the Adfin results.
 */
tr.AdFinSubscription.prototype._processQueue = function() {
	/** @type {Array.<tr.AdFinSubscription.Request>} */ var chunckQueue = [];
	for(var i = 0, length = this._queue.length; i < length; ++i) {
		chunckQueue.push(this._queue[i]);
		if(chunckQueue.length === tr.AdFinSubscription._chunckSize) {
			this._requestAdFin(chunckQueue);
			chunckQueue = [];
		}
	}
	if(chunckQueue.length) {
		this._requestAdFin(chunckQueue);
		chunckQueue = [];
	}
	this._queue = []; // Clear queue
	this._timerId = null; // Clear timeout id
};
/** @private
 * @description Publish the Adfin request to the Adfin plugin.
 * @param {Array.<tr.AdFinSubscription.Request>} requests AdFin functions to be called (with parameters)
 */
tr.AdFinSubscription.prototype._requestAdFin = function(requests) {

	var http = new HttpRequest(this._adfinUrl, requests);
	http.send({
        "X-TRACS-ApplicationId": this._productId,       // Application ID for the Thin Monitor
        "X-TRACS-UserId": this._userId                  // UUID
    })
	.then(function(json) {
		json["signatures"] = [];
		json["functions"] = [];
		for(var i = 0, length = requests.length; i < length; ++i) {
			json["signatures"].push(JSON.stringify(requests[i]));		// Attach the signature of the requests
			json["functions"].push(requests[i]["Function"]);			// Attach the function of the requests
		}
		return json;
	})
	.then(this.onAdFinUpdate)
	.catch(function(err) {
//		console.error(err);
	});
};
/** @private
 * @description Function for processing the Adfin response and publish the event to the listener.
 * @param {Object} json JSON object of Adfin response
 */
tr.AdFinSubscription.prototype.onAdFinUpdate = function(json) {
	var signatures = json["signatures"];
	var functions = json["functions"];
	var toDispatch = {};
	for(var i = 0, length = signatures.length; i < length; ++i) {
		var value;
		var signature = signatures[i];
		var func = functions[i];
		var entry = json[i];
		if(!entry) {
			continue;
		}
		for(var j = 0, length2 = entry.length; j < length2; ++j) {
			if((entry[j].length >= 1) && (typeof (entry[j][0]) === "number")) {
				value = entry[j][0];
				break;
			} else if((entry[j].length === 2) && (typeof(entry[j][0]) === "string")) {
				value = entry[j][1];
				break;
			}
		}
		if(value === undefined) {
			delete this._cache[signature];
		} else {
			this._cache[signature]["pending"] = false;
			this._cache[signature]["value"] = value;
		}
		toDispatch[func] = true;
	}

	for(var entry in toDispatch) {
		var rowData = {};
		var functionName = entry;
//		rowData[functionName] = value;
		this._dispatch("dataChanged", { "function": functionName, "values": rowData, "sender": this }); // Publish data
	}
};
