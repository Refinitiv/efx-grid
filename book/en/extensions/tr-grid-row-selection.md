## Row Selection

The Row Selection Extension will add the ability to select a single row or multiple rows on Grid. 

```live
<style>
	atlas-blotter {
		height: 169px;
	}
</style>

<atlas-blotter id="grid"></atlas-blotter>

<script>
	var fields = ["companyName", "market", "CF_LAST", "CF_NETCHNG", "industry"];
	var records = tr.DataGenerator.generateRecords(fields, { numRows: 15 });
	var configObj = {
		sorting: {
			sortableColumns: true
		},
		columns: [
			{ name: "Company", field: fields[0] },
			{ name: "Market", field: fields[1], width: 120 },
			{ name: "Last", field: fields[2], width: 100 },
			{ name: "Net. Chng", field: fields[3], width: 100 },
			{ name: "Industry", field: fields[4] }
		],
		staticDataRows: records,
		extensions: [
			new tr.RowSelectionExtension()
		]
	};

	var grid = document.getElementById("grid");
	grid.config = configObj;
</script>
```

### basedOnContent mode

Normally, when content is sorted and a row changes position, row selection will not change position like the content does (`try example above`). But if you do want the row selection to relocate itself with the content, you can turn on basedOnContent mode.

For actual result, try to select one or more rows, then click the column header to sort data and see what happens with row selection.

```live
<style>
	atlas-blotter {
		height: 169px;
	}
</style>

<atlas-blotter id="grid"></atlas-blotter>

<script>
	var fields = ["companyName", "market", "CF_LAST", "CF_NETCHNG", "industry"];
	var records = tr.DataGenerator.generateRecords(fields, { numRows: 15 });
	var configObj = {
		sorting: {
			sortableColumns: true
		},
		columns: [
			{ name: "Company", field: fields[0] },
			{ name: "Market", field: fields[1], width: 120 },
			{ name: "Last", field: fields[2], width: 100 },
			{ name: "Net. Chng", field: fields[3], width: 100 },
			{ name: "Industry", field: fields[4] }
		],
		staticDataRows: records,
		extensions: [

		],
		rowSelection: {
			basedOnContent: true
```

### Copy text content from seletected rows

You can directly hook [native copy event](https://developer.mozilla.org/en-US/docs/Web/API/Element/copy_event) on Grid element to retrieve data and put it on system clipboard. Use the extension's APIs to get currently selected rows.  

```live
<style>
	html hr {
		margin: 5px;
	}
	atlas-blotter {
		height: 169px;
	}
	textarea {
		width: 100%;
		height: 100px;
	}
</style>

<big>Try copy (<b>Ctrl/Command + C</b>) text and paste the content on the text box below</big>
<hr>
<atlas-blotter id="grid"></atlas-blotter>
<hr>
<textarea placeholder="Paste Clipboard Content Here"></textarea>

<script>
	function onCopy(e) {
		var selectedRowIndices = rowSelExt.getSelectedRows();
		var text = "No Selected Row";
		if(selectedRowIndices) {
			var rows = grid.api.getMultipleRowData(selectedRowIndices);
			var values = rows.map(function getValueInTSV(row) {
				return [
					row["companyName"], 
					row["market"], 
					row["CF_LAST"], 
					row["CF_NETCHNG"], 
					row["industry"]
				].join("\t");
			});
			text = values.join("\n");
		}
		e.clipboardData.setData("text/plain", text);
    e.preventDefault();
	}

	var rowSelExt = new tr.RowSelectionExtension();
	var fields = ["companyName", "market", "CF_LAST", "CF_NETCHNG", "industry"];
	var records = tr.DataGenerator.generateRecords(fields, { numRows: 5 });
	var configObj = {
		sorting: {
			sortableColumns: true
		},
		columns: [
			{ name: "Company", field: fields[0] },
			{ name: "Market", field: fields[1], width: 120 },
			{ name: "Last", field: fields[2], width: 100 },
			{ name: "Net. Chng", field: fields[3], width: 100 },
			{ name: "Industry", field: fields[4] }
		],
		staticDataRows: records,
		extensions: [
			rowSelExt
		],
		rowSelection: {
			basedOnContent: true
		}
	};

	var grid = document.getElementById("grid");
	grid.config = configObj;
	grid.addEventListener("copy", onCopy);
</script>
```
