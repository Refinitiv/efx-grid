Basic Grid creation

To initialize Grid you can put the element tag on the HTML page, just like any other native element. The element works as if it were a div element (that is, an element with 100% width and the same height as its content). 

<!-- For ELF v4 -->
<atlas-blotter></atlas-blotter>

<!-- For ELF v3 -->
<emerald-grid></emerald-grid>

Or you can dynamically create the element at runtime by using a standard native method like document.createElement(tag-name).

// For ELF v4
var blotter = document.createElement("atlas-blotter");
document.body.appendChild(blotter);

// For ELF v3
var grid = document.createElement("emerald-grid");
document.body.appendChild(grid);

The element is used for positioning and sizing the grid in the document. Anything else should be done by passing a configuration object to the grid.

See all available options for atlas-blotter and emerald-grid.

To define how the grid renders its layout and content use the columns property, as shown below:

// For Atlas Blotter
var configObj = {
    columns: [
        {name: "Header Text", field: "For Mapping Data"},
        {name: "Column 2",    field: "Field 2"}
    ] // Two columns will be shown
};

// For Emerald Grid
var configObj = {
    columns: [
        {title: "Header Text", field: "For Mapping Data"},
        {title: "Column 2",    field: "Field 2"}
    ] // Two columns will be shown
};

To define initial or static data for rendering use the staticDataRows for Atlas Blotter or dataModel for Emerald Grid, as shown below:

// For Atlas Blotter
var configObj = {
    staticDataRows: [
        {"Field 1": "a", "Field 2": "b", "Field 3": "c"},
        {"ABC": "d",     "Field 2": "e", "XYZ": "f"}
    ] // Two rows will be shown
};

// For Emerald Grid
var configObj = {
    dataModel: {
        data: [
            {"Field 1": "a", "Field 2": "b", "Field 3": "c"},
            {"ABC": "d",     "Field 2": "e", "XYZ": "f"}
        ] // Two rows will be shown
    }
};

Each column definition should have a link to the data in Grid through the field property for rendering. Doing it this way separates the UI from the data. The data can have any number of columns, regardless of the actual number of columns being displayed in the grid. This means extra information can be stored, such as row metadata, display states, and so on. The same field can also be used to display multiple columns.

The basic Grid configuration object is usually something like the following: 

var records = [
    {companyName: "Company A", market: "Market 1"},
    {companyName: "Company B", market: "Market 2"},
];
// For Atlas Blotter
var configObj = {
    columns: [
        {name: "Company Name", field: "companyName"},
        {name: "Market", field: "market"}
    ],
    staticDataRows: records
};

// For Emerald Grid
var configObj = {
    columns: [
        {title: "Company Name", field: "companyName"},
        {title: "Market", field: "market"}
    ],
    dataModel: {
        data: records
    }
};

Example

<atlas-blotter id="grid"></atlas-blotter>

import './resources/extensions.js'; // This line is only required for demo purpose. It is not relevant for your application.
import { halo } from './theme-loader.js'; // This line is only required for demo purpose. It is not relevant for your application.
await halo(); // This line is only required for demo purpose. It is not relevant for your application.

/* ---------------------------------- Note ----------------------------------
  DataGenerator, Formatters and extensions are exposed to global scope
  in the bundle file to make it easier to create live examples.
  Importing formatters and extensions is still required in your application. 
  Please see the document for further information.
---------------------------------------------------------------------------*/
var fields = ["companyName", "market", "CF_LAST", "CF_NETCHNG", "industry"];
var records = tr.DataGenerator.generateRecords(fields, { seed: 0, numRows: 5 });

// Drag column to move
var columnDraggingExt = new tr.ColumnDraggingExtension();

// Drag column edge to resize the column
var columnResizingExt = new tr.ColumnResizingExtension();

// Double click column edge to fit column to its content
var columnFitterExt = new tr.ColumnFitterExtension();

var configObj = {
    sorting: { // Grid level options
        sortableColumns: true // Make all column sortable
    },
    columns: [
        { name: "Company", field: fields[0] },
        { name: "Market", field: fields[1], width: 120 },
        { name: "Last", field: fields[2], width: 100 },
        { name: "Net. Chng", field: fields[3], width: 100 },
        { name: "Industry", field: fields[4] }
    ],
    staticDataRows: records,
    extensions: [
        columnDraggingExt,
        columnResizingExt,
        columnFitterExt
    ]
};

var grid = document.getElementById("grid");
grid.config = configObj;
