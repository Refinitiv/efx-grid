Simple Toggle Formatter

This formatter creates a toggle column with a default "eye" icon.

Specific options

There are no specific options for this formatter.

Example

atlas-blotter {
    height: 320px;
}

<atlas-blotter id="grid"></atlas-blotter>

var fields = ["toggle", "number_1"];
var records = new Array(40);
for(var i = 0; i < 40; ++i) {
    var record = records[i] = {};
    record[fields[0]] = i % 2 === 0;
    record[fields[1]] = i;
}
var configObj = {
    sorting: {
        sortableColumns: true
    },
    columns: [
        { 
            name: "",
            field: fields[0],
            alignment: "c",
            binding: tr.SimpleToggleFormatter.create()
        },
        { 
            name: "Column 2",
            field: fields[1],
            alignment: "c",
        }
    ],
    staticDataRows: records
};

var grid = document.getElementById("grid");
grid.config = configObj;
