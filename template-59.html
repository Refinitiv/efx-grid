Custom UI for filter input

You can use inputCreated event to replace default filter input by your custom UI element. In this case, you will need to directly interact with Row Filtering Extension and manually wire your custom UI element with filtering logic. The following example shows how to use date range element in place of default filter input:

efx-grid {
    height: 250px;
}

<efx-grid id="grid"></efx-grid>

await (await import("./theme-loader.js")).default; // This line is only required for demo purpose. It is not relevant for using EFX Grid in your application.

/* ---------------------------------- Note ----------------------------------
  DataGenerator, Formatters and extensions are exposed to global scope
  in the bundle file to make it easier to create live examples.
  Importing formatters and extensions is still required in your application. 
  Please see the document for further information.
---------------------------------------------------------------------------*/
var filterInputExt = new FilterInput();
var rowFilteringExt = new RowFiltering();

var dateFormatter = function (e) {
    var value = e.data;
    var cell = e.cell;
    var parts = value.toDateString().split(" ");
    cell.setContent(parts[2] + "-" + parts[1] + "-" + parts[3]);
};
var onInputCreated = function (e) {
    if (e.colIndex !== Date_Index) {
        return;
    }

    var dateRange = document.createElement("ef-datetime-picker");

    dateRange.range = true;
    dateRange.duplex = "split";
    dateRange.addEventListener("value-changed", onDateRangeChange);

    var parentInput = e.input.parentNode;
    parentInput.removeChild(e.input); // Completely ignore default UI input
    parentInput.appendChild(dateRange);
};
var onDateRangeChange = function (e) {
    var dateRange = e.currentTarget;
    var rangeValues = dateRange.values;
    if (rangeValues.length == 2) {
        var startDate = new Date(rangeValues[0]);
        var endDate = new Date(rangeValues[1]);
        rowFilteringExt.setColumnFilter(
            Date_Index,
            myFilterLogic.bind(null, startDate, endDate)
        );
    } else {
        rowFilteringExt.setColumnFilter(Date_Index, null);
    }
};
var myFilterLogic = function (startDate, endDate, rowData) {
    var dateObj = rowData.date;
    return dateObj >= startDate && dateObj <= endDate;
};

const fields = ["date", "country", "currency", "rate"];
const Date_Index = 0;
const currencies = ["PLN", "EUR", "SEK", "GBP", "DKK"];

// Build data information

Using filter input for fetching data

If you want to fetch data from server side based on the value of the filter input, you can achieve this by listening to the event of the filter input element that we provide in the inputCreated event. You can follow the example below for guidance.

efx-grid {
    height: 250px;
}

/* ---------------------------------- Note ----------------------------------
  DataGenerator, Formatters and extensions are exposed to global scope
  in the bundle file to make it easier to create live examples.
  Importing formatters and extensions is still required in your application. 
  Please see the document for further information.
---------------------------------------------------------------------------*/
var fields = ["id", "market", "sentence", "CF_NETCHNG", "CF_LAST"];
var filterInputExt = new FilterInput();
var rowFilteringExt = new RowFiltering();

var configObj = {
    columns: [
        {
            field: fields[0],
            name: "Id",
            width: 70
        },
        {
            field: fields[1],
            name: "Market",
            filterInput: {
                defaultLogic: function () {
                    // Disable default logic (disable filter to use server side filtering)
                    return true;
                }
            }
        },
        {
            field: fields[2],
            name: "Description"
        },
        {
            field: fields[3],
            name: "Net Chng."
        },
        {
            field: fields[4],
            name: "Last"
        }
    ],
    defaultColumnOptions: {
        filterInput: {
            disabled: true
        }
    },
    extensions: [filterInputExt, rowFilteringExt]
};

var userInput = "";
var timerId = 0; 
var grid = document.getElementById("grid");
grid.config = configObj;
requestData();

filterInputExt.addEventListener("inputCreated", function (e) {
    var inputEl = e.input;
    inputEl.addEventListener("value-changed", onFilterInputKeyUp);
});

function onFilterInputKeyUp(e) {
    userInput = e.detail.value;
    userInput = userInput.toLowerCase(); // For easy text comparison
    requestData();
}
function byMatchedMarket(record) {
    var marketText = record.market;
    if(marketText) {
        return marketText.toLowerCase().indexOf(userInput) >= 0;
    }
    return false;
}

// ========== Simulate server side in below ==========
function requestData() {
    if(timerId) { // Prevent excessive data calls
        clearTimeout(timerId);
        timerId = 0;
    }
    
    // Simulate delay from server request and response
    console.log("Requesting data from the server...");
    timerId = setTimeout(onServerResponse, 1000);
}

function onServerResponse() {
    timerId = 0;
    // This part should be done on the server side
    var records = DataGenerator.generateRecords(fields, {
        numRows: 100,
        seed: null
    });

    if (userInput) {
        records = records.filter(byMatchedMarket);
    }
    
    // This part is done on the client side
    console.log("Data received");
    grid.data = records;
}
