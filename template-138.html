Coral Radio Button Formatter

This formatter creates a radio button which will be unchecked by default.

Specific options

Name Type Attributes Description
initialIndex number optional Initial checked row index

Example

efx-grid {
    height: 320px;
}

<efx-grid id="grid"></efx-grid>

import { halo } from './theme-loader.js'; // This line is only required for demo purpose. It is not relevant for your application.
await halo(); // This line is only required for demo purpose. It is not relevant for your application.

/* ---------------------------------- Note ----------------------------------
  DataGenerator, Formatters and extensions are exposed to global scope
  in the bundle file to make it easier to create live examples.
  Importing formatters and extensions is still required in your application. 
  Please see the document for further information.
---------------------------------------------------------------------------*/
var fields = ["number", "word"];
var records = DataGenerator.generateRecords(fields, { seed: 1, numRows: 100 });

var configObj = {
    sorting: {
        sortableColumns: true
    },
    columns: [
        {
            alignment: "c",
            field: "radio",
            width: 34,
            binding: EFRadioButtonFormatter.create({
                initialIndex: 0,
                events: {
                    "click": function(e) {
                        console.log("clicked");
                    }
                }
            })
        },
        {
            name: "Number",
            field: fields[0],
            alignment: "c",
        },
        {
            name: "Description",
            field: fields[1],
            alignment: "c",
        }
    ],
    staticDataRows: records
};

var grid = document.getElementById("grid");
grid.config = configObj;
